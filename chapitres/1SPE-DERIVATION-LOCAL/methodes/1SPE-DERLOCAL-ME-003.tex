\begin{fichemethode}{M3}{Lire ou construire une tangente}
\quandUtiliser{Mots-clés : lire graphiquement $f'(a)$, pente de la tangente, tracer la tangente.}
\pasApas{\begin{enumerate}\item Je place $A(a;f(a))$.\item Je prends deux points de la tangente pour lire sa pente.\item Pour construire, j'utilise un déplacement horizontal simple puis le déplacement vertical imposé par la pente.\end{enumerate}}
\exempleRedige{La tangente en $A(2;1)$ a une pente $\frac12$. \commentaireMarge{J'avance de 2 et je monte de 1.} Je place $(4;2)$ puis je trace la droite passant par ces deux points.}
\pieges{Lire deux points de la courbe au lieu de la droite ; oublier que la tangente passe par $A$ ; inverser montée et déplacement horizontal.}
\verifier{Je contrôle que la droite tracée contient le point de contact.}

\end{fichemethode}
