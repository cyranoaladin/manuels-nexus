\begin{fichemethode}{M2}{Interpréter un nombre dérivé}
\quandUtiliser{Mots-clés : vitesse instantanée, pente au point, coût marginal, nombre dérivé.}
\pasApas{\begin{enumerate}\item Je repère le point ou l'instant $a$.\item Je lis ou utilise $f'(a)$.\item Je donne son sens : pente de tangente ou variation instantanée.\item J'ajoute l'unité adaptée.\end{enumerate}}
\exempleRedige{Une distance $d$ est en mètres et $d'(5)=7$. À $5$ s, la vitesse instantanée est $7$ m/s. \commentaireMarge{Je ne confonds pas cette vitesse avec $d(5)$.}}
\pieges{Ne pas donner $d(5)$ à la place de $d'(5)$ ; ne pas répondre « 7 mètres » ; ne pas parler de vitesse moyenne sans intervalle.}
\verifier{Une vitesse positive correspond à une position qui augmente localement.}

\end{fichemethode}
