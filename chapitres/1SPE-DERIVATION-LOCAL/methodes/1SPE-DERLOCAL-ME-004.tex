\begin{fichemethode}{M4}{Écrire l'équation d'une tangente}
\quandUtiliser{Mots-clés : équation de la tangente en $a$, droite tangente à $\mathcal C_f$.}
\pasApas{\begin{enumerate}\item Je calcule ou lis $f(a)$.\item Je repère $f'(a)$.\item J'écris $y=f(a)+f'(a)(x-a)$.\item Je développe seulement si une forme réduite est demandée.\end{enumerate}}
\exempleRedige{Pour $f(2)=-1$ et $f'(2)=3$, la tangente est $y=-1+3(x-2)=3x-7$. \commentaireMarge{Je vérifie : pour $x=2$, j'obtiens $-1$.}}
\pieges{Oublier $f(a)$ ; écrire $x+a$ ; utiliser $f'(x)$ au lieu de $f'(a)$.}
\verifier{Je remplace $x$ par $a$ : l'ordonnée doit redevenir $f(a)$.}
\sEntrainer{\refExos{M4}}
\end{fichemethode}
