\begin{corrige}{1SPE-DERLOCAL-EX-010}
\textbf{1.} $(1+h)^3=1+3h+3h^2+h^3$.

\textbf{2.} $f(1+h)-f(1)=(1+3h+3h^2+h^3)-1=3h+3h^2+h^3$.

$\dfrac{f(1+h)-f(1)}{h}=\dfrac{3h+3h^2+h^3}{h}=3+3h+h^2$.

\textbf{3.} Lorsque $h$ se rapproche de $0$, $3+3h+h^2$ se rapproche de $3$. Donc $f'(1)=3$.

\textbf{4.} La tangente à la courbe de $f$ au point d'abscisse $1$ a pour coefficient directeur $f'(1)=3>0$. La tangente est donc croissante (elle monte de gauche à droite).
\end{corrige}
