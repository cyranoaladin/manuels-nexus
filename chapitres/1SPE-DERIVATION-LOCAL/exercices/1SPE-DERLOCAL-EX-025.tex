\begin{exercice}{1SPE-DERLOCAL-EX-025}{1}{8}
Soit $f(x)=-x^2+6x-3$. On admet que $f'(x)=-2x+6$.

\begin{enumerate}
  \item Calculer $f(4)$ et $f'(4)$.
  \item En utilisant l'approximation linéaire, donner une valeur approchée de $f(4{,}03)$.
  \item Calculer la valeur exacte de $f(4{,}03)$ et comparer.
\end{enumerate}
\end{exercice}
