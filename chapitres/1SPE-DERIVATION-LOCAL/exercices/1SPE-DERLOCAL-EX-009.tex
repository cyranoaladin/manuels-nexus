\begin{exercice}{1SPE-DERLOCAL-EX-009}{2}{15}
Un objet en chute libre a parcouru une distance $d(t)=5t^2$ mètres après $t$ secondes.

\begin{enumerate}
  \item Calculer $d(3)$ et $d(3{,}1)$. En déduire le taux de variation de $d$ entre $3$ et $3{,}1$.
  \item Exprimer $\dfrac{d(3+h)-d(3)}{h}$ en fonction de $h$ pour $h\ne 0$.
  \item En déduire $d'(3)$.
  \item Interpréter $d'(3)$ dans le contexte de la chute libre. Préciser l'unité.
\end{enumerate}
\end{exercice}
