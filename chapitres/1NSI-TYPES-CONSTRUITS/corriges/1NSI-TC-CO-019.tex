\begin{corrige}{1NSI-TC-EX-019}
\begin{enumerate}
  \item Le programme affiche \lstinline{[3, 1, 2]}.

  Détail du parcours :
  \begin{itemize}
    \item $i=0$ : \lstinline{3} n'est pas dans \lstinline{resultat}, on l'ajoute $\to$ \lstinline{[3]}.
    \item $i=1$ : \lstinline{1} n'est pas dans \lstinline{resultat}, on l'ajoute $\to$ \lstinline{[3, 1]}.
    \item $i=2$ : \lstinline{3} est déjà dans \lstinline{resultat}, on ne l'ajoute pas.
    \item $i=3$ : \lstinline{2} n'est pas dans \lstinline{resultat}, on l'ajoute $\to$ \lstinline{[3, 1, 2]}.
    \item $i=4$ : \lstinline{1} est déjà dans \lstinline{resultat}, on ne l'ajoute pas.
    \item $i=5$ : \lstinline{3} est déjà dans \lstinline{resultat}, on ne l'ajoute pas.
  \end{itemize}

  \item La fonction supprime les doublons d'un tableau en conservant l'ordre de première apparition de chaque élément.

  \item Version parcourant directement les éléments :
\begin{python}
def mystere(tab):
    resultat = []
    for element in tab:
        if element not in resultat:
            resultat.append(element)
    return resultat
\end{python}
\end{enumerate}

Vérification : les résultats ont été confirmés par exécution.
\end{corrige}
