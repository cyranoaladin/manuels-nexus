\begin{python}
def moyennes_glissantes(releves, k):
    if len(releves) < k:
        return []
    resultat = []
    for i in range(len(releves) - k + 1):
        # Calcul de la moyenne des k elements a partir de i
        somme = 0
        for j in range(k):
            somme = somme + releves[i + j]
        resultat.append(somme / k)
    return resultat
\end{python}

\textbf{Principe :} pour chaque position $i$ de 0 à \lstinline{len(releves) - k}, on calcule la moyenne des $k$ éléments consécutifs \lstinline{releves[i]}, \lstinline{releves[i+1]}, \dots, \lstinline{releves[i+k-1]}. Le tableau résultat contient \lstinline{len(releves) - k + 1} valeurs.

\textbf{Exemple détaillé} avec \lstinline{releves = [10, 20, 30, 40, 50]} et $k = 3$ :
\begin{itemize}
  \item $i=0$ : moyenne de $10, 20, 30$ $= 20.0$
  \item $i=1$ : moyenne de $20, 30, 40$ $= 30.0$
  \item $i=2$ : moyenne de $30, 40, 50$ $= 40.0$
\end{itemize}

Vérification : \lstinline{moyennes_glissantes([10, 20, 30, 40, 50], 3)} renvoie \lstinline{[20.0, 30.0, 40.0]}. Résultat confirmé par exécution.
\end{corrige}
