\begin{corrige}{1NSI-TC-EX-002}
\begin{enumerate}
  \item \lstinline{type(station)} renvoie \lstinline{<class 'tuple'>}.
        Le tuple contient 3 éléments, donc \lstinline{len(station)} vaut \lstinline{3}.

  \item \lstinline{station[0]} vaut \lstinline{"Tunis"} (premier élément, indice 0).
        \lstinline{station[2]} vaut \lstinline{10.2} (troisième élément, indice 2).

  \item Non. Un tuple est non mutable : toute tentative de modification provoque
        \lstinline{TypeError: 'tuple' object does not support item assignment}.
        Si l'on a besoin de modifier une valeur, il faut construire un nouveau tuple
        ou utiliser une liste.
\end{enumerate}
\end{corrige}
