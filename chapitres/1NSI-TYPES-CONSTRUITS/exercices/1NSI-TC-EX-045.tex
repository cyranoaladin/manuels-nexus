\begin{exercice}{1NSI-TC-EX-045}{3}{20}
On dispose d'une liste de dictionnaires représentant des stations météo, chacun
ayant les clés \lstinline{"nom"}, \lstinline{"region"} et \lstinline{"temp"}.

\begin{enumerate}
  \item Écrire une fonction \lstinline{stations_par_region(stations)} qui renvoie un
        dictionnaire dont les clés sont les régions et les valeurs sont les listes
        des noms de stations dans chaque région.
  \item Compléter par une fonction \lstinline{region_la_plus_chaude(stations)} qui
        renvoie le nom de la région ayant la température moyenne la plus élevée.
\end{enumerate}

\textbf{Exemple :}
\begin{console}
>>> stations = [{"nom": "Lyon", "region": "ARA", "temp": 30},
...             {"nom": "Paris", "region": "IDF", "temp": 28},
...             {"nom": "Grenoble", "region": "ARA", "temp": 27}]
>>> stations_par_region(stations)
{'ARA': ['Lyon', 'Grenoble'], 'IDF': ['Paris']}
\end{console}
\end{exercice}
