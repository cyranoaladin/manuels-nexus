\begin{exercice}{1NSI-TC-EX-010}{2}{8}
On considère le programme suivant :
\begin{python}
def swap(t):
    return (t[1], t[0])

a = (3, 7)
b = swap(a)
print(a)
print(b)
c = (1, 2, 3)
print(swap(c))
\end{python}
\begin{enumerate}
  \item Donner, sans exécuter, les trois affichages produits.
  \item Le tuple \lstinline{a} est-il modifié par l'appel \lstinline{swap(a)} ? Justifier.
  \item Que se passe-t-il si on appelle \lstinline{swap(c)} avec \lstinline{c = (1, 2, 3)} ?
        Le troisième élément est-il perdu ?
\end{enumerate}
\end{exercice}
