\section{Mutabilité et références}
\label{sec:mutabilite}

\margeAppui{Comprendre la mutabilité est essentiel pour éviter des bugs subtils.
Un objet \emph{mutable} (liste, dictionnaire) peut être modifié en place ;
un objet \emph{non mutable} (tuple, chaîne, entier) ne le peut pas.}

\begin{nxdef}[D11]
  Lorsqu'on écrit \lstinline{b = a} pour un objet mutable, \lstinline{b} ne contient
  pas une copie de \lstinline{a} : les deux variables désignent \textbf{le même objet}
  en mémoire. On dit que \lstinline{b} est un \textbf{alias} de \lstinline{a}.
  Toute modification via l'une des variables est visible via l'autre.
\end{nxdef}

\exemple{L'affectation crée un alias, pas une copie :}

\begin{python}
notes = [12, 15, 9]
copie = notes          # alias : même objet
copie.append(18)
print(notes)           # [12, 15, 9, 18]
print(len(copie))      # 4
\end{python}

\erreurFrequente{Croire que \lstinline{copie = notes} crée une copie indépendante.
C'est l'erreur la plus fréquente en NSI sur les types mutables.}

\begin{nxdef}[D12]
  Pour obtenir une \textbf{copie indépendante} d'une liste, on utilise :
  \begin{itemize}
    \item \lstinline{copie = list(original)} ou \lstinline{copie = original[:]}
      (copie \emph{superficielle}).
    \item \lstinline{copie = [list(ligne) for ligne in original]} pour une grille
      (copie de chaque sous-liste).
  \end{itemize}
\end{nxdef}

\exemple{Copie indépendante :}

\begin{python}
notes = [12, 15, 9]
copie = list(notes)    # copie indépendante
copie.append(18)
print(notes)           # [12, 15, 9] (inchangé)
print(copie)           # [12, 15, 9, 18]
\end{python}

\contreexemple{Copie superficielle insuffisante pour les listes imbriquées :}

\begin{python}
grille = [[1, 2], [3, 4]]
copie = list(grille)       # copie superficielle
copie[0][0] = 99
print(grille)              # [[99, 2], [3, 4]] — modifié !
\end{python}

% BEGIN-TRACE
% grille = [[1, 2], [3, 4]]
% copie = list(grille)
% copie[0][0] = 99
% print(grille)
% EXPECTED
% [[99, 2], [3, 4]]
% END-TRACE

\erreurFrequente{Copier une liste imbriquée seulement au premier niveau (EF4).
Avec \lstinline{list(grille)}, les sous-listes restent partagées.
Pour une copie profonde, copier chaque sous-liste :
\lstinline{[list(ligne) for ligne in grille]}.}

\begin{codereference}{Effet de bord dans une fonction}
\begin{python}
def ajouter_note(bulletin, note):
    """Ajoute une note au bulletin.

    ATTENTION : modifie le bulletin passé en argument
    (effet de bord).
    """
    bulletin.append(note)

notes_ali = [12, 15]
ajouter_note(notes_ali, 18)
print(notes_ali)  # [12, 15, 18]
\end{python}
\end{codereference}

% BEGIN-VERIFY
% def ajouter_note(b, n):
%     b.append(n)

% assert notes == [12, 15, 18]
% END-VERIFY

\margeAppui{Quand une fonction modifie un argument mutable, on parle d'\emph{effet de bord}.
C'est parfois voulu, mais il faut le documenter clairement dans la docstring.}

\approfondissement{L'opérateur \lstinline{is} teste si deux variables désignent le
\emph{même objet} (même adresse mémoire), tandis que \lstinline{==} teste l'égalité
des \emph{valeurs}. Après \lstinline{b = a}, on a \lstinline{b is a} qui vaut
\lstinline{True}. Après \lstinline{b = list(a)}, on a \lstinline{b is a} qui vaut
\lstinline{False} mais \lstinline{b == a} qui vaut \lstinline{True}.}

\alamachine{Crée une liste \lstinline{original = [1, 2, 3]}. Fais un alias
\lstinline{alias = original} et une copie \lstinline{copie = list(original)}.
Modifie \lstinline{alias[0] = 99}. Affiche les trois variables et vérifie :
\lstinline{original} est modifié, \lstinline{copie} ne l'est pas.
Teste \lstinline{alias is original} et \lstinline{copie is original}.}
