\section{Les tableaux (listes Python)}
\label{sec:tableaux}

\margeAppui{En Python, un tableau se représente par une \emph{liste} (\lstinline{list}).
Contrairement au tuple, une liste est \textbf{mutable} : on peut modifier ses éléments.}

\begin{nxdef}[D4]
  Un \textbf{tableau} est une séquence ordonnée d'éléments accessibles par leur indice,
  en commençant à 0. En Python, il est représenté par le type \lstinline{list},
  délimité par des crochets. Un tableau est \textbf{mutable} : on peut modifier, ajouter
  ou supprimer des éléments.
\end{nxdef}

\exemple{Créer et manipuler un tableau de relevés de températures :}

\begin{python}
releves = [18, 20, 19, 22, 17]
print(releves[0])      # 18  (premier élément)
print(releves[-1])     # 17  (dernier élément)
print(len(releves))    # 5
releves[2] = 21        # modification autorisée
print(releves)         # [18, 20, 21, 22, 17]
\end{python}

\contreexemple{Accéder à un indice inexistant provoque une erreur :}

\begin{console}
>>> releves = [18, 20, 19]
>>> releves[3]
IndexError: list index out of range
\end{console}

\erreurFrequente{Indice hors limites (off-by-one). Un tableau de $n$ éléments a des indices
de 0 à $n-1$. L'indice $n$ n'existe pas.}

\begin{nxdef}[D5]
  Les méthodes principales d'une liste :
  \begin{itemize}
    \item \lstinline{t.append(v)} : ajoute \lstinline{v} en fin de liste.
    \item \lstinline{t.pop()} : supprime et renvoie le dernier élément.
    \item \lstinline{t.insert(i, v)} : insère \lstinline{v} à l'indice \lstinline{i}.
    \item \lstinline{v in t} : teste si \lstinline{v} est dans \lstinline{t}.
  \end{itemize}
\end{nxdef}

\begin{codereference}{Parcours d'un tableau}
\begin{python}
releves = [18, 20, 19, 22, 17]

# Parcours par valeur (préféré quand l'indice est inutile)
for temperature in releves:
    print(temperature)

# Parcours par indice (quand on a besoin de la position)
for i in range(len(releves)):
    print(f"Relevé {i} : {releves[i]}")
\end{python}
\end{codereference}

\erreurFrequente{Parcourir les indices quand les valeurs suffisent (EF2).
Écrire \lstinline{for i in range(len(t)): print(t[i])} quand
\lstinline{for v in t: print(v)} suffit. Le parcours par valeur est plus lisible
et moins sujet aux erreurs d'indice.}

\begin{nxdef}[D6]
  Un \textbf{tableau en compréhension} se construit en une seule expression :
  \lstinline{[expression for variable in itérable]} avec un filtre optionnel
  \lstinline{if condition}.
\end{nxdef}

\exemple{Construire un tableau des carrés des entiers de 0 à 9 :}

\begin{python}
carres = [x ** 2 for x in range(10)]
print(carres)  # [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]

# Avec filtre : garder uniquement les carrés pairs
pairs = [x ** 2 for x in range(10) if x % 2 == 0]
print(pairs)   # [0, 4, 16, 36, 64]
\end{python}

\margeAppui{La compréhension remplace avantageusement une boucle \lstinline{for}
avec \lstinline{append} : elle est plus concise et plus rapide.}

\begin{codereference}{Moyenne d'un tableau}
\begin{python}
def moyenne(notes):
    """Renvoie la moyenne d'un tableau de nombres.

    Précondition : notes est un tableau non vide de nombres.
    Lève ValueError si le tableau est vide.
    """
    if len(notes) == 0:
        raise ValueError("le tableau est vide")
    return sum(notes) / len(notes)

print(moyenne([12, 14, 16]))  # 14.0
\end{python}
\end{codereference}
