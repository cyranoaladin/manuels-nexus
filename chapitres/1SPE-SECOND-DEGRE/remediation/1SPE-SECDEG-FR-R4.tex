\subsection*{Rappel — Fonctions de reference}

\textbf{Fonction carre} : $f(x) = x^2$, definie sur $\mathbb{R}$.
\begin{itemize}
  \item Decroissante sur $]-\infty\,;\,0]$, croissante sur $[0\,;\,+\infty[$.
  \item Minimum en $x = 0$ : $f(0) = 0$.
  \item Parabole symetrique par rapport a l'axe $Oy$.
\end{itemize}

\textbf{Translations} : la courbe de $f(x-\alpha) + \beta$ est obtenue en translatant celle de $f$ de vecteur $(\alpha\,;\,\beta)$.

Donc $y = (x-\alpha)^2 + \beta$ est une parabole de sommet $(\alpha\,;\,\beta)$.

\begin{enumerate}

  \item Verifier que $f(-x) = f(x)$ pour tout $x$ (symetrie).
  \item Montrer que $f$ est decroissante sur $]-\infty\,;\,0]$ en calculant $f(a) - f(b)$ pour $a < b \leq 0$.
  \item Tracer l'allure de la courbe (parabole en U de sommet $O$).
\end{enumerate}
\end{exercice}

\textbf{Question 2.} $f(-x) = (-x)^2 = x^2 = f(x)$ pour tout $x \in \mathbb{R}$ : la courbe est symetrique par rapport a l'axe $Oy$.

\textbf{Question 3.} Soient $a < b \leq 0$. Alors :
\[
f(b) - f(a) = b^2 - a^2 = (b-a)(b+a).
\]
On a $b - a > 0$ (car $b > a$) et $b + a \leq 0$ (car $a < b \leq 0$, donc $a + b < 0$). Donc $(b-a)(b+a) \leq 0$, i.e. $f(b) \leq f(a)$ : $f$ est decroissante sur $]-\infty\,;\,0]$.

\textbf{Question 4.} (Trace libre — parabole en U, sommet $O(0\,;\,0)$, passant par $(\pm 1\,;\,1)$, $(\pm 2\,;\,4)$.)

\end{corrige}

\begin{corrige}{1SPE-SECDEG-FR-R4-EX3}

\textbf{Question 1.} $a = 1$, $b = -4$, $c = 7$.
$\alpha = -(-4)/(2 \times 1) = 2$.
$\beta = f(2) = 4 - 8 + 7 = 3$.
Forme canonique : $f(x) = (x-2)^2 + 3$.

\textbf{Question 2.} $a = 1$, $b = 6$, $c = 11$.
$\alpha = -6/2 = -3$.
$\beta = g(-3) = 9 - 18 + 11 = 2$.
Forme canonique : $g(x) = (x+3)^2 + 2$.

\commentaireMarge{On factorise par $a = 2$ avant de completer le carre.}

\textbf{Question 3.} $h(x) = 2(x^2 - 6x + 10)$.
Dans $x^2 - 6x + 10$ : $\alpha = 3$, $\beta = 9 - 18 + 20 = 11$... Retour : $\alpha = -(-6)/(2 \times 1) = 3$, $f_{\text{int}}(3) = 9 - 18 + 10 = 1$, donc $x^2 - 6x + 10 = (x-3)^2 + 1$.
Ainsi : $h(x) = 2\bigl[(x-3)^2 + 1\bigr] = 2(x-3)^2 + 2$.

\end{corrige}

\begin{exercice}{1SPE-SECDEG-FR-R4-EX4}{1}{6}

\begin{enumerate}
  \item Montrer que $f(x) = x^2 - 6x + 10 \geq 1$ pour tout $x \in \mathbb{R}$.
  \item Trouver la valeur de $x$ qui minimise $g(x) = (x-1)^2 + 4$, et preciser la valeur minimale.
  \item Montrer que $h(x) = -3(x+2)^2 + 1 \leq 1$ pour tout $x$, et preciser quand l'egalite a lieu.

\begin{corrige}{1SPE-SECDEG-FR-R4-EX4}

\textbf{Question 1.} $f(x) = x^2 - 6x + 10 = (x-3)^2 + 1$. Comme $(x-3)^2 \geq 0$, on a $f(x) \geq 1$ pour tout $x \in \mathbb{R}$. L'egalite a lieu pour $x = 3$ : $f(3) = 1$.

\textbf{Question 2.} $g(x) = (x-1)^2 + 4 \geq 4$ pour tout $x$ (car $(x-1)^2 \geq 0$). Le minimum est $g(1) = 4$, atteint en $x = 1$.

\textbf{Question 3.} $h(x) = -3(x+2)^2 + 1$. Comme $(x+2)^2 \geq 0$, on a $-3(x+2)^2 \leq 0$, donc $h(x) \leq 1$ pour tout $x$. L'egalite $h(x) = 1$ a lieu quand $(x+2)^2 = 0$, c'est-a-dire pour $x = -2$.

\end{corrige}
