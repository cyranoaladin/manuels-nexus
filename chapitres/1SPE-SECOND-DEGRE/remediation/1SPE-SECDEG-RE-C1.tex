\begin{exercice}{1SPE-SECDEG-RE-C1-EX1}{1}{8}
Soit $f(x) = 3x^2 - 12x + 9$.
\begin{enumerate}
  \item Identifier $a$, $b$, $c$.
  \item Calculer $\alpha = -b/(2a)$ et $\beta = f(\alpha)$.
  \item Ecrire la forme canonique de $f$.
  \item Developper pour verifier que la forme canonique est bien equivalente a la forme developpee.
\end{enumerate}
\end{exercice}

% assert alpha == 2

% canon = 3*(x - 2)**2 - 3
% assert expand(canon) == f

\begin{corrige}{1SPE-SECDEG-RE-C1-EX1}

\commentaireMarge{Identifier $a$, $b$, $c$ dans $ax^2 + bx + c$ : le coefficient de $x^2$ est $a$.}

\textbf{Question 1.} On a $f(x) = 3x^2 - 12x + 9$, donc $a = 3$, $b = -12$, $c = 9$.

\commentaireMarge{Formule : $\alpha = -b/(2a) = -(-12)/(2 \times 3)$.}

\textbf{Question 2.}
\[
\alpha = \frac{-b}{2a} = \frac{-(-12)}{2 \times 3} = \frac{12}{6} = 2.
\]
\[
\beta = f(2) = 3 \times 4 - 12 \times 2 + 9 = 12 - 24 + 9 = -3.
\]

\commentaireMarge{Forme canonique : $f(x) = a(x - \alpha)^2 + \beta$.}

\textbf{Question 3.}
\[
f(x) = 3(x - 2)^2 - 3.
\]

\textbf{Question 4. Verification :}
\[
3(x-2)^2 - 3 = 3(x^2 - 4x + 4) - 3 = 3x^2 - 12x + 12 - 3 = 3x^2 - 12x + 9. \quad \checkmark
\]

\end{corrige}

\begin{exercice}{1SPE-SECDEG-RE-C1-EX2}{1}{8}
Soit $g(x) = -2x^2 + 8x - 5$.
\begin{enumerate}
  \item Donner la forme canonique de $g$.
  \item Ecrire sous forme developpee l'expression $-2(x - 2)^2 + 3$, et verifier que c'est bien $g(x)$.
  \item La parabole representant $g$ est-elle en U ou en $\cap$ ? Justifier.
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Pour calculer $\alpha$ avec $a = -2$ : $\alpha = -b/(2a) = -8/(2 \times (-2)) = -8/(-4) = 2$. Attention au signe de $a$.}

% BEGIN-VERIFY
% from sympy import symbols, expand, Rational
% x = symbols('x')
% g = -2*x**2 + 8*x - 5
% a, b, c = -2, 8, -5
% alpha = Rational(-b, 2*a)
% assert alpha == 2
% beta = g.subs(x, 2)
% assert beta == 3
% canon = -2*(x - 2)**2 + 3
% assert expand(canon) == g
% END-VERIFY

\begin{corrige}{1SPE-SECDEG-RE-C1-EX2}

\commentaireMarge{$a = -2$ : attention, le denominateur est $2a = -4$.}

\textbf{Question 1.} $a = -2$, $b = 8$, $c = -5$.
\[
\alpha = \frac{-8}{2 \times (-2)} = \frac{-8}{-4} = 2. \qquad \beta = g(2) = -8 + 16 - 5 = 3.
\]
Forme canonique : $g(x) = -2(x-2)^2 + 3$.

\textbf{Question 2.}
\[
-2(x-2)^2 + 3 = -2(x^2 - 4x + 4) + 3 = -2x^2 + 8x - 8 + 3 = -2x^2 + 8x - 5. \quad \checkmark
\]

\commentaireMarge{Signe de $a$ : $a < 0$ donne une parabole en $\cap$ (maximum).}

\textbf{Question 3.} Comme $a = -2 < 0$, la parabole est en $\cap$ (ouverte vers le bas). Le sommet $S(2\,;\,3)$ est un maximum.

\end{corrige}

\begin{exercice}{1SPE-SECDEG-RE-C1-EX3}{1}{10}
Parmi les expressions suivantes, identifier celles qui sont des polynomes du second degre. Pour chacun, donner les coefficients $a$, $b$, $c$, puis la forme canonique.
\begin{enumerate}
  \item $f(x) = (x+3)(x-1)$
  \item $g(x) = x^2 - x + \dfrac{1}{4}$
  \item $h(x) = (x+2)^2 - (x-1)^2$
  \item $k(x) = x^3 - 4x^2$
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-RE-C1-EX3}

\textbf{Question 1.} $f(x) = (x+3)(x-1) = x^2 + 2x - 3$ : polynome du second degre, $a = 1$, $b = 2$, $c = -3$.
$\alpha = -2/(2) = -1$, $\beta = f(-1) = 1 - 2 - 3 = -4$.
Forme canonique : $f(x) = (x+1)^2 - 4$.

\textbf{Question 2.} $g(x) = x^2 - x + \frac{1}{4}$ : polynome du second degre, $a = 1$, $b = -1$, $c = \frac{1}{4}$.
$\alpha = \frac{1}{2}$, $\beta = g\!\left(\frac{1}{2}\right) = \frac{1}{4} - \frac{1}{2} + \frac{1}{4} = 0$.
Forme canonique : $g(x) = \left(x - \dfrac{1}{2}\right)^2$.

\commentaireMarge{Developper pour verifier le degre : $(x+2)^2 - (x-1)^2$ peut se simplifier.}

\textbf{Question 3.} $h(x) = (x+2)^2 - (x-1)^2 = (x^2+4x+4) - (x^2-2x+1) = 6x + 3$.
C'est un polynome du \textbf{premier degre} ($a = 0$) : pas un second degre.

\textbf{Question 4.} $k(x) = x^3 - 4x^2$ est un polynome de \textbf{degre 3} : pas un second degre.

\end{corrige}
