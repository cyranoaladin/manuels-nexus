\subsection*{Rappel — Equations du premier degre et produit nul}

\textbf{Equation du premier degre} : $ax + b = 0$ (avec $a \neq 0$) a pour unique solution $x = -b/a$.

\textbf{Regle du produit nul} : pour tous reels $A$ et $B$,
\[
  A \times B = 0 \quad \Longleftrightarrow \quad A = 0 \quad \text{ou} \quad B = 0.
\]
Cette regle est fondamentale pour resoudre les equations factorisees du second degre.

%---------------------------------------------------------------

\begin{exercice}{1SPE-SECDEG-FR-R2-EX1}{1}{5}
Resoudre les equations du premier degre suivantes.
\begin{enumerate}
  \item $3x - 6 = 0$
  \item $2x + 8 = 0$
  \item $-x + 4 = 0$
  \item $5x - 15 = 0$
\end{enumerate}
\end{exercice}

% assert solve(2*x + 8, x) == [-4]
% assert solve(-x + 4, x) == [4]
% assert solve(5*x - 15, x) == [3]
% END-VERIFY

\begin{corrige}{1SPE-SECDEG-FR-R2-EX1}

\commentaireMarge{On isole $x$ : $ax = -b$, donc $x = -b/a$.}

\textbf{Question 1.} $3x - 6 = 0 \Leftrightarrow 3x = 6 \Leftrightarrow x = 2$.

\textbf{Question 2.} $2x + 8 = 0 \Leftrightarrow 2x = -8 \Leftrightarrow x = -4$.

\textbf{Question 3.} $-x + 4 = 0 \Leftrightarrow x = 4$.

\textbf{Question 4.} $5x - 15 = 0 \Leftrightarrow 5x = 15 \Leftrightarrow x = 3$.

\end{corrige}

\begin{exercice}{1SPE-SECDEG-FR-R2-EX2}{1}{5}
Resoudre les equations suivantes en appliquant la regle du produit nul.
\begin{enumerate}
  \item $(x - 3)(x + 5) = 0$
  \item $2x(x - 7) = 0$
  \item $(3x - 1)(x + 2) = 0$
  \item $(x - 4)^2 = 0$
\end{enumerate}
\end{exercice}

% assert sorted(solve((x - 3)*(x + 5), x)) == [-5, 3]
% assert sorted(solve(2*x*(x - 7), x)) == [0, 7]
% assert sorted(solve((3*x - 1)*(x + 2), x)) == [-2, 1/3]
% # (x-4)^2 = 0 => double root
% assert solve((x-4)**2, x) == [4]
% END-VERIFY

\begin{corrige}{1SPE-SECDEG-FR-R2-EX2}

\commentaireMarge{$AB = 0 \Leftrightarrow A = 0$ ou $B = 0$. On resout chaque facteur separement.}

\textbf{Question 1.}
\[
(x-3)(x+5) = 0 \Leftrightarrow x - 3 = 0 \text{ ou } x + 5 = 0 \Leftrightarrow x = 3 \text{ ou } x = -5.
\]

\textbf{Question 2.}
\[
2x(x-7) = 0 \Leftrightarrow x = 0 \text{ ou } x - 7 = 0 \Leftrightarrow x = 0 \text{ ou } x = 7.
\]

\textbf{Question 3.}
\[
(3x-1)(x+2) = 0 \Leftrightarrow 3x - 1 = 0 \text{ ou } x + 2 = 0 \Leftrightarrow x = \frac{1}{3} \text{ ou } x = -2.
\]

\commentaireMarge{Un carre est nul ssi sa base est nulle : $(x-4)^2 = 0 \Leftrightarrow x - 4 = 0$.}

\textbf{Question 4.}
\[
(x-4)^2 = 0 \Leftrightarrow x - 4 = 0 \Leftrightarrow x = 4 \quad (\text{racine double}).
\]

\end{corrige}

\begin{exercice}{1SPE-SECDEG-FR-R2-EX3}{1}{6}
Resoudre les equations en se ramenant a un produit nul.
\begin{enumerate}
  \item $x^2 - 9 = 0$
  \item $x^2 - 5x = 0$
  \item $x^2 - 5x + 6 = 0$ \quad (aide : $(x-2)(x-3)$)
  \item $2x^2 - 8 = 0$
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% from sympy import symbols, solve
% x = symbols('x')
% assert sorted(solve(x**2 - 9, x)) == [-3, 3]
% assert sorted(solve(x**2 - 5*x, x)) == [0, 5]
% assert sorted(solve(x**2 - 5*x + 6, x)) == [2, 3]
% assert sorted(solve(2*x**2 - 8, x)) == [-2, 2]
% END-VERIFY

\begin{corrige}{1SPE-SECDEG-FR-R2-EX3}

\textbf{Question 1.} $x^2 - 9 = (x-3)(x+3) = 0 \Leftrightarrow x = 3$ ou $x = -3$.

\textbf{Question 2.} $x^2 - 5x = x(x-5) = 0 \Leftrightarrow x = 0$ ou $x = 5$.

\textbf{Question 3.} $x^2 - 5x + 6 = (x-2)(x-3) = 0 \Leftrightarrow x = 2$ ou $x = 3$.

\textbf{Question 4.} $2x^2 - 8 = 0 \Leftrightarrow x^2 = 4 \Leftrightarrow x = 2$ ou $x = -2$.

(Ou : $2x^2 - 8 = 2(x^2-4) = 2(x-2)(x+2) = 0$.)

\end{corrige}

\begin{exercice}{1SPE-SECDEG-FR-R2-EX4}{1}{6}
Pour chacune des situations, ecrire une equation, puis la resoudre.
\begin{enumerate}
  \item Le double d'un nombre augmente de $3$ est egal a $11$. Trouver ce nombre.
  \item La surface d'un carre de cote $x$ est $25$. Trouver $x$ (avec $x > 0$).
  \item Le produit de deux entiers consecutifs $n$ et $n+1$ est nul. Quelles sont les valeurs de $n$ ?
  \item On cherche $x > 0$ tel que $(x-1)(x+3) = 5$. Montrer que cela revient a $x^2 + 2x - 8 = 0$, puis resoudre.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-FR-R2-EX4}

\textbf{Question 1.} Equation : $2x + 3 = 11 \Leftrightarrow 2x = 8 \Leftrightarrow x = 4$.

\textbf{Question 2.} $x^2 = 25 \Leftrightarrow x = 5$ (car $x > 0$).

\textbf{Question 3.} $n(n+1) = 0 \Leftrightarrow n = 0$ ou $n + 1 = 0 \Leftrightarrow n = 0$ ou $n = -1$.

\textbf{Question 4.} On developpe : $(x-1)(x+3) = x^2 + 2x - 3$. L'equation donne $x^2 + 2x - 3 = 5$, soit $x^2 + 2x - 8 = 0$.

On cherche $p$, $q$ tels que $p + q = 2$ et $pq = -8$ : $p = -2$, $q = 4$... non ($pq = -8$ et $p+q = 2$) : $p = 4$, $q = -2$ convient. Donc $(x + 4)(x - 2) = 0$, d'ou $x = -4$ ou $x = 2$.

Comme $x > 0$, la solution est $x = 2$.

\end{corrige}
