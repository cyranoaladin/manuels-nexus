\begin{exercice}{1SPE-SECDEG-RE-C2-EX1}{1}{8}
Soit $f(x) = 2x^2 - 8x + 6$.
\begin{enumerate}
  \item Calculer les coordonnees du sommet $S(\alpha\,;\,\beta)$.
  \item Ecrire l'equation de l'axe de symetrie de la parabole.
  \item Dresser le tableau de variations de $f$ sur $\mathbb{R}$.
  \item Preciser si $f$ admet un minimum ou un maximum, et donner sa valeur.
\end{enumerate}
\end{exercice}

% alpha = Rational(-b, 2*a)
% assert alpha == 2
% beta = f.subs(x, alpha)
% assert beta == -2
% END-VERIFY

\begin{corrige}{1SPE-SECDEG-RE-C2-EX1}

\commentaireMarge{$a = 2 > 0$ : parabole en U, minimum au sommet.}

\textbf{Question 1.} $a = 2$, $b = -8$.
\[
\alpha = \frac{-(-8)}{2 \times 2} = \frac{8}{4} = 2. \qquad
\beta = f(2) = 2 \times 4 - 8 \times 2 + 6 = 8 - 16 + 6 = -2.
\]
Sommet : $S(2\,;\,-2)$.

\textbf{Question 2.} L'axe de symetrie est la droite d'equation $x = \alpha = 2$.

\textbf{Question 3. Tableau de variations :}

\begin{center}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $-\infty$ & & $2$ & & $+\infty$ \\
\hline
$f(x)$ & $+\infty$ & $\searrow$ & $-2$ & $\nearrow$ & $+\infty$ \\
\hline
\end{tabular}
\end{center}

\textbf{Question 4.} Comme $a = 2 > 0$, $f$ admet un \textbf{minimum} en $x = 2$. Ce minimum vaut $f(2) = -2$.

\end{corrige}

\begin{exercice}{1SPE-SECDEG-RE-C2-EX2}{1}{8}
Soit $g(x) = -x^2 + 4x - 3$.
\begin{enumerate}
  \item Montrer que la forme canonique est $g(x) = -(x-2)^2 + 1$.
  \item En deduire les coordonnees du sommet.
  \item Dresser le tableau de variations de $g$.
  \item Sur quel intervalle $g$ est-elle positive ? (Repondre par observation du tableau et des racines $x = 1$ et $x = 3$.)
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Pour la forme canonique avec $a = -1$ : $\alpha = -4/(2 \times (-1)) = 2$ et $\beta = g(2) = -4 + 8 - 3 = 1$.}

\begin{corrige}{1SPE-SECDEG-RE-C2-EX2}

\textbf{Question 1.} $\alpha = 2$, $\beta = g(2) = -4 + 8 - 3 = 1$. On verifie : $-(x-2)^2 + 1 = -(x^2-4x+4)+1 = -x^2+4x-3 = g(x)$. \checkmark

\textbf{Question 2.} Sommet : $S(2\,;\,1)$.

\commentaireMarge{$a = -1 < 0$ : parabole en $\cap$, maximum au sommet.}

\textbf{Question 3. Tableau de variations :}

\begin{center}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $-\infty$ & & $2$ & & $+\infty$ \\
\hline
$g(x)$ & $-\infty$ & $\nearrow$ & $1$ & $\searrow$ & $-\infty$ \\
\hline
\end{tabular}
\end{center}

\textbf{Question 4.} Les racines de $g$ sont $x = 1$ et $x = 3$. Comme $a < 0$, $g(x) \geq 0$ entre les racines : sur $[1\,;\,3]$.

\end{corrige}

\begin{exercice}{1SPE-SECDEG-RE-C2-EX3}{1}{10}
Pour chacune des fonctions suivantes, donner le sommet, l'axe de symetrie, le tableau de variations, et preciser si la fonction admet un minimum ou un maximum.
\begin{enumerate}
  \item $p(x) = x^2 + 4x + 1$
  \item $q(x) = -3x^2 + 6x + 2$
  \item $r(x) = (x-1)^2 + 3$ \quad (forme canonique directement lisible)
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-RE-C2-EX3}

\textbf{Question 1.} $p(x) = x^2 + 4x + 1$ : $a = 1 > 0$.
$\alpha = -4/2 = -2$, $\beta = p(-2) = 4 - 8 + 1 = -3$.
Sommet $S(-2\,;\,-3)$, axe $x = -2$. Minimum $-3$ en $x = -2$.

\begin{center}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $-\infty$ & & $-2$ & & $+\infty$ \\
\hline
$p(x)$ & $+\infty$ & $\searrow$ & $-3$ & $\nearrow$ & $+\infty$ \\
\hline
\end{tabular}
\end{center}

\textbf{Question 2.} $q(x) = -3x^2 + 6x + 2$ : $a = -3 < 0$.
$\alpha = -6/(2 \times (-3)) = 1$, $\beta = q(1) = -3 + 6 + 2 = 5$.
Sommet $S(1\,;\,5)$, axe $x = 1$. Maximum $5$ en $x = 1$.

\begin{center}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $-\infty$ & & $1$ & & $+\infty$ \\
\hline
$q(x)$ & $-\infty$ & $\nearrow$ & $5$ & $\searrow$ & $-\infty$ \\
\hline
\end{tabular}
\end{center}

\commentaireMarge{Forme canonique directement : sommet $= (\alpha\,;\,\beta)$ du terme $(x-\alpha)^2 + \beta$.}

\textbf{Question 3.} $r(x) = (x-1)^2 + 3$ : $a = 1 > 0$, sommet $S(1\,;\,3)$, axe $x = 1$. Minimum $3$ en $x = 1$.

\end{corrige}
