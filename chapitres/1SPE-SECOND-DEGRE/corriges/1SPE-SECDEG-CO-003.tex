\begin{corrige}{1SPE-SECDEG-EX-003}

\textbf{1. Identification des trinômes du second degré.}

\commentaireMarge{Un trinôme du 2\textsuperscript{nd} degré vérifie $a \neq 0$.}

\begin{itemize}
  \item $p(x) = 4x^2 - x + 7$ : \textbf{oui}, c'est un trinôme du second degré. On a $a = 4$, $b = -1$, $c = 7$.
  \item $q(x) = 3x - 2$ : \textbf{non}, c'est un polynôme de degré 1 (pas de terme en $x^2$).
  \item $r(x) = -x^2 + 5$ : \textbf{oui}, c'est un trinôme du second degré. On a $a = -1$, $b = 0$, $c = 5$.
\end{itemize}

\medskip
\textbf{2a. Développement de $f(x) = 2\!\left(x - \dfrac{1}{2}\right)\!(x - 3)$.}

\commentaireMarge{Distribuer le produit, puis multiplier par $2$.}

\[\left(x - \frac{1}{2}\right)(x - 3) = x^2 - 3x - \frac{x}{2} + \frac{3}{2} = x^2 - \frac{7}{2}x + \frac{3}{2}.\]

\[f(x) = 2\!\left(x^2 - \frac{7}{2}x + \frac{3}{2}\right) = 2x^2 - 7x + 3.\]

\[\boxed{f(x) = 2x^2 - 7x + 3.}\]

\medskip
\textbf{2b. Coordonnées du sommet.}

On lit $a = 2$, $b = -7$. L'abscisse du sommet est :
\[\alpha = -\frac{b}{2a} = -\frac{-7}{4} = \frac{7}{4}.\]

L'ordonnée du sommet est :
\[\beta = f\!\left(\frac{7}{4}\right) = 2\times\frac{49}{16} - 7\times\frac{7}{4} + 3 = \frac{49}{8} - \frac{49}{4} + 3 = \frac{49}{8} - \frac{98}{8} + \frac{24}{8} = -\frac{25}{8}.\]

Le sommet est $S\!\left(\dfrac{7}{4}\,;\,-\dfrac{25}{8}\right)$.

\medskip
\textbf{2c. Forme canonique.}

\commentaireMarge{$f(x) = a(x-\alpha)^2 + \beta$.}

\[\boxed{f(x) = 2\left(x - \frac{7}{4}\right)^2 - \frac{25}{8}.}\]
