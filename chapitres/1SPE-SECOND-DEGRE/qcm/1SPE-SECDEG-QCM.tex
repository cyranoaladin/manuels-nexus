% ------------------------------------------------------------
\subsection*{C1 — Reconnaitre un polynome du second degre et passer d'une forme a une autre}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} Parmi les expressions suivantes, laquelle est un polynome du second degre ?

  \begin{enumerate}[label=\Alph*.]
    \item $f(x) = 3x + 7$
    \item $f(x) = 2x^2 - 5x + 1$
    \item $f(x) = x^3 - 2x + 4$
    \item $f(x) = \dfrac{1}{x^2} + 3$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} Soit $f(x) = 2x^2 - 8x + 6$. Quelle est la forme canonique de $f$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $f(x) = 2(x - 2)^2 - 2$
    \item $f(x) = 2(x - 4)^2 - 10$
    \item $f(x) = 2(x - 2)^2 + 2$
    \item $f(x) = (x - 2)^2 - 2$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} On developpe $3(x - 1)^2 + 5$. Quel resultat obtient-on ?

  \begin{enumerate}[label=\Alph*.]
    \item $3x^2 - 6x + 8$
    \item $3x^2 - 6x + 3$
    \item $3x^2 - 3x + 8$
    \item $3x^2 + 8$
  \end{enumerate}

\end{enumerate}

\bigskip

% ------------------------------------------------------------
\subsection*{C2 — Determiner le sommet, l'axe de symetrie et dresser le tableau de variations}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q4 ----
\item \textbf{[Q4]} Soit $f(x) = -x^2 + 4x - 3$.
  Quelle est l'abscisse du sommet de la parabole representant $f$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $x = -2$
    \item $x = 2$
    \item $x = -4$
    \item $x = 4$
  \end{enumerate}

\bigskip

% ---- Q5 ----
\item \textbf{[Q5]} Soit $f(x) = x^2 - 6x + 5$.
  Sur quel intervalle $f$ est-elle decroissante ?

  \begin{enumerate}[label=\Alph*.]
    \item $[3\,;\,+\infty[$
    \item $]-\infty\,;\,3]$
    \item $]-\infty\,;\,5]$
    \item $[5\,;\,+\infty[$
  \end{enumerate}

\bigskip

% ---- Q6 ----
\item \textbf{[Q6]} Parmi les assertions suivantes concernant $f(x) = 2x^2 - 4x + 7$,
  laquelle est vraie ?

  \begin{enumerate}[label=\Alph*.]
    \item Le minimum de $f$ vaut $7$.
    \item La parabole est tournee vers le bas (coefficient directeur negatif).
    \item Le minimum de $f$ vaut $5$.
    \item La fonction $f$ n'admet pas de minimum.
  \end{enumerate}

\end{enumerate}

\bigskip

\subsection*{C3 — Calculer le discriminant et resoudre une equation du second degre}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} On considere $f(x) = x^2 - 5x + 6$.
  Quelle est la valeur du discriminant $\Delta$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\Delta = 1$
    \item $\Delta = 49$
    \item $\Delta = -11$
    \item $\Delta = 25$
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} L'equation $2x^2 - 4x + 2 = 0$ admet :

  \begin{enumerate}[label=\Alph*.]
    \item Deux racines reelles distinctes.
    \item Une racine double, egale a $1$.
    \item Aucune racine reelle.
    \item Une racine double, egale a $2$.
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} Les solutions de $x^2 - 3x - 4 = 0$ sont :

  \begin{enumerate}[label=\Alph*.]
    \item $x_1 = -1$ et $x_2 = 4$
    \item $x_1 = 1$ et $x_2 = -4$
    \item $x_1 = -1$ et $x_2 = -4$
    \item $x_1 = 1$ et $x_2 = 4$
  \end{enumerate}

\end{enumerate}

% ------------------------------------------------------------
\subsection*{C4 — Factoriser un trinome et etudier son signe}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q10 ----
\item \textbf{[Q10]} Soit $f(x) = 2x^2 - 8x + 6$, dont les racines sont $x_1 = 1$ et $x_2 = 3$.
  Quelle est la forme factorisee de $f(x)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $f(x) = (x - 1)(x - 3)$
    \item $f(x) = 2(x - 1)(x - 3)$
    \item $f(x) = 2(x + 1)(x + 3)$
    \item $f(x) = (2x - 1)(2x - 3)$
  \end{enumerate}

\bigskip

% ---- Q11 ----
\item \textbf{[Q11]} On sait que $f(x) = (x - 2)(x - 5)$.
  Sur quel intervalle $f(x)$ est-il negatif ou nul ?

  \begin{enumerate}[label=\Alph*.]
    \item $]-\infty\,;\,2] \cup [5\,;\,+\infty[$
    \item $[2\,;\,5]$
    \item $]-\infty\,;\,2]$
    \item $[5\,;\,+\infty[$
  \end{enumerate}

\bigskip

% ---- Q12 ----
\item \textbf{[Q12]} Le trinome $f(x) = x^2 + 2x + 5$ a pour discriminant $\Delta = -16 < 0$.
  Quel est son signe sur $\mathbb{R}$ ?

  \begin{enumerate}[label=\Alph*.]
    \item Il est positif pour $x > 0$ et negatif pour $x < 0$.
    \item Il est negatif pour tout $x \in \mathbb{R}$.
    \item Il est positif pour tout $x \in \mathbb{R}$.
    \item Il change de signe en $x = -1$.
  \end{enumerate}

\end{enumerate}

\bigskip

\subsection*{C5 — Resoudre une inequation du second degre}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q13 ----
\item \textbf{[Q13]} Resoudre $x^2 - x - 6 \leq 0$
  (les racines de $x^2 - x - 6$ sont $-2$ et $3$).

  \begin{enumerate}[label=\Alph*.]
    \item $x \in [-2\,;\,3]$
    \item $x \in ]-\infty\,;\,-2] \cup [3\,;\,+\infty[$
    \item $x \in [-2\,;\,0]$
    \item $x \in [0\,;\,3]$
  \end{enumerate}

\bigskip

% ---- Q14 ----
\item \textbf{[Q14]} L'ensemble de solutions de $-x^2 + 4x - 3 \geq 0$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $]-\infty\,;\,1] \cup [3\,;\,+\infty[$
    \item $[1\,;\,3]$
    \item $]-\infty\,;\,-1] \cup [3\,;\,+\infty[$
    \item $[-1\,;\,3]$
  \end{enumerate}

\bigskip

% ---- Q15 ----
\item \textbf{[Q15]} Pour quelle valeur de $k$ l'equation $x^2 - 2x + k = 0$
  admet-elle exactement une solution reelle ?

  \begin{enumerate}[label=\Alph*.]
    \item $k = 0$
    \item $k = 2$
    \item $k = 1$
    \item $k = -1$
  \end{enumerate}

\end{enumerate}

\bigskip

% ------------------------------------------------------------
\subsection*{C6 — Modeliser un probleme concret par un polynome du second degre et trouver un optimum}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q16 ----
\item \textbf{[Q16]} On decoupe des carres de cote $x$ aux coins d'une feuille $30 \times 20$
  pour former une boite ouverte. Le volume est $V(x) = x(30 - 2x)(20 - 2x)$.
  Quelle est la valeur de $V(4)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $V(4) = 1\,344$
    \item $V(4) = 1\,280$
    \item $V(4) = 2\,640$
    \item $V(4) = 1\,056$
  \end{enumerate}

\bigskip

% ---- Q17 ----
\item \textbf{[Q17]} On cherche a maximiser $f(x) = -2x^2 + 8x + 3$ sur $\mathbb{R}$.
  Quelle est la valeur du maximum ?

  \begin{enumerate}[label=\Alph*.]
    \item $f_{\max} = 3$
    \item $f_{\max} = 11$
    \item $f_{\max} = 8$
    \item $f_{\max} = 2$
  \end{enumerate}

\bigskip

% ---- Q18 ----
\item \textbf{[Q18]} Une balle est lancee verticalement. Sa hauteur est
  $h(t) = -5t^2 + 20t$ (en metres, $t$ en secondes).
  A quel instant atteint-elle sa hauteur maximale ?

  \begin{enumerate}[label=\Alph*.]
    \item $t = 4$~s
    \item $t = 1$~s
    \item $t = 2$~s
    \item $t = 20$~s
  \end{enumerate}

\end{enumerate}
