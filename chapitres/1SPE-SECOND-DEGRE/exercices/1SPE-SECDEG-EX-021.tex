\begin{exercice}{1SPE-SECDEG-EX-021}{2}{20}
On considère le trinôme $f(x) = x^2 - 4x + 1$.

\begin{enumerate}
  \item \textbf{(C1)} Écrire $f$ sous forme canonique $f(x) = (x - \alpha)^2 + \beta$.
  \item \textbf{(C3)} Calculer $\Delta$ et résoudre $f(x) = 0$ dans $\mathbb{R}$. Donner les solutions sous forme exacte (avec $\sqrt{3}$).
  \item \textbf{(C3)} Résoudre $f(x) = -2$ dans $\mathbb{R}$.

  \textit{On ramènera à la forme $ax^2 + bx + c = 0$.}

  \item \textbf{(C1)} La valeur $\beta$ est-elle bien le minimum de $f$ ? Justifier à l'aide de la forme canonique.
\end{enumerate}
\end{exercice}
