\begin{exercice}{1SPE-SECDEG-EX-026}{2}{20}
\begin{enumerate}
  \item \textbf{(C4)} Factoriser et étudier le signe de $g(x) = 3x^2 - 5x - 2$.
  \item \textbf{(C5)} Résoudre dans $\mathbb{R}$ l'inéquation $3x^2 - 5x - 2 < 0$.

  \item Un jardinier dispose de deux parcelles :
  \begin{itemize}
    \item un carré de côté $x$ mètres ($x > 1$) ;
    \item un rectangle de dimensions $(x+3)$~m et $(x-1)$~m.
  \end{itemize}
  \textbf{(C5)} Pour quelles valeurs de $x$ le rectangle a-t-il une aire strictement supérieure à celle du carré ?

  \textit{Montrer que l'inéquation se simplifie et n'est pas du second degré.}
\end{enumerate}
\end{exercice}
