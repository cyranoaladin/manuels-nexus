\begin{exercice}{1SPE-SECDEG-EX-010}{1}{10}
Factoriser chacune des expressions suivantes.

\begin{enumerate}
  \item $x^2 - 9$
  \item $2x^2 - 8x$
  \item $x^2 - x - 6$
  \item $3x^2 - 12x + 12$
\end{enumerate}

\textit{On précisera à chaque fois la méthode utilisée : identité remarquable, facteur commun, ou utilisation du discriminant.}
\end{exercice}
