\begin{exercice}{1SPE-SECDEG-EX-023}{2}{20}
On considère le trinôme $f(x) = 6x^2 - x - 2$.

\begin{enumerate}
  \item \textbf{(C3)} Calculer $\Delta$ et déterminer les racines de $f$.
  \item \textbf{(C4)} Factoriser $f(x)$. On vérifiera que $f(x) = (2x+1)(3x-2)$.
  \item \textbf{(C4)} Dresser le tableau de signes de $f$ sur $\mathbb{R}$.
  \item \textbf{(C3)} Résoudre dans $\mathbb{R}$ l'équation $6x^2 - x = 2$.
  \item \textbf{(C4)} En déduire, sans nouveau calcul, le signe de $6x^2 - x - 2$ sur $\left[-\dfrac{1}{2}\,;\,\dfrac{2}{3}\right]$.
\end{enumerate}
\end{exercice}
