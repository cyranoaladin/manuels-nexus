\begin{exercice}{1SPE-SECDEG-EX-022}{2}{20}
On pose $p(x) = 2x^2 - 4x - 6$.

\begin{enumerate}
  \item \textbf{(C1)} Écrire $p(x)$ sous forme canonique, puis sous forme factorisée. Détailler les calculs.
  \item \textbf{(C3)} Résoudre dans $\mathbb{R}$ l'équation $p(x) = 10$.

  \textit{Simplifier l'équation avant de calculer $\Delta$.}

  \item \textbf{(C3)} Résoudre dans $\mathbb{R}$ l'équation $p(x) = -8$.

  \textit{Utiliser la forme canonique pour conclure directement.}

  \item \textbf{(C1)} Vérifier que les trois formes de $p(x)$ donnent le même résultat pour $p(2)$.
\end{enumerate}
\end{exercice}
