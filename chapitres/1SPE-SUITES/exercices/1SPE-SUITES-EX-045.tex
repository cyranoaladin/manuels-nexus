\begin{exercice}{1SPE-SUITES-EX-045}{3}{35}
On considère la suite $(u_n)$ définie par $u_0 = 5$ et, pour tout entier naturel $n$ :
\[
  u_{n+1} = \frac{1}{2}\,u_n + 3.
\]

\begin{enumerate}
  \item Calculer $u_1$, $u_2$ et $u_3$ (valeurs exactes). Ces valeurs semblent-elles converger ?

  \item \textbf{Point fixe.} On cherche un réel $\ell$ tel que $\dfrac{1}{2}\,\ell + 3 = \ell$.
  \begin{enumerate}
    \item Résoudre cette équation.
    \item On pose $v_n = u_n - \ell$. Montrer que $(v_n)$ est géométrique, préciser sa raison et son premier terme.
  \end{enumerate}

  \item Exprimer $v_n$ puis $u_n$ en fonction de $n$.

  \item \textbf{Variations.}
  \begin{enumerate}
    \item Calculer $u_{n+1} - u_n$ en fonction de $n$.
    \item En déduire le sens de variation de $(u_n)$.
  \end{enumerate}

  \item \textbf{Encadrement.} Montrer par récurrence que, pour tout entier naturel $n$ : $5 \leq u_n < 6$.

  \item \textbf{Généralisation.} On remplace $u_0 = 5$ par $u_0 = 8$. Sans refaire tous les calculs, donner la nouvelle expression de $u_n$ et le nouveau sens de variation. Justifier.
\end{enumerate}
\end{exercice}
