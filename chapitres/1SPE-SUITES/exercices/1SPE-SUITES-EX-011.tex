\begin{exercice}{1SPE-SUITES-EX-011}{1}{10}
On considère la suite arithmétique $(u_n)$ de premier terme $u_0 = 2$ et de raison $r = 3$.
\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$, $u_3$, $u_4$ et $u_5$.
  \item En appliquant la formule
        $S = (n+1) \times \dfrac{u_0 + u_n}{2}$,
        calculer $S = u_0 + u_1 + \cdots + u_5$.
  \item Vérifier le résultat en additionnant directement les six termes.
\end{enumerate}
\end{exercice}
