\begin{exercice}{1SPE-SUITES-EX-036}{2}{22}
Soit $(u_n)$ la suite définie pour tout entier naturel $n$ par :
\[
  u_n = n^2 - 5n + 8.
\]

\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$, $u_3$ et $u_4$.

  \item La suite $(u_n)$ est-elle définie par une formule explicite ou par une relation de récurrence ? Justifier.

  \item Étudier le sens de variation de $(u_n)$ en calculant $u_{n+1} - u_n$ et en étudiant son signe.

  \item En déduire pour quels entiers $n$ la suite est décroissante, puis croissante. Préciser le(s) terme(s) minimal/minimaux.

  \item On pose $f(x) = x^2 - 5x + 8$ pour tout réel $x \geq 0$. Calculer la dérivée $f'(x)$, étudier son signe, et comparer les variations de $f$ avec celles de $(u_n)$.
\end{enumerate}
\end{exercice}
