\begin{exercice}{1SPE-SUITES-EX-030}{2}{22}
Soit $(u_n)$ la suite géométrique de premier terme $u_0 = 5$ et de raison $q = \dfrac{3}{4}$.

\begin{enumerate}
  \item Calculer $u_1$, $u_2$ et $u_3$ (en fractions).

  \item Démontrer que la suite $(u_n)$ est strictement décroissante en utilisant la méthode du quotient. Justifier que $u_n > 0$ pour tout $n \in \mathbb{N}$.

  \item Exprimer $u_n$ en fonction de $n$.

  \item On pose $v_n = \dfrac{u_{n+1}}{u_n} - 1$ pour tout $n \in \mathbb{N}$.
  \begin{enumerate}
    \item Calculer $v_n$.
    \item Interpréter le signe de $v_n$ au regard du sens de variation de $(u_n)$.
  \end{enumerate}

  \item On considère la suite $(w_n)$ définie par $w_n = \dfrac{1}{u_n}$ pour tout $n \in \mathbb{N}$. Montrer que $(w_n)$ est une suite géométrique et préciser sa raison.
\end{enumerate}
\end{exercice}
