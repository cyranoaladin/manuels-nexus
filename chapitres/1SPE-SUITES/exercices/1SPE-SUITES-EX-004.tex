\begin{exercice}{1SPE-SUITES-EX-004}{1}{10}
Amira définit la suite $(u_n)$ pour tout entier naturel $n$ par
\[
  u_n = n^2 - 2n + 3.
\]
\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$ et $u_3$.
  \item Calculer $u_5$.
  \item Amira affirme que $u_1 = u_2$. A-t-elle raison ? Justifier par le calcul.
\end{enumerate}
\end{exercice}
