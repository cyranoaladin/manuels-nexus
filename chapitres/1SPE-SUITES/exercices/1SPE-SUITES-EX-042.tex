\begin{exercice}{1SPE-SUITES-EX-042}{3}{35}
Pour tout entier naturel $n \geq 1$, on pose $u_n = \dfrac{1}{n(n+1)}$ et $S_n = \displaystyle\sum_{k=1}^{n} u_k$.

\begin{enumerate}
  \item Calculer $u_1$, $u_2$, $u_3$, puis $S_1$, $S_2$, $S_3$. Émettre une conjecture sur la forme de $S_n$.

  \item \textbf{Décomposition.} Montrer que, pour tout entier $k \geq 1$ :
  \[
    u_k = \frac{1}{k} - \frac{1}{k+1}.
  \]

  \item \textbf{Somme télescopique.} En déduire, pour tout entier $n \geq 1$, que :
  \[
    S_n = \frac{n}{n+1}.
  \]
  \textit{Détailler soigneusement le mécanisme de la somme télescopique.}

  \item \textbf{Monotonie.} Montrer que la suite $(S_n)$ est strictement croissante et majorée par $1$. En déduire sa nature.

  \item \textbf{Généralisation.} On pose, pour $n \geq 1$ :
  \[
    T_n = \sum_{k=1}^{n} \frac{1}{k(k+2)}.
  \]
  \begin{enumerate}
    \item Trouver des réels $A$ et $B$ tels que $\dfrac{1}{k(k+2)} = \dfrac{A}{k} + \dfrac{B}{k+2}$.
    \item En déduire une expression simplifiée de $T_n$.
  \end{enumerate}
\end{enumerate}
\end{exercice}
