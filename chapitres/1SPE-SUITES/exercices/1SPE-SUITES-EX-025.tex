\begin{exercice}{1SPE-SUITES-EX-025}{2}{22}
Soit $(u_n)$ une suite arithmétique telle que $u_1 = 8$ et $u_5 = 20$.

\begin{enumerate}
  \item Déterminer la raison $r$ et le premier terme $u_0$ de la suite $(u_n)$.

  \item Exprimer $u_n$ en fonction de $n$ pour tout $n \in \mathbb{N}$.

  \item Calculer la somme $S = u_1 + u_2 + \cdots + u_{10}$.

  \item On pose $T_n = u_1 + u_2 + \cdots + u_n$ pour tout entier $n \geq 1$. Exprimer $T_n$ en fonction de $n$, puis déterminer le plus petit entier $n$ tel que $T_n > 500$.
\end{enumerate}
\end{exercice}
