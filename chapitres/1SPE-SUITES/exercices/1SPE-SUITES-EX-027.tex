\begin{exercice}{1SPE-SUITES-EX-027}{2}{20}
Soit $(u_n)$ la suite géométrique de premier terme $u_0 = 256$ et de raison $q = \dfrac{1}{2}$.

\begin{enumerate}
  \item Vérifier que $u_n \neq 0$ pour tout $n \in \mathbb{N}$, puis calculer $\dfrac{u_{n+1}}{u_n}$ et conclure.

  \item Exprimer $u_n$ en fonction de $n$.

  \item Calculer $u_3$ et $u_7$.

  \item Calculer la somme $S = u_0 + u_1 + \cdots + u_7$. En déduire la somme $T = u_1 + u_2 + \cdots + u_7$.

  \item Montrer que pour tout entier $n \geq 1$ :
  \[
    u_0 + u_1 + \cdots + u_n = 512\left(1 - \left(\frac{1}{2}\right)^{n+1}\right).
  \]
  Que se passe-t-il quand $n$ devient très grand ?
\end{enumerate}
\end{exercice}
