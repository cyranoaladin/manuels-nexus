\begin{exercice}{1SPE-SUITES-EX-048}{3}{35}
On considère la suite $(u_n)$ définie par $u_0 = 2$ et, pour tout entier naturel $n$ :
\[
  u_{n+1} = \frac{u_n}{3}.
\]

\begin{enumerate}
  \item Calculer $u_1$, $u_2$ et $u_3$.

  \item \textbf{Nature de la suite.} Montrer que $(u_n)$ est géométrique. Préciser son premier terme et sa raison. En déduire l'expression de $u_n$ en fonction de $n$.

  \item \textbf{Variations.}
  \begin{enumerate}
    \item Montrer, par récurrence, que $u_n > 0$ pour tout entier naturel $n$.
    \item En déduire que la suite $(u_n)$ est strictement décroissante.
  \end{enumerate}

  \item \textbf{Convergence.}
  \begin{enumerate}
    \item Montrer que $(u_n)$ est minorée par $0$.
    \item La suite $(u_n)$ est donc décroissante et minorée : admettre qu'elle converge. Déterminer sa limite en utilisant l'expression explicite de $u_n$.
  \end{enumerate}

  \item \textbf{Seuil.} Déterminer le plus petit entier $n$ tel que $u_n < 10^{-3}$.
  \textit{On utilisera le fait que $\ln(3) \approx 1{,}099$ et $\ln(2) \approx 0{,}693$.}

  \item \textbf{Généralisation.} On remplace la relation $u_{n+1} = \dfrac{u_n}{3}$ par $u_{n+1} = r\,u_n$, où $r$ est un réel.
  Pour quelles valeurs de $r$ la suite $(u_n)$ converge-t-elle vers $0$ (en supposant $u_0 \neq 0$) ? Justifier sans calcul de limite.
\end{enumerate}
\end{exercice}
