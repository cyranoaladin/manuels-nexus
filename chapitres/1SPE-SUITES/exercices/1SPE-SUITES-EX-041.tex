\begin{exercice}{1SPE-SUITES-EX-041}{3}{35}
On considère deux suites $(a_n)$ et $(b_n)$ définies pour tout entier naturel $n$ par :
\[
  a_n = 3n + 2 \qquad \text{et} \qquad b_n = 2 \times 3^n.
\]

\begin{enumerate}
  \item \textbf{Nature des suites.}
  \begin{enumerate}
    \item Montrer que $(a_n)$ est arithmétique. Préciser son premier terme et sa raison.
    \item Montrer que $(b_n)$ est géométrique. Préciser son premier terme et sa raison.
    \item Comparer $a_0$ et $b_0$.
  \end{enumerate}

  \item \textbf{Comparaison terme à terme.} On cherche à déterminer les entiers $n$ pour lesquels $b_n > a_n$.
  \begin{enumerate}
    \item Vérifier que $b_1 > a_1$.
    \item Montrer que, pour tout entier $n \geq 1$, on a $b_n > a_n$. \textit{On pourra raisonner par récurrence.}
  \end{enumerate}

  \item \textbf{Sommes.}
  \begin{enumerate}
    \item Calculer $S_A = \displaystyle\sum_{k=0}^{9} a_k$.
    \item Calculer $S_G = \displaystyle\sum_{k=0}^{9} b_k$.
    \item Calculer $S_G - S_A$ et interpréter ce résultat.
  \end{enumerate}

  \item \textbf{Question ouverte.} Trouver la plus petite valeur de $n$ pour laquelle $S_G^{(n)} > 100 \times S_A^{(n)}$, où $S_A^{(n)} = \displaystyle\sum_{k=0}^{n} a_k$ et $S_G^{(n)} = \displaystyle\sum_{k=0}^{n} b_k$. Justifier la démarche.
\end{enumerate}
\end{exercice}
