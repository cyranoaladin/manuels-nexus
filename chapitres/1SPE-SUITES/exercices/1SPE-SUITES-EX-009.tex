\begin{exercice}{1SPE-SUITES-EX-009}{1}{10}
On considère la suite $(u_n)$ définie pour tout entier naturel $n$ par
$u_n = 6 \times 2^n$.
\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$ et $u_3$.
  \item Montrer que la suite $(u_n)$ est géométrique et préciser sa raison.
  \item Exprimer $u_{n+1}$ en fonction de $u_n$.
\end{enumerate}
\end{exercice}
