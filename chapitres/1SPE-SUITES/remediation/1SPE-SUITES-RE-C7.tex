\begin{exercice}{1SPE-SUITES-RE-C7-EX1}{1}{10}
On considère la suite $(u_n)$ définie par $u_0 = 1$ et $u_{n+1} = 2u_n + 3$ pour tout $n \in \mathbb{N}$.

On veut écrire un programme Python calculant $u_n$ pour un rang $n$ donné.

\begin{enumerate}
  \item Dérouler à la main : calculer $u_1$, $u_2$, $u_3$.
  \item Lire et analyser le programme suivant, ligne par ligne :

\medskip
\verb|def terme(n):|

\verb|    u = 1|

\verb|    for k in range(n):|

\verb|        u = 2 * u + 3|

\verb|    return u|

  \item Vérifier que \texttt{terme(3)} renvoie bien la valeur de $u_3$ calculée en question 1.
  \item Que renvoie \texttt{terme(0)} ? Est-ce cohérent ?
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SUITES-RE-C7-EX1}

\commentaireMarge{On applique la relation de récurrence $u_{n+1} = 2u_n + 3$ à partir de $u_0 = 1$.}

\textbf{Question 1.}
\[
u_1 = 2 \times 1 + 3 = 5. \quad u_2 = 2 \times 5 + 3 = 13. \quad u_3 = 2 \times 13 + 3 = 29.
\]

\commentaireMarge{La variable \texttt{u} joue le rôle du terme courant. La boucle \texttt{for} effectue exactement $n$ mises à jour.}

\textbf{Question 2.} Analyse ligne par ligne :
\begin{itemize}
  \item \texttt{u = 1} : initialise $u$ à $u_0 = 1$.
  \item \texttt{for k in range(n):} : répète $n$ fois (pour $k = 0, 1, \ldots, n-1$).
  \item \texttt{u = 2 * u + 3} : applique la relation $u_{n+1} = 2u_n + 3$.
  \item \texttt{return u} : renvoie le terme $u_n$ après $n$ mises à jour.
\end{itemize}

\textbf{Question 3.} Pour \texttt{terme(3)} : on applique $3$ fois la relation en partant de $u_0 = 1$.
Après $3$ étapes, \texttt{u} vaut $29 = u_3$. \checkmark

\commentaireMarge{\texttt{range(0)} est vide : la boucle ne s'exécute pas. On renvoie directement $u_0$.}

\textbf{Question 4.} \texttt{terme(0)} : la boucle ne s'exécute pas (\texttt{range(0)} est vide). On renvoie \texttt{u = 1}. C'est bien $u_0 = 1$. Cohérent. \checkmark

\end{corrige}

\begin{exercice}{1SPE-SUITES-RE-C7-EX2}{1}{10}
On considère la suite géométrique $(v_n)$ de premier terme $v_0 = 2$ et de raison $q = 3$.

On veut calculer la somme $S_n = v_0 + v_1 + \cdots + v_n$.

\begin{enumerate}
  \item Calculer $S_3$ directement (sans programme).
  \item Lire le programme suivant :

\medskip
\verb|def somme(n):|

\verb|    v = 2|

\verb|    S = 2|

\verb|    for k in range(n):|

\verb|        v = 3 * v|

\verb|        S = S + v|

\verb|    return S|

  \item Dérouler le programme pour $n = 3$ (tableau de trace) et vérifier que \texttt{somme(3)} renvoie $S_3$.
  \item Modifier ce programme pour calculer la somme $v_0 + v_1 + \cdots + v_n$ si $v_0 = 5$ et $q = 4$.
\end{enumerate}
\end{exercice}

\[

\textbf{Question 2.} Analyse :
\begin{itemize}
  \item \texttt{v = 2} : terme courant, initialisé à $v_0 = 2$.
  \item \texttt{S = 2} : accumulateur, initialisé à $S = v_0 = 2$ (le premier terme est déjà compté).
  \item \texttt{for k in range(n):} : $n$ itérations pour ajouter $v_1, v_2, \ldots, v_n$.
  \item \texttt{v = 3 * v} : calcule le terme suivant.
  \item \texttt{S = S + v} : ajoute ce terme à l'accumulateur.
\end{itemize}

\begin{exercice}{1SPE-SUITES-RE-C7-EX3}{1}{12}
Un capital de $1\,000$~€ est placé avec un taux d'intérêt de $5\,\%$ par an. On cherche le plus petit rang $n$ tel que le capital dépasse $1\,500$~€.
\begin{enumerate}
  \item Écrire la relation de récurrence donnant le capital $C_{n+1}$ en fonction de $C_n$.
  \item Analyser le programme suivant et expliquer ce qu'il calcule :

\medskip
\verb|C = 1000|

\verb|n = 0|

\verb|while C <= 1500:|

\verb|    C = 1.05 * C|

\verb|    n = n + 1|

\verb|print(n, C)|

  \item Dérouler les $3$ premiers tours de boucle (tableau de trace).
  \item Déterminer la valeur de $n$ affichée. Vérifier que $C_{n-1} \leq 1\,500$ et $C_n > 1\,500$.
\end{enumerate}
\end{exercice}
