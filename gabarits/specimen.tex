\section{Les p-uplets (tuples)}

\margeAppui{Un p-uplet est une collection \emph{ordonnée} et \emph{non modifiable} de valeurs.
En Python, on utilise le mot anglais \emph{tuple}.}

\begin{nxdef}[D1]
  Un \textbf{p-uplet} (ou \emph{tuple}) est une séquence ordonnée de $p$ éléments,
  éventuellement de types différents, séparés par des virgules et entourés de parenthèses.
  Un p-uplet est \textbf{non mutable} : une fois créé, on ne peut ni ajouter, ni retirer,
  ni modifier ses éléments.
\end{nxdef}

\exemple{Le tuple \lstinline{station = ("Tunis-Carthage", 36.8, 10.2)} contient trois éléments :
une chaîne de caractères et deux flottants.}

\begin{python}
station = ("Tunis-Carthage", 36.8, 10.2)
nom = station[0]        # "Tunis-Carthage"
latitude = station[1]   # 36.8
\end{python}

\contreexemple{Toute tentative de modification provoque une erreur.}

\erreurFrequente{Modifier un tuple comme une liste (EF1). Un tuple est immuable
par conception : si l'on a besoin de modifier des données, il faut utiliser une liste.
\par
\textcolor{nxGris}{\sout{Copie fautive : \lstinline{station[1] = 37.0}}}
\par
\textcolor{nxVert}{Correction : créer un nouveau tuple ou utiliser une liste.}}

\begin{nxdef}[D2]
  La fonction \lstinline{len(t)} renvoie le nombre d'éléments d'un tuple \lstinline{t}.
  L'opérateur \lstinline{in} teste l'appartenance d'une valeur à un tuple.
\end{nxdef}

\begin{nxthm}[Déballage]
  Une fonction peut renvoyer un tuple pour communiquer \emph{plusieurs résultats} à la fois.
  On parle de \textbf{déballage} (\emph{unpacking}) lorsqu'on affecte chaque élément du tuple
  à une variable distincte : \lstinline{x, y = milieu(a, b)}.
\end{nxthm}

\approfondissement{Un tuple de longueur 1 s'écrit \lstinline{(42,)} avec une virgule
obligatoire. Sans virgule, \lstinline{(42)} est simplement l'entier 42 entre parenthèses.
Le tuple vide s'écrit \lstinline{()}.}

% ===================== PAGE 5 : FICHE MÉTHODE ==============================
\begin{fichemethode}{M1}{Choisir entre tuple et liste}

\quandUtiliser{Tu hésites entre un tuple et une liste pour stocker tes données.}

\pasApas{%
\begin{enumerate}
  \item Les données doivent-elles rester fixes après création ? $\to$ \textbf{tuple}.
  \item Faut-il ajouter, supprimer ou modifier des éléments ? $\to$ \textbf{liste}.
  \item Les éléments sont-ils de types différents (nom + âge + note) ? $\to$ \textbf{tuple}.
  \item S'agit-il d'une collection homogène (liste de scores) ? $\to$ \textbf{liste}.
\end{enumerate}}

\exempleRedige{%
Coordonnées GPS d'une station : \lstinline{("Tunis", 36.8, 10.2)} $\to$ tuple
(la position ne change pas). Relevés de température horaires : \lstinline{[18, 20, 19]}
$\to$ liste (on ajoute des mesures).}

\pieges{%
\begin{itemize}
  \item Ne pas utiliser une liste quand les données sont fixes.
  \item Ne pas oublier la virgule pour un tuple d'un seul élément.
\end{itemize}}

\verifier{Si tu as choisi un tuple, vérifie que tu ne tentes jamais de le modifier.}

\sEntrainer{Exercices 2, 4, 8.}

\end{fichemethode}

  \item \lstinline{type(station)} renvoie \lstinline{<class 'tuple'>}.
        Le tuple contient 3 éléments : \lstinline{len(station)} vaut 3.
  \item \lstinline{station[0]} vaut \lstinline{"Tunis"}, \lstinline{station[2]} vaut 10.2.
  \item Non : un tuple est non mutable. L'instruction provoque \lstinline{TypeError}.

\section*{Code de référence}

\begin{codereference}{Fonction renvoyant un tuple}
\begin{python}
def milieu(a, b):
    """Renvoie le milieu de deux points."""
    mx = (a[0] + b[0]) / 2
    my = (a[1] + b[1]) / 2
    return (mx, my)

x, y = milieu((1, 2), (5, 8))
print(x, y)  # 3.0 5.0
\end{python}
\end{codereference}

\begin{console}
>>> milieu((0, 0), (6, 8))
(3.0, 4.0)
>>> type(milieu((0, 0), (6, 8)))
<class 'tuple'>
\end{console}
