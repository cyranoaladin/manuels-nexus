\begin{exercice}{1NSI-TC-EX-040}{2}{10}
On considère la fonction suivante :
\begin{python}
def fusionner_dicos(d1, d2):
    resultat = {}
    for cle in d1:
        resultat[cle] = d1[cle]
    for cle in d2:
        resultat[cle] = d2[cle]
    return resultat

a = {"x": 1, "y": 2}
b = {"y": 10, "z": 3}
print(fusionner_dicos(a, b))
\end{python}
\begin{enumerate}
  \item Donner, sans exécuter, l'affichage produit.
  \item Que se passe-t-il lorsqu'une clé est présente dans les deux dictionnaires ?
  \item Les dictionnaires \lstinline{a} et \lstinline{b} sont-ils modifiés ?
\end{enumerate}
\end{exercice}
