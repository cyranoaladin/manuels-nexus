\begin{python}
def decaler(tab):
    dernier = tab[-1]
    for i in range(len(tab) - 1, 0, -1):
        tab[i] = tab[i - 1]
    tab[0] = dernier

scores = [10, 20, 30, 40]
decaler(scores)
print(scores)
\end{python}
