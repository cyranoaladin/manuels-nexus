\coupDePouce{1}{On te demande de prévoir les affichages d'un parcours de dictionnaire avec une boucle \lstinline{for}, puis de réécrire ce parcours avec \lstinline{.items()}.}
\coupDePouce{2}{\lstinline{for cle in station} parcourt uniquement les \emph{clés}. Pour obtenir la valeur associée, on écrit \lstinline{station[cle]}. Avec \lstinline{.items()}, la boucle devient \lstinline{for cle, val in station.items()}.}
