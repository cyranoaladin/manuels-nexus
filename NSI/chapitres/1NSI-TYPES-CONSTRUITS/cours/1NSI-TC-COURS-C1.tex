\section{Les p-uplets (tuples)}
\label{sec:tuples}

\margeAppui{Un p-uplet est une collection \emph{ordonnée} et \emph{non modifiable} de valeurs.
En Python, on utilise le mot anglais \emph{tuple}.}

\begin{nxdef}[D1]
  Un \textbf{p-uplet} (ou \emph{tuple}) est une séquence ordonnée de $p$ éléments, éventuellement
  de types différents, séparés par des virgules et entourés de parenthèses.
  Un p-uplet est \textbf{non mutable} : une fois créé, on ne peut ni ajouter, ni retirer,
  ni modifier ses éléments.
\end{nxdef}

\exemple{Le tuple \lstinline{station = ("Tunis-Carthage", 36.8, 10.2)} contient trois éléments :
une chaîne de caractères et deux flottants. On accède à chaque élément par son indice,
en commençant à 0 :}

\begin{python}
station = ("Tunis-Carthage", 36.8, 10.2)
nom = station[0]        # "Tunis-Carthage"
latitude = station[1]   # 36.8
longitude = station[2]  # 10.2
\end{python}

\contreexemple{Toute tentative de modification provoque une erreur :}

\begin{console}
>>> station[1] = 37.0
TypeError: 'tuple' object does not support item assignment
\end{console}

\erreurFrequente{Modifier un tuple comme une liste (EF1). Un tuple est immuable par conception :
si l'on a besoin de modifier des données, il faut utiliser une liste.}

\begin{nxdef}[D2]
  La fonction \lstinline{len(t)} renvoie le nombre d'éléments d'un tuple \lstinline{t}.
  L'opérateur \lstinline{in} teste l'appartenance d'une valeur à un tuple.
\end{nxdef}

\exemple{Vérifier qu'une station est dans une zone :}

\begin{python}
coordonnees = (36.8, 10.2)
print(len(coordonnees))      # 2
print(36.8 in coordonnees)   # True
\end{python}

\begin{nxdef}[D3]
  Une fonction peut renvoyer un tuple pour communiquer \emph{plusieurs résultats} à la fois.
  On parle de \textbf{déballage} (\emph{unpacking}) lorsqu'on affecte chaque élément du tuple
  à une variable distincte.
\end{nxdef}

\begin{codereference}{Fonction renvoyant un tuple}
\begin{python}
def milieu(a, b):
    """Renvoie le milieu de deux points (tuples de coordonnées).

    Préconditions : a et b sont des tuples de 2 flottants.
    """
    mx = (a[0] + b[0]) / 2
    my = (a[1] + b[1]) / 2
    return (mx, my)

# Déballage du résultat
x, y = milieu((1, 2), (5, 8))
print(x, y)  # 3.0 5.0
\end{python}
\end{codereference}

\margeAppui{Le déballage est très utilisé en Python. On écrit
\lstinline{x, y = milieu(a, b)} plutôt que
\lstinline{r = milieu(a, b); x = r[0]; y = r[1]}.}

\approfondissement{Un tuple de longueur 1 s'écrit \lstinline{(42,)} avec une virgule
obligatoire. Sans virgule, \lstinline{(42)} est simplement l'entier 42 entre parenthèses.
Le tuple vide s'écrit \lstinline{()}.}
