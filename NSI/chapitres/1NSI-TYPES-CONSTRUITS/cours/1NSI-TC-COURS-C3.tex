\section{Les grilles : tableaux de tableaux}
\label{sec:grilles}

\margeAppui{Une grille (ou matrice) se représente en Python par un tableau dont chaque
élément est lui-même un tableau. On parle de \emph{tableau à deux dimensions}.}

\begin{nxdef}[D7]
  Une \textbf{grille} est un tableau de tableaux où chaque sous-tableau représente une ligne.
  On accède à l'élément de la ligne \lstinline{i} et de la colonne \lstinline{j}
  par \lstinline{grille[i][j]}.
\end{nxdef}

\exemple{Une grille de pixels en niveaux de gris (0 = noir, 255 = blanc) :}

\begin{python}
image = [
    [0,   0,   0],
    [0, 255,   0],
    [0,   0,   0],
]
print(image[1][1])  # 255 (pixel central)
\end{python}

\begin{codereference}{Parcours d'une grille}
\begin{python}
def afficher_grille(grille):
    """Affiche une grille ligne par ligne."""
    for ligne in grille:
        for valeur in ligne:
            print(f"{valeur:4d}", end="")
        print()

def nb_lignes(grille):
    """Renvoie le nombre de lignes."""
    return len(grille)

def nb_colonnes(grille):
    """Renvoie le nombre de colonnes (suppose grille non vide)."""
    return len(grille[0])
\end{python}
\end{codereference}

\erreurFrequente{Confondre lignes et colonnes. Dans \lstinline{grille[i][j]},
\lstinline{i} est le numéro de \emph{ligne} et \lstinline{j} le numéro de \emph{colonne}.}

\begin{nxdef}[D8]
  Pour construire une grille $n \times m$ remplie de zéros, on utilise une
  \textbf{compréhension imbriquée} :
  \lstinline{[[0 for j in range(m)] for i in range(n)]}.
\end{nxdef}

\contreexemple{Ne jamais écrire \lstinline{[[0] * m] * n} : toutes les lignes
seraient des \emph{alias} du même objet (voir section~\ref{sec:mutabilite}).}

\begin{python}
# CORRECT : chaque ligne est un objet distinct
grille = [[0 for j in range(3)] for i in range(3)]
grille[0][0] = 1
print(grille)  # [[1, 0, 0], [0, 0, 0], [0, 0, 0]]

# INCORRECT : alias dangereux
mauvaise = [[0] * 3] * 3
mauvaise[0][0] = 1
print(mauvaise)  # [[1, 0, 0], [1, 0, 0], [1, 0, 0]]
\end{python}
