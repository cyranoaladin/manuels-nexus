\section*{Définition D1 — p-uplet}
\begin{nxdef}[D1]
Un p-uplet est une séquence ordonnée d'éléments, éventuellement de types différents,
séparés par des virgules et entourés de parenthèses.
\end{nxdef}
\exemple{La station est définie dans le texte et le listing par la même valeur :
\lstinline{station = ("Tunis-Carthage", 36.8, 10.2)}.}
\begin{python}
station = ("Tunis-Carthage", 36.8, 10.2)
nom = station[0]
latitude = station[1]
longitude = station[2]
\end{python}
\begin{codereference}{Déballer une station}
\begin{python}
nom, latitude, longitude = station
print(nom)
\end{python}
\end{codereference}
\begin{center}\memtable{nom,lat.,long.}{0}\end{center}

\section*{QCM diagnostique}
\textbf{Q2.} Que se passe-t-il lors de l'exécution de ce code ?
\begin{python}
t = (1, 2)
t[0] = 5
\end{python}
\begin{enumerate}[label=\Alph*.]
  \item \lstinline{t} vaut \lstinline{(5, 2)}.
  \item Une erreur \lstinline{TypeError} est levée.
  \item \lstinline{t} vaut \lstinline{(1, 2, 5)}.
\end{enumerate}
\begin{nxcard}[diagnostic séparé]
Si tu as choisi A, tu confonds un p-uplet non mutable avec une liste modifiable.
\end{nxcard}

\section*{Accents et code exact}
La méthode crée une séquence de données ; la réponse est vérifiée à chaque étape.
L'accès aux éléments nécessite une référence, sans modifier les données précédentes.
\begin{console}
>>> donnees = [3, 5, 8]
>>> [x for x in donnees if x > 4]
[5, 8]
\end{console}

\section*{Contrôle transversal}
\begin{nxthm}[T1]
Un tableau mutable partage ses effets lorsque deux variables référencent le même objet.
\end{nxthm}
\begin{nxprop}[P1]
Une copie explicite évite les effets de bord dans les données modifiables.
\end{nxprop}
\approfondissement{La complexité d'un accès dépend de la structure choisie.}
