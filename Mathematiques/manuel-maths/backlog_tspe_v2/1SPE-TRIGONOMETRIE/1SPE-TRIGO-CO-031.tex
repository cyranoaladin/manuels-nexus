\begin{corrige}{1SPE-TRIGO-EX-031}

\textbf{1.} $\cos x = \dfrac{1}{2}$. On a $\cos\dfrac{\pi}{3} = \dfrac{1}{2}$, donc $x = \dfrac{\pi}{3} + 2k\pi$ ou $x = -\dfrac{\pi}{3} + 2k\pi$, $k \in \mathbb{Z}$.

\textbf{2.} $\cos x = -1$. On a $\cos\pi = -1$, donc $x = \pi + 2k\pi$, $k \in \mathbb{Z}$.

\textbf{3.} $\sin x = 0$. On a $\sin 0 = 0$, donc $x = k\pi$, $k \in \mathbb{Z}$.

\end{corrige}
