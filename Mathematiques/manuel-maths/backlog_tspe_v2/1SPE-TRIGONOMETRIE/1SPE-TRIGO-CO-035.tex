\begin{corrige}{1SPE-TRIGO-EX-035}

\textbf{1.} $\cos(2x) = \dfrac{1}{2}$ : $2x = \dfrac{\pi}{3} + 2k\pi$ ou $2x = -\dfrac{\pi}{3} + 2k\pi$, donc $x = \dfrac{\pi}{6} + k\pi$ ou $x = -\dfrac{\pi}{6} + k\pi$.

Sur $[0;2\pi]$ : $x \in \left\{\dfrac{\pi}{6},\;\dfrac{5\pi}{6},\;\dfrac{7\pi}{6},\;\dfrac{11\pi}{6}\right\}$.

\textbf{2.} $\sin(2x) = -\dfrac{\sqrt{3}}{2}$ : $2x = -\dfrac{\pi}{3} + 2k\pi$ ou $2x = \pi + \dfrac{\pi}{3} + 2k\pi = \dfrac{4\pi}{3} + 2k\pi$.

Donc $x = -\dfrac{\pi}{6} + k\pi$ ou $x = \dfrac{2\pi}{3} + k\pi$.

Sur $[0;2\pi]$ : $x \in \left\{\dfrac{2\pi}{3},\;\dfrac{5\pi}{6},\;\dfrac{5\pi}{3},\;\dfrac{11\pi}{6}\right\}$.

\end{corrige}
