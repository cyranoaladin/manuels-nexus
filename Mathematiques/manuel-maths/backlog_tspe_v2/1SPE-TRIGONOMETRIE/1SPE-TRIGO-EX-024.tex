\begin{exercice}{1SPE-TRIGO-EX-024}{1}{10}
\begin{enumerate}
  \item A l'aide de $\cos(2a) = 2\cos^2 a - 1$, montrer que $\cos^2 a = \dfrac{1 + \cos(2a)}{2}$.
  \item De meme, montrer que $\sin^2 a = \dfrac{1 - \cos(2a)}{2}$.
  \item En deduire la valeur exacte de $\cos^2\!\left(\dfrac{\pi}{8}\right)$.
\end{enumerate}
\end{exercice}
