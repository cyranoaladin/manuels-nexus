\begin{exercice}{1SPE-TRIGO-EX-049}{3}{18}
On pose $f(x) = \cos(2x) + \cos x$.
\begin{enumerate}
  \item Montrer que $f(x) = 2\cos^2 x + \cos x - 1$.
  \item Factoriser l'expression $2u^2 + u - 1$ pour $u = \cos x$.
  \item En deduire les solutions de $f(x) = 0$ sur $[0\,;\,2\pi]$.
  \item Dresser le tableau de signes de $f$ sur $[0\,;\,2\pi]$.
\end{enumerate}
\end{exercice}
