\subsection*{Rappel — Resolution d'equations du second degre}

Pour $au^2 + bu + c = 0$ : calculer $\Delta = b^2 - 4ac$, puis $u = \frac{-b \pm \sqrt{\Delta}}{2a}$.

\begin{exercice}{1SPE-TRIGO-FR-R5-EX1}{1}{5}
Resoudre les equations suivantes.
\begin{enumerate}
  \item $2u^2 - 3u + 1 = 0$
  \item $u^2 + u - 2 = 0$
  \item $4u^2 - 1 = 0$
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-TRIGO-FR-R5-EX1}
\textbf{1.} $\Delta = 9 - 8 = 1$. $u = \frac{3 \pm 1}{4}$, d'ou $u = 1$ ou $u = \frac{1}{2}$.

\textbf{2.} $\Delta = 1 + 8 = 9$. $u = \frac{-1 \pm 3}{2}$, d'ou $u = 1$ ou $u = -2$.

\textbf{3.} $4u^2 = 1$, $u^2 = \frac{1}{4}$, $u = \frac{1}{2}$ ou $u = -\frac{1}{2}$.
\end{corrige}
