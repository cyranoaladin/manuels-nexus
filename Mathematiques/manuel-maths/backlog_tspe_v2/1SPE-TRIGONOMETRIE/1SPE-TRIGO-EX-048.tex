\begin{exercice}{1SPE-TRIGO-EX-048}{2}{15}
On pose $f(x) = 2\sin x - \sqrt{3}$.
\begin{enumerate}
  \item Resoudre $f(x) = 0$ sur $[0\,;\,2\pi]$.
  \item Dresser le tableau de variations de $f$ sur $[0\,;\,2\pi]$.
  \item En deduire le signe de $f(x)$ sur $[0\,;\,2\pi]$.
  \item Resoudre l'inequation $2\sin x \geqslant \sqrt{3}$ sur $[0\,;\,2\pi]$.
\end{enumerate}
\end{exercice}
