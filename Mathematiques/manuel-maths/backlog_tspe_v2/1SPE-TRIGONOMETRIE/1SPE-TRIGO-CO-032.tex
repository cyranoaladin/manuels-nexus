\begin{corrige}{1SPE-TRIGO-EX-032}

\textbf{1.} $\sin x = \dfrac{\sqrt{2}}{2}$. $\alpha = \dfrac{\pi}{4}$.
$x = \dfrac{\pi}{4} + 2k\pi$ ou $x = \pi - \dfrac{\pi}{4} + 2k\pi = \dfrac{3\pi}{4} + 2k\pi$, $k \in \mathbb{Z}$.

\textbf{2.} $\sin x = \dfrac{1}{2}$. $\alpha = \dfrac{\pi}{6}$.
$x = \dfrac{\pi}{6} + 2k\pi$ ou $x = \dfrac{5\pi}{6} + 2k\pi$, $k \in \mathbb{Z}$.

\textbf{3.} $\sin x = -\dfrac{\sqrt{3}}{2}$. $\alpha = -\dfrac{\pi}{3}$.
$x = -\dfrac{\pi}{3} + 2k\pi$ ou $x = \pi + \dfrac{\pi}{3} + 2k\pi = \dfrac{4\pi}{3} + 2k\pi$, $k \in \mathbb{Z}$.

\end{corrige}
