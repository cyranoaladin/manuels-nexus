\begin{fichemethode}{M4}{Resoudre une equation trigonometrique}

\quandUtiliser{%
  Utiliser cette methode lorsqu'on doit resoudre une equation du type $\cos x = a$ ou $\sin x = a$,
  ou une equation qui s'y ramene apres transformation.
}

\pasApas{%
  \begin{enumerate}
    \item Verifier que $a \in [-1\,;\,1]$ (sinon pas de solution).
    \item Identifier l'angle remarquable $\alpha$ tel que $\cos\alpha = a$ (ou $\sin\alpha = a$).
    \item Ecrire les deux familles de solutions :
      \begin{itemize}
        \item Pour $\cos x = a$ : $x = \alpha + 2k\pi$ ou $x = -\alpha + 2k\pi$.
        \item Pour $\sin x = a$ : $x = \alpha + 2k\pi$ ou $x = \pi - \alpha + 2k\pi$.
      \end{itemize}
    \item Si l'enonce restreint a un intervalle $[a\,;\,b]$, selectionner les valeurs de $k$ qui conviennent.
  \end{enumerate}
}

\exempleRedige{%
  Resoudre $\sin x = -\dfrac{\sqrt{3}}{2}$ sur $[0\,;\,2\pi]$.

  \medskip
  \commentaireMarge{Identifier $\alpha$ tel que $\sin\alpha = -\frac{\sqrt{3}}{2}$.}%
  On a $\sin\!\left(-\dfrac{\pi}{3}\right) = -\dfrac{\sqrt{3}}{2}$, donc $\alpha = -\dfrac{\pi}{3}$.

  \commentaireMarge{Ecrire les solutions generales.}%
  Les solutions sont :
  \[
    x = -\frac{\pi}{3} + 2k\pi \quad \text{ou} \quad x = \pi + \frac{\pi}{3} + 2k\pi = \frac{4\pi}{3} + 2k\pi.
  \]

  \commentaireMarge{Selectionner dans $[0\,;\,2\pi]$.}%
  \begin{itemize}
    \item $x = -\dfrac{\pi}{3} + 2\pi = \dfrac{5\pi}{3} \in [0\,;\,2\pi]$. \checkmark
    \item $x = \dfrac{4\pi}{3} \in [0\,;\,2\pi]$. \checkmark
  \end{itemize}

  Les solutions sur $[0\,;\,2\pi]$ sont $x = \dfrac{4\pi}{3}$ et $x = \dfrac{5\pi}{3}$.
}

\pieges{%
  \begin{itemize}
    \item \textbf{Piege 1.} Oublier la deuxieme famille de solutions. Il y a toujours \textbf{deux} familles
      (sauf cas particulier $a = \pm 1$).
    \item \textbf{Piege 2.} Oublier le ``$+ 2k\pi$'' dans les solutions generales.
    \item \textbf{Piege 3.} Confondre les formules pour $\cos$ et $\sin$ :
      pour $\cos$ c'est $\pm\alpha$, pour $\sin$ c'est $\alpha$ et $\pi - \alpha$.
  \end{itemize}
}

\verifier{%
  Substituer chaque solution trouvee dans l'equation initiale et verifier que l'egalite est satisfaite.
}
