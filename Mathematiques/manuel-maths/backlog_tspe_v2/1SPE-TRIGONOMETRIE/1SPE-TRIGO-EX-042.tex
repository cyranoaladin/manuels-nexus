\begin{exercice}{1SPE-TRIGO-EX-042}{1}{10}

\begin{enumerate}
  \item Quelle est la periode de $g$ ?
  \item Dresser le tableau de variations de $g$ sur $[0\,;\,2\pi]$.
  \item Donner le maximum et le minimum de $g$ sur $\mathbb{R}$.
\end{enumerate}
