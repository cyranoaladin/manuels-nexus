\begin{corrige}{1SPE-TRIGO-EX-025}

\textbf{1.} $\cos(a+b) + \cos(a-b) = [\cos a\cos b - \sin a\sin b] + [\cos a\cos b + \sin a\sin b] = 2\cos a\cos b$.

\textbf{2.} $\sin(a+b) + \sin(a-b) = [\sin a\cos b + \cos a\sin b] + [\sin a\cos b - \cos a\sin b] = 2\sin a\cos b$.

\textbf{3.} Avec $a = \dfrac{\pi}{4}$ et $b = \dfrac{\pi}{6}$ : $\cos\!\left(\dfrac{5\pi}{12}\right) + \cos\!\left(\dfrac{\pi}{12}\right) = 2\cos\dfrac{\pi}{4}\cos\dfrac{\pi}{6} = 2 \times \dfrac{\sqrt{2}}{2} \times \dfrac{\sqrt{3}}{2} = \dfrac{\sqrt{6}}{2}$.

\end{corrige}
