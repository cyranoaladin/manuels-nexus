\begin{exercice}{1SPE-TRIGO-EX-041}{1}{10}
On considere la fonction $f$ definie sur $\mathbb{R}$ par $f(x) = 2\cos x - 1$.
\begin{enumerate}
  \item Calculer $f(0)$, $f\!\left(\dfrac{\pi}{3}\right)$, $f(\pi)$ et $f(2\pi)$.
  \item Dresser le tableau de variations de $f$ sur $[0\,;\,2\pi]$.
  \item Determiner les extremums de $f$ sur $[0\,;\,2\pi]$.
  \item Resoudre $f(x) = 0$ sur $[0\,;\,2\pi]$.
\end{enumerate}
\end{exercice}
