\begin{exercice}{1SPE-TRIGO-EX-044}{2}{15}
On pose $f(x) = \cos x + \sin x$.
\begin{enumerate}
  \item Montrer que $f(x) = \sqrt{2}\cos\!\left(x - \dfrac{\pi}{4}\right)$.
  \item En deduire la periode, le maximum et le minimum de $f$.
  \item Dresser le tableau de variations de $f$ sur $[0\,;\,2\pi]$.
  \item Resoudre $f(x) = 0$ sur $[0\,;\,2\pi]$.
\end{enumerate}
\end{exercice}
