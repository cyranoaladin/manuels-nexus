\begin{exercice}{1SPE-TRIGO-EX-027}{2}{12}
\textbf{Linearisation de $\cos^4 x$.}
\begin{enumerate}
  \item Exprimer $\cos^2 x$ en fonction de $\cos(2x)$ (formule de linearisation).
  \item En deduire $\cos^4 x = \left(\cos^2 x\right)^2$ en fonction de $\cos(2x)$ et $\cos(4x)$.
  \item Montrer que $\cos^4 x = \dfrac{3}{8} + \dfrac{1}{2}\cos(2x) + \dfrac{1}{8}\cos(4x)$.
\end{enumerate}
\end{exercice}
