\begin{ouverture}[\decorGrille]{Programmation\\ dynamique}
  \item Reconnaître les sous-problèmes qui se recouvrent.
  \item Mémoïser pour éviter les recalculs.

\section{Mémoïsation}

\begin{definition}[Mémoïsation]
La \emph{mémoïsation} consiste à stocker les résultats déjà calculés
d'une fonction pour les réutiliser, transformant un calcul
exponentiel en calcul linéaire.
\end{definition}

\begin{methode}[Rendre une fonction dynamique]
\begin{enumerate}
  \item Identifier la relation de récurrence.
  \item Choisir une structure de mémoire (tableau, dictionnaire).
  \item Remplir la mémoire des cas de base vers le cas cherché.
\end{enumerate}
\end{methode}

\begin{python}[Fibonacci mémoïsé]
def fib(n: int, memo: dict = {}) -> int:
    if n < 2:
        return n
    if n not in memo:
        memo[n] = fib(n - 1) + fib(n - 2)   # sous-problèmes
    return memo[n]

print(fib(50))   # instantané : 12586269025
\end{python}

\begin{retenir}
Sans mémoïsation, \code{fib} effectue $O(\varphi^{\,n})$ appels ;
avec, $O(n)$. Le gain est spectaculaire dès $n \geq 40$.
\end{retenir}
