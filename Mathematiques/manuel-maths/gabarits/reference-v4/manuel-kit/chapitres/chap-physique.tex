\section{Onde et célérité}

\begin{definition}[Onde progressive]
Une \emph{onde mécanique progressive} est la propagation d'une
perturbation dans un milieu matériel, sans transport de matière mais
avec transport d'énergie.
\end{definition}

\begin{propriete}
Pour une onde périodique, la longueur d'onde $\lambda$, la période
$T$ et la célérité $v$ vérifient $\lambda = v\,T = \dfrac{v}{f}$.
\end{propriete}

\begin{retenir}
La célérité dépend du \emph{milieu}, pas de la source : changer la
fréquence modifie $\lambda$, jamais $v$.
\end{retenir}

\begin{exemple}
Un son de fréquence \SI{440}{\hertz} se propage à
\SI{340}{\meter\per\second} : $\lambda = \frac{340}{440}
\approx \SI{0.77}{\meter}$.
\end{exemple}

\begin{histoire}[Doppler et les étoiles (1842)]
Christian Doppler relie le décalage de fréquence perçu au mouvement
de la source ; l'effet, d'abord contesté, fonde aujourd'hui la mesure
des vitesses en astronomie.
\end{histoire}
