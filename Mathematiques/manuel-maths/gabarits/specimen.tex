\section{Les p-uplets (tuples)}

\margeAppui{Un p-uplet est une collection \emph{ordonnee} et \emph{non modifiable} de valeurs. En Python, on utilise le mot anglais \emph{tuple}.}

\begin{nxdef}[D1]
  Un \textbf{p-uplet} (ou \emph{tuple}) est une sequence ordonnee de $p$ elements,
  eventuellement de types differents, separes par des virgules et entoures de parentheses.
  Un p-uplet est \textbf{non mutable} : une fois cree, on ne peut ni ajouter, ni retirer,
  ni modifier ses elements.
\end{nxdef}

\exemple{Le tuple \lstinline{station = ("Tunis", 36.8, 10.2)} contient trois elements.}

\begin{python}
station = ("Tunis-Carthage", 36.8, 10.2)
nom = station[0]        # "Tunis-Carthage"
latitude = station[1]   # 36.8
\end{python}

\contreexemple{Toute tentative de modification provoque une erreur \lstinline{TypeError}.}

\begin{nxthm}[Deballage]
  Une fonction peut renvoyer un tuple pour communiquer \emph{plusieurs resultats} a la fois.
  On parle de \textbf{deballage} (\emph{unpacking}) lorsqu'on affecte chaque element du tuple
  a une variable distincte : \lstinline{x, y = f(a, b)}.
\end{nxthm}

\margeAppui{Le deballage est tres utilise en Python. On ecrit
\lstinline{x, y = milieu(a, b)} plutot que
\lstinline{r = milieu(a, b)} puis \lstinline{x = r[0]}.}

\erreurFrequente{Modifier un tuple comme une liste (EF1).
Un tuple est immuable par conception.
\par\textcolor{nxGris}{\sout{Copie fautive : \lstinline{station[1] = 37.0}}}
\par\textcolor{nxVert}{Correction : creer un nouveau tuple ou utiliser une liste.}}

\approfondissement{Un tuple de longueur 1 s'ecrit \lstinline{(42,)} avec une virgule
obligatoire. Sans virgule, \lstinline{(42)} est simplement l'entier 42 entre parentheses.
Le tuple vide s'ecrit \lstinline{()}.}

% ===================== PAGE 5 : FICHE METHODE ===============================
\begin{fichemethode}{M1}{Choisir entre tuple et liste}

\quandUtiliser{Tu hesites entre un tuple et une liste pour stocker tes donnees.}

\pasApas{%
\begin{enumerate}
  \item Les donnees doivent-elles rester fixes apres creation ? $\to$ \textbf{tuple}.
  \item Faut-il ajouter, supprimer ou modifier des elements ? $\to$ \textbf{liste}.
  \item Les elements sont-ils de types differents (nom + age) ? $\to$ \textbf{tuple}.
  \item S'agit-il d'une collection homogene (scores) ? $\to$ \textbf{liste}.
\end{enumerate}}

\exempleRedige{%
Coordonnees GPS : \lstinline{("Tunis", 36.8, 10.2)} $\to$ tuple.
Releves horaires : \lstinline{[18, 20, 19]} $\to$ liste.}

\pieges{%
\begin{itemize}
  \item Ne pas oublier la virgule pour un tuple d'un seul element.
\end{itemize}}

\verifier{Si tu as choisi un tuple, verifie que tu ne tentes jamais de le modifier.}

\sEntrainer{Exercices 2, 4, 8.}

\end{fichemethode}

\section*{Code de reference}

\begin{codereference}{Fonction renvoyant un tuple}
\begin{python}
def milieu(a, b):
    """Renvoie le milieu de deux points."""
    mx = (a[0] + b[0]) / 2
    my = (a[1] + b[1]) / 2
    return (mx, my)

x, y = milieu((1, 2), (5, 8))
print(x, y)  # 3.0 5.0
\end{python}
\end{codereference}

\begin{console}
>>> milieu((0, 0), (6, 8))
(3.0, 4.0)
>>> type(milieu((0, 0), (6, 8)))
<class 'tuple'>
\end{console}
