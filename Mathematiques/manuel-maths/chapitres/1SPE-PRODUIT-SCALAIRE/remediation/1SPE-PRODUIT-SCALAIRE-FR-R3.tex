\subsection*{Rappel — Valeurs remarquables de cos et sin}

\begin{center}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
$\theta$ & $0$ & $\frac{\pi}{6}$ & $\frac{\pi}{4}$ & $\frac{\pi}{3}$ & $\frac{\pi}{2}$ \\
\hline
$\cos\theta$ & $1$ & $\frac{\sqrt{3}}{2}$ & $\frac{\sqrt{2}}{2}$ & $\frac{1}{2}$ & $0$ \\
\hline
$\sin\theta$ & $0$ & $\frac{1}{2}$ & $\frac{\sqrt{2}}{2}$ & $\frac{\sqrt{3}}{2}$ & $1$ \\
\hline
\end{tabular}
\end{center}

\begin{exercice}{1SPE-PRODSCAL-FR-R3-EX1}{1}{5}
Donner les valeurs exactes.
\begin{enumerate}
  \item $\cos\!\left(\frac{\pi}{3}\right)$ et $\sin\!\left(\frac{\pi}{3}\right)$
  \item $\cos\!\left(\frac{2\pi}{3}\right)$ et $\sin\!\left(\frac{2\pi}{3}\right)$
  \item $\cos\!\left(\frac{\pi}{4}\right)$
\end{enumerate}
\end{exercice}
