\subsection*{Rappel — Identites remarquables}

$(a+b)^2 = a^2 + 2ab + b^2$ \quad $(a-b)^2 = a^2 - 2ab + b^2$ \quad $(a+b)(a-b) = a^2 - b^2$

\begin{exercice}{1SPE-PRODSCAL-FR-R4-EX1}{1}{6}
Developper et reduire.
\begin{enumerate}
  \item $(x+3)^2$
  \item $(2x-1)^2$
  \item $(x+2)(x-2)$
\end{enumerate}
\end{exercice}
