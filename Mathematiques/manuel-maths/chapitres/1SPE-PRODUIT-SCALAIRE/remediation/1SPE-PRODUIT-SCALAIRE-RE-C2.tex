\subsection*{Remediation C2 — Proprietes du produit scalaire}

\begin{exercice}{1SPE-PRODSCAL-RE-C2-EX1}{1}{6}
$\|\vec{u}\| = 2$, $\|\vec{v}\| = 3$, $\vec{u} \cdot \vec{v} = 4$. Calculer $\|\vec{u} + \vec{v}\|^2$.
\end{exercice}

\begin{corrige}{1SPE-PRODSCAL-RE-C2-EX1}
$\|\vec{u}+\vec{v}\|^2 = 4 + 8 + 9 = 21$, donc $\|\vec{u}+\vec{v}\| = \sqrt{21}$.
\end{corrige}
