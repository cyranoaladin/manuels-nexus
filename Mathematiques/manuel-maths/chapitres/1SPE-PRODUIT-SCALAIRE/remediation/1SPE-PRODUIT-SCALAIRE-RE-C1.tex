\subsection*{Remediation C1 — Calculer un produit scalaire}

\begin{exercice}{1SPE-PRODSCAL-RE-C1-EX1}{1}{6}
Calculer $\vec{u} \cdot \vec{v}$ dans chaque cas.
\begin{enumerate}
  \item $\vec{u}\begin{pmatrix} 2 \\ 3 \end{pmatrix}$, $\vec{v}\begin{pmatrix} 1 \\ -1 \end{pmatrix}$
  \item $\|\vec{u}\| = 6$, $\|\vec{v}\| = 2$, angle $= \frac{\pi}{4}$
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-PRODSCAL-RE-C1-EX1}
\textbf{1.} $\vec{u} \cdot \vec{v} = 2 - 3 = -1$. \textbf{2.} $\vec{u} \cdot \vec{v} = 12 \times \frac{\sqrt{2}}{2} = 6\sqrt{2}$.
\end{corrige}
