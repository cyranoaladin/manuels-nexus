\subsection*{C1 — Produit scalaire : definition et calcul}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} Soient $\vec{u}\begin{pmatrix} 3 \\ -2 \end{pmatrix}$ et $\vec{v}\begin{pmatrix} 1 \\ 4 \end{pmatrix}$. Que vaut $\vec{u} \cdot \vec{v}$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $-5$
    \item $5$
    \item $-11$
    \item $11$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} Si $\|\vec{u}\| = 5$, $\|\vec{v}\| = 4$ et $(\vec{u}, \vec{v}) = \dfrac{\pi}{3}$, alors $\vec{u} \cdot \vec{v}$ vaut :

  \begin{enumerate}[label=\Alph*.]
    \item $20$
    \item $10$
    \item $10\sqrt{3}$
    \item $10\sqrt{2}$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} Le produit scalaire $\vec{u} \cdot \vec{v}$ est :

  \begin{enumerate}[label=\Alph*.]
    \item un vecteur
    \item un nombre reel
    \item toujours positif
    \item defini uniquement si $\vec{u}$ et $\vec{v}$ sont non nuls
  \end{enumerate}

\end{enumerate}

\subsection*{C2 — Proprietes du produit scalaire}

\begin{enumerate}[resume]

% ---- Q4 ----
\item \textbf{[Q4]} Que vaut $\|\vec{u} + \vec{v}\|^2$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\|\vec{u}\|^2 + \|\vec{v}\|^2$
    \item $\|\vec{u}\|^2 + 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2$
    \item $\|\vec{u}\|^2 - 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2$
    \item $(\vec{u} \cdot \vec{v})^2$
  \end{enumerate}

\bigskip

  \end{enumerate}

    \item $\sqrt{35}$

\item \textbf{[Q7]} Deux vecteurs $\vec{u}$ et $\vec{v}$ sont orthogonaux si et seulement si :

  \begin{enumerate}[label=\Alph*.]

    \item $\vec{u} \cdot \vec{v} < 0$
    \item $\vec{u} \cdot \vec{v} = 0$

\subsection*{C4 — Applications geometriques}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q10 ----
\item \textbf{[Q10]} $M$ appartient a la mediatrice de $[AB]$ si et seulement si :

  \begin{enumerate}[label=\Alph*.]
    \item $\overrightarrow{MA} \cdot \overrightarrow{MB} = 0$
    \item $MA = MB$
    \item $\overrightarrow{MA} + \overrightarrow{MB} = \vec{0}$
    \item $MA + MB = AB$
  \end{enumerate}

\bigskip

% ---- Q11 ----
\item \textbf{[Q11]} L'aire d'un triangle de cotes $a$ et $b$ formant un angle $\theta$ vaut :

  \begin{enumerate}[label=\Alph*.]
    \item $ab\cos\theta$
    \item $\dfrac{1}{2}ab\cos\theta$
    \item $\dfrac{1}{2}ab\sin\theta$
    \item $ab\sin\theta$
  \end{enumerate}

\bigskip

% ---- Q12 ----
\item \textbf{[Q12]} Le pied $H$ de la hauteur issue de $A$ dans un triangle $ABC$ verifie :

  \begin{enumerate}[label=\Alph*.]
    \item $\overrightarrow{AH} \cdot \overrightarrow{AB} = 0$
    \item $\overrightarrow{AH} \cdot \overrightarrow{BC} = 0$
    \item $\overrightarrow{AH} \cdot \overrightarrow{AC} = 0$
    \item $AH = BH$
  \end{enumerate}

\end{enumerate}

\subsection*{C5 — Formule d'Al-Kashi}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q13 ----
\item \textbf{[Q13]} La formule d'Al-Kashi s'ecrit :

  \begin{enumerate}[label=\Alph*.]
    \item $a^2 = b^2 + c^2 + 2bc\cos\widehat{A}$
    \item $a^2 = b^2 + c^2 - 2bc\cos\widehat{A}$
    \item $a^2 = b^2 - c^2 - 2bc\cos\widehat{A}$
    \item $a = b + c - 2bc\cos\widehat{A}$
  \end{enumerate}

\bigskip

% ---- Q14 ----
\item \textbf{[Q14]} Dans un triangle avec $b = 5$, $c = 8$ et $\widehat{A} = \dfrac{\pi}{3}$, que vaut $a^2$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $49$
    \item $89$
    \item $129$
    \item $9$
  \end{enumerate}

\bigskip

% ---- Q15 ----
\item \textbf{[Q15]} Si $\widehat{A} = \dfrac{\pi}{2}$, la formule d'Al-Kashi donne :

  \begin{enumerate}[label=\Alph*.]
    \item $a^2 = b^2 - c^2$
    \item $a^2 = b^2 + c^2$
    \item $a^2 = 2bc$
    \item $a = b + c$
  \end{enumerate}

\end{enumerate}
