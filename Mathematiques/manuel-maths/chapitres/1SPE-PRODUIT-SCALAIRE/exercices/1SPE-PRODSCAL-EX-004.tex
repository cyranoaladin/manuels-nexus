\begin{exercice}{1SPE-PRODSCAL-EX-004}{1}{6}

\begin{enumerate}
  \item $\vec{u} \cdot \vec{v}$ avec $\|\vec{u}\| = 4$, $\|\vec{v}\| = 6$ et $(\vec{u}, \vec{v}) = \dfrac{\pi}{4}$.

  \item $\vec{p} \cdot \vec{q}$ avec $\|\vec{p}\| = 7$, $\|\vec{q}\| = 1$ et $(\vec{p}, \vec{q}) = \dfrac{\pi}{2}$.
\end{enumerate}
\end{exercice}
