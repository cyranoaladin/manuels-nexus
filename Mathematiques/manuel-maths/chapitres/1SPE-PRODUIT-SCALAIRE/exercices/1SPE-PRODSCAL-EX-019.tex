\begin{exercice}{1SPE-PRODSCAL-EX-019}{3}{10}
\textbf{Identite du parallelogramme.}
Demontrer que pour tous vecteurs $\vec{u}$ et $\vec{v}$ :
\[
\|\vec{u} + \vec{v}\|^2 + \|\vec{u} - \vec{v}\|^2 = 2\left(\|\vec{u}\|^2 + \|\vec{v}\|^2\right).
\]
\textit{Interpretation geometrique : dans un parallelogramme, la somme des carres des diagonales est egale a la somme des carres des quatre cotes.}
\end{exercice}
