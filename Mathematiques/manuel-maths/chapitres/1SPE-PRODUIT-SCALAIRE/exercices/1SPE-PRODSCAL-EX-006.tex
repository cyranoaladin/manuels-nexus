\begin{exercice}{1SPE-PRODSCAL-EX-006}{2}{8}
Soient $\vec{u}\begin{pmatrix} 2 \\ 1 \end{pmatrix}$ et $\vec{v}\begin{pmatrix} 3 \\ -2 \end{pmatrix}$.
\begin{enumerate}
  \item Calculer $\vec{u} \cdot \vec{v}$.
  \item Calculer les coordonnees de $\vec{u} + \vec{v}$ et $\vec{u} - \vec{v}$.
  \item Calculer $(\vec{u} + \vec{v}) \cdot (\vec{u} - \vec{v})$.
  \item Verifier que ce resultat est egal a $\|\vec{u}\|^2 - \|\vec{v}\|^2$.
\end{enumerate}
\end{exercice}
