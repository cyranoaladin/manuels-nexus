\begin{exercice}{1SPE-PRODSCAL-EX-014}{1}{8}
\begin{enumerate}
  \item Demontrer l'\textbf{identite de polarisation} : pour tous vecteurs $\vec{u}$ et $\vec{v}$,
  \[
  \vec{u} \cdot \vec{v} = \frac{1}{2}\left(\|\vec{u} + \vec{v}\|^2 - \|\vec{u}\|^2 - \|\vec{v}\|^2\right).
  \]
  \item En deduire une autre forme : $\vec{u} \cdot \vec{v} = \dfrac{1}{4}\left(\|\vec{u} + \vec{v}\|^2 - \|\vec{u} - \vec{v}\|^2\right)$.
\end{enumerate}
\end{exercice}
