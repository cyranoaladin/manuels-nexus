\begin{exercice}{1SPE-PRODSCAL-EX-011}{1}{5}
Sachant que $\|\vec{u}\| = 3$, $\|\vec{v}\| = 4$ et $\vec{u} \cdot \vec{v} = -2$, calculer :
\begin{enumerate}
  \item $\|\vec{u} + \vec{v}\|^2$
  \item $\|\vec{u} - \vec{v}\|^2$
  \item $(\vec{u} + \vec{v}) \cdot (\vec{u} - \vec{v})$
\end{enumerate}
\end{exercice}
