\begin{exercice}{1SPE-PRODSCAL-EX-036}{2}{8}
Soient $\vec{u}\begin{pmatrix} 1 \\ 2 \end{pmatrix}$ et $\vec{v}\begin{pmatrix} 3 \\ -1 \end{pmatrix}$.
\begin{enumerate}
  \item Calculer le projete orthogonal de $\vec{v}$ sur $\vec{u}$ : $\text{proj}_{\vec{u}}\vec{v} = \dfrac{\vec{u} \cdot \vec{v}}{\|\vec{u}\|^2}\,\vec{u}$.
  \item Verifier que $\vec{v} - \text{proj}_{\vec{u}}\vec{v}$ est orthogonal a $\vec{u}$.
\end{enumerate}
\end{exercice}
