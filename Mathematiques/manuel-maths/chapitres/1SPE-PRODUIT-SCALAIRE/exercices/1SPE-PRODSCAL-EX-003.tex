\begin{exercice}{1SPE-PRODSCAL-EX-003}{1}{5}
Soit $\vec{u}\begin{pmatrix} 1 \\ 1 \end{pmatrix}$ et $\vec{v}\begin{pmatrix} -3 \\ 4 \end{pmatrix}$.
\begin{enumerate}
  \item Calculer $\vec{u} \cdot \vec{u}$ et $\vec{v} \cdot \vec{v}$.
  \item En deduire $\|\vec{u}\|$ et $\|\vec{v}\|$.
  \item Calculer $\vec{u} \cdot \vec{v}$.
\end{enumerate}
\end{exercice}
