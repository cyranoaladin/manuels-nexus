\begin{exercice}{1SPE-PRODSCAL-EX-017}{2}{8}
Soient $\vec{u}\begin{pmatrix} 1 \\ 2 \end{pmatrix}$ et $\vec{v}\begin{pmatrix} -2 \\ 1 \end{pmatrix}$.
\begin{enumerate}
  \item Montrer que $\vec{u}$ et $\vec{v}$ sont orthogonaux.
  \item Calculer $\|\vec{u} + \vec{v}\|^2$ par l'identite remarquable.
  \item Verifier que le theoreme de Pythagore est bien verifie : $\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + \|\vec{v}\|^2$.
\end{enumerate}
\end{exercice}
