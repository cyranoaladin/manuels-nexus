\begin{exercice}{1SPE-PRODSCAL-EX-012}{1}{6}
Demontrer que pour tous vecteurs $\vec{u}$ et $\vec{v}$ :
\[
\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2.
\]
\textit{Indication : developper $(\vec{u} + \vec{v}) \cdot (\vec{u} + \vec{v})$ par bilinearite.}
\end{exercice}
