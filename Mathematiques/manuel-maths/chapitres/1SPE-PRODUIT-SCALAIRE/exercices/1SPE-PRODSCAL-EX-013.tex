\begin{exercice}{1SPE-PRODSCAL-EX-013}{1}{6}
Soient $\vec{u}\begin{pmatrix} 2 \\ -1 \end{pmatrix}$ et $\vec{v}\begin{pmatrix} 1 \\ 3 \end{pmatrix}$.
\begin{enumerate}
  \item Calculer $\|\vec{u}\|^2$, $\|\vec{v}\|^2$ et $\vec{u} \cdot \vec{v}$.
  \item Calculer $\|\vec{u} + \vec{v}\|^2$ de deux manieres : directement et par l'identite remarquable.
  \item Calculer $\|\vec{u} + \vec{v}\|$.
\end{enumerate}
\end{exercice}
