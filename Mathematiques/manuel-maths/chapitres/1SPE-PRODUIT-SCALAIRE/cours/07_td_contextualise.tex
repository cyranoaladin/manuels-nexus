\section*{TD — Navigation et cap}

\textit{Duree : 50 minutes.}

\bigskip

\begin{center}
\textbf{Situation}

\medskip
\parbox{0.9\textwidth}{Un bateau quitte le port $P$ et navigue $8$ km dans la direction du vecteur $\vec{u}$ (cap nord-est, angle $\frac{\pi}{4}$ avec l'axe est). Il change ensuite de cap et parcourt $5$ km dans la direction du vecteur $\vec{v}$ (cap sud-est, angle $-\frac{\pi}{6}$ avec l'axe est). On souhaite calculer la distance totale au port et l'angle entre les deux trajets.}
\end{center}
