\begin{corrige}{1SPE-PRODSCAL-EX-020}
\begin{enumerate}
\item $\|\vec{u} - \vec{v}\|^2 = \|\vec{u}\|^2 - 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2$, donc $49 = 25 - 2\,\vec{u} \cdot \vec{v} + 9$, d'ou $\vec{u} \cdot \vec{v} = -\dfrac{15}{2}$.

\item $\|\vec{u} + \vec{v}\|^2 = 25 + 2 \times (-\frac{15}{2}) + 9 = 25 - 15 + 9 = 19$, donc $\|\vec{u} + \vec{v}\| = \sqrt{19}$.
\end{enumerate}
