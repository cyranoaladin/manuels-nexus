\begin{corrige}{1SPE-PRODSCAL-EX-012}
On a $\|\vec{u} + \vec{v}\|^2 = (\vec{u} + \vec{v}) \cdot (\vec{u} + \vec{v})$.

Par bilinearite :
\begin{align*}
(\vec{u} + \vec{v}) \cdot (\vec{u} + \vec{v}) &= \vec{u} \cdot \vec{u} + \vec{u} \cdot \vec{v} + \vec{v} \cdot \vec{u} + \vec{v} \cdot \vec{v} \\
&= \|\vec{u}\|^2 + 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2.
\end{align*}
On a utilise la symetrie $\vec{v} \cdot \vec{u} = \vec{u} \cdot \vec{v}$ et la propriete $\vec{u} \cdot \vec{u} = \|\vec{u}\|^2$.
\end{corrige}
