\begin{corrige}{1SPE-PRODSCAL-EX-014}
\begin{enumerate}
\item On part de $\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2$.

En isolant $\vec{u} \cdot \vec{v}$ :
$\vec{u} \cdot \vec{v} = \dfrac{\|\vec{u} + \vec{v}\|^2 - \|\vec{u}\|^2 - \|\vec{v}\|^2}{2}$.

\item On a $\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2$ et $\|\vec{u} - \vec{v}\|^2 = \|\vec{u}\|^2 - 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2$.

Par soustraction : $\|\vec{u} + \vec{v}\|^2 - \|\vec{u} - \vec{v}\|^2 = 4\,\vec{u} \cdot \vec{v}$, d'ou le resultat.
\end{enumerate}
\end{corrige}
