\begin{corrige}{1SPE-PRODSCAL-EX-010}
\begin{enumerate}
\item $\vec{u} = \begin{pmatrix} 4\cos\frac{\pi}{3} \\ 4\sin\frac{\pi}{3} \end{pmatrix} = \begin{pmatrix} 2 \\ 2\sqrt{3} \end{pmatrix}$ et $\vec{v} = \begin{pmatrix} 3\cos(-\frac{\pi}{4}) \\ 3\sin(-\frac{\pi}{4}) \end{pmatrix} = \begin{pmatrix} \frac{3\sqrt{2}}{2} \\ -\frac{3\sqrt{2}}{2} \end{pmatrix}$.
\item $\vec{u} \cdot \vec{v} = 2 \times \dfrac{3\sqrt{2}}{2} + 2\sqrt{3} \times \left(-\dfrac{3\sqrt{2}}{2}\right) = 3\sqrt{2} - 3\sqrt{6}$.
\item Angle entre $\vec{u}$ et $\vec{v}$ : $\dfrac{\pi}{3} - \left(-\dfrac{\pi}{4}\right) = \dfrac{7\pi}{12}$.
$\vec{u} \cdot \vec{v} = 4 \times 3 \times \cos\!\left(\dfrac{7\pi}{12}\right) = 12\cos\!\left(\dfrac{7\pi}{12}\right) = 3\sqrt{2} - 3\sqrt{6}$.
\item Les deux resultats coincident.
\end{enumerate}
\end{corrige}
