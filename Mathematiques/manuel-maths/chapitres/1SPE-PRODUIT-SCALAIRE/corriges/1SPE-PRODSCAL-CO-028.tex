\begin{corrige}{1SPE-PRODSCAL-EX-028}
\begin{enumerate}
\item $\overrightarrow{AB}\begin{pmatrix} 1 \\ 2 \end{pmatrix}$ et $\overrightarrow{CD}\begin{pmatrix} -4 \\ 2 \end{pmatrix}$.

$\overrightarrow{AB} \cdot \overrightarrow{CD} = -4 + 4 = 0$. Les droites sont perpendiculaires.

\item $\overrightarrow{AC}\begin{pmatrix} 3 \\ 1 \end{pmatrix}$. $\overrightarrow{AB} \cdot \overrightarrow{AC} = 3 + 2 = 5$.

$\|\overrightarrow{AB}\| = \sqrt{5}$, $\|\overrightarrow{AC}\| = \sqrt{10}$.

$\cos\theta = \dfrac{5}{\sqrt{5}\sqrt{10}} = \dfrac{5}{\sqrt{50}} = \dfrac{5}{5\sqrt{2}} = \dfrac{\sqrt{2}}{2}$, donc $\theta = \dfrac{\pi}{4}$.
\end{enumerate}
