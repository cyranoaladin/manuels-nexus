\begin{corrige}{1SPE-PRODSCAL-EX-011}
\begin{enumerate}
\item $\|\vec{u} + \vec{v}\|^2 = \|\vec{u}\|^2 + 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2 = 9 + 2 \times (-2) + 16 = 21$. Donc $\|\vec{u} + \vec{v}\| = \sqrt{21}$.
\item $\|\vec{u} - \vec{v}\|^2 = 9 - 2 \times (-2) + 16 = 29$. Donc $\|\vec{u} - \vec{v}\| = \sqrt{29}$.
\item $(\vec{u} + \vec{v}) \cdot (\vec{u} - \vec{v}) = \|\vec{u}\|^2 - \|\vec{v}\|^2 = 9 - 16 = -7$.
\end{enumerate}
\end{corrige}
