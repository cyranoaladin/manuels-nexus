\begin{corrige}{1SPE-PRODSCAL-EX-030}
\begin{enumerate}
\item $C = (1; \sqrt{3})$ (milieu en abscisse, $y = \sqrt{4 - 1} = \sqrt{3}$).
\item $\overrightarrow{AB} \cdot \overrightarrow{AC} = 2 \times 1 + 0 \times \sqrt{3} = 2$. $\|\overrightarrow{AB}\| = 2$, $\|\overrightarrow{AC}\| = 2$.
$\cos\widehat{BAC} = \dfrac{2}{4} = \dfrac{1}{2}$, donc $\widehat{BAC} = \dfrac{\pi}{3}$.
\item Par symetrie du triangle equilateral, les trois angles sont egaux a $\dfrac{\pi}{3}$.
\end{enumerate}
