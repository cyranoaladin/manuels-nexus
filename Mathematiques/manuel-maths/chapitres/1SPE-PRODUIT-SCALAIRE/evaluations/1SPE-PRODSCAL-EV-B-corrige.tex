\subsection*{Exercice 1 — 5 points}
%---------------------------------------------------------------

\textbf{Question 1.}
$\vec{u} \cdot \vec{v} = 4 \times (-1) + 2 \times 3 = -4 + 6 = 2$.

\textbf{Question 2.}
$\|\vec{u}\| = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt{5}$ et $\|\vec{v}\| = \sqrt{1 + 9} = \sqrt{10}$.

\textbf{Question 3.}
$\|\vec{u} - \vec{v}\|^2 = \|\vec{u}\|^2 - 2\,\vec{u} \cdot \vec{v} + \|\vec{v}\|^2 = 20 - 4 + 10 = 26$.

Donc $\|\vec{u} - \vec{v}\| = \sqrt{26}$.

Verification : $\vec{u} - \vec{v} = \begin{pmatrix} 5 \\ -1 \end{pmatrix}$, $\|\vec{u} - \vec{v}\|^2 = 25 + 1 = 26$. \checkmark

\textbf{Question 4.}
$(\vec{u} + \vec{v}) \cdot (\vec{u} - \vec{v}) = \|\vec{u}\|^2 - \|\vec{v}\|^2 = 20 - 10 = 10$.

Verification : $\vec{u} + \vec{v} = \begin{pmatrix} 3 \\ 5 \end{pmatrix}$, $\vec{u} - \vec{v} = \begin{pmatrix} 5 \\ -1 \end{pmatrix}$. $15 - 5 = 10$. \checkmark

\subsection*{Exercice 2 — 5 points}
%---------------------------------------------------------------

\textbf{Question 1.}
$\overrightarrow{AB}\begin{pmatrix} 3 \\ -2 \end{pmatrix}$ et $\overrightarrow{AC}\begin{pmatrix} 2 \\ 3 \end{pmatrix}$.

$\overrightarrow{AB} \cdot \overrightarrow{AC} = 6 - 6 = 0$.

Le triangle est \textbf{rectangle en $A$}.

\textbf{Question 2.}
$\overrightarrow{BA}\begin{pmatrix} -3 \\ 2 \end{pmatrix}$, $\overrightarrow{BC}\begin{pmatrix} -1 \\ 5 \end{pmatrix}$.

$\overrightarrow{BA} \cdot \overrightarrow{BC} = 3 + 10 = 13$, $\|\overrightarrow{BA}\| = \sqrt{13}$, $\|\overrightarrow{BC}\| = \sqrt{26}$.

$\cos\widehat{ABC} = \dfrac{13}{\sqrt{13} \times \sqrt{26}} = \dfrac{13}{13\sqrt{2}} = \dfrac{\sqrt{2}}{2}$.

$\widehat{ABC} = \dfrac{\pi}{4}$.

\textbf{Question 3.}
$\widehat{ACB} = \pi - \dfrac{\pi}{2} - \dfrac{\pi}{4} = \dfrac{\pi}{4}$. Triangle rectangle isocele en $A$.

\subsection*{Exercice 3 — 5 points}

\textbf{Question 1.}
$I = (4; 0)$, $\overrightarrow{AB}\begin{pmatrix} 8 \\ 0 \end{pmatrix}$. Mediatrice : $8(x - 4) = 0$, soit $x = 4$.

\textbf{Question 2.}
$\overrightarrow{BC}\begin{pmatrix} -5 \\ 6 \end{pmatrix}$. $H = B + t\,\overrightarrow{BC} = (8 - 5t; 6t)$.

$\overrightarrow{AH} \cdot \overrightarrow{BC} = 0$ : $(8-5t)(-5) + (6t)(6) = 0$, soit $-40 + 25t + 36t = 0$, $61t = 40$, $t = \dfrac{40}{61}$.

$H = \left(\dfrac{288}{61}; \dfrac{240}{61}\right)$.

\textbf{Question 3.}
$\mathcal{A} = \dfrac{1}{2} |x_B \cdot y_C - x_C \cdot y_B| = \dfrac{1}{2} |48 - 0| = 24$.

\subsection*{Exercice 4 — 5 points}
%---------------------------------------------------------------

\textbf{Question 1.}
$BC^2 = 36 + 100 - 120\cos\!\left(\dfrac{\pi}{4}\right) = 136 - 60\sqrt{2}$.

$BC = \sqrt{136 - 60\sqrt{2}} \approx 7{,}15$.

\textbf{Question 2.}
$\cos\widehat{ABC} = \dfrac{BC^2 + AB^2 - AC^2}{2 \cdot BC \cdot AB} = \dfrac{136 - 60\sqrt{2} + 36 - 100}{2 \cdot BC \cdot 6} = \dfrac{72 - 60\sqrt{2}}{12\,BC}$.

Numeriquement : $\cos\widehat{ABC} \approx \dfrac{72 - 84{,}85}{12 \times 7{,}15} \approx \dfrac{-12{,}85}{85{,}8} \approx -0{,}150$.

$\widehat{ABC} \approx 98{,}6°$.

\textbf{Question 3.}
$\mathcal{A} = \dfrac{1}{2} \times 6 \times 10 \times \sin\!\left(\dfrac{\pi}{4}\right) = 30 \times \dfrac{\sqrt{2}}{2} = 15\sqrt{2} \approx 21{,}2$.
