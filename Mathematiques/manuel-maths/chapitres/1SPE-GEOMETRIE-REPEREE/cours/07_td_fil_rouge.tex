\section*{Temps 7 — TD fil rouge : Le câble et la zone interdite}

\textbf{Situation.} Un ingénieur cartographe travaille sur un plan quadrillé (repère orthonormé, unité : 100~m). Un câble rectiligne relie deux pylônes $A(1 ; 7)$ et $B(9 ; 1)$. Une zone circulaire interdite a pour centre $\Omega(5 ; 2)$ et pour rayon $3$.

\bigskip

%---------------------------------------------------------------
% PARTIE A — Equation de la droite (AB)
%---------------------------------------------------------------

\begin{exercice}{1SPE-GEOREP-COURS-07-FR-EX1}{2}{20}

\textbf{Partie A — Équation de la droite $(AB)$ \hfill [C1]}

\begin{enumerate}
  \item Calculer les coordonnées du vecteur $\vec{AB}$.
  \item Déterminer le coefficient directeur de la droite $(AB)$.
  \item En déduire une équation cartésienne de $(AB)$.
  \item Vérifier que $A$ et $B$ vérifient cette équation.
\end{enumerate}

\textbf{Partie B — Équation du cercle \hfill [C3]}

\begin{enumerate}[resume]
  \item Écrire l'équation du cercle $\mathcal{C}$ délimitant la zone interdite.
\end{enumerate}

\textbf{Partie C — Le câble traverse-t-il la zone ? \hfill [C4]}

\begin{enumerate}[resume]
  \item Calculer la distance du centre $\Omega$ à la droite $(AB)$.
  \item Comparer cette distance au rayon $r$. Conclure.
  \item Déterminer les coordonnées exactes des points d'intersection du câble et de la frontière de la zone.
\end{enumerate}

\textbf{Partie D — Longueur du tronçon interdit \hfill [C5]}

\begin{enumerate}[resume]
  \item Calculer la longueur exacte du tronçon de câble situé dans la zone interdite.
  \item L'ingénieur envisage de déplacer le pylône $B$ en $B'(9 ; y')$ pour que le câble soit tangent à la zone. Déterminer la ou les valeurs possibles de $y'$.
\end{enumerate}

\end{exercice}
