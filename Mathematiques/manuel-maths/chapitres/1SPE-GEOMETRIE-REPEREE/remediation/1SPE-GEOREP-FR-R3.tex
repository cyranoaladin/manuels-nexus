\subsection*{Rappel — Produit scalaire}

$\vec{u}(x ; y) \cdot \vec{v}(x' ; y') = xx' + yy'$.

$\vec{u} \perp \vec{v} \iff \vec{u} \cdot \vec{v} = 0$.

\begin{exercice}{1SPE-GEOREP-FR-R3-EX1}{1}{5}
\begin{enumerate}
  \item Calculer $\vec{u} \cdot \vec{v}$ avec $\vec{u}(4 ; -1)$ et $\vec{v}(2 ; 3)$.
  \item Les vecteurs $\vec{u}(5 ; 2)$ et $\vec{v}(-2 ; 5)$ sont-ils orthogonaux ?
  \item Déterminer $a$ tel que $\vec{u}(3 ; a)$ et $\vec{v}(2 ; -6)$ soient orthogonaux.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-GEOREP-FR-R3-EX1}

\textbf{1.} $\vec{u} \cdot \vec{v} = 4 \times 2 + (-1) \times 3 = 8 - 3 = 5$.

\textbf{2.} $\vec{u} \cdot \vec{v} = 5 \times (-2) + 2 \times 5 = -10 + 10 = 0$ : oui, orthogonaux.

\textbf{3.} $3 \times 2 + a \times (-6) = 0$, soit $6 - 6a = 0$, $a = 1$.

\end{corrige}
