\subsection*{C3 — Équation du cercle}
% ------------------------------------------------------------

% ---- Q7 ----
\item \textbf{[Q7]} L'équation $(x - 3)^2 + (y + 1)^2 = 16$ représente un cercle de :

  \begin{enumerate}[label=\Alph*.]
    \item centre $(3 ; -1)$ et rayon $16$
    \item centre $(3 ; 1)$ et rayon $4$
    \item centre $(3 ; -1)$ et rayon $4$
    \item centre $(-3 ; 1)$ et rayon $4$
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} Le cercle de centre $(0 ; 0)$ et de rayon $3$ a pour équation :

  \begin{enumerate}[label=\Alph*.]
    \item $x^2 + y^2 = 3$
    \item $x^2 + y^2 = 9$
    \item $(x + 3)^2 + (y + 3)^2 = 0$
    \item $x + y = 3$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} L'équation $x^2 + y^2 - 2x + 4y - 4 = 0$ représente :

  \begin{enumerate}[label=\Alph*.]
    \item un cercle de centre $(1 ; -2)$ et de rayon $3$
    \item un cercle de centre $(-1 ; 2)$ et de rayon $9$
    \item un cercle de centre $(1 ; -2)$ et de rayon $9$
    \item aucun cercle
  \end{enumerate}
