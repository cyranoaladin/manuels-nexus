\subsection*{C1 — Calcul de termes, formule explicite et recurrence}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = 2n^2 - 3n + 1$.
  Quelle est la valeur de $u_3$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $10$
    \item $19$
    \item $8$
    \item $7$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} Soit $(v_n)$ la suite definie par $v_0 = 4$ et $v_{n+1} = 3v_n - 2$
  pour tout $n \in \mathbb{N}$. Quelle est la valeur de $v_2$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $10$
    \item $28$
    \item $34$
    \item $22$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} Parmi les assertions suivantes, laquelle est exacte concernant
  une suite definie par recurrence ?

  \begin{enumerate}[label=\Alph*.]
    \item On peut calculer $u_{10}$ directement en substituant $n = 10$ dans la relation
      $u_{n+1} = f(u_n)$.
    \item Pour calculer $u_1$, on substitue $n = 0$ dans la relation $u_{n+1} = f(u_n)$.
    \item Une suite definie par formule explicite necessite de connaitre le terme precedent.
    \item La notation $u(n)$ avec des parentheses est la notation mathematique standard
      pour une suite.
  \end{enumerate}

\end{enumerate}

\subsection*{C3 — Identification geometrique, raison, formule explicite}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} Soit $(u_n)$ la suite geometrique de raison $q = 2$ et de premier
  terme $u_0 = 3$. Quelle est la formule explicite de $u_n$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $u_n = 3 + 2n$
    \item $u_n = 3 \times 2^{n-1}$
    \item $u_n = 2 \times 3^n$
    \item $u_n = 3 \times 2^n$
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} La suite $(v_n)$ est definie par $v_0 = 6$ et
  $v_{n+1} = \dfrac{v_n}{3}$ pour tout $n \in \mathbb{N}$. Quelle est la raison
  de cette suite ?

  \begin{enumerate}[label=\Alph*.]
    \item $3$
    \item $-3$
    \item $\dfrac{1}{3}$
    \item $2$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} Parmi les assertions suivantes, laquelle est exacte pour
  identifier une suite geometrique ?

  \begin{enumerate}[label=\Alph*.]
    \item Il suffit de verifier que $u_1/u_0 = u_2/u_1$ pour conclure que la suite
      est geometrique.
    \item Une suite geometrique peut comporter un terme nul.
    \item Pour montrer qu'une suite est geometrique, il faut calculer $u_{n+1}/u_n$
      et verifier que ce quotient est constant pour tout $n$.
    \item La raison d'une suite geometrique est toujours un entier positif.
  \end{enumerate}

\end{enumerate}

\subsection*{C5 — Sens de variation, methode adaptee, quantificateur}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q13 ----
\item \textbf{[Q13]} Soit $(u_n)$ la suite arithmetique de raison $r = -4$ et de
  premier terme $u_0 = 10$. Quel est le sens de variation de cette suite ?

  \begin{enumerate}[label=\Alph*.]
    \item Croissante
    \item Constante
    \item Strictement decroissante
    \item D'abord croissante, puis decroissante
  \end{enumerate}

\bigskip

% ---- Q14 ----
\item \textbf{[Q14]} Pour etudier le sens de variation de la suite $(u_n)$ definie
  par $u_n = 5 \times (0{,}8)^n$ (avec $u_n > 0$ pour tout $n$), quelle est la
  methode la plus efficace ?

  \begin{enumerate}[label=\Alph*.]
    \item Methode A : calculer le signe de $u_{n+1} - u_n$.
    \item Methode B : comparer le quotient $\dfrac{u_{n+1}}{u_n}$ a $1$.
    \item Methode C : etudier la derivee de la fonction $f(x) = 5 \times (0{,}8)^x$.
    \item Calculer les 5 premiers termes et observer la tendance.
  \end{enumerate}

\bigskip

% ---- Q15 ----
\item \textbf{[Q15]} Soit $(u_n)$ la suite definie par $u_n = n^2 - 6n + 5$.
  A partir de quel rang cette suite est-elle strictement croissante ?

  \begin{enumerate}[label=\Alph*.]
    \item Pour tout $n \in \mathbb{N}$
    \item A partir de $n = 3$
    \item A partir de $n = 2$
    \item A partir de $n = 6$
  \end{enumerate}

\end{enumerate}

\subsection*{C6 — Coefficient multiplicateur, modele adapte, interpretation}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q16 ----
\item \textbf{[Q16]} Un capital de $2\,000$~euros est place a un taux annuel de
  $4\,\%$. Quel est le coefficient multiplicateur applicable chaque annee ?

  \begin{enumerate}[label=\Alph*.]
    \item $0{,}04$
    \item $1{,}4$
    \item $1{,}04$
    \item $4$
  \end{enumerate}

\bigskip

% ---- Q17 ----
\item \textbf{[Q17]} Une ville compte $50\,000$ habitants et perd $300$ habitants
  par an. Quel modele est le plus adapte pour representer l'evolution de sa
  population ?

  \begin{enumerate}[label=\Alph*.]
    \item Une suite geometrique, car la population diminue.
    \item Une suite arithmetique, car la variation absolue annuelle est constante
      ($-300$).
    \item Une suite geometrique, car on multiplie chaque annee par un coefficient.
    \item Ni une suite arithmetique ni une suite geometrique.
  \end{enumerate}

\bigskip

% ---- Q18 ----
\item \textbf{[Q18]} Un placement diminue de $5\,\%$ par an. Si le capital initial
  est $u_0 = 10\,000$~euros, quelle est l'expression de $u_n$ (capital apres
  $n$ annees) ?

  \begin{enumerate}[label=\Alph*.]
    \item $u_n = 10\,000 - 500n$
    \item $u_n = 10\,000 \times 1{,}05^n$
    \item $u_n = 10\,000 \times 0{,}95^n$
    \item $u_n = 10\,000 \times 0{,}05^n$
  \end{enumerate}

\end{enumerate}

\subsection*{C7 — Lecture de code, boucle for/while, erreur dans un script}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q19 ----
\item \textbf{[Q19]} On considere le programme Python suivant :

\begin{verbatim}
def terme(n):
    u = 2
    for k in range(n):
        u = 3 * u + 1
    return u
\end{verbatim}

  Quelle valeur renvoie l'appel \texttt{terme(3)} ?

  \begin{enumerate}[label=\Alph*.]
    \item $7$
    \item $21$
    \item $64$
    \item $22$
  \end{enumerate}

\bigskip

% ---- Q20 ----
\item \textbf{[Q20]} On cherche le plus petit entier $n$ tel que $u_n > 100$,
  ou $u_0 = 1$ et $u_{n+1} = 2u_n$. Parmi les codes suivants, lequel est
  correct ?

  \begin{enumerate}[label=\Alph*.]
    \item \begin{verbatim}
u = 1
n = 0
while u > 100:
    u = 2 * u
    n = n + 1\end{verbatim}

    \item \begin{verbatim}
u = 1
n = 0
while u <= 100:
    u = 2 * u
    n = n + 1\end{verbatim}

    \item \begin{verbatim}
u = 1
n = 0
for k in range(100):
    u = 2 * u
    n = n + 1\end{verbatim}

    \item \begin{verbatim}
u = 0
n = 0
while u <= 100:
    u = 2 * u
    n = n + 1\end{verbatim}
  \end{enumerate}

\bigskip

% ---- Q21 ----
\item \textbf{[Q21]} On veut calculer $S = u_0 + u_1 + \cdots + u_5$ ou
  $u_k = 2^k$. Un eleve ecrit :

\begin{verbatim}
S = 0
for k in range(5):
    S = S + 2**k
\end{verbatim}

  Quelle erreur contient ce code ?

  \begin{enumerate}[label=\Alph*.]
    \item L'accumulateur devrait etre initialise a $1$.
    \item \texttt{range(5)} produit les indices $0, 1, 2, 3, 4$ : le terme
      $u_5 = 2^5$ est omis. Il faudrait \texttt{range(6)}.
    \item Il faut ecrire $2^k$ et non \texttt{2**k} en Python.
    \item Il faut une boucle \texttt{while} et non \texttt{for} pour calculer
      une somme.
  \end{enumerate}

\end{enumerate}
