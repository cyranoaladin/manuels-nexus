\begin{exercice}{1SPE-SUITES-RE-C3-EX1}{1}{10}
On considère la suite $(u_n)$ définie pour tout $n \in \mathbb{N}$ par $u_n = 3 \times 2^n$.
\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$, $u_3$.
  \item Calculer $\dfrac{u_{n+1}}{u_n}$ pour tout $n \in \mathbb{N}$.
  \item En déduire que $(u_n)$ est géométrique. Préciser la raison $q$ et le premier terme $u_0$.
  \item Exprimer $u_{n+1}$ en fonction de $u_n$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SUITES-RE-C3-EX1}

\textbf{Question 1.}
\[
u_0 = 3 \times 2^0 = 3. \quad u_1 = 3 \times 2^1 = 6. \quad u_2 = 3 \times 2^2 = 12. \quad u_3 = 3 \times 2^3 = 24.
\]

\commentaireMarge{Pour une suite géométrique, on calcule le quotient $u_{n+1}/u_n$ et on vérifie qu'il est constant. Attention : tous les termes doivent être non nuls.}

\textbf{Question 2.} Pour tout $n \in \mathbb{N}$, $u_n = 3 \times 2^n \neq 0$. Le quotient est :
\[
\frac{u_{n+1}}{u_n} = \frac{3 \times 2^{n+1}}{3 \times 2^n} = 2^{n+1-n} = 2.
\]

\commentaireMarge{Le quotient est constant égal à $2$, indépendant de $n$ : c'est la définition d'une suite géométrique.}

\textbf{Question 3.} Le quotient $\dfrac{u_{n+1}}{u_n} = 2$ est constant. La suite $(u_n)$ est donc géométrique de raison $q = 2$ et de premier terme $u_0 = 3$.

\textbf{Question 4.} $u_{n+1} = q \times u_n = 2 u_n$.

\end{corrige}

\begin{exercice}{1SPE-SUITES-RE-C3-EX2}{1}{10}
On donne les premiers termes : $v_0 = 2$, $v_1 = 6$, $v_2 = 18$, $v_3 = 54$.
\begin{enumerate}
  \item Calculer $\dfrac{v_1}{v_0}$, $\dfrac{v_2}{v_1}$, $\dfrac{v_3}{v_2}$.
  \item La suite est-elle géométrique ? Si oui, donner la raison $q$.
  \item Exprimer le terme général $v_n$ en fonction de $n$.
  \item Calculer $v_6$.
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Une suite $(v_n)$ est géométrique si le quotient $\dfrac{v_{n+1}}{v_n}$ est le même pour tous les termes (et si les termes sont tous non nuls). Calculer les trois quotients demandés et les comparer.}

\begin{corrige}{1SPE-SUITES-RE-C3-EX2}

\textbf{Question 1.}
\[
\frac{v_1}{v_0} = \frac{6}{2} = 3. \quad \frac{v_2}{v_1} = \frac{18}{6} = 3. \quad \frac{v_3}{v_2} = \frac{54}{18} = 3.
\]

\commentaireMarge{Les quotients sont tous égaux à $3$ : la suite est géométrique de raison $3$.}

\textbf{Question 2.} Les quotients consécutifs sont tous égaux à $3$. La suite $(v_n)$ est géométrique de raison $q = 3$.

\commentaireMarge{Formule : $v_n = v_0 \times q^n$.}

\textbf{Question 3.} $v_n = v_0 \times q^n = 2 \times 3^n$.

\textbf{Question 4.}
\[
v_6 = 2 \times 3^6 = 2 \times 729 = 1\,458.
\]

\end{corrige}

\begin{exercice}{1SPE-SUITES-RE-C3-EX3}{1}{12}
Une suite géométrique $(w_n)$ vérifie $w_0 = 5$ et $w_3 = 40$.
\begin{enumerate}
  \item Montrer que $w_3 = w_0 \times q^3$, puis calculer $q$.
  \item Exprimer $w_n$ en fonction de $n$.
  \item Calculer $w_5$.
  \item Montrer que $(w_n)$ est une suite géométrique en vérifiant que le quotient $\dfrac{w_{n+1}}{w_n}$ est bien constant.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SUITES-RE-C3-EX3}

\commentaireMarge{$w_3 = w_0 \times q^3$ car on multiplie par $q$ à trois reprises depuis $w_0$.}

\textbf{Question 1.} Pour une suite géométrique de raison $q$ : $w_n = w_0 \times q^n$. Donc $w_3 = 5q^3$.
\[
5q^3 = 40 \implies q^3 = 8 \implies q = 2.
\]

\textbf{Question 2.} $w_n = 5 \times 2^n$.

\textbf{Question 3.}
\[
w_5 = 5 \times 2^5 = 5 \times 32 = 160.
\]

\textbf{Question 4.} Pour tout $n \in \mathbb{N}$, $w_n = 5 \times 2^n \neq 0$. Le quotient :
\[
\frac{w_{n+1}}{w_n} = \frac{5 \times 2^{n+1}}{5 \times 2^n} = 2.
\]
Le quotient est constant égal à $2$ : la suite $(w_n)$ est bien géométrique de raison $2$.

\end{corrige}
