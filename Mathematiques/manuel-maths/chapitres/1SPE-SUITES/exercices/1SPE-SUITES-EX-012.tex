\begin{exercice}{1SPE-SUITES-EX-012}{1}{10}
On considère la suite géométrique $(u_n)$ de premier terme $u_0 = 1$ et de raison $q = 2$.
\begin{enumerate}
  \item Écrire la formule du terme général $u_n$.
  \item En appliquant la formule de somme géométrique
        $S = u_0 \times \dfrac{1 - q^{n+1}}{1 - q}$,
        calculer $S = u_0 + u_1 + \cdots + u_7$.
  \item Calculer directement $u_0 + u_1 + \cdots + u_7$ pour vérifier.
\end{enumerate}
\end{exercice}
