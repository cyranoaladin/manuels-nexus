\begin{exercice}{1SPE-SUITES-EX-047}{3}{35}
On consid\`ere la suite $(u_n)$ d\'efinie pour tout entier naturel $n$ par $u_n = n^2 + n + 1$.

On pose $d_n = u_{n+1} - u_n$ pour tout $n \in \mathbb{N}$.

\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$, $u_3$ et $u_4$.

  \item La suite $(u_n)$ est-elle arithm\'etique ? Est-elle g\'eom\'etrique ? Justifier.

  \item Calculer $d_0$, $d_1$, $d_2$ et $d_3$. \'Emettre une conjecture sur la nature de la suite $(d_n)$.

  \item D\'emontrer cette conjecture en calculant $d_n$ en fonction de $n$, puis $d_{n+1} - d_n$.

  \item En d\'eduire le sens de variation de la suite $(u_n)$.

  \item \textbf{Exploration Python.} \'Ecrire un programme Python qui affiche, pour $n$ allant de $0$ \`a $20$, les valeurs de $u_n$ et $d_n$. V\'erifier les r\'esultats des questions pr\'ec\'edentes.
\end{enumerate}
\end{exercice}
