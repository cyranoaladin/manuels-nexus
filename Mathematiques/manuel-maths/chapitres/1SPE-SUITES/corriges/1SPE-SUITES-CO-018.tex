\begin{corrige}{1SPE-SUITES-EX-018}

\commentaireMarge{Un taux d'intérêt annuel de $5\,\%$ signifie que le capital est multiplié par $1{,}05$ chaque année : c'est le coefficient multiplicateur.}

\textbf{Question 1.} Un taux annuel de $5\,\%$ correspond au coefficient multiplicateur :
\[
1 + \frac{5}{100} = 1{,}05.
\]

\commentaireMarge{La relation de récurrence $P_{n+1} = q \times P_n$ caractérise une suite géométrique de raison $q$.}

\textbf{Question 2.} Chaque année, le capital est multiplié par $1{,}05$, donc :
\[
P_{n+1} = 1{,}05 \times P_n.
\]

Par application répétée à partir de $P_0 = 2000$ :
\[
P_n = P_0 \times (1{,}05)^n = 2000 \times (1{,}05)^n.
\]

\textbf{Question 3.} La suite $(P_n)$ est \textbf{géométrique} car $\dfrac{P_{n+1}}{P_n} = 1{,}05$ est constant pour tout $n \in \mathbb{N}$.

Elle a pour :
\begin{itemize}
  \item premier terme $P_0 = 2000$~€ ;
  \item raison $q = 1{,}05$.
\end{itemize}

\commentaireMarge{On applique la formule $P_n = P_0 \times q^n$ avec $n = 3$. On arrondit au centime.}

\textbf{Question 4.} On calcule $P_3$ :
\[
P_3 = 2000 \times (1{,}05)^3 = 2000 \times 1{,}157625 = 2315{,}25.
\]

Après $3$ ans, le capital est de $\mathbf{2\,315{,}25}$~\textbf{€}.

\end{corrige}
