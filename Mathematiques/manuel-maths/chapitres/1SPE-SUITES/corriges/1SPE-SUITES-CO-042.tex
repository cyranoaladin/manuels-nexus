\begin{corrige}{1SPE-SUITES-EX-042}

\textbf{1. Premiers termes et conjecture.}

\[
u_1 = \frac{1}{1 \times 2} = \frac{1}{2} \quad;\quad u_2 = \frac{1}{2 \times 3} = \frac{1}{6} \quad;\quad u_3 = \frac{1}{3 \times 4} = \frac{1}{12}.
\]
\[
S_1 = \frac{1}{2} \quad;\quad S_2 = \frac{1}{2} + \frac{1}{6} = \frac{2}{3} \quad;\quad S_3 = \frac{2}{3} + \frac{1}{12} = \frac{3}{4}.
\]

\textbf{Conjecture :} $S_n = \dfrac{n}{n+1}$.

\textbf{2. Décomposition.}

\commentaireMarge{Mettre les deux fractions au même dénominateur $k(k+1)$ pour vérifier l'identité.}

\[
\frac{1}{k} - \frac{1}{k+1} = \frac{(k+1) - k}{k(k+1)} = \frac{1}{k(k+1)} = u_k. \quad \checkmark
\]

\textbf{3. Somme télescopique.}

\commentaireMarge{Dans une somme télescopique, les termes intermédiaires s'annulent deux à deux.}

\[
S_n = \sum_{k=1}^{n} u_k = \sum_{k=1}^{n} \left(\frac{1}{k} - \frac{1}{k+1}\right).
\]

En développant et en groupant (chaque $\frac{1}{k+1}$ s'annule avec le $\frac{1}{k}$ du terme suivant) :
\[
S_n = \left(\frac{1}{1} - \frac{1}{2}\right) + \left(\frac{1}{2} - \frac{1}{3}\right) + \cdots + \left(\frac{1}{n} - \frac{1}{n+1}\right) = 1 - \frac{1}{n+1} = \frac{n}{n+1}.
\]

\textbf{4. Monotonie et convergence.}

$S_{n+1} - S_n = u_{n+1} = \dfrac{1}{(n+1)(n+2)} > 0$ pour tout $n \geq 1$ : la suite $(S_n)$ est \textbf{strictement croissante}.

De plus, $S_n = \dfrac{n}{n+1} = 1 - \dfrac{1}{n+1} < 1$ pour tout $n \geq 1$ : $(S_n)$ est \textbf{majorée par $1$}.

La suite $(S_n)$ est croissante et majorée, donc elle converge. Sa limite est $\lim_{n \to +\infty} \dfrac{n}{n+1} = 1$.

\textbf{5. Généralisation.}

\textbf{a)} On cherche $A$ et $B$ tels que $\dfrac{1}{k(k+2)} = \dfrac{A}{k} + \dfrac{B}{k+2}$, soit $A(k+2) + Bk = 1$ pour tout $k$.

\begin{itemize}
  \item $k = 0$ : $2A = 1 \Rightarrow A = \dfrac{1}{2}$.
  \item $k = -2$ : $-2B = 1 \Rightarrow B = -\dfrac{1}{2}$.
\end{itemize}

Donc $\dfrac{1}{k(k+2)} = \dfrac{1}{2}\left(\dfrac{1}{k} - \dfrac{1}{k+2}\right)$.

\textbf{b)} Somme télescopique avec un décalage de $2$ :

\commentaireMarge{Les termes $\frac{1}{k}$ survivent pour $k = 1, 2$ ; les termes $\frac{1}{k+2}$ survivent pour $k = n-1, n$.}

\[
T_n = \frac{1}{2} \sum_{k=1}^{n} \left(\frac{1}{k} - \frac{1}{k+2}\right) = \frac{1}{2}\left[\left(\frac{1}{1} + \frac{1}{2}\right) - \left(\frac{1}{n+1} + \frac{1}{n+2}\right)\right].
\]
\[
\boxed{T_n = \frac{3}{4} - \frac{1}{2(n+1)} - \frac{1}{2(n+2)}}.
\]

\end{corrige}
