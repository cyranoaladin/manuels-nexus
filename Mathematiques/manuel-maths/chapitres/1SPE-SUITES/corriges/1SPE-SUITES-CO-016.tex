\begin{corrige}{1SPE-SUITES-EX-016}

\commentaireMarge{Méthode : étudier les variations d'une suite $u_n = f(n)$ via les variations de la fonction associée $f$ sur $[0\,;\,+\infty[$.}

\textbf{Question 1.} On calcule la dérivée de $f(x) = x^2 - 4x + 7$ :
\[
f'(x) = 2x - 4.
\]

\textbf{Question 2.} On étudie le signe de $f'(x) = 2x - 4 = 2(x - 2)$ sur $[0\,;\,+\infty[$ :
\begin{itemize}
  \item $f'(x) < 0$ si $x < 2$, donc $f$ est \textbf{strictement décroissante} sur $[0\,;\,2]$ ;
  \item $f'(x) = 0$ si $x = 2$ (minimum de $f$) ;
  \item $f'(x) > 0$ si $x > 2$, donc $f$ est \textbf{strictement croissante} sur $[2\,;\,+\infty[$.
\end{itemize}

Tableau de variations de $f$ sur $[0\,;\,+\infty[$ :

\medskip
\begin{center}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $0$ & & $2$ & & $+\infty$ \\
\hline
$f'(x)$ & & $-$ & $0$ & $+$ & \\
\hline
$f(x)$ & $7$ & $\searrow$ & $3$ & $\nearrow$ & $+\infty$ \\
\hline
\end{tabular}
\end{center}
\medskip

\commentaireMarge{Les variations de $u_n = f(n)$ suivent celles de $f$, à condition de restreindre aux entiers naturels.}

\textbf{Question 3.} On calcule les premiers termes de $(u_n)$ :
\[
u_0 = 0^2 - 4 \times 0 + 7 = 7, \quad u_1 = 1 - 4 + 7 = 4, \quad u_2 = 4 - 8 + 7 = 3, \quad u_3 = 9 - 12 + 7 = 4.
\]

D'après les variations de $f$ :
\begin{itemize}
  \item Pour $n \in \{0, 1, 2\}$, on a $n \leq 2$, donc $f$ est décroissante : la suite $(u_n)$ est \textbf{strictement décroissante} sur $\{0, 1, 2\}$ (on vérifie : $u_0 = 7 > u_1 = 4 > u_2 = 3$).
  \item Pour $n \geq 2$, on a $n \geq 2$, donc $f$ est croissante : la suite $(u_n)$ est \textbf{strictement croissante} pour $n \geq 2$ (on vérifie : $u_2 = 3 < u_3 = 4$).
\end{itemize}

La suite $(u_n)$ est donc décroissante puis croissante, avec un minimum en $n = 2$ : $u_2 = 3$.

\end{corrige}
