\begin{corrige}{1SPE-SUITES-EX-010}

\commentaireMarge{Pour une suite géométrique, on applique la relation $u_{n+1} = q \times u_n$ de proche en proche.}

\textbf{Question 1.} La suite $(u_n)$ est géométrique de premier terme $u_0 = 81$ et de raison $q = \dfrac{1}{3}$. On calcule :
\[
u_1 = u_0 \times q = 81 \times \frac{1}{3} = 27.
\]
\[
u_2 = u_1 \times q = 27 \times \frac{1}{3} = 9.
\]
\[
u_3 = u_2 \times q = 9 \times \frac{1}{3} = 3.
\]
\[
u_4 = u_3 \times q = 3 \times \frac{1}{3} = 1.
\]

Donc $u_1 = 27$, $u_2 = 9$, $u_3 = 3$ et $u_4 = 1$.

\commentaireMarge{La formule du terme général d'une suite géométrique est $u_n = u_0 \times q^n$.}

\textbf{Question 2.} La formule du terme général est :
\[
u_n = u_0 \times q^n = 81 \times \left(\frac{1}{3}\right)^n.
\]

On peut aussi l'écrire $u_n = 3^4 \times 3^{-n} = 3^{4-n}$.

\commentaireMarge{Le sens de variation d'une suite géométrique à termes positifs dépend de la comparaison de la raison $q$ à $1$.}

\textbf{Question 3.} Tous les termes sont positifs ($u_0 = 81 > 0$ et $q = \dfrac{1}{3} > 0$, donc $u_n = 81 \times \left(\dfrac{1}{3}\right)^n > 0$ pour tout $n$).

De plus, $0 < q = \dfrac{1}{3} < 1$, donc pour tout $n \in \mathbb{N}$ :
\[
u_{n+1} = \frac{1}{3} \times u_n < u_n.
\]

La suite $(u_n)$ est donc \textbf{strictement décroissante}.

\end{corrige}
