\begin{fichemethode}{M5}{Utiliser l'inegalite de Bernoulli et les croissances comparees}

\quandUtiliser{%
  Utiliser cette methode lorsque l'enonce fait intervenir $(1+a)^n$, $q^n$ avec $q > 1$ ou $|q| < 1$, ou des expressions melant exponentielle et polynomes.
}

\pasApas{%
  \begin{enumerate}
    \item \textbf{Inegalite de Bernoulli.} Pour minorer $(1+a)^n$ avec $a > 0$ :
      \[
        (1+a)^n \geq 1 + na.
      \]
      Le membre de droite tend vers $+\infty$ si $a > 0$, ce qui prouve que $(1+a)^n \to +\infty$.
    \item \textbf{Limite de $q^n$.} Appliquer le tableau :
      \begin{itemize}
        \item $q > 1$ : $q^n \to +\infty$.
        \item $|q| < 1$ : $q^n \to 0$.
        \item $q = 1$ : $q^n = 1$.
        \item $q \leq -1$ : $(q^n)$ diverge sans limite.
      \end{itemize}
    \item \textbf{Croissances comparees.} Pour lever une forme indeterminee $\dfrac{\mathrm{e}^n}{P(n)}$ :
      \[
        \lim_{n \to +\infty} \frac{\mathrm{e}^n}{n^k} = +\infty \quad \text{pour tout } k \in \mathbb{N}.
      \]
      L'exponentielle l'emporte sur tout polynome.
  \end{enumerate}
}

\exempleRedige{%
  \textbf{Exemple 1.} Montrer que $\lim_{n \to +\infty} 1{,}05^n = +\infty$.

  \medskip
  \commentaireMarge{Ecrire $1{,}05 = 1 + 0{,}05$ et appliquer Bernoulli.}%
  On ecrit $1{,}05^n = (1 + 0{,}05)^n$. Par l'inegalite de Bernoulli :
  \[
    1{,}05^n \geq 1 + 0{,}05n.
  \]
  Or $\lim(1 + 0{,}05n) = +\infty$. Par comparaison, $\lim 1{,}05^n = +\infty$.

  \bigskip
  \textbf{Exemple 2.} Determiner $\lim_{n \to +\infty} n^3 \times 0{,}9^n$.

  \medskip
  \commentaireMarge{Ecrire le produit comme quotient pour utiliser la croissance comparee.}%
  On pose $q = 0{,}9$, soit $\dfrac{1}{q} = \dfrac{10}{9} > 1$. On a :
  \[
    n^3 \times 0{,}9^n = \frac{n^3}{(10/9)^n}.
  \]

  Or $(10/9)^n$ croit plus vite que tout polynome (car $10/9 > 1$). Plus precisement, $\dfrac{n^3}{(10/9)^n} \to 0$.

  Donc $\lim_{n \to +\infty} n^3 \times 0{,}9^n = 0$.
}

\pieges{%
  \begin{itemize}
    \item \textbf{Piege 1.} Appliquer Bernoulli avec $a < -1$ : l'inegalite n'est valide que pour $a \geq -1$.
    \item \textbf{Piege 2.} Confondre « $q^n \to 0$ quand $|q| < 1$ » avec « $q^n$ est nul ». La suite tend vers $0$ mais n'est jamais nulle.
    \item \textbf{Piege 3.} Oublier que la croissance comparee $\mathrm{e}^n / n^k \to +\infty$ vaut aussi pour $q^n / n^k$ quand $q > 1$.
  \end{itemize}
}

\verifier{%
  Verifier numeriquement : $1{,}05^{100} \approx 131{,}5$ (confirme la divergence). Pour $0{,}9^{20} \approx 0{,}12$ : les termes $n^3 \times 0{,}9^n$ decroissent bien vers $0$ pour $n$ grand.
}
