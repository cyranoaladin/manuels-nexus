\textbf{Exercice 1.}

\textbf{1.} $u_n = \dfrac{n^2(4 - 1/n)}{n^2(2 + 5/n^2)} = \dfrac{4 - 1/n}{2 + 5/n^2} \to \dfrac{4}{2} = \boxed{2}$.

\textbf{2.} $|2/5| < 1$, donc $(2/5)^n \to 0$. $v_n = 3 \times 0 + 7 = \boxed{7}$.

\textbf{3.} $w_n = n^2(2 - 5/n + 3/n^2)$. Pour $n \geq 5$ : $2 - 5/n + 3/n^2 \geq 2 - 1 + 0 = 1$, donc $w_n \geq n^2$. Or $n^2 \to +\infty$. $\boxed{\lim w_n = +\infty}$.

\textbf{Exercice 2.}

\textbf{1.} $\mathcal{P}(n)$ : « $3^n \geq 2n + 1$ ».

\textit{Init.} $3^0 = 1 \geq 1 = 2 \times 0 + 1$. Vrai.

\textit{Heredite.} Si $3^n \geq 2n+1$ : $3^{n+1} = 3 \times 3^n \geq 3(2n+1) = 6n + 3$. Or $6n + 3 \geq 2(n+1) + 1 = 2n + 3$ (car $4n \geq 0$). Donc $3^{n+1} \geq 2(n+1)+1$.

\textit{Conclusion.} Par recurrence, $3^n \geq 2n+1$ pour tout $n$.

\textbf{2.} $2n+1 \to +\infty$ et $3^n \geq 2n+1$. $\boxed{\lim 3^n = +\infty}$.

\textbf{Exercice 3.}

\textbf{1.} $u_0 = 200$. $80\,\%$ reste + $60$~L : $u_{n+1} = 0{,}8\,u_n + 60$.

\textbf{2.} $u_1 = 0{,}8 \times 200 + 60 = 220$. $u_2 = 0{,}8 \times 220 + 60 = 236$.

\textbf{3.} $\ell = 60/0{,}2 = 300$. $v_n = u_n - 300$, $v_{n+1} = 0{,}8\,v_n$. $v_0 = -100$. $v_n = -100 \times 0{,}8^n$. $u_n = 300 - 100 \times 0{,}8^n$.

\textbf{4.} $0{,}8^n \to 0$. $\lim u_n = \boxed{300}$~litres. Le volume se stabilise a $300$~L.

\textbf{Exercice 4.}

\textbf{1.} $u_1 = \sqrt{2 \times 2 + 3} = \sqrt{7} \approx 2{,}65$. $u_2 = \sqrt{2\sqrt{7}+3} \approx \sqrt{8{,}29} \approx 2{,}88$.

\textbf{2.} $u_0 = 2 \in [0, 3]$. Si $0 \leq u_n \leq 3$ : $3 \leq 2u_n+3 \leq 9$, $\sqrt{3} \leq u_{n+1} \leq 3$. Donc $0 \leq u_{n+1} \leq 3$.

\textbf{3.} $u_{n+1} \geq u_n \iff \sqrt{2u_n+3} \geq u_n \iff 2u_n+3 \geq u_n^2$ (car tout $> 0$) $\iff u_n^2 - 2u_n - 3 \leq 0 \iff (u_n-3)(u_n+1) \leq 0$. Vrai car $-1 \leq 0 \leq u_n \leq 3$.

\textbf{4.} Croissante majoree : converge. $\ell = \sqrt{2\ell+3}$, $\ell^2 = 2\ell+3$, $\ell^2 - 2\ell - 3 = 0$, $(\ell-3)(\ell+1) = 0$. $\ell \geq 0$, donc $\boxed{\ell = 3}$.
