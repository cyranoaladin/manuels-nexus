\begin{exercice}{TSPE-SUITLIM-EX-039}{2}{15}
Soit $(u_n)$ la suite definie pour $n \geq 0$ par $u_n = \dfrac{2^n + n}{2^n + 1}$.
\begin{enumerate}
  \item Montrer que $u_n = 1 + \dfrac{n - 1}{2^n + 1}$.
  \item Montrer que $0 \leq \dfrac{n-1}{2^n + 1} \leq \dfrac{n}{2^n}$ pour $n \geq 1$.
  \item En utilisant la limite de $\dfrac{n}{q^n}$ pour $q > 1$, determiner $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
