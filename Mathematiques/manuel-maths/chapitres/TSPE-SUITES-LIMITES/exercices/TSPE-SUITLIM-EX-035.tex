\begin{exercice}{TSPE-SUITLIM-EX-035}{3}{20}
Soit $q$ un reel tel que $|q| < 1$. On pose $S_n = \displaystyle\sum_{k=0}^{n} q^k$.
\begin{enumerate}
  \item Rappeler l'expression de $S_n$ en fonction de $n$ et $q$.
  \item En utilisant la limite de $q^n$, determiner $\lim_{n \to +\infty} S_n$.
  \item Calculer $\lim S_n$ pour $q = \dfrac{1}{3}$ et $q = \dfrac{2}{5}$.
  \item Que se passe-t-il si $q \geq 1$ ?
\end{enumerate}
\end{exercice}
