\begin{exercice}{TSPE-SUITLIM-EX-032}{2}{15}
Soit $(u_n)$ la suite definie par $u_n = \dfrac{2^n + 3^n}{4^n}$.
\begin{enumerate}
  \item Montrer que $u_n = \left(\dfrac{1}{2}\right)^n + \left(\dfrac{3}{4}\right)^n$.
  \item Determiner $\lim_{n \to +\infty} u_n$.
  \item Montrer que $(u_n)$ est decroissante.
\end{enumerate}
\end{exercice}
