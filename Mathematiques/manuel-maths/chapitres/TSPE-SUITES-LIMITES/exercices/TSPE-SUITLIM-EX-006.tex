\begin{exercice}{TSPE-SUITLIM-EX-006}{3}{15}
Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = \dfrac{2n+3}{n+1}$.
\begin{enumerate}
  \item Montrer que pour tout $n \in \mathbb{N}$, $u_n - 2 = \dfrac{1}{n+1}$.
  \item Soit $\varepsilon > 0$. Determiner, en fonction de $\varepsilon$, un rang $N$ a partir duquel $|u_n - 2| \leq \varepsilon$.
  \item En deduire, avec la definition formelle, que $\lim_{n \to +\infty} u_n = 2$.
\end{enumerate}
\end{exercice}
