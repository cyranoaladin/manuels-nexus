\begin{exercice}{TSPE-SUITLIM-EX-007}{3}{15}
Soit $(u_n)$ la suite definie pour tout $n \geq 1$ par $u_n = \dfrac{n\cos(n)}{n^2 + 1}$.
\begin{enumerate}
  \item Montrer que pour tout $n \geq 1$, $|u_n| \leq \dfrac{n}{n^2 + 1}$.
  \item Montrer que pour tout $n \geq 1$, $\dfrac{n}{n^2 + 1} \leq \dfrac{1}{n}$.
  \item En deduire $\lim_{n \to +\infty} u_n$ a l'aide du theoreme des gendarmes.
\end{enumerate}
\end{exercice}
