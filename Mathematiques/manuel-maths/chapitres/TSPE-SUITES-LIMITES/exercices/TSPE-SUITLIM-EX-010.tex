\begin{exercice}{TSPE-SUITLIM-EX-010}{1}{12}
Soit $(u_n)$ la suite definie par $u_0 = 2$ et $u_{n+1} = 3u_n - 4$ pour tout $n \in \mathbb{N}$.
\begin{enumerate}
  \item Calculer $u_1$, $u_2$ et $u_3$. Que constate-t-on ?
  \item Demontrer par recurrence que pour tout $n \in \mathbb{N}$, $u_n = 2$.
\end{enumerate}
\end{exercice}
