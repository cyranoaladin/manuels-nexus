\begin{exercice}{TSPE-SUITLIM-EX-040}{2}{15}
Soit $(u_n)$ la suite definie par $u_n = n + (-1)^n\sqrt{n}$.
\begin{enumerate}
  \item La suite est-elle monotone ?
  \item Montrer que pour tout $n \geq 1$, $u_n \geq n - \sqrt{n}$.
  \item Montrer que $\lim_{n \to +\infty}(n - \sqrt{n}) = +\infty$ (factoriser par $n$).
  \item Conclure sur la limite de $(u_n)$.
\end{enumerate}
\end{exercice}
