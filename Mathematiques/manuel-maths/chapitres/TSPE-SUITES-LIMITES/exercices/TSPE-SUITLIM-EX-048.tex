\begin{exercice}{TSPE-SUITLIM-EX-048}{3}{20}
\begin{enumerate}
  \item Soit $g(x) = \mathrm{e}^x - 1 - x$. Etudier les variations de $g$ et montrer que $g(x) \geq 0$ pour tout $x \in \mathbb{R}$.
  \item En deduire que $\mathrm{e}^x \geq 1 + x$ pour tout $x \in \mathbb{R}$.
  \item En appliquant cette inegalite a $x = \dfrac{1}{2}$ repete $n$ fois, montrer que $\mathrm{e}^{n/2} \geq \left(\dfrac{3}{2}\right)^n$.
  \item En deduire une nouvelle preuve que $\lim_{x \to +\infty} \mathrm{e}^x = +\infty$.
\end{enumerate}
\end{exercice}
