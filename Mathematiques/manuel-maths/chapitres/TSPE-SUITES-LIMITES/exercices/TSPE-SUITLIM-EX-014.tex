\begin{exercice}{TSPE-SUITLIM-EX-014}{3}{20}
Soit $(u_n)$ definie par $u_0 = 3$ et $u_{n+1} = \sqrt{2u_n + 3}$ pour tout $n \in \mathbb{N}$.
\begin{enumerate}
  \item Montrer par recurrence que pour tout $n \in \mathbb{N}$, $1 \leq u_n \leq 3$.
  \item Montrer que la suite $(u_n)$ est decroissante.
  \item En deduire que $(u_n)$ converge et determiner sa limite.
\end{enumerate}
\end{exercice}
