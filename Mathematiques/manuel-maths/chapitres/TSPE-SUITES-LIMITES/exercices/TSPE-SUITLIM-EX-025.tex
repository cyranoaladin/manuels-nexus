\begin{exercice}{TSPE-SUITLIM-EX-025}{2}{15}
Soit $(u_n)$ definie par $u_0 = 1$ et $u_{n+1} = u_n + \dfrac{1}{u_n}$.
\begin{enumerate}
  \item Montrer que pour tout $n$, $u_n > 0$.
  \item Montrer que $(u_n)$ est strictement croissante.
  \item Montrer que $u_n^2 \geq n + 1$ pour tout $n$ (par recurrence).
  \item En deduire que $(u_n)$ diverge vers $+\infty$.
\end{enumerate}
\end{exercice}
