\begin{exercice}{TSPE-SUITLIM-EX-013}{3}{20}
\begin{enumerate}
  \item Demontrer par recurrence que pour tout entier $n \geq 1$,
    \[
      \sum_{k=1}^{n} k^2 = \frac{n(n+1)(2n+1)}{6}.
    \]
  \item En deduire $\lim_{n \to +\infty} \dfrac{1}{n^3}\sum_{k=1}^{n} k^2$.
\end{enumerate}
