\begin{exercice}{TSPE-SUITLIM-EX-023}{1}{12}
Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = \sqrt{n}$.
\begin{enumerate}
  \item Montrer que $(u_n)$ est croissante.
  \item Soit $M > 0$ un reel quelconque. Montrer qu'il existe $N$ tel que $u_N > M$.
  \item Conclure que $\lim_{n \to +\infty} u_n = +\infty$.
\end{enumerate}
\end{exercice}
