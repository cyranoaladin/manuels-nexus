\begin{exercice}{TSPE-SUITLIM-EX-031}{1}{10}
Soit $(u_n)$ la suite definie par $u_n = 3 \times \left(\dfrac{2}{3}\right)^n + 5$.
\begin{enumerate}
  \item Calculer $u_0$ et $u_1$.
  \item Justifier que $\left(\dfrac{2}{3}\right)^n \to 0$ quand $n \to +\infty$.
  \item En deduire $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
