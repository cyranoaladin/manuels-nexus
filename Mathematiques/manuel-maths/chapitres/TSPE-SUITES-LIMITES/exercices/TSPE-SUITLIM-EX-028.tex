\begin{exercice}{TSPE-SUITLIM-EX-028}{3}{20}
Soit $(u_n)$ definie par $u_0 = 0$ et $u_{n+1} = \sin(u_n) + 1$.
\begin{enumerate}
  \item Calculer $u_1$, $u_2$ et $u_3$ (arrondir a $10^{-2}$).
  \item Montrer que pour tout $n$, $0 \leq u_n \leq 2$.
  \item Montrer que $(u_n)$ est croissante (on pourra etudier $f(x) = \sin(x) + 1 - x$ sur $[0, 2]$).
  \item En deduire que $(u_n)$ converge. Ecrire l'equation verifiee par la limite.
\end{enumerate}
\end{exercice}
