\begin{exercice}{TSPE-SUITLIM-EX-034}{3}{20}
Soit $a \geq 0$ un reel fixe.
\begin{enumerate}
  \item Demontrer par recurrence que pour tout $n \geq 2$,
    \[
      (1+a)^n \geq 1 + na + \frac{n(n-1)}{2}\,a^2.
    \]
  \item En deduire que pour $q > 1$, $\lim_{n \to +\infty} \dfrac{q^n}{n} = +\infty$.
\end{enumerate}
\end{exercice}
