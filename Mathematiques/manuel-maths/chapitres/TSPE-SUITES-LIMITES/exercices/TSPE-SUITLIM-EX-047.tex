\begin{exercice}{TSPE-SUITLIM-EX-047}{2}{15}
Soit $(u_n)$ la suite definie par $u_n = \dfrac{\mathrm{e}^n}{n^2 + 1}$.
\begin{enumerate}
  \item Montrer que $u_n \geq \dfrac{\mathrm{e}^n}{2n^2}$ pour $n \geq 1$.
  \item En deduire que $\lim_{n \to +\infty} u_n = +\infty$.
  \item Montrer que $(u_n)$ est croissante a partir d'un certain rang.
\end{enumerate}
\end{exercice}
