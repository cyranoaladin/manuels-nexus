\begin{exercice}{TSPE-SUITLIM-EX-005}{2}{12}
Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = \sqrt{n^2 + 3n} - n$.
\begin{enumerate}
  \item Montrer que pour tout $n \geq 1$, $u_n = \dfrac{3n}{\sqrt{n^2 + 3n} + n}$.
  \item En deduire $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
