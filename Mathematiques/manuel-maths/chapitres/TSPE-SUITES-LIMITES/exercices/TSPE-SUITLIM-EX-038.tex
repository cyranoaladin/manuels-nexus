\begin{exercice}{TSPE-SUITLIM-EX-038}{1}{10}
Soit $(u_n)$ la suite definie pour $n \geq 1$ par $u_n = \dfrac{n + \sin(n)}{n + 1}$.
\begin{enumerate}
  \item Montrer que pour tout $n \geq 1$, $\dfrac{n - 1}{n + 1} \leq u_n \leq \dfrac{n + 1}{n + 1} = 1$.
  \item Determiner $\lim_{n \to +\infty} \dfrac{n-1}{n+1}$.
  \item En deduire $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
