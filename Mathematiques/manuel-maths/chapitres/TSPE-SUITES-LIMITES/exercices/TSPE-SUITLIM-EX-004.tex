\begin{exercice}{TSPE-SUITLIM-EX-004}{2}{12}
Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = \dfrac{3n + (-1)^n}{n + 2}$.
\begin{enumerate}
  \item Calculer $u_0$, $u_1$, $u_2$ et $u_3$.
  \item Montrer que pour tout $n \in \mathbb{N}$, $\dfrac{3n - 1}{n + 2} \leq u_n \leq \dfrac{3n + 1}{n + 2}$.
  \item En deduire $\lim_{n \to +\infty} u_n$ a l'aide du theoreme des gendarmes.
\end{enumerate}
\end{exercice}
