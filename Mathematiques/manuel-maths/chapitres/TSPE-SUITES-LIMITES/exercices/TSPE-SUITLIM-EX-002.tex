\begin{exercice}{TSPE-SUITLIM-EX-002}{1}{8}
Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = \dfrac{n^2 - 4}{2n^2 + 1}$.
\begin{enumerate}
  \item Calculer $u_0$ et $u_2$.
  \item Determiner $\lim_{n \to +\infty} u_n$ en factorisant par le terme dominant.
\end{enumerate}
\end{exercice}
