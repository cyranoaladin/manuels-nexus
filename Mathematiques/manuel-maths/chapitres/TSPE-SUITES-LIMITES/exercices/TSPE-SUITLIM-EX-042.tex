\begin{exercice}{TSPE-SUITLIM-EX-042}{3}{20}
\textbf{Theoreme de Cesaro.} Soit $(u_n)$ une suite convergente de limite $\ell$. On pose $S_n = \dfrac{u_0 + u_1 + \cdots + u_n}{n + 1}$.
\begin{enumerate}
  \item Soit $\varepsilon > 0$. Justifier qu'il existe $N$ tel que pour tout $n \geq N$, $|u_n - \ell| \leq \varepsilon$.
  \item Pour $n > N$, decomposer $S_n - \ell$ en
    \[
      S_n - \ell = \frac{1}{n+1}\sum_{k=0}^{N}(u_k - \ell) + \frac{1}{n+1}\sum_{k=N+1}^{n}(u_k - \ell).
    \]
  \item Majorer chacune des deux sommes pour montrer que $|S_n - \ell| \leq 2\varepsilon$ a partir d'un certain rang.
  \item Conclure que $\lim_{n \to +\infty} S_n = \ell$.
\end{enumerate}
\end{exercice}
