\begin{exercice}{TSPE-SUITLIM-EX-027}{3}{20}
Soit $(u_n)$ definie par $u_0 = 1$ et $u_{n+1} = u_n + \sqrt{u_n}$.
\begin{enumerate}
  \item Montrer que pour tout $n$, $u_n \geq 1$.
  \item Montrer que $(u_n)$ est strictement croissante.
  \item Montrer par recurrence que $u_n \geq \dfrac{n^2}{4}$ pour tout $n \geq 2$.
  \item En deduire le comportement de $(u_n)$ en $+\infty$.
\end{enumerate}
\end{exercice}
