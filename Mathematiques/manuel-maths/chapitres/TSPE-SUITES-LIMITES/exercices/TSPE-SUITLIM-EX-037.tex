\begin{exercice}{TSPE-SUITLIM-EX-037}{1}{10}
Soit $(u_n)$ la suite definie pour $n \geq 1$ par $u_n = \dfrac{\cos(n)}{n}$.
\begin{enumerate}
  \item Justifier que pour tout $n \geq 1$, $-\dfrac{1}{n} \leq u_n \leq \dfrac{1}{n}$.
  \item En deduire $\lim_{n \to +\infty} u_n$ par le theoreme des gendarmes.
\end{enumerate}
\end{exercice}
