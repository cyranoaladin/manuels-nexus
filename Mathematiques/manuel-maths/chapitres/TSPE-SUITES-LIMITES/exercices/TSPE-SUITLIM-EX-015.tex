\begin{exercice}{TSPE-SUITLIM-EX-015}{1}{12}
Un reservoir contient $100$~litres d'eau. Chaque jour, $20\,\%$ de l'eau s'evapore et on ajoute $50$~litres. On note $u_n$ le volume d'eau (en litres) au debut du jour $n$.
\begin{enumerate}
  \item Justifier que $u_0 = 100$ et que $u_{n+1} = 0{,}8\,u_n + 50$.
  \item Calculer $u_1$ et $u_2$.
  \item Determiner le reel $\ell$ tel que $v_n = u_n - \ell$ soit geometrique.
  \item Exprimer $u_n$ en fonction de $n$ et determiner $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
