\begin{exercice}{TSPE-SUITLIM-EX-049}{3}{25}
On pose $v_n = \left(1 + \dfrac{1}{n}\right)^n$ pour $n \geq 1$.
\begin{enumerate}
  \item Calculer $v_1$, $v_2$, $v_5$ et $v_{10}$ (arrondir a $10^{-4}$).
  \item En utilisant l'inegalite de Bernoulli, montrer que $v_n \geq 2$ pour tout $n \geq 1$.
  \item Montrer que pour tout $n \geq 1$, $v_n \leq 3$ (on pourra utiliser $\left(1 + \dfrac{1}{n}\right)^n \leq \mathrm{e}$, admis).
  \item Conjecturer la limite de $(v_n)$.
\end{enumerate}
(On admet que $\lim_{n \to +\infty} v_n = \mathrm{e}$.)
\end{exercice}
