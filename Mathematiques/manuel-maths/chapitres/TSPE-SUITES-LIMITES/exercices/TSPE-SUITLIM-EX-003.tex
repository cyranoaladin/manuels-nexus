\begin{exercice}{TSPE-SUITLIM-EX-003}{1}{10}
Soit $(u_n)$ la suite definie pour tout $n \in \mathbb{N}$ par $u_n = 3n^2 - 5n + 2$.
\begin{enumerate}
  \item Calculer $u_0$, $u_1$ et $u_2$.
  \item Montrer que pour tout $n \geq 2$, $u_n \geq n^2$.
  \item En deduire $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
