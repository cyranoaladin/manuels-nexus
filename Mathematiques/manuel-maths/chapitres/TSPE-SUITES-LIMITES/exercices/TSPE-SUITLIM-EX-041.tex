\begin{exercice}{TSPE-SUITLIM-EX-041}{3}{20}
On pose $S_n = \displaystyle\sum_{k=1}^{n} \dfrac{1}{k^2}$ pour $n \geq 1$.
\begin{enumerate}
  \item Montrer que $(S_n)$ est croissante.
  \item Montrer que pour tout $k \geq 2$, $\dfrac{1}{k^2} \leq \dfrac{1}{k(k-1)} = \dfrac{1}{k-1} - \dfrac{1}{k}$.
  \item En deduire que $S_n \leq 2 - \dfrac{1}{n}$ pour tout $n \geq 1$.
  \item Conclure que $(S_n)$ converge. (On admettra que $\lim S_n = \dfrac{\pi^2}{6}$.)
\end{enumerate}
\end{exercice}
