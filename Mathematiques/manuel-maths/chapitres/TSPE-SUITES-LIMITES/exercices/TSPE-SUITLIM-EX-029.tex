\begin{exercice}{TSPE-SUITLIM-EX-029}{1}{8}

\begin{enumerate}
  \item $u_n = \left(\dfrac{3}{4}\right)^n$
  \item $v_n = \left(\dfrac{5}{3}\right)^n$
  \item $w_n = \left(-\dfrac{1}{2}\right)^n$
  \item $t_n = (-1)^n$
\end{enumerate}
