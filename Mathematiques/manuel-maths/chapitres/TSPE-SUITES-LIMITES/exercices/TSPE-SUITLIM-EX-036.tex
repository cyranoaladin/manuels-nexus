\begin{exercice}{TSPE-SUITLIM-EX-036}{1}{8}
Soit $(u_n)$ la suite definie par $u_n = n^2 + 3n$.
\begin{enumerate}
  \item Montrer que pour tout $n \in \mathbb{N}$, $u_n \geq n^2$.
  \item En deduire que $\lim_{n \to +\infty} u_n = +\infty$.
\end{enumerate}
\end{exercice}
