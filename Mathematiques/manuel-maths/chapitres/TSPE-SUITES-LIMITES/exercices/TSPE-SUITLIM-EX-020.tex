\begin{exercice}{TSPE-SUITLIM-EX-020}{3}{20}
Soit $(u_n)$ definie par $u_0 = 2$ et $u_{n+1} = \sqrt{u_n + 6}$ pour tout $n \in \mathbb{N}$.
\begin{enumerate}
  \item Calculer $u_1$, $u_2$, $u_3$ (valeurs exactes).
  \item Montrer par recurrence que pour tout $n$, $0 \leq u_n \leq 3$.
  \item Montrer que la suite est croissante.
  \item En deduire que $(u_n)$ converge et determiner sa limite.
\end{enumerate}
\end{exercice}
