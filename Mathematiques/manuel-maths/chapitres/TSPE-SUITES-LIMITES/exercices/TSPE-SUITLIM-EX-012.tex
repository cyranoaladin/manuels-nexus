\begin{exercice}{TSPE-SUITLIM-EX-012}{2}{15}
Soit $(u_n)$ definie par $u_0 = 1$ et $u_{n+1} = \dfrac{u_n}{2u_n + 1}$ pour tout $n \in \mathbb{N}$.
\begin{enumerate}
  \item Calculer $u_1$, $u_2$ et $u_3$.
  \item Demontrer par recurrence que pour tout $n \in \mathbb{N}$, $0 < u_n \leq \dfrac{1}{2n+1}$.
  \item En deduire $\lim_{n \to +\infty} u_n$.
\end{enumerate}
\end{exercice}
