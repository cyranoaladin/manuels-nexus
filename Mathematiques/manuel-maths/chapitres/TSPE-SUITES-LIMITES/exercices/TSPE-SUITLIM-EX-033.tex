\begin{exercice}{TSPE-SUITLIM-EX-033}{2}{15}
Soit $(u_n)$ la suite definie par $u_n = n \times 0{,}9^n$.
\begin{enumerate}
  \item Calculer $u_1$, $u_5$ et $u_{10}$ (arrondir a $10^{-2}$).
  \item On pose $q = 0{,}9$ et on ecrit $\dfrac{1}{q} = \dfrac{10}{9} = 1 + \dfrac{1}{9}$. En utilisant l'inegalite de Bernoulli, montrer que $\left(\dfrac{1}{q}\right)^n \geq 1 + \dfrac{n}{9}$.
  \item En deduire que $0 \leq u_n \leq \dfrac{9n}{9 + n}$ et conclure sur $\lim u_n$.
\end{enumerate}
\end{exercice}
