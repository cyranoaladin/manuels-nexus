\begin{exercice}{TSPE-SUITLIM-EX-024}{1}{12}
On considere la suite des sommes partielles harmoniques $S_n = \displaystyle\sum_{k=1}^{n} \dfrac{1}{k}$.
\begin{enumerate}
  \item Calculer $S_1$, $S_2$, $S_3$ et $S_4$.
  \item Montrer que $(S_n)$ est croissante.
  \item Montrer que $S_{2^p} \geq 1 + \dfrac{p}{2}$ pour tout $p \geq 1$ (par groupement de termes).
  \item En deduire que $(S_n)$ diverge vers $+\infty$.
\end{enumerate}
\end{exercice}
