\subsection*{Rappel — Suites arithmetiques et geometriques}

\textbf{Suite arithmetique :} $u_{n+1} = u_n + r$. Terme general : $u_n = u_0 + nr$.

\textbf{Suite geometrique :} $u_{n+1} = q \cdot u_n$. Terme general : $u_n = u_0 \cdot q^n$.

\textbf{Sommes :} $\sum_{k=0}^{n} (u_0 + kr) = (n+1)u_0 + \dfrac{n(n+1)}{2}r$ et $\sum_{k=0}^{n} u_0 q^k = u_0 \cdot \dfrac{1-q^{n+1}}{1-q}$ (si $q \neq 1$).

\begin{exercice}{TSPE-SUITLIM-FR-R1-EX1}{1}{8}
\begin{enumerate}
  \item La suite $(u_n)$ definie par $u_n = 3 + 5n$ est-elle arithmetique ? Preciser la raison.
  \item Exprimer $u_{20}$ et $\sum_{k=0}^{10} u_k$.
  \item La suite $(v_n)$ definie par $v_n = 4 \times 2^n$ est-elle geometrique ? Preciser la raison.
\end{enumerate}
\end{exercice}
