\subsection*{Rappel — Derivation et variations}

\textbf{Si $f'(x) > 0$} sur un intervalle, $f$ est strictement croissante.

\textbf{Si $f'(x) < 0$} sur un intervalle, $f$ est strictement decroissante.

\textbf{Derivees usuelles :} $(x^n)' = nx^{n-1}$, $(\mathrm{e}^x)' = \mathrm{e}^x$, $(uv)' = u'v + uv'$.

\begin{exercice}{TSPE-SUITLIM-FR-R4-EX1}{1}{10}
Calculer la derivee et etudier les variations.
\begin{enumerate}
  \item $f(x) = x^2 - 6x + 5$
  \item $g(x) = x\,\mathrm{e}^x$
\end{enumerate}
\end{exercice}
