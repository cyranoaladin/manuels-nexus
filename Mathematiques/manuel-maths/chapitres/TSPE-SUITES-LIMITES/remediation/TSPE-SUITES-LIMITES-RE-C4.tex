\begin{exercice}{TSPE-SUITLIM-RE-C4-EX1}{1}{10}
Soit $(u_n)$ definie par $u_n = \dfrac{n^2}{n+1}$.
\begin{enumerate}
  \item Montrer que $(u_n)$ est croissante (pour $n \geq 1$).
  \item Est-elle majoree ? Conclure.
\end{enumerate}
\end{exercice}

\begin{corrige}{TSPE-SUITLIM-RE-C4-EX1}
$u_n = n - \dfrac{n}{n+1}$. $u_{n+1}-u_n > 0$ pour $n \geq 1$ (calcul). $u_n \geq n/2 \to +\infty$. Non majoree. $\lim u_n = +\infty$.
\end{corrige}
