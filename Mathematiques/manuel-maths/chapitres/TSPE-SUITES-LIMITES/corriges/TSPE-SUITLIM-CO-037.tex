\begin{corrige}{TSPE-SUITLIM-EX-037}

\textbf{Q1.} $|\cos(n)| \leq 1$, donc $|u_n| = |\cos(n)|/n \leq 1/n$, soit $-1/n \leq u_n \leq 1/n$.

\textbf{Q2.} $\lim(\pm 1/n) = 0$. Par les gendarmes, $\boxed{\lim u_n = 0}$.
