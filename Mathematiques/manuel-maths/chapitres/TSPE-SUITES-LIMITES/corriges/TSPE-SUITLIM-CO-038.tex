\begin{corrige}{TSPE-SUITLIM-EX-038}

\textbf{Q1.} $-1 \leq \sin(n) \leq 1$, donc $n-1 \leq n+\sin(n) \leq n+1$. En divisant par $n+1>0$ : $\dfrac{n-1}{n+1} \leq u_n \leq 1$.

\textbf{Q2.} $\dfrac{n-1}{n+1} = \dfrac{1-1/n}{1+1/n} \to 1$.

\textbf{Q3.} Par les gendarmes (bornes $\to 1$), $\boxed{\lim u_n = 1}$.

\end{corrige}
