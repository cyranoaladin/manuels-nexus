\begin{corrige}{TSPE-SUITLIM-EX-006}

\textbf{Question 1.} $u_n - 2 = \dfrac{2n+3}{n+1} - 2 = \dfrac{2n+3-2(n+1)}{n+1} = \dfrac{1}{n+1}$.

\commentaireMarge{Preuve $\varepsilon$-$N$ : resoudre $\dfrac{1}{n+1} \leq \varepsilon$.}

\textbf{Question 2.} $|u_n - 2| = \dfrac{1}{n+1} \leq \varepsilon \iff n+1 \geq \dfrac{1}{\varepsilon} \iff n \geq \dfrac{1}{\varepsilon} - 1$.

On pose $N = \left\lfloor \dfrac{1}{\varepsilon} \right\rfloor$. Pour tout $n \geq N$, $|u_n - 2| \leq \varepsilon$.

\textbf{Question 3.} Pour tout $\varepsilon > 0$, il existe $N = \lfloor 1/\varepsilon \rfloor$ tel que pour tout $n \geq N$, $|u_n - 2| \leq \varepsilon$. Par definition, $\boxed{\lim_{n \to +\infty} u_n = 2}$.

\end{corrige}
