\textbf{Question 1.} Pour $n \geq 1$ :
\[
u_n = \frac{(\sqrt{n^2+3n} - n)(\sqrt{n^2+3n} + n)}{\sqrt{n^2+3n} + n} = \frac{n^2+3n - n^2}{\sqrt{n^2+3n}+n} = \frac{3n}{\sqrt{n^2+3n}+n}.
\]

\textbf{Question 2.} Factorisons par $n$ : $u_n = \dfrac{3}{\sqrt{1+3/n}+1} \to \dfrac{3}{1+1} = \boxed{\dfrac{3}{2}}$.
