\begin{corrige}{TSPE-SUITLIM-EX-030}

\textbf{Q1.} Pour tout $a \geq -1$ et tout $n \in \mathbb{N}$ : $(1+a)^n \geq 1+na$.

\textbf{Q2.} On prend $a = 0{,}1 \geq -1$. Bernoulli donne $(1+0{,}1)^n \geq 1+0{,}1n$, soit $1{,}1^n \geq 1+0{,}1n$.

\textbf{Q3.} $1+0{,}1n \to +\infty$ et $1{,}1^n \geq 1+0{,}1n$. Par comparaison, $\boxed{\lim 1{,}1^n = +\infty}$.

\end{corrige}
