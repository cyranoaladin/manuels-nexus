\begin{corrige}{TSPE-SUITLIM-EX-034}

\textbf{Q1.} $\mathcal{P}(n)$ : « $(1+a)^n \geq 1+na+\dfrac{n(n-1)}{2}a^2$ ».

$n=2$ : $(1+a)^2 = 1+2a+a^2 \geq 1+2a+a^2$. Vrai (egalite).

Heredite : $(1+a)^{n+1} = (1+a)^n(1+a) \geq (1+na+\dfrac{n(n-1)}{2}a^2)(1+a)$. En developpant et minorant (les termes en $a^3$ sont positifs), on obtient $\geq 1+(n+1)a+\dfrac{n(n+1)}{2}a^2$.

\textbf{Q2.} $q = 1+a$, $a>0$. $q^n \geq 1+na+\dfrac{n(n-1)}{2}a^2 \geq \dfrac{n(n-1)}{2}a^2$. $\dfrac{q^n}{n} \geq \dfrac{(n-1)a^2}{2} \to +\infty$. $\boxed{\lim q^n/n = +\infty}$.

\end{corrige}
