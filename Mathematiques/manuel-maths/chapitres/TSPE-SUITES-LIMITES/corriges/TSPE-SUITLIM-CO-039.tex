\begin{corrige}{TSPE-SUITLIM-EX-039}

\textbf{Q1.} $u_n = \dfrac{2^n+n}{2^n+1} = 1 + \dfrac{n-1}{2^n+1}$.

\textbf{Q2.} Pour $n \geq 1$ : $0 \leq \dfrac{n-1}{2^n+1} \leq \dfrac{n}{2^n}$ (car $n-1 \leq n$ et $2^n+1 \geq 2^n$).

\textbf{Q3.} $n/2^n \to 0$ (croissance comparee : $2>1$). Par les gendarmes, $\dfrac{n-1}{2^n+1} \to 0$. $\boxed{\lim u_n = 1}$.

\end{corrige}
