\begin{corrige}{TSPE-SUITLIM-EX-042}

\textbf{Q1.} $\lim u_n = \ell$, donc $\exists N : \forall n \geq N, |u_n-\ell| \leq \varepsilon$.

\textbf{Q2.} $S_n - \ell = \dfrac{1}{n+1}\sum_{k=0}^N(u_k-\ell) + \dfrac{1}{n+1}\sum_{k=N+1}^n(u_k-\ell)$.

\textbf{Q3.} Premiere somme : $\left|\dfrac{1}{n+1}\sum_{k=0}^N(u_k-\ell)\right| \leq \dfrac{C}{n+1}$ ou $C = \sum_{k=0}^N|u_k-\ell|$ est fixe. Pour $n$ assez grand, $\leq \varepsilon$.

Deuxieme somme : $\left|\dfrac{1}{n+1}\sum_{k=N+1}^n(u_k-\ell)\right| \leq \dfrac{(n-N)\varepsilon}{n+1} \leq \varepsilon$.

Total : $|S_n-\ell| \leq 2\varepsilon$ pour $n$ assez grand.

\textbf{Q4.} $\forall \varepsilon > 0$, $|S_n - \ell| \leq 2\varepsilon$ a partir d'un certain rang. $\boxed{\lim S_n = \ell}$.

\end{corrige}
