\textbf{Question 1.} $u_0 = \dfrac{1}{2}$, $u_1 = \dfrac{2}{3}$, $u_2 = \dfrac{7}{4}$, $u_3 = \dfrac{8}{5}$.

\commentaireMarge{Encadrer $(-1)^n$ entre $-1$ et $1$.}

\textbf{Question 2.} Pour tout $n$, $-1 \leq (-1)^n \leq 1$, donc $3n - 1 \leq 3n + (-1)^n \leq 3n + 1$. Comme $n + 2 > 0$ :
\[
\frac{3n-1}{n+2} \leq u_n \leq \frac{3n+1}{n+2}.
\]

\textbf{Question 3.} $\lim \dfrac{3n \pm 1}{n+2} = 3$. Par les gendarmes, $\boxed{\lim u_n = 3}$.
