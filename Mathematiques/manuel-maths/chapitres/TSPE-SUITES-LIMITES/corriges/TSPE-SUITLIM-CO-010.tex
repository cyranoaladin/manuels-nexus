\begin{corrige}{TSPE-SUITLIM-EX-010}

\textbf{Question 1.} $u_1 = 3 \times 2 - 4 = 2$, $u_2 = 3 \times 2 - 4 = 2$, $u_3 = 2$. La suite semble constante egale a $2$.

\commentaireMarge{Le point fixe de $f(x)=3x-4$ est $x=2$.}

\textbf{Question 2.} Soit $\mathcal{P}(n)$ : « $u_n = 2$ ».

\textit{Init.} $u_0 = 2$. $\mathcal{P}(0)$ vraie.

\textit{Heredite.} Si $u_n = 2$, alors $u_{n+1} = 3 \times 2 - 4 = 2$. $\mathcal{P}(n+1)$ vraie.

\textit{Conclusion.} $\boxed{u_n = 2}$ pour tout $n$.

\end{corrige}
