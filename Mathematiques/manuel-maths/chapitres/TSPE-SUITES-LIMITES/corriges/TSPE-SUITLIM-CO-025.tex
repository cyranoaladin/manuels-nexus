\begin{corrige}{TSPE-SUITLIM-EX-025}

\textbf{Q1.} $u_0=1>0$. Si $u_n>0$, $u_{n+1}=u_n+1/u_n>0$. Par recurrence, $u_n>0$.

\textbf{Q2.} $u_{n+1}-u_n=1/u_n>0$. Strictement croissante.

\textbf{Q3.} $\mathcal{P}(n)$ : « $u_n^2 \geq n+1$ ». $u_0^2=1 \geq 1$. Si $u_n^2 \geq n+1$ : $u_{n+1}^2 = u_n^2 + 2 + 1/u_n^2 \geq n+1+2 = n+3 > n+2$. Vrai.

\textbf{Q4.} $u_n \geq \sqrt{n+1} \to +\infty$. $\boxed{\lim u_n = +\infty}$.

\end{corrige}
