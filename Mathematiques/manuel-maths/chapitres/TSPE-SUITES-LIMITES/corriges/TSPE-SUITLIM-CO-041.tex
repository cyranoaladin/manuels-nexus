\begin{corrige}{TSPE-SUITLIM-EX-041}

\textbf{Q1.} $S_{n+1} = S_n + 1/(n+1)^2 > S_n$. Croissante.

\textbf{Q2.} $\dfrac{1}{k^2} \leq \dfrac{1}{k(k-1)}$ car $k(k-1) \leq k^2$. Et $\dfrac{1}{k(k-1)} = \dfrac{1}{k-1}-\dfrac{1}{k}$ (decomposition en elements simples).

\textbf{Q3.} $S_n = 1 + \sum_{k=2}^n \dfrac{1}{k^2} \leq 1 + \sum_{k=2}^n \left(\dfrac{1}{k-1}-\dfrac{1}{k}\right) = 1 + 1 - \dfrac{1}{n} = 2-\dfrac{1}{n}$.

\textbf{Q4.} Croissante et majoree par $2$ : $\boxed{(S_n) \text{ converge}}$.
