\begin{corrige}{TSPE-SUITLIM-EX-044}

\textbf{a.} $\mathrm{e}^x - x = x(\mathrm{e}^x/x - 1)$. $\mathrm{e}^x/x \to +\infty$, donc $\boxed{\lim = +\infty}$.

\textbf{b.} $\dfrac{\mathrm{e}^x+1}{\mathrm{e}^x-1} = \dfrac{1+\mathrm{e}^{-x}}{1-\mathrm{e}^{-x}} \to \dfrac{1+0}{1-0} = \boxed{1}$.

\textbf{c.} $\mathrm{e}^{-x} \to 0$, donc $\boxed{\lim = 3}$.

\end{corrige}
