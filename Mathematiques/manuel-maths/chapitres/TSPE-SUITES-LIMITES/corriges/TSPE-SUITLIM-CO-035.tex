\begin{corrige}{TSPE-SUITLIM-EX-035}

\textbf{Q1.} $S_n = \dfrac{1-q^{n+1}}{1-q}$.

\textbf{Q2.} $|q|<1 \Rightarrow q^{n+1} \to 0$. $\lim S_n = \dfrac{1}{1-q}$.

\textbf{Q3.} $q=1/3$ : $\lim S_n = \dfrac{1}{2/3} = \boxed{3/2}$. $q=2/5$ : $\lim S_n = \dfrac{1}{3/5} = \boxed{5/3}$.

\textbf{Q4.} Si $q \geq 1$, $q^{n+1} \to +\infty$ (ou $= 1$), donc $S_n \to +\infty$. Pas de somme finie.

\end{corrige}
