\textbf{Question 1.} $|\cos(n)| \leq 1$, donc $|u_n| = \dfrac{|n\cos(n)|}{n^2+1} = \dfrac{n|\cos(n)|}{n^2+1} \leq \dfrac{n}{n^2+1}$.

\textbf{Question 2.} Pour $n \geq 1$ : $\dfrac{n}{n^2+1} \leq \dfrac{n}{n^2} = \dfrac{1}{n}$.

\commentaireMarge{Appliquer les gendarmes avec $-1/n \leq u_n \leq 1/n$.}

\textbf{Question 3.} On a $-\dfrac{1}{n} \leq u_n \leq \dfrac{1}{n}$ et $\lim(\pm 1/n) = 0$. Par les gendarmes, $\boxed{\lim u_n = 0}$.
