\begin{exercice}{1SPE-TRIGO-EX-018}{3}{15}
\begin{enumerate}
  \item Developper $(\cos x + \sin x)^2$ et simplifier a l'aide de $\cos^2 x + \sin^2 x = 1$.
  \item En deduire que $(\cos x + \sin x)^2 = 1 + 2\cos x \sin x$.
  \item De meme, montrer que $(\cos x - \sin x)^2 = 1 - 2\cos x \sin x$.
  \item On sait que $\cos x + \sin x = \dfrac{7}{5}$. Calculer $\cos x \cdot \sin x$, puis $\cos x - \sin x$ (sachant que $x \in \left]0;\dfrac{\pi}{4}\right[$).
\end{enumerate}
\end{exercice}
