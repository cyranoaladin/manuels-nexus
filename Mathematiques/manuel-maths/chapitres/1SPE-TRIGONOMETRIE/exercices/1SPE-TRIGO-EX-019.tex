\begin{exercice}{1SPE-TRIGO-EX-019}{3}{15}
\begin{enumerate}
  \item A l'aide des formules d'addition (vues en C3) ou a l'aide du cercle trigonometrique, demontrer que :
  \[
    \cos\!\left(\frac{\pi}{2} - x\right) = \sin x \quad \text{et} \quad \sin\!\left(\frac{\pi}{2} - x\right) = \cos x.
  \]
  \item En deduire que $\cos\!\left(\frac{\pi}{2} + x\right) = -\sin x$ et $\sin\!\left(\frac{\pi}{2} + x\right) = \cos x$.
  \item Application : calculer $\cos\!\left(\dfrac{\pi}{2} + \dfrac{\pi}{6}\right)$ et $\sin\!\left(\dfrac{\pi}{2} - \dfrac{\pi}{3}\right)$.
\end{enumerate}
\end{exercice}
