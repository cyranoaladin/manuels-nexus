\begin{exercice}{1SPE-TRIGO-EX-029}{3}{15}
On admet que $\cos\!\left(\dfrac{\pi}{5}\right) = \dfrac{1+\sqrt{5}}{4}$.
\begin{enumerate}
  \item Verifier que $\cos^2\!\left(\dfrac{\pi}{5}\right) = \dfrac{6+2\sqrt{5}}{16} = \dfrac{3+\sqrt{5}}{8}$.
  \item En deduire $\cos\!\left(\dfrac{2\pi}{5}\right)$ a l'aide de la formule de duplication.
  \item Calculer $\sin\!\left(\dfrac{\pi}{5}\right)$ (valeur exacte).
\end{enumerate}
\end{exercice}
