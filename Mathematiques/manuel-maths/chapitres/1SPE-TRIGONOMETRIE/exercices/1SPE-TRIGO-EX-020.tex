\begin{exercice}{1SPE-TRIGO-EX-020}{3}{15}
\textbf{Retrouver les valeurs remarquables par la geometrie.}
\begin{enumerate}
  \item Dans un triangle rectangle isocele d'hypotenuse $1$, calculer la longueur des cotes egaux. En deduire $\cos\!\left(\dfrac{\pi}{4}\right)$ et $\sin\!\left(\dfrac{\pi}{4}\right)$.
  \item Dans un triangle equilateral de cote $1$, tracer une hauteur. Calculer la longueur de cette hauteur. En deduire $\cos\!\left(\dfrac{\pi}{6}\right)$, $\sin\!\left(\dfrac{\pi}{6}\right)$, $\cos\!\left(\dfrac{\pi}{3}\right)$ et $\sin\!\left(\dfrac{\pi}{3}\right)$.
\end{enumerate}
\end{exercice}
