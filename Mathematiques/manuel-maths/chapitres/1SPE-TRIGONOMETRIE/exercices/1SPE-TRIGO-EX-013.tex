\begin{exercice}{1SPE-TRIGO-EX-013}{1}{10}
\begin{enumerate}
  \item Sachant que $\cos x = \dfrac{3}{5}$ et que $x \in \left[0\,;\,\dfrac{\pi}{2}\right]$, calculer $\sin x$.
  \item Sachant que $\sin x = -\dfrac{5}{13}$ et que $x \in \left[\pi\,;\,\dfrac{3\pi}{2}\right]$, calculer $\cos x$.
  \item Verifier dans chaque cas que $\cos^2 x + \sin^2 x = 1$.
\end{enumerate}
