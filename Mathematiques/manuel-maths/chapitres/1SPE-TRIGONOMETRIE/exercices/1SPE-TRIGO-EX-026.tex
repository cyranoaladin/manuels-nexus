\begin{exercice}{1SPE-TRIGO-EX-026}{2}{12}
\begin{enumerate}
  \item En ecrivant $3x = 2x + x$, developper $\cos(3x)$ a l'aide de la formule d'addition $\cos(a+b)$.
  \item Utiliser la formule de duplication pour exprimer $\cos(2x)$ et $\sin(2x)$ en fonction de $\cos x$ et $\sin x$.
  \item A l'aide de $\sin^2 x = 1 - \cos^2 x$, montrer que :
  \[
    \cos(3x) = 4\cos^3 x - 3\cos x.
  \]
\end{enumerate}
\end{exercice}
