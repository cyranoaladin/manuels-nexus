\begin{exercice}{1SPE-TRIGO-EX-046}{3}{18}
On pose $f(x) = \sin^2 x - \cos x$.
\begin{enumerate}
  \item Montrer que $f(x) = -\cos^2 x - \cos x + 1$.
  \item Poser $u = \cos x$ et etudier la fonction $g(u) = -u^2 - u + 1$ sur $[-1\,;\,1]$.
  \item En deduire le maximum de $f$ sur $\mathbb{R}$ et la (les) valeur(s) de $x$ ou il est atteint.
\end{enumerate}
\end{exercice}
