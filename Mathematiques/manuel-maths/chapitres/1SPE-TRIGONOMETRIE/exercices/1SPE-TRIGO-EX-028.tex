\begin{exercice}{1SPE-TRIGO-EX-028}{3}{15}
\begin{enumerate}
  \item Developper $\cos(a+b)\cos(a-b)$ en utilisant les formules d'addition.
  \item Montrer que $\cos(a+b)\cos(a-b) = \cos^2 a - \sin^2 b$.
  \item Application : en posant $a = \dfrac{\pi}{4}$ et $b = \dfrac{\pi}{12}$, calculer $\cos\!\left(\dfrac{\pi}{3}\right)\cos\!\left(\dfrac{\pi}{6}\right)$ et verifier le resultat.
\end{enumerate}
\end{exercice}
