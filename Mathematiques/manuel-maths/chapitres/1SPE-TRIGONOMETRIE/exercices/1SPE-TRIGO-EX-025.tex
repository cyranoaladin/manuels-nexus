\begin{exercice}{1SPE-TRIGO-EX-025}{2}{12}
\begin{enumerate}
  \item A l'aide des formules d'addition, montrer que pour tous reels $a$ et $b$ :
  \[
    \cos(a+b) + \cos(a-b) = 2\cos a \cos b.
  \]
  \item De meme, montrer que :
  \[
    \sin(a+b) + \sin(a-b) = 2\sin a \cos b.
  \]
  \item Application : calculer $\cos\!\left(\dfrac{5\pi}{12}\right) + \cos\!\left(\dfrac{\pi}{12}\right)$ en posant $a = \dfrac{\pi}{4}$ et $b = \dfrac{\pi}{6}$.
\end{enumerate}
\end{exercice}
