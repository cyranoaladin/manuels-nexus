\begin{exercice}{1SPE-TRIGO-EX-021}{1}{10}
En ecrivant $\dfrac{\pi}{12} = \dfrac{\pi}{3} - \dfrac{\pi}{4}$, calculer :
\begin{enumerate}
  \item $\cos\!\left(\dfrac{\pi}{12}\right)$
  \item $\sin\!\left(\dfrac{\pi}{12}\right)$
\end{enumerate}
Verifier que $\cos^2\!\left(\dfrac{\pi}{12}\right) + \sin^2\!\left(\dfrac{\pi}{12}\right) = 1$.
\end{exercice}
