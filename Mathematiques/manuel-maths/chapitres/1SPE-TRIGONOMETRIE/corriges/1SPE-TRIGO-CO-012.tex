\begin{corrige}{1SPE-TRIGO-EX-012}

\commentaireMarge{$\cos(-x)=\cos x$, $\sin(-x)=-\sin x$, $\cos(\pi-x)=-\cos x$, $\sin(\pi+x)=-\sin x$.}

\textbf{1.} $\cos\!\left(-\dfrac{\pi}{3}\right) = \cos\!\left(\dfrac{\pi}{3}\right) = \dfrac{1}{2}$ (cosinus est pair).

\textbf{2.} $\sin\!\left(-\dfrac{\pi}{6}\right) = -\sin\!\left(\dfrac{\pi}{6}\right) = -\dfrac{1}{2}$ (sinus est impair).

\textbf{3.} $\cos\!\left(\dfrac{3\pi}{4}\right) = \cos\!\left(\pi - \dfrac{\pi}{4}\right) = -\cos\!\left(\dfrac{\pi}{4}\right) = -\dfrac{\sqrt{2}}{2}$.

\textbf{4.} $\sin\!\left(\dfrac{4\pi}{3}\right) = \sin\!\left(\pi + \dfrac{\pi}{3}\right) = -\sin\!\left(\dfrac{\pi}{3}\right) = -\dfrac{\sqrt{3}}{2}$.

\end{corrige}
