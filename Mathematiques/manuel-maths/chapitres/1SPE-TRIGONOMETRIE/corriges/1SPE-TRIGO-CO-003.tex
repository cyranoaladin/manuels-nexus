\begin{corrige}{1SPE-TRIGO-EX-003}

\commentaireMarge{Retrancher ou ajouter des multiples de $2\pi$ pour obtenir un angle dans $]-\pi;\pi]$.}

\textbf{1.} $\dfrac{13\pi}{3} = \dfrac{13\pi}{3} - 2 \times 2\pi = \dfrac{13\pi}{3} - \dfrac{12\pi}{3} = \dfrac{\pi}{3}$.

La mesure principale est $\dfrac{\pi}{3}$.

\medskip
\textbf{2.} $-\dfrac{7\pi}{4} = -\dfrac{7\pi}{4} + 2\pi = -\dfrac{7\pi}{4} + \dfrac{8\pi}{4} = \dfrac{\pi}{4}$.

La mesure principale est $\dfrac{\pi}{4}$.

\medskip
\textbf{3.} $\dfrac{17\pi}{6} = \dfrac{17\pi}{6} - 2\pi = \dfrac{17\pi}{6} - \dfrac{12\pi}{6} = \dfrac{5\pi}{6}$.

On a $\dfrac{5\pi}{6} \in {]-\pi;\pi]}$, donc la mesure principale est $\dfrac{5\pi}{6}$.

\medskip
\textbf{4.} $-\dfrac{11\pi}{3} = -\dfrac{11\pi}{3} + 2 \times 2\pi = -\dfrac{11\pi}{3} + \dfrac{12\pi}{3} = \dfrac{\pi}{3}$.

La mesure principale est $\dfrac{\pi}{3}$.

\end{corrige}
