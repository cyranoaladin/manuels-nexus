\begin{corrige}{1SPE-TRIGO-EX-034}

\textbf{1.} $\cos x \geqslant 0$ sur $[0;2\pi]$ : $\cos x = 0$ en $x = \dfrac{\pi}{2}$ et $x = \dfrac{3\pi}{2}$. Le cosinus est positif pour $x \in \left[0;\dfrac{\pi}{2}\right] \cup \left[\dfrac{3\pi}{2};2\pi\right]$.

\textbf{2.} $\sin x > \dfrac{1}{2}$ sur $[0;2\pi]$ : $\sin x = \dfrac{1}{2}$ en $x = \dfrac{\pi}{6}$ et $x = \dfrac{5\pi}{6}$. Le sinus est strictement superieur a $\dfrac{1}{2}$ pour $x \in \left]\dfrac{\pi}{6};\dfrac{5\pi}{6}\right[$.

\end{corrige}
