\begin{corrige}{1SPE-TRIGO-EX-019}

\textbf{1.} Par la formule d'addition :
$\cos\!\left(\dfrac{\pi}{2} - x\right) = \cos\dfrac{\pi}{2}\cos x + \sin\dfrac{\pi}{2}\sin x = 0 \cdot \cos x + 1 \cdot \sin x = \sin x$.

$\sin\!\left(\dfrac{\pi}{2} - x\right) = \sin\dfrac{\pi}{2}\cos x - \cos\dfrac{\pi}{2}\sin x = 1 \cdot \cos x - 0 \cdot \sin x = \cos x$.

\textbf{2.} En remplacant $x$ par $-x$ dans $\cos\!\left(\dfrac{\pi}{2} - x\right) = \sin x$ :
$\cos\!\left(\dfrac{\pi}{2} + x\right) = \sin(-x) = -\sin x$.

En remplacant $x$ par $-x$ dans $\sin\!\left(\dfrac{\pi}{2} - x\right) = \cos x$ :
$\sin\!\left(\dfrac{\pi}{2} + x\right) = \cos(-x) = \cos x$.

\textbf{3.} $\cos\!\left(\dfrac{\pi}{2} + \dfrac{\pi}{6}\right) = -\sin\dfrac{\pi}{6} = -\dfrac{1}{2}$.

$\sin\!\left(\dfrac{\pi}{2} - \dfrac{\pi}{3}\right) = \cos\dfrac{\pi}{3} = \dfrac{1}{2}$.
