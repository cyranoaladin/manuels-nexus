\textbf{1.} $\cos\!\left(\dfrac{\pi}{6}\right) = \dfrac{\sqrt{3}}{2}$.

\textbf{2.} $\sin\!\left(\dfrac{\pi}{4}\right) = \dfrac{\sqrt{2}}{2}$.

\textbf{3.} $\cos\!\left(\dfrac{\pi}{3}\right) + \sin\!\left(\dfrac{\pi}{6}\right) = \dfrac{1}{2} + \dfrac{1}{2} = 1$.

\textbf{4.} $\cos^2\!\left(\dfrac{\pi}{4}\right) + \sin^2\!\left(\dfrac{\pi}{3}\right) = \left(\dfrac{\sqrt{2}}{2}\right)^2 + \left(\dfrac{\sqrt{3}}{2}\right)^2 = \dfrac{1}{2} + \dfrac{3}{4} = \dfrac{5}{4}$.
