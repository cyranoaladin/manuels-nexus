\textbf{2.} $B = \sin\!\left(\dfrac{5\pi}{6}\right)\cos\!\left(\dfrac{\pi}{3}\right) + \cos\!\left(\dfrac{5\pi}{6}\right)\sin\!\left(\dfrac{\pi}{4}\right) = \dfrac{1}{2} \times \dfrac{1}{2} + \left(-\dfrac{\sqrt{3}}{2}\right) \times \dfrac{\sqrt{2}}{2} = \dfrac{1}{4} - \dfrac{\sqrt{6}}{4} = \dfrac{1 - \sqrt{6}}{4}$.

\textbf{3.} $C = \cos^2\!\left(\dfrac{\pi}{6}\right) - \sin^2\!\left(\dfrac{\pi}{6}\right) = \dfrac{3}{4} - \dfrac{1}{4} = \dfrac{1}{2}$.
