\begin{corrige}{1SPE-TRIGO-EX-017}

\textbf{1.} $\sin^2 x = 1 - \cos^2 x = 1 - \dfrac{49}{625} = \dfrac{576}{625}$. Comme $x \in \left]0;\dfrac{\pi}{2}\right[$, $\sin x > 0$, donc $\sin x = \dfrac{24}{25}$.

\textbf{2.} $\cos(\pi - x) = -\cos x = -\dfrac{7}{25}$ et $\sin(\pi - x) = \sin x = \dfrac{24}{25}$.

\textbf{3.} $\cos(\pi + x) = -\cos x = -\dfrac{7}{25}$ et $\sin(\pi + x) = -\sin x = -\dfrac{24}{25}$.

\textbf{4.} $\cos(-x) = \cos x = \dfrac{7}{25}$ et $\sin(-x) = -\sin x = -\dfrac{24}{25}$.

\end{corrige}
