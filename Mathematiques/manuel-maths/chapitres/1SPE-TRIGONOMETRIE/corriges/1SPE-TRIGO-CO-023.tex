\begin{corrige}{1SPE-TRIGO-EX-023}

\textbf{1.} $\cos\!\left(\dfrac{2\pi}{3}\right) = 2\cos^2\!\left(\dfrac{\pi}{3}\right) - 1 = 2\left(\dfrac{1}{2}\right)^2 - 1 = \dfrac{1}{2} - 1 = -\dfrac{1}{2}$.

\textbf{2.} $\sin\!\left(\dfrac{2\pi}{3}\right) = 2\sin\!\left(\dfrac{\pi}{3}\right)\cos\!\left(\dfrac{\pi}{3}\right) = 2 \times \dfrac{\sqrt{3}}{2} \times \dfrac{1}{2} = \dfrac{\sqrt{3}}{2}$.

\textbf{3.} $\sin\!\left(\dfrac{\pi}{3}\right) = \sin\!\left(2 \times \dfrac{\pi}{6}\right) = 2\sin\!\left(\dfrac{\pi}{6}\right)\cos\!\left(\dfrac{\pi}{6}\right) = 2 \times \dfrac{1}{2} \times \dfrac{\sqrt{3}}{2} = \dfrac{\sqrt{3}}{2}$. \checkmark

\end{corrige}
