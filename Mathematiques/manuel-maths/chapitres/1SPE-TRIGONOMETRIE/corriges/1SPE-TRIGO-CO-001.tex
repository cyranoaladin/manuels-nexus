\commentaireMarge{Multiplier par $\frac{\pi}{180}$.}

\textbf{1.} $120° = 120 \times \dfrac{\pi}{180} = \dfrac{120\pi}{180} = \dfrac{2\pi}{3}$.

\textbf{2.} $45° = 45 \times \dfrac{\pi}{180} = \dfrac{45\pi}{180} = \dfrac{\pi}{4}$.

\textbf{3.} $270° = 270 \times \dfrac{\pi}{180} = \dfrac{270\pi}{180} = \dfrac{3\pi}{2}$.

\textbf{4.} $330° = 330 \times \dfrac{\pi}{180} = \dfrac{330\pi}{180} = \dfrac{11\pi}{6}$.
