\begin{corrige}{1SPE-TRIGO-EX-027}

\textbf{1.} $\cos^2 x = \dfrac{1 + \cos(2x)}{2}$.

\textbf{2.} $\cos^4 x = \left(\dfrac{1+\cos(2x)}{2}\right)^2 = \dfrac{1 + 2\cos(2x) + \cos^2(2x)}{4}$.

Or $\cos^2(2x) = \dfrac{1+\cos(4x)}{2}$.

\textbf{3.} $\cos^4 x = \dfrac{1 + 2\cos(2x) + \frac{1+\cos(4x)}{2}}{4} = \dfrac{\frac{3}{2} + 2\cos(2x) + \frac{\cos(4x)}{2}}{4} = \dfrac{3}{8} + \dfrac{1}{2}\cos(2x) + \dfrac{1}{8}\cos(4x)$. \hfill $\square$

\end{corrige}
