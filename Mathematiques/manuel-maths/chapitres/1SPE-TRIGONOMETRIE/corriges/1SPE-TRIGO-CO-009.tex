\begin{corrige}{1SPE-TRIGO-EX-009}

\textbf{1.} L'aire du disque est $\pi R^2$ pour un angle de $2\pi$. Par proportionnalite :
$\mathcal{A} = \dfrac{\theta}{2\pi} \times \pi R^2 = \dfrac{1}{2}R^2\theta$.

\textbf{2.} $\mathcal{A} = \dfrac{1}{2} \times 5^2 \times \dfrac{\pi}{3} = \dfrac{25\pi}{6} \approx 13{,}09$~cm$^2$.

\textbf{3.} On veut $\dfrac{1}{2}R^2\theta = \dfrac{1}{4}\pi R^2$, soit $\theta = \dfrac{\pi}{2}$ rad $= 90°$.

\end{corrige}
