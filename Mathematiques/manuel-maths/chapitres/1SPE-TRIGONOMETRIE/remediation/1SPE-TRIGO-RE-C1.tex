\begin{exercice}{1SPE-TRIGO-RE-C1-EX1}{1}{8}
\begin{enumerate}
  \item Convertir en radians : $90°$, $135°$, $240°$, $300°$.
  \item Convertir en degres : $\dfrac{3\pi}{4}$, $\dfrac{5\pi}{3}$, $\dfrac{7\pi}{6}$.
  \item Determiner la mesure principale de $\dfrac{15\pi}{4}$ et de $-\dfrac{10\pi}{3}$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-TRIGO-RE-C1-EX1}
\textbf{1.} $90° = \dfrac{\pi}{2}$, $135° = \dfrac{3\pi}{4}$, $240° = \dfrac{4\pi}{3}$, $300° = \dfrac{5\pi}{3}$.

\textbf{2.} $\dfrac{3\pi}{4} = 135°$, $\dfrac{5\pi}{3} = 300°$, $\dfrac{7\pi}{6} = 210°$.

\textbf{3.} $\dfrac{15\pi}{4} - 2\pi = \dfrac{7\pi}{4}$, puis $\dfrac{7\pi}{4} - 2\pi = -\dfrac{\pi}{4}$. Mesure principale : $-\dfrac{\pi}{4}$.

$-\dfrac{10\pi}{3} + 4\pi = -\dfrac{10\pi}{3} + \dfrac{12\pi}{3} = \dfrac{2\pi}{3}$. Mesure principale : $\dfrac{2\pi}{3}$.
\end{corrige}
