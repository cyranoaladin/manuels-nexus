\subsection*{C1 — Cercle trigonometrique, radian, angles orientes}

% ---- Q1 ----
\item \textbf{[Q1]} Quelle est la mesure en radians de $60°$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{\pi}{6}$
    \item $\dfrac{\pi}{4}$
    \item $\dfrac{\pi}{3}$
    \item $\dfrac{\pi}{2}$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} Quelle est la mesure principale de $\dfrac{9\pi}{4}$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{\pi}{4}$
    \item $\dfrac{9\pi}{4}$
    \item $-\dfrac{7\pi}{4}$
    \item $\dfrac{5\pi}{4}$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} Combien vaut $180°$ en radians ?

  \begin{enumerate}[label=\Alph*.]
    \item $2\pi$
    \item $\pi$
    \item $\dfrac{\pi}{2}$
    \item $\dfrac{3\pi}{2}$
  \end{enumerate}

\end{enumerate}

\item \textbf{[Q4]} Quelle est la valeur de $\cos\!\left(\dfrac{\pi}{3}\right)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{\sqrt{3}}{2}$
    \item $\dfrac{\sqrt{2}}{2}$
    \item $\dfrac{1}{2}$
    \item $0$
  \end{enumerate}

\bigskip

% ---- Q5 ----
\item \textbf{[Q5]} Quelle est la valeur de $\sin\!\left(\dfrac{5\pi}{6}\right)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $-\dfrac{1}{2}$
    \item $\dfrac{1}{2}$
    \item $\dfrac{\sqrt{3}}{2}$
    \item $-\dfrac{\sqrt{3}}{2}$
  \end{enumerate}

\bigskip

% ---- Q6 ----
\item \textbf{[Q6]} Que vaut $\cos^2\!\left(\dfrac{\pi}{4}\right) + \sin^2\!\left(\dfrac{\pi}{4}\right)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{1}{2}$
    \item $1$
    \item $\sqrt{2}$
    \item $2$
  \end{enumerate}

\end{enumerate}
