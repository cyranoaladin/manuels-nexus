  \end{enumerate}

\subsection*{C3 — Formules d'addition et de duplication}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} Que vaut $\cos(a+b)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\cos a \cos b + \sin a \sin b$
    \item $\cos a \cos b - \sin a \sin b$
    \item $\sin a \cos b + \cos a \sin b$
    \item $\sin a \cos b - \cos a \sin b$
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} Que vaut $\sin(2a)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $2\sin a$
    \item $\sin^2 a$
    \item $2\sin a \cos a$
    \item $\cos^2 a - \sin^2 a$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} Quelle est la valeur de $\cos(2a)$ si $\cos a = \dfrac{1}{2}$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $1$
    \item $\dfrac{1}{2}$
    \item $-\dfrac{1}{2}$
    \item $0$
  \end{enumerate}

\end{enumerate}

  \end{enumerate}

\bigskip

% ---- Q11 ----
\item \textbf{[Q11]} Les solutions de $\sin x = 0$ dans $\mathbb{R}$ sont :

  \begin{enumerate}[label=\Alph*.]
    \item $x = 2k\pi$, $k \in \mathbb{Z}$
    \item $x = k\pi$, $k \in \mathbb{Z}$
    \item $x = \dfrac{\pi}{2} + k\pi$, $k \in \mathbb{Z}$
    \item $x = \dfrac{k\pi}{2}$, $k \in \mathbb{Z}$
  \end{enumerate}

\bigskip

% ---- Q12 ----
\item \textbf{[Q12]} Sur $[0\,;\,2\pi]$, l'ensemble des solutions de $\cos x \geqslant 0$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $\left[0\,;\,\dfrac{\pi}{2}\right]$
    \item $\left[0\,;\,\dfrac{\pi}{2}\right] \cup \left[\dfrac{3\pi}{2}\,;\,2\pi\right]$
    \item $\left[\dfrac{\pi}{2}\,;\,\dfrac{3\pi}{2}\right]$
    \item $[0\,;\,\pi]$
  \end{enumerate}

\end{enumerate}
