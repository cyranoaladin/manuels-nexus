\begin{corrige}{1SPE-VARALEA-EX-001}

\textbf{1.} $X$ prend les valeurs $1, 2, 3, 4, 5, 6$.

\medskip
\textbf{2.} Le dé est équilibré, donc chaque face a la probabilité $\frac{1}{6}$.
\[
\begin{array}{|c|c|c|c|c|c|c|}
\hline
x_i & 1 & 2 & 3 & 4 & 5 & 6 \\
\hline
P(X = x_i) & \frac{1}{6} & \frac{1}{6} & \frac{1}{6} & \frac{1}{6} & \frac{1}{6} & \frac{1}{6} \\
\hline
\end{array}
\]

\medskip
\textbf{3.} $\frac{1}{6} \times 6 = 1$. \checkmark

\medskip
\textbf{4.} $P(X \leq 3) = 3 \times \frac{1}{6} = \frac{1}{2}$.

$P(X \geq 5) = 2 \times \frac{1}{6} = \frac{1}{3}$.

\end{corrige}
