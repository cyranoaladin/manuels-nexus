\begin{corrige}{1SPE-VARALEA-EX-008}

\textbf{1.} $\{X \leq k\}$ signifie que les deux dés donnent au plus $k$. Chaque dé a $k$ faces favorables sur $6$, donc $P(X \leq k) = \frac{k}{6} \times \frac{k}{6} = \frac{k^2}{36}$.

\medskip
\textbf{2.} $P(X=k) = P(X \leq k) - P(X \leq k-1) = \frac{k^2}{36} - \frac{(k-1)^2}{36} = \frac{2k-1}{36}$.

\medskip
\textbf{3.}
\[
\begin{array}{|c|c|c|c|c|c|c|}
\hline
k & 1 & 2 & 3 & 4 & 5 & 6 \\
\hline
P(X=k) & \frac{1}{36} & \frac{3}{36} & \frac{5}{36} & \frac{7}{36} & \frac{9}{36} & \frac{11}{36} \\
\hline
\end{array}
\]
Vérification : $1+3+5+7+9+11=36$, donc $\sum = 1$. \checkmark

\end{corrige}
