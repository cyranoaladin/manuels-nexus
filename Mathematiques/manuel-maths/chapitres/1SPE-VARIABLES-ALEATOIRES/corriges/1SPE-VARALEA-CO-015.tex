\begin{corrige}{1SPE-VARALEA-EX-015}

\textbf{1.} $E(X) = 0 \times \frac{1}{2} + 1 \times \frac{1}{2} = \frac{1}{2}$.

$V(X) = E(X^2) - [E(X)]^2 = \frac{1}{2} - \frac{1}{4} = \frac{1}{4}$.

\medskip
\textbf{2.} $E(Y) = 10 \times \frac{1}{2} - 5 = 0$. Le jeu est équitable si la mise est $m = 5$~euros (car $Y = 10 - 5 = 5$ si Pile, $Y = 0 - 5 = -5$ si Face, et $E(Y) = 0$).

\end{corrige}
