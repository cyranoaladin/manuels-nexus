\subsection*{C1 — Loi de probabilité d'une variable aléatoire}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} On lance un dé équilibré. Soit $X$ le résultat. Quelle est $P(X = 3)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\frac{1}{3}$
    \item $\frac{1}{6}$
    \item $\frac{3}{6}$
    \item $\frac{1}{2}$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} La somme des probabilités d'un tableau de loi vaut toujours :

  \begin{enumerate}[label=\Alph*.]
    \item $0$
    \item $\frac{1}{2}$
    \item $1$
    \item Le nombre de valeurs prises par $X$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} On lance deux pièces. Soit $X$ le nombre de Pile. Quelle est $P(X = 1)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\frac{1}{4}$
    \item $\frac{1}{2}$
    \item $\frac{1}{3}$
    \item $\frac{3}{4}$
  \end{enumerate}

\end{enumerate}

\subsection*{C3 — Loi de Bernoulli et loi binomiale}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} $X \sim \mathcal{B}(p)$ est une loi de Bernoulli. Combien de valeurs $X$ peut-elle prendre ?

  \begin{enumerate}[label=\Alph*.]
    \item $1$
    \item $2$
    \item $p$
    \item Une infinité
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} $\binom{7}{3}$ vaut :

  \begin{enumerate}[label=\Alph*.]
    \item $21$
    \item $35$
    \item $42$
    \item $7$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} $X \sim \mathcal{B}(6, \frac{1}{3})$. Quelle est $P(X = 0)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\left(\frac{1}{3}\right)^6$
    \item $\left(\frac{2}{3}\right)^6$
    \item $0$
    \item $1$
  \end{enumerate}

\end{enumerate}

\item \textbf{[Q13]} Un joueur paie $3$~euros et reçoit $15$~euros s'il obtient un $6$ au dé. Quel est son gain algébrique s'il obtient un $6$ ?

    \item Défavorable au joueur
    \item Impossible à déterminer
  \end{enumerate}

\bigskip

  \begin{enumerate}[label=\Alph*.]
    \item $60$~euros
    \item $20$~euros
    \item $-40$~euros
    \item $40$~euros
  \end{enumerate}
