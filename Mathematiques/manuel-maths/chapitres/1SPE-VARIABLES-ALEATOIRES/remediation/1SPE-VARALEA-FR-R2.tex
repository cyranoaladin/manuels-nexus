\subsection*{Rappel — Probabilités conditionnelles}

$P(A|B) = \frac{P(A \cap B)}{P(B)}$ si $P(B) > 0$.

\textbf{Formule des probabilités totales} : si $B_1, \ldots, B_n$ forment une partition de $\Omega$ :
$P(A) = \sum_{i=1}^{n} P(A|B_i) P(B_i)$.

\textbf{Indépendance} : $A$ et $B$ sont indépendants si $P(A \cap B) = P(A) \times P(B)$.

\begin{exercice}{1SPE-VARALEA-FR-R2-EX1}{1}{5}
Une urne contient $4$ boules rouges et $6$ boules bleues. On tire deux boules successivement sans remise.
\begin{enumerate}
  \item Calculer la probabilité que la première boule soit rouge.
  \item Sachant que la première est rouge, calculer la probabilité que la deuxième soit rouge.
\end{enumerate}
\end{exercice}
