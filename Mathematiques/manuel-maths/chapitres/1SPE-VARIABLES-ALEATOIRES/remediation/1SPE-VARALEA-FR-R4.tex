\subsection*{Rappel — Suites et sommes}

\textbf{Somme des entiers} : $1 + 2 + \cdots + n = \frac{n(n+1)}{2}$.

\textbf{Notation sigma} : $\sum_{k=1}^{n} a_k = a_1 + a_2 + \cdots + a_n$.

\textbf{Somme géométrique} : $\sum_{k=0}^{n} q^k = \frac{1-q^{n+1}}{1-q}$ si $q \neq 1$.

\begin{exercice}{1SPE-VARALEA-FR-R4-EX1}{1}{5}
\begin{enumerate}
  \item Calculer $\sum_{k=1}^{5} k$.
  \item Calculer $\sum_{k=0}^{4} \left(\frac{1}{2}\right)^k$.
\end{enumerate}
\end{exercice}
