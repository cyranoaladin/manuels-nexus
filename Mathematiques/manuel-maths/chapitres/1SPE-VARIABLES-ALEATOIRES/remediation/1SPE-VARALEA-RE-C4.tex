\begin{exercice}{1SPE-VARALEA-RE-C4-EX1}{1}{10}
Soit $X \sim \mathcal{B}(12, \frac{1}{4})$.
\begin{enumerate}
  \item Calculer $E(X)$ à l'aide de la formule $np$.
  \item Calculer $V(X)$ et $\sigma(X)$.
  \item Retrouver $n$ et $p$ à partir de $E(X)$ et $V(X)$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-VARALEA-RE-C4-EX1}
\textbf{1.} $E(X) = 12 \times \frac{1}{4} = 3$.
\textbf{2.} $V(X) = 12 \times \frac{1}{4} \times \frac{3}{4} = \frac{9}{4} = 2{,}25$. $\sigma(X) = \frac{3}{2} = 1{,}5$.
\textbf{3.} $1-p = \frac{9/4}{3} = \frac{3}{4}$, donc $p = \frac{1}{4}$. $n = \frac{3}{1/4} = 12$. \checkmark
\end{corrige}
