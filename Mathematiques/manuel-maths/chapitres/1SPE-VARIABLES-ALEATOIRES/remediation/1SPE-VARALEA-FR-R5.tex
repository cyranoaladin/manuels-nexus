\subsection*{Rappel — Puissances}

Règles :
\begin{itemize}
  \item $a^m \times a^n = a^{m+n}$
  \item $(a^m)^n = a^{mn}$
  \item $\left(\frac{a}{b}\right)^n = \frac{a^n}{b^n}$
  \item $a^0 = 1$ (pour $a \neq 0$)
\end{itemize}

\begin{exercice}{1SPE-VARALEA-FR-R5-EX1}{1}{5}
Calculer :
\begin{enumerate}
  \item $\left(\frac{1}{2}\right)^4$
  \item $\left(\frac{3}{4}\right)^2$
  \item $2^3 \times 2^4$
  \item $\left(\frac{2}{3}\right)^0$
\end{enumerate}
\end{exercice}
