\begin{exercice}{1SPE-VARALEA-EX-028}{3}{20}
\begin{enumerate}
  \item Démontrer la formule de Pascal : $\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}$ pour $1 \leq k \leq n-1$.
  \item Démontrer que $\sum_{k=0}^{n} \binom{n}{k} = 2^n$.
  \item En déduire que la somme des probabilités d'une loi $\mathcal{B}(n, \frac{1}{2})$ vaut $1$.
\end{enumerate}
\end{exercice}
