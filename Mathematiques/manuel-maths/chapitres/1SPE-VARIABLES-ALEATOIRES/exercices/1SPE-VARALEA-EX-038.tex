\begin{exercice}{1SPE-VARALEA-EX-038}{3}{20}
\begin{enumerate}
  \item Démontrer que pour $X \sim \mathcal{B}(n,p)$ : $E[X(X-1)] = n(n-1)p^2$.
  \item En déduire $E(X^2) = n(n-1)p^2 + np$.
  \item Montrer que $V(X) = np(1-p)$.
  \item Vérifier pour $X \sim \mathcal{B}(5, \frac{1}{3})$.
\end{enumerate}
\end{exercice}
