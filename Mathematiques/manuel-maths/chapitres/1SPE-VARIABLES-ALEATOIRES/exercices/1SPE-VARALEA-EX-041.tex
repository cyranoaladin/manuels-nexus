\begin{exercice}{1SPE-VARALEA-EX-041}{1}{12}

\begin{enumerate}
  \item Dresser le tableau de loi de $G$.
  \item Calculer $E(G)$. Le jeu est-il favorable au joueur ?
\end{enumerate}
