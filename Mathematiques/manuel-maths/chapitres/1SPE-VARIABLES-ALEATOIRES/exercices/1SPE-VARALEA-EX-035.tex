\begin{exercice}{1SPE-VARALEA-EX-035}{3}{20}
\begin{enumerate}
  \item Démontrer que pour $1 \leq k \leq n$ : $k \binom{n}{k} = n \binom{n-1}{k-1}$.
  \item En utilisant cette identité, démontrer que si $X \sim \mathcal{B}(n,p)$, alors $E(X) = np$.
  \item Application : vérifier pour $X \sim \mathcal{B}(4, \frac{1}{3})$.
\end{enumerate}
\end{exercice}
