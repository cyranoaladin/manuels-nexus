\begin{exercice}{1SPE-VARALEA-EX-039}{1}{10}
Soit $X \sim \mathcal{B}(\frac{2}{5})$ (loi de Bernoulli).

\begin{enumerate}
  \item Vérifier que $E(X) = p$ en utilisant la formule $np$ avec $n=1$.
  \item Vérifier que $V(X) = p(1-p)$.
\end{enumerate}
\end{exercice}
