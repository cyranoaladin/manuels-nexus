\begin{exercice}{1SPE-VARALEA-EX-008}{3}{15}
On lance deux dés équilibrés. Soit $X$ le maximum des deux résultats.

\begin{enumerate}
  \item Montrer que $P(X \leq k) = \left(\dfrac{k}{6}\right)^2$ pour $k \in \{1,\ldots,6\}$.
  \item En déduire $P(X = k)$ pour chaque $k$.
  \item Dresser le tableau de loi de $X$.
\end{enumerate}
\end{exercice}
