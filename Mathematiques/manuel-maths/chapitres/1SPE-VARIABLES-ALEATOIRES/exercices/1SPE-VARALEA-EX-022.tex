\begin{exercice}{1SPE-VARALEA-EX-022}{1}{12}
On lance $5$ fois une pièce équilibrée. Soit $X$ le nombre de Pile.

\begin{enumerate}
  \item Justifier que $X \sim \mathcal{B}(5, \frac{1}{2})$.
  \item Calculer $P(X = 3)$.
  \item Calculer $P(X = 0)$ et $P(X = 5)$.
\end{enumerate}
\end{exercice}
