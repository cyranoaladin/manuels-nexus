\begin{exercice}{1SPE-VARALEA-EX-020}{3}{20}
Soit $X$ une variable aléatoire telle que $E(X) = 3$ et $V(X) = 2$. On pose $Y = 2X + 1$.

\begin{enumerate}
  \item Calculer $E(Y)$.
  \item Calculer $V(Y)$ et $\sigma(Y)$.
  \item Si $Z = -X + 5$, calculer $E(Z)$ et $V(Z)$.
\end{enumerate}
\end{exercice}
