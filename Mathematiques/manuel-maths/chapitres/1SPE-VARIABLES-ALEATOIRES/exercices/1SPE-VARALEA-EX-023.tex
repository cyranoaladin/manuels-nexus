\begin{exercice}{1SPE-VARALEA-EX-023}{1}{12}
On lance $4$ fois un dé. Soit $X$ le nombre de $6$ obtenus. On a $X \sim \mathcal{B}(4, \frac{1}{6})$.

\begin{enumerate}
  \item Calculer $P(X = 0)$.
  \item Calculer $P(X = 2)$.
  \item En déduire $P(X \geq 1)$.
\end{enumerate}
\end{exercice}
