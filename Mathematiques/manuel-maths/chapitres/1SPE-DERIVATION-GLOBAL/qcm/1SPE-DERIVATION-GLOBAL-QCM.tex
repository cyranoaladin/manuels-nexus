\subsection*{C1 — Dérivées des fonctions de référence}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} La dérivée de $f(x) = x^4$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $f'(x) = x^3$
    \item $f'(x) = 4x^3$
    \item $f'(x) = 4x^4$
    \item $f'(x) = 3x^3$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} La dérivée de $f(x) = \sqrt{x}$ pour $x > 0$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $f'(x) = 2\sqrt{x}$
    \item $f'(x) = \dfrac{1}{\sqrt{x}}$
    \item $f'(x) = \dfrac{1}{2\sqrt{x}}$
    \item $f'(x) = \dfrac{\sqrt{x}}{2}$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} La dérivée de $f(x) = \dfrac{1}{x}$ pour $x \neq 0$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $f'(x) = \dfrac{1}{x^2}$
    \item $f'(x) = -x^2$
    \item $f'(x) = -\dfrac{1}{x^2}$
    \item $f'(x) = \ln x$
  \end{enumerate}

\end{enumerate}

\subsection*{C2 — Règles de dérivation (somme, produit, quotient)}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q4 ----
\item \textbf{[Q4]} Soit $f(x) = 3x^2 - 5x + 2$. Alors $f'(x)$ vaut :

  \begin{enumerate}[label=\Alph*.]
    \item $6x - 5$
    \item $3x - 5$
    \item $6x^2 - 5$
    \item $6x + 2$
  \end{enumerate}

\bigskip

% ---- Q5 ----
\item \textbf{[Q5]} Soit $f(x) = x^2 \cdot \sqrt{x}$ pour $x > 0$. La règle à utiliser pour calculer $f'$ est :

  \begin{enumerate}[label=\Alph*.]
    \item La règle du quotient.
    \item La règle de la somme.
    \item La règle du produit : $(uv)' = u'v + uv'$.
    \item Aucune règle : $f$ n'est pas dérivable.
  \end{enumerate}

\bigskip

% ---- Q6 ----
\item \textbf{[Q6]} Soit $f(x) = \dfrac{x+1}{x-1}$ pour $x \neq 1$. La règle du quotient donne $f'(x) = \dfrac{u'v - uv'}{v^2}$ avec $u = x+1$ et $v = x-1$. Alors $f'(x)$ vaut :

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{0}{(x-1)^2} = 0$
    \item $\dfrac{(x-1)-(x+1)}{(x-1)^2} = \dfrac{-2}{(x-1)^2}$
    \item $\dfrac{(x+1)+(x-1)}{(x-1)^2} = \dfrac{2x}{(x-1)^2}$
    \item $\dfrac{1}{(x-1)^2}$
  \end{enumerate}

\end{enumerate}

\subsection*{C3 — Signe de la dérivée et tableau de variations}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} Si $f'(x) > 0$ sur un intervalle $I$, alors $f$ est :

  \begin{enumerate}[label=\Alph*.]
    \item Décroissante sur $I$.
    \item Constante sur $I$.
    \item Croissante sur $I$.
    \item Nulle sur $I$.
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} Soit $f(x) = x^2 - 4x + 1$. On a $f'(x) = 2x - 4$. Sur quel intervalle $f$ est-elle décroissante ?

  \begin{enumerate}[label=\Alph*.]
    \item $[2\,;\,+\infty[$
    \item $]-\infty\,;\,4]$
    \item $]-\infty\,;\,2]$
    \item $[4\,;\,+\infty[$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} La dérivée $f'$ s'annule en $x = 3$ et change de signe de $+$ à $-$ en ce point. Cela signifie que $f$ :

  \begin{enumerate}[label=\Alph*.]
    \item Admet un minimum en $x = 3$.
    \item Est constante au voisinage de $x = 3$.
    \item Admet un maximum en $x = 3$.
    \item Est nulle en $x = 3$.
  \end{enumerate}

\end{enumerate}

\subsection*{C4 — Extremums d'une fonction}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q10 ----
\item \textbf{[Q10]} Soit $f(x) = -x^2 + 6x - 5$. On a $f'(x) = -2x + 6$. La valeur $f'(x) = 0$ donne $x = $ :

  \begin{enumerate}[label=\Alph*.]
    \item $x = -3$
    \item $x = 3$
    \item $x = 6$
    \item $x = 0$
  \end{enumerate}

\bigskip

% ---- Q11 ----
\item \textbf{[Q11]} Dans la situation de la question précédente, $f$ admet en $x = 3$ :

  \begin{enumerate}[label=\Alph*.]
    \item Un minimum global, car $f''(3) > 0$.
    \item Un maximum global, car le coefficient dominant de $f$ est négatif et $f'(3) = 0$.
    \item Un point d'inflexion.
    \item Aucun extremum.
  \end{enumerate}

\bigskip

% ---- Q12 ----
\item \textbf{[Q12]} Pour trouver les extremums d'une fonction dérivable sur $\mathbb{R}$, on doit :

  \begin{enumerate}[label=\Alph*.]
    \item Résoudre $f(x) = 0$.
    \item Résoudre $f'(x) = 0$ puis étudier le signe de $f'$ autour des solutions.
    \item Calculer $f(0)$.
    \item Vérifier que $f$ est croissante.
  \end{enumerate}

\end{enumerate}

\subsection*{C5 — Problèmes d'optimisation}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q13 ----
\item \textbf{[Q13]} Pour résoudre un problème d'optimisation, la première étape est de :

  \begin{enumerate}[label=\Alph*.]
    \item Calculer $f'(x)$ directement.
    \item Définir la quantité à optimiser et l'exprimer en fonction d'une seule variable.
    \item Résoudre une équation du second degré.
    \item Tracer la courbe de $f$.
  \end{enumerate}

\bigskip

% ---- Q14 ----
\item \textbf{[Q14]} Une entreprise fabrique $x$ articles. Son bénéfice est $B(x) = -2x^2 + 80x - 300$ (en euros). La valeur de $x$ qui maximise le bénéfice vérifie $B'(x) = 0$, soit $-4x + 80 = 0$. On trouve $x = $ :

  \begin{enumerate}[label=\Alph*.]
    \item $x = 80$
    \item $x = 4$
    \item $x = 20$
    \item $x = 40$
  \end{enumerate}

\bigskip

% ---- Q15 ----
\item \textbf{[Q15]} Dans un problème où $x$ doit vérifier $0 \leqslant x \leqslant 10$, on trouve que $f'(x) = 0$ pour $x = 5$. Parmi les valeurs à tester pour chercher le maximum de $f$, lesquelles doit-on obligatoirement inclure ?

  \begin{enumerate}[label=\Alph*.]
    \item Uniquement $x = 5$.
    \item $x = 5$ uniquement si $f'$ change de signe.
    \item $x = 0$, $x = 5$ et $x = 10$.
    \item $x = 0$ et $x = 10$ uniquement.
  \end{enumerate}

\end{enumerate}
