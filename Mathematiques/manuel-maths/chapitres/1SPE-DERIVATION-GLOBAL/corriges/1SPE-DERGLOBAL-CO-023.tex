\begin{corrige}{1SPE-DERGLOBAL-EX-023}
\textbf{1.} On dérive $h(x) = x^3 - 3x$ :
\[
h'(x) = 3x^2 - 3 = 3(x^2 - 1) = 3(x-1)(x+1)
\]

\textbf{2.} On résout $h'(x) = 0$ :
\[
3(x-1)(x+1) = 0 \iff x = -1 \text{ ou } x = 1
\]

Signe de $h'(x) = 3(x+1)(x-1)$ :
\begin{itemize}
\item Si $x < -1$ : $(x+1) < 0$ et $(x-1) < 0$, donc $h'(x) > 0$.
\item Si $-1 < x < 1$ : $(x+1) > 0$ et $(x-1) < 0$, donc $h'(x) < 0$.
\item Si $x > 1$ : $(x+1) > 0$ et $(x-1) > 0$, donc $h'(x) > 0$.
\end{itemize}

\textbf{3.} Tableau de variations de $h$ :

\[
\begin{array}{|c|ccccccc|}
\hline
x & -\infty & & -1 & & 1 & & +\infty \\
\hline
h'(x) & & + & 0 & - & 0 & + & \\
\hline
h & -\infty & \nearrow & 2 & \searrow & -2 & \nearrow & +\infty \\
\hline
\end{array}
\]

avec $h(-1) = -1 + 3 = 2$ et $h(1) = 1 - 3 = -2$.
\end{corrige}
