\begin{corrige}{1SPE-DERGLOBAL-EX-041}
\textbf{1.} On dérive terme à terme :
\[
f'(x) = -2x + 6.
\]

\textbf{2.} On résout $f'(x) = 0$ :
\[
-2x + 6 = 0 \iff x = 3.
\]
$f'(x) > 0$ pour $x < 3$ et $f'(x) < 0$ pour $x > 3$.

Tableau de variations de $f$ sur $\mathbb{R}$ :
\[
\begin{array}{|c|ccccc|}
\hline
x     & -\infty & & 3 & & +\infty \\
\hline
f'(x) &         & + & 0 & - & \\
\hline
f(x)  & -\infty & \nearrow & 4 & \searrow & -\infty \\
\hline
\end{array}
\]

\textbf{3.} $f'$ change de signe de $+$ à $-$ en $x = 3$, donc $f$ admet un \textbf{maximum} en $x = 3$.
Ce maximum vaut $f(3) = -9 + 18 - 5 = \boxed{4}$.
\end{corrige}
