\begin{corrige}{1SPE-DERGLOBAL-EX-014}
On applique la règle du quotient $\left(\dfrac{u}{v}\right)' = \dfrac{u'v - uv'}{v^2}$.

\textbf{1.} On pose $u = x$, $v = x^2+4$, donc $u' = 1$, $v' = 2x$.
\[
f'(x) = \frac{1 \cdot (x^2+4) - x \cdot 2x}{(x^2+4)^2} = \frac{x^2+4-2x^2}{(x^2+4)^2} = \boxed{\frac{4-x^2}{(x^2+4)^2}}.
\]

\textbf{2.} On pose $u = x^2-1$, $v = 2x+1$, donc $u' = 2x$, $v' = 2$.
\[
g'(x) = \frac{2x(2x+1) - (x^2-1) \cdot 2}{(2x+1)^2} = \frac{4x^2+2x - 2x^2+2}{(2x+1)^2} = \boxed{\frac{2x^2+2x+2}{(2x+1)^2}}.
\]
\end{corrige}
