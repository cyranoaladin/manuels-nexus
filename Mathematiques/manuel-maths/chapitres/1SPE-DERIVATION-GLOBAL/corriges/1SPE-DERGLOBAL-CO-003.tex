\begin{corrige}{1SPE-DERGLOBAL-EX-003}
On applique la règle $(af)' = af'$ pour une constante $a$.

\textbf{1.} $f'(x) = 3 \times 2x = 6x$.

\textbf{2.} $g'(x) = 5 \times \dfrac{1}{2\sqrt{x}} = \dfrac{5}{2\sqrt{x}}$.

\textbf{3.} $h'(x) = -2 \times \left(-\dfrac{1}{x^2}\right) = \dfrac{2}{x^2}$.
\end{corrige}
