\begin{corrige}{1SPE-DERGLOBAL-EX-013}

\textbf{1.} On pose $u = x^2-2x+3$ et $v = x+1$, donc $u' = 2x-2$ et $v' = 1$.
\[
f'(x) = (2x-2)(x+1) + (x^2-2x+3)(1) = 2x^2+2x-2x-2 + x^2-2x+3 = \boxed{3x^2-2x+1}.
\]

\textit{Vérification :} $f(x) = x^3 - x^2 + x + 3$, donc $f'(x) = 3x^2 - 2x + 1$. \checkmark

\textbf{2.} On pose d'abord $p(x) = x(x-1) = x^2-x$, donc $p'(x) = 2x-1$. Puis avec $q = x+1$, $q' = 1$ :
\[
g'(x) = p'(x)q(x) + p(x)q'(x) = (2x-1)(x+1) + (x^2-x)(1) = 2x^2+x-x-1 + x^2-x = 3x^2-x-1.
\]

\textit{Vérification plus rapide :} $g(x) = x(x-1)(x+1) = x(x^2-1) = x^3-x$, donc $g'(x) = \boxed{3x^2-1}$.
\end{corrige}
