\begin{corrige}{1SPE-DERGLOBAL-EX-031}
\textbf{1.} On calcule la dérivée de $f(x) = x^2 - 6x + 8$ :
\[
f'(x) = 2x - 6
\]
On résout $f'(x) = 0$ :
\[
2x - 6 = 0 \iff x = 3
\]

\textbf{2.} On vérifie le changement de signe de $f'(x) = 2(x - 3)$ :
\begin{itemize}
\item Pour $x < 3$ : $x - 3 < 0$, donc $f'(x) < 0$.
\item Pour $x > 3$ : $x - 3 > 0$, donc $f'(x) > 0$.
\end{itemize}
La dérivée change de signe de $-$ à $+$ en $x = 3$.

\textbf{3.} Puisque $f'$ passe de négative à positive en $x = 3$, $f$ admet un \textbf{minimum} en $x = 3$.

La valeur minimale est :
\[
f(3) = 9 - 18 + 8 = \boxed{-1}
\]
\end{corrige}
