\begin{corrige}{1SPE-DERGLOBAL-EX-020}

\textbf{Question 1.} Le discriminant de $x^2-x+1$ est $\Delta = 1-4 = -3 < 0$. Avec un coefficient dominant positif, on a donc $x^2-x+1 > 0$ pour tout $x \in \mathbb{R}$. La fonction $f$ est bien définie sur $\mathbb{R}$.

\textbf{Question 2.} On pose $u = x+1$, $v = x^2-x+1$, $u' = 1$, $v' = 2x-1$.
\[
f'(x) = \frac{(x^2-x+1) - (x+1)(2x-1)}{(x^2-x+1)^2}.
\]
On développe le numérateur :
\[
(x^2-x+1) - (2x^2-x+2x-1) = x^2-x+1 - 2x^2-x+1 = -x^2-2x+2.
\]
Donc $\displaystyle f'(x) = \frac{-x^2-2x+2}{(x^2-x+1)^2}$.

\textbf{Question 3.} $f'(x) = 0 \iff -x^2-2x+2 = 0 \iff x^2+2x-2=0$.

Discriminant : $\Delta = 4+8 = 12$, donc $x = \dfrac{-2 \pm 2\sqrt{3}}{2} = -1 \pm \sqrt{3}$.

On note $x_1 = -1-\sqrt{3} \approx -2{,}73$ et $x_2 = -1+\sqrt{3} \approx 0{,}73$.

\textbf{Question 4.} Le maximum de $f$ est atteint en $x_2 = -1+\sqrt{3}$.
\[
f(x_2) = \frac{(-1+\sqrt{3})+1}{(-1+\sqrt{3})^2-(-1+\sqrt{3})+1} = \frac{\sqrt{3}}{1-2\sqrt{3}+3+1-\sqrt{3}+1} = \frac{\sqrt{3}}{6-3\sqrt{3}} = \frac{\sqrt{3}}{3(2-\sqrt{3})}.
\]
En rationalisant : $\dfrac{\sqrt{3}(2+\sqrt{3})}{3(4-3)} = \dfrac{2\sqrt{3}+3}{3} = 1 + \dfrac{2\sqrt{3}}{3} \approx 2{,}155$.

\textit{Vérification numérique :} $f(0{,}73) \approx \dfrac{1{,}73}{0{,}53-0{,}73+1} \approx \dfrac{1{,}73}{0{,}80} \approx 2{,}16$. \checkmark
