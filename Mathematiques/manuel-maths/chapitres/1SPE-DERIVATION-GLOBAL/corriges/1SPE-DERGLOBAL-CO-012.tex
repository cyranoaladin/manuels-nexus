\begin{corrige}{1SPE-DERGLOBAL-EX-012}
On applique la règle du quotient $\left(\dfrac{u}{v}\right)' = \dfrac{u'v - uv'}{v^2}$.

\textbf{1.} On pose $u = x^2+1$, $v = x-2$, donc $u' = 2x$, $v' = 1$.
\[
f'(x) = \frac{2x(x-2) - (x^2+1) \cdot 1}{(x-2)^2} = \frac{2x^2 - 4x - x^2 - 1}{(x-2)^2} = \boxed{\frac{x^2 - 4x - 1}{(x-2)^2}}.
\]

\textbf{2.} On pose $u = 3x-1$, $v = x^2+1$, donc $u' = 3$, $v' = 2x$.
\[
g'(x) = \frac{3(x^2+1) - (3x-1)(2x)}{(x^2+1)^2} = \frac{3x^2+3 - 6x^2+2x}{(x^2+1)^2} = \boxed{\frac{-3x^2+2x+3}{(x^2+1)^2}}.
\]
\end{corrige}
