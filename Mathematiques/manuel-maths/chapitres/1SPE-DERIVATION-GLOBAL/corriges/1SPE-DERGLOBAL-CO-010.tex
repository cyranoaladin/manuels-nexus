\begin{corrige}{1SPE-DERGLOBAL-EX-010}

\textbf{Question 1.} L'élève a commis une erreur de signe sur le terme $-\dfrac{1}{x}$. La dérivée de $-\dfrac{1}{x}$ est $-\left(-\dfrac{1}{x^2}\right) = +\dfrac{1}{x^2}$, et non $-\dfrac{1}{x^2}$.

Calcul correct :
\[
f'(x) = (\sqrt{x})' + \left(-\frac{1}{x}\right)' = \frac{1}{2\sqrt{x}} + \frac{1}{x^2}.
\]

\textbf{Question 2.}
\[
f'(9) = \frac{1}{2\sqrt{9}} + \frac{1}{9^2} = \frac{1}{6} + \frac{1}{81} = \frac{27}{162} + \frac{2}{162} = \boxed{\frac{29}{162}}.
\]

\textbf{Question 3.} Sur $]0\,;\,+\infty[$, on a $\dfrac{1}{2\sqrt{x}} > 0$ et $\dfrac{1}{x^2} > 0$, donc leur somme $f'(x) > 0$. Ainsi $f$ est strictement croissante sur $]0\,;\,+\infty[$.

\textbf{Question 4.} Puisque $f$ est strictement croissante sur $]0\,;\,+\infty[$ et $4 < 9$, on conclut directement $f(4) < f(9)$.

\textit{Vérification numérique :} $f(4) = 2 - \tfrac{1}{4} = \tfrac{7}{4}$ et $f(9) = 3 - \tfrac{1}{9} = \tfrac{26}{9} \approx 2{,}89$. On a bien $\tfrac{7}{4} = 1{,}75 < 2{,}89$. \checkmark
\end{corrige}
