\begin{exercice}{1SPE-DERGLOBAL-EX-047}{3}{20}
\textbf{Synthèse C1 + C2 + C3} — Étude d'une fonction sur $\left]0\,;\,+\infty\right[$

Soit $f(x) = \dfrac{x^2 - 3}{\sqrt{x}}$ définie sur $\left]0\,;\,+\infty\right[$.

\begin{enumerate}
  \item Réécrire $f(x)$ sous la forme $f(x) = x^{3/2} - 3x^{-1/2}$ et vérifier
        que ces deux expressions coïncident.
  \item En utilisant les dérivées des fonctions de référence et la règle de la somme,
        calculer $f'(x)$. Simplifier le résultat sous la forme
        $f'(x) = \dfrac{3(x^2+1)}{2x^{3/2}}$.
  \item Étudier le signe de $f'(x)$ pour tout $x > 0$ et conclure sur le sens de
        variation de $f$.
  \item Dresser le tableau de variations complet de $f$ sur $\left]0\,;\,+\infty\right[$,
        en précisant les limites aux bornes.
  \item Calculer $f(1)$ et $f(4)$. Interpréter ces valeurs dans le tableau de variations.
\end{enumerate}
\end{exercice}
