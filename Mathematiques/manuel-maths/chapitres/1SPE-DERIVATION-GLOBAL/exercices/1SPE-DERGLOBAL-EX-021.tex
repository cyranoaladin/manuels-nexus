\begin{exercice}{1SPE-DERGLOBAL-EX-021}{1}{5}

\begin{enumerate}
\item Calculer $f'(x)$.
\item Résoudre $f'(x) = 0$.
\item En déduire le signe de $f'(x)$ sur $\mathbb{R}$ et dresser le tableau de variations de $f$.
\end{enumerate}
