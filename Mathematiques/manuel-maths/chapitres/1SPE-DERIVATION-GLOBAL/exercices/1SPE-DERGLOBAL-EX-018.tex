\begin{exercice}{1SPE-DERGLOBAL-EX-018}{2}{12}
Soit $f(x) = \dfrac{2x+3}{x^2-1}$ définie sur $\mathbb{R} \setminus \{-1\,;\,1\}$.
\begin{enumerate}
  \item Calculer $f'(x)$ et le simplifier.
  \item Résoudre $f'(x) = 0$.
  \item Étudier le signe du numérateur de $f'(x)$ sur $]-\infty\,;\,-1[$.
\end{enumerate}
\end{exercice}
