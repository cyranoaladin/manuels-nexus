\begin{exercice}{1SPE-DERGLOBAL-EX-008}{2}{12}
Soit $f(x) = x^3 - \dfrac{3}{x}$ définie sur $]0\,;\,+\infty[$.
\begin{enumerate}
  \item Calculer $f'(x)$.
  \item Calculer $f'(2)$.
  \item Montrer que $f'(x) > 0$ pour tout $x \in ]0\,;\,+\infty[$, et conclure sur le sens de variation de $f$.
\end{enumerate}
\end{exercice}
