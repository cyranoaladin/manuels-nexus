\subsection*{Rappel ã Sens de variation}

\textbf{Fonction croissante} sur $I$ : pour tous $a < b$ dans $I$, on a $f(a) \leqslant f(b)$. La courbe ô¨ monte ô£ de gauche û  droite.

\textbf{Fonction dûˋcroissante} sur $I$ : pour tous $a < b$ dans $I$, on a $f(a) \geqslant f(b)$. La courbe ô¨ descend ô£ de gauche û  droite.

\textbf{Tableau de variations} : tableau rûˋcapitulatif du sens de variation, avec les valeurs aux bornes ou aux extremums.

\begin{exercice}{1SPE-DERGLOBAL-FR-R5-EX1}{1}{5}
La courbe d'une fonction $f$ passe par les points $A(0\,;\,3)$, $B(2\,;\,7)$, $C(5\,;\,1)$.
\begin{enumerate}
  \item Entre $x = 0$ et $x = 2$, $f$ est-elle croissante ou dûˋcroissante ? Justifier.
  \item Entre $x = 2$ et $x = 5$, $f$ est-elle croissante ou dûˋcroissante ? Justifier.
  \item Dresser un tableau de variations partiel de $f$ sur $[0\,;\,5]$.
\end{enumerate}
\end{exercice}

% assert 3 < 7

% assert 7 > 1

\begin{corrige}{1SPE-DERGLOBAL-FR-R5-EX1}

\textbf{Question 1.}
$f(0) = 3 < f(2) = 7$ : $f$ est croissante sur $[0\,;\,2]$.

\textbf{Question 2.}
$f(2) = 7 > f(5) = 1$ : $f$ est dûˋcroissante sur $[2\,;\,5]$.

\textbf{Question 3.}

\commentaireMarge{Le maximum de $f$ sur cet intervalle est $7$, atteint en $x=2$.}

\end{tabular}

\end{corrige}

\begin{exercice}{1SPE-DERGLOBAL-FR-R5-EX2}{1}{6}
On donne le tableau de variations de $g$ :

\begin{center}
\begin{tabular}{|c|ccccccc|}
\hline
$x$ & $-3$ & & $-1$ & & $2$ & & $4$ \\
\hline
$g(x)$ & $1$ & $\searrow$ & $-4$ & $\nearrow$ & $6$ & $\searrow$ & $0$ \\
\hline
\end{tabular}
\end{center}

\begin{enumerate}
  \item Sur quel(s) intervalle(s) $g$ est-elle croissante ?
  \item Donner le maximum de $g$ sur $[-3\,;\,4]$.
  \item Donner le minimum de $g$ sur $[-3\,;\,4]$.
  \item Comparer $g(-2)$ et $g(0)$ (sans calculer leur valeur exacte).
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% # Vûˋrifications logiques sur le tableau de variations
% # max = 6 en x=2, min = -4 en x=-1
% assert 6 > 1 and 6 > 0  # max global
% assert -4 < 1 and -4 < 0  # min global
% END-VERIFY

\begin{corrige}{1SPE-DERGLOBAL-FR-R5-EX2}

\textbf{Question 1.}
$g$ est croissante sur $[-1\,;\,2]$.

\textbf{Question 2.}
Le maximum de $g$ sur $[-3\,;\,4]$ est $6$, atteint en $x = 2$.

\textbf{Question 3.}
Le minimum de $g$ sur $[-3\,;\,4]$ est $-4$, atteint en $x = -1$.

\textbf{Question 4.}
$-2 \in [-3\,;\,-1]$ et $0 \in [-1\,;\,2]$. Sur $[-3\,;\,-1]$, $g$ est dûˋcroissante donc $g(-2) > g(-1) = -4$, et sur $[-1\,;\,2]$, $g$ est croissante donc $g(0) > g(-1) = -4$. En comparant : $-2 < 0$, et sur $[-1\,;\,2]$, $g$ est croissante, donc $g(-1) \leqslant g(-2)$ n'est pas directement applicable. Cependant, on sait uniquement que $g(-2)$ est entre $-4$ et $1$, et $g(0)$ est entre $-4$ et $6$. \textit{On ne peut pas conclure sans plus d'information.}

\end{corrige}

\begin{exercice}{1SPE-DERGLOBAL-FR-R5-EX3}{1}{5}
Voici une liste de fonctions et leurs intervalles de dûˋfinition. Pour chacune, indiquer si elle est croissante, dûˋcroissante, ou ni l'un ni l'autre, sur l'intervalle donnûˋ.

  \item $g(x) = -x^2$ sur $[0\,;\,+\infty[$.
  \item $h(x) = \dfrac{1}{x}$ sur $]0\,;\,+\infty[$.

\begin{corrige}{1SPE-DERGLOBAL-FR-R5-EX3}

\textbf{Question 1.}
$f(x) = 3x + 1$ est une fonction affine de coefficient directeur $3 > 0$ : elle est \textbf{croissante} sur $\mathbb{R}$.

\commentaireMarge{$-x^2$ est concave, donc dûˋcroissante pour $x>0$.}

\textbf{Question 2.}
Pour $0 \leqslant a < b$ : $a^2 < b^2$ donc $-a^2 > -b^2$ : $g(a) > g(b)$. $g$ est \textbf{dûˋcroissante} sur $[0\,;\,+\infty[$.

\textbf{Question 3.}
Pour $0 < a < b$ : $\frac{1}{a} > \frac{1}{b}$ donc $h(a) > h(b)$. $h$ est \textbf{dûˋcroissante} sur $]0\,;\,+\infty[$.

\textbf{Question 4.}
$k(x) = (x-1)^2 \geqslant 0$. Pour $x < 1$, $k$ est dûˋcroissante ; pour $x > 1$, $k$ est croissante. $k$ n'est donc \textbf{ni croissante ni dûˋcroissante} sur $\mathbb{R}$ entier.

\end{corrige}
