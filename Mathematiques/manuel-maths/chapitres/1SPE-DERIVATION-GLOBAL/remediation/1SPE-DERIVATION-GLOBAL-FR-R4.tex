\subsection*{Rappel — Polynômes du second degré}

Soit $f(x) = ax^2 + bx + c$ avec $a \neq 0$.

\textbf{Discriminant} : $\Delta = b^2 - 4ac$.
\begin{itemize}
  \item Si $\Delta > 0$ : deux racines réelles $x_1 = \dfrac{-b - \sqrt{\Delta}}{2a}$ et $x_2 = \dfrac{-b + \sqrt{\Delta}}{2a}$.
  \item Si $\Delta = 0$ : racine double $x_0 = \dfrac{-b}{2a}$.
  \item Si $\Delta < 0$ : pas de racine réelle.
\end{itemize}

\textbf{Forme factorisée} (si $\Delta \geqslant 0$) : $f(x) = a(x - x_1)(x - x_2)$.

\textbf{Signe} : du signe de $a$ à l'extérieur des racines, opposé entre les racines.

\begin{exercice}{1SPE-DERGLOBAL-FR-R4-EX1}{1}{7}
Soit $f(x) = x^2 - 5x + 6$.
\begin{enumerate}
  \item Calculer le discriminant $\Delta$.
  \item Donner les racines de $f$.
  \item Factoriser $f(x)$.
\end{enumerate}
\end{exercice}

\textbf{Question 1.}
$\Delta = (-5)^2 - 4 \times 1 \times 6 = 25 - 24 = 1 > 0.$

\textbf{Question 2.}
$x_1 = \dfrac{5 - 1}{2} = 2$ et $x_2 = \dfrac{5 + 1}{2} = 3.$

\textbf{Question 3.}
$f(x) = (x - 2)(x - 3).$

\begin{corrige}{1SPE-DERGLOBAL-FR-R4-EX3}

\textbf{Question 1.}
$h'(x) = -x^2 + 4 = -(x^2 - 4) = -(x-2)(x+2).$

\textbf{Question 2.}
\begin{center}
\begin{tabular}{|c|cccccc|}
\hline
$x$ & $-\infty$ & $-2$ & & $2$ & & $+\infty$ \\
\hline
$-(x-2)$ & $+$ & $+$ & $+$ & $0$ & $-$ & $-$ \\
$(x+2)$ & $-$ & $0$ & $+$ & $+$ & $+$ & $+$ \\
$h'(x)$ & $-$ & $0$ & $+$ & $0$ & $-$ & $-$ \\
\hline
\end{tabular}
\end{center}

\textbf{Question 3.}
$h'(x) > 0$ sur $]-2\,;\,2[$, donc $h$ est croissante sur $[-2\,;\,2]$.
$h'(x) < 0$ sur $]-\infty\,;\,-2[$ et $]2\,;\,+\infty[$, donc $h$ est décroissante sur ces intervalles.

\end{corrige}
