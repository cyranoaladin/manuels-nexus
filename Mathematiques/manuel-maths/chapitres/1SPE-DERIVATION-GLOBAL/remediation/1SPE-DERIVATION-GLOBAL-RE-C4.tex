\begin{corrige}{1SPE-DERGLOBAL-RE-C4-EX1}

\textbf{Question 1.}
$f'(x) = 3x^2 - 3 = 3(x^2-1) = 3(x-1)(x+1).$
$f'(x) = 0 \Leftrightarrow x = -1$ ou $x = 1.$

\textbf{Question 3.}

\begin{corrige}{1SPE-DERGLOBAL-RE-C4-EX2}

\textbf{Question 1.}
Comme $a > 0$ (parabole ouverte vers le haut), la fonction $g$ est d'abord dûˋcroissante puis croissante. En $x = 2$, $g'(2) = 0$ et $g'$ passe de $-$ û  $+$, donc $g$ admet un \textbf{minimum} en $x = 2$.

$g(0) = 3 \Rightarrow C = 3$.

\begin{corrige}{1SPE-DERGLOBAL-RE-C4-EX3}

\textbf{Question 1.}
$u = x^2 - 4x + 8$, $v = x-1$, $u' = 2x-4$, $v' = 1$.
\[
h'(x) = \frac{(2x-4)(x-1) - (x^2-4x+8)}{(x-1)^2}.
\]

\textbf{Question 2.}

\commentaireMarge{$(2x-4)(x-1) = 2x^2 - 6x + 4$.}

Numûˋrateur : $2x^2 - 6x + 4 - x^2 + 4x - 8 = x^2 - 2x - 4.$

\[
h'(x) = \frac{x^2 - 2x - 4}{(x-1)^2}.
\]

$h'(x) = 0 \Leftrightarrow x^2 - 2x - 4 = 0.$
$\Delta = 4 + 16 = 20$, $\sqrt{\Delta} = 2\sqrt{5}$.
$x = \dfrac{2 \pm 2\sqrt{5}}{2} = 1 \pm \sqrt{5}.$

Seule $x_0 = 1 + \sqrt{5} \approx 3{,}24$ appartient û  $]1\,;\,+\infty[$.

\textbf{Question 3.}
Sur $]1\,;\,+\infty[$ : $(x-1)^2 > 0$ toujours. Le signe de $h'$ est celui de $x^2-2x-4$, nûˋgatif entre les racines.

\begin{tabular}{|c|ccccc|}
\hline
$x$ & $1$ & & $1+\sqrt{5}$ & & $+\infty$ \\

$h'(x)$ & & $-$ & $0$ & $+$ & \\
\hline
$h(x)$ & & $\searrow$ & $2\sqrt{5}-2$ & $\nearrow$ & \\

\end{tabular}
\end{center}

$h(1+\sqrt{5}) = \dfrac{(1+\sqrt{5})^2 - 4(1+\sqrt{5})+8}{\sqrt{5}} = \dfrac{10-2\sqrt{5}}{\sqrt{5}} = 2\sqrt{5}-2 \approx 2{,}47.$

Le minimum de $h$ sur $]1\,;\,+\infty[$ est $2\sqrt{5}-2$.

\end{corrige}
