\subsection*{Rappel — Calcul littéral}

\textbf{Développer} : $(a+b)^2 = a^2 + 2ab + b^2$, $(a-b)(a+b) = a^2 - b^2$.

\textbf{Factoriser} : extraire le facteur commun, identifier une forme remarquable.

\textbf{Signe d'un produit} : $A \times B > 0 \Leftrightarrow (A>0 \text{ et } B>0)$ ou $(A<0 \text{ et } B<0)$.

\textbf{Signe d'un quotient} : $\dfrac{A}{B} > 0 \Leftrightarrow A$ et $B$ ont le même signe ($B \neq 0$).

\begin{exercice}{1SPE-DERGLOBAL-FR-R3-EX1}{1}{6}
Développer et réduire chaque expression.
\begin{enumerate}
  \item $(x+3)(x-3)$
  \item $(2x-1)^2$
  \item $x(x-4) + 3(x^2 - 1)$
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-DERGLOBAL-FR-R3-EX1}

\textbf{Question 1.}
$(x+3)(x-3) = x^2 - 3^2 = x^2 - 9.$

\commentaireMarge{Identité : $(a-b)(a+b) = a^2 - b^2$.}

\textbf{Question 2.}
$(2x-1)^2 = 4x^2 - 2 \times 2x \times 1 + 1 = 4x^2 - 4x + 1.$

\textbf{Question 3.}
$x(x-4) + 3(x^2-1) = x^2 - 4x + 3x^2 - 3 = 4x^2 - 4x - 3.$

\end{corrige}

$6x^2 - 4x = 2x(3x - 2).$

\textbf{Question 2.}

\textbf{Question 3.}
