\begin{fichemethode}{M2}{Calculer la dérivée d'une expression}
\quandUtiliser{J'utilise cette méthode lorsque la fonction est une combinaison de fonctions de base : somme, produit ou quotient. L'énoncé peut demander de « dériver $f$ » ou « calculer $f'$ ».}
\pasApas{\begin{enumerate}
  \item J'identifie la structure globale : somme, produit ou quotient.
  \item J'applique la règle adaptée :
    \begin{itemize}
      \item Somme : $(u+v)'=u'+v'$ ;
      \item Produit : $(uv)'=u'v+uv'$ ;
      \item Quotient : $\left(\dfrac{u}{v}\right)'=\dfrac{u'v-uv'}{v^2}$.
    \end{itemize}
  \item Je développe et je réduis l'expression obtenue.
\end{enumerate}}
\exempleRedige{Calculons la dérivée de $f(x)=(2x+1)(x^2-3)$. \commentaireMarge{Structure produit : $u=2x+1$ et $v=x^2-3$.} On a $u'=2$ et $v'=2x$. Par la règle du produit :
\[f'(x)=u'v+uv'=2(x^2-3)+(2x+1)(2x).\]
\commentaireMarge{Je développe chaque terme.}
\[f'(x)=2x^2-6+4x^2+2x=6x^2+2x-6.\]}
\pieges{\begin{itemize}
  \item Écrire $(uv)'=u'v'$ : la règle du produit n'est pas la multiplication des dérivées.
  \item Oublier un des deux termes $u'v$ ou $uv'$ dans la règle du produit.
  \item Inverser l'ordre au numérateur du quotient : écrire $uv'-u'v$ au lieu de $u'v-uv'$.
\end{itemize}}
\verifier{Je peux contrôler le degré : si $f$ est de degré $n$, alors $f'$ doit être de degré $n-1$.}

\end{fichemethode}
