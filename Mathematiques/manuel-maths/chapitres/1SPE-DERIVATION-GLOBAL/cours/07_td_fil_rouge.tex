\section*{TD fil rouge — La boîte de conserve optimale}

\textit{Durée : 50 minutes — Toutes les capacités C1 à C5 sont mobilisées.}

\bigskip

\begin{center}
\textbf{Accroche}

\medskip
\parbox{0.9\textwidth}{Un fabricant de boîtes de conserve veut concevoir une boîte cylindrique de volume $V = 500$ mL $= 500$ cm³. Il souhaite \textbf{minimiser la quantité de métal utilisée}, c'est-à-dire minimiser l'aire totale de la surface de la boîte. Quelles dimensions choisir ?}
\end{center}

\bigskip

\textit{Rappels :} pour un cylindre droit de rayon $r$ et de hauteur $h$ :
\[
  \text{Volume} : V = \pi r^2 h \qquad \text{Aire totale} : S = 2\pi r^2 + 2\pi r h.
\]

\bigskip
