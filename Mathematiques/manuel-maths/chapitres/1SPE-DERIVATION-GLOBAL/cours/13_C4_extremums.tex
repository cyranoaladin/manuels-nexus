\sectionCapacite{Extremums d'une fonction}{Je sais trouver les extremums d'une fonction en annulant sa dérivée.}

% ── Définitions ──────────────────────────────────────────────────────────────

\begin{nxdef}[Extremum local]
Soit $f$ une fonction définie sur un intervalle $I$ et $a \in I$.

\begin{itemize}
  \item $f$ admet un \textbf{maximum local} en $a$ si $f(x) \leq f(a)$ pour tout $x$
        au voisinage de $a$.
  \item $f$ admet un \textbf{minimum local} en $a$ si $f(x) \geq f(a)$ pour tout $x$
        au voisinage de $a$.
\end{itemize}

On dit que $f(a)$ est un \textbf{extremum local} de $f$.
\end{nxdef}

\begin{nxdef}[Extremum global]
$f$ admet un \textbf{maximum global} en $a$ sur $I$ si $f(x) \leq f(a)$ pour tout
$x \in I$ (et de même pour le minimum global). Un extremum global est aussi un
extremum local, mais la réciproque est fausse en général.
\end{nxdef}

\margeAppui{On distingue : extremum \emph{local} (autour d'un point) et extremum
\emph{global} (sur tout le domaine). En optimisation, c'est l'extremum global
qui répond au problème.}

% ── Théorème principal ───────────────────────────────────────────────────────

\begin{nxthm}[Condition nécessaire d'extremum]
Si $f$ est dérivable en $a$ et admet un extremum local en $a$, alors
\[
  f'(a) = 0.
\]
\end{nxthm}

\erreurFrequente{La réciproque est \textbf{fausse} : $f'(a)=0$ ne garantit pas
l'existence d'un extremum en $a$. Il faut vérifier le changement de signe de
$f'$ autour de $a$.}

% ── Propriété : critère du changement de signe ───────────────────────────────

\begin{nxprop}[Critère du changement de signe]
Soit $f$ dérivable sur un intervalle $I$ et $a \in I$ tel que $f'(a)=0$.

\begin{itemize}
  \item Si $f'$ change de signe de $+$ à $-$ en $a$, alors $f$ admet un
        \textbf{maximum local} en $a$.
  \item Si $f'$ change de signe de $-$ à $+$ en $a$, alors $f$ admet un
        \textbf{minimum local} en $a$.
  \item Si $f'$ ne change pas de signe en $a$, alors $f$ \textbf{n'admet pas
        d'extremum} en $a$.
\end{itemize}
\end{nxprop}

\margeAppui{Méthode en trois étapes : (1) résoudre $f'(x)=0$ ; (2) dresser le
tableau de signes de $f'$ ; (3) conclure sur la nature de chaque point critique.}

% ── Exemple complet ──────────────────────────────────────────────────────────

\exemple{
Soit $f(x) = 2x^3 - 9x^2 + 12x - 2$, définie sur $\mathbb{R}$.

\textbf{Étape 1 — Calcul de $f'$.}
\[
  f'(x) = 6x^2 - 18x + 12 = 6(x^2 - 3x + 2) = 6(x-1)(x-2).
\]

\textbf{Étape 2 — Résoudre $f'(x) = 0$.}

$f'(x)=0 \Leftrightarrow x=1$ ou $x=2$.

\textbf{Étape 3 — Tableau de signes de $f'$ et tableau de variations.}

\[
\begin{array}{|c|ccccccc|}
\hline
x      & -\infty & & 1 & & 2 & & +\infty \\
\hline
f'(x)  &  & + & 0 & - & 0 & + & \\
\hline
f      & \nearrow & & 3 & & 2 & & \nearrow \\
       &          &\searrow& &\searrow& & & \\
\hline
\end{array}
\]

\textbf{Conclusions :}
\begin{itemize}
  \item En $x=1$ : $f'$ passe de $+$ à $-$, donc $f(1)=3$ est un
        \textbf{maximum local}.
  \item En $x=2$ : $f'$ passe de $-$ à $+$, donc $f(2)=2$ est un
        \textbf{minimum local}.
\end{itemize}

Ces extremums sont locaux : $f$ tend vers $+\infty$ quand $x \to +\infty$,
il n'y a donc pas d'extremum global sur $\mathbb{R}$.
}

% ── Contre-exemple : annulation sans extremum ─────────────────────────────────

\contreexemple{
Pour $f(x) = x^3$, on a $f'(x) = 3x^2$, donc $f'(0) = 0$.

Pourtant, $f'(x) = 3x^2 \geq 0$ pour tout $x$, et $f'$ \textbf{ne change pas
de signe} en $0$ : $f'$ est positive avant et après $0$.

Donc $f$ est croissante sur $\mathbb{R}$ et \textbf{n'admet pas d'extremum} en
$0$. L'annulation de la dérivée est nécessaire mais pas suffisante.
}

\margeAppui{Le point $(0\,;\,0)$ de la courbe de $x^3$ est un \emph{point
d'inflexion} : la tangente (horizontale) traverse la courbe. Ce concept est
hors programme en première ; on retient uniquement que $f'(a)=0$ seul ne suffit
pas.}

\erreurFrequente{Conclure « $f'(1)=0$ donc $f$ a un extremum en $1$ » sans
dresser le tableau de signes. Il faut toujours vérifier le changement de signe
de $f'$.}

\approfondissement{
Pour les fonctions deux fois dérivables, le signe de $f''(a)$ donne
l'information manquante : si $f'(a)=0$ et $f''(a)>0$, alors $f(a)$ est un
minimum local ; si $f''(a)<0$, c'est un maximum local. Si $f''(a)=0$, on ne
peut pas conclure directement (cas de $x^3$ en $0$). Ce critère de la dérivée
seconde est étudié en terminale.
}
