\subsection*{Rappel — Derivation et etude de variations}

\textbf{Derivees usuelles :} $(x^n)' = nx^{n-1}$, $(\mathrm{e}^x)' = \mathrm{e}^x$, $(1/x)' = -1/x^2$.

\textbf{Regles :} $(u+v)' = u'+v'$, $(ku)' = ku'$, $(uv)' = u'v + uv'$, $(u/v)' = (u'v-uv')/v^2$.

\begin{exercice}{TSPE-LIMFCT-FR-R2-EX1}{1}{8}
\begin{enumerate}
  \item Deriver $f(x) = 3x^2 - 5x + 2$.
  \item Deriver $g(x) = x\,\mathrm{e}^x$.
  \item Dresser le tableau de variations de $f$.
\end{enumerate}
\end{exercice}
