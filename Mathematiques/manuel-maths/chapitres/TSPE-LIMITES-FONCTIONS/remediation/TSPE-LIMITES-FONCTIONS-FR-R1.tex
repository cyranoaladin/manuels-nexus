\subsection*{Rappel — Limites de suites}

\textbf{Limite finie :} $\lim_{n \to +\infty} u_n = \ell$ signifie que $u_n$ se rapproche de $\ell$.

\textbf{Limites usuelles :} $1/n \to 0$, $q^n \to 0$ si $|q| < 1$, $n^k \to +\infty$.

\textbf{Operations :} somme, produit, quotient de limites (sauf formes indeterminees).

\begin{exercice}{TSPE-LIMFCT-FR-R1-EX1}{1}{8}
\begin{enumerate}
  \item Determiner $\lim_{n \to +\infty} \dfrac{3n+1}{n+2}$.
  \item Determiner $\lim_{n \to +\infty} 5 \times (0{,}7)^n$.
  \item Determiner $\lim_{n \to +\infty} (n^2 - 3n)$.
\end{enumerate}
\end{exercice}
