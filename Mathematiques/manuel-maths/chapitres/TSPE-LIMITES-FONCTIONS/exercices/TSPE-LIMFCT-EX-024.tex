\begin{exercice}{TSPE-LIMFCT-EX-024}{1}{10}
Soit $f(x) = \dfrac{2x-1}{x+1}$.
\begin{enumerate}
  \item Determiner les asymptotes de $\mathcal{C}_f$.
  \item Montrer que $f(x) = 2 - \dfrac{3}{x+1}$.
  \item En deduire la position de $\mathcal{C}_f$ par rapport a la droite $y = 2$.
\end{enumerate}
\end{exercice}
