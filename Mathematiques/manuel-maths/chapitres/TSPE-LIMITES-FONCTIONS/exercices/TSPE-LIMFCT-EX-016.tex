\begin{exercice}{TSPE-LIMFCT-EX-016}{3}{15}
Soit $f(x) = x + 2 + \dfrac{1}{x-1}$ definie sur $\mathbb{R} \setminus \{1\}$.
\begin{enumerate}
  \item Determiner les limites de $f$ en $+\infty$ et en $-\infty$.
  \item Determiner les limites de $f$ en $1^+$ et $1^-$.
  \item Montrer que la droite $y = x+2$ est asymptote oblique a $\mathcal{C}_f$.
\end{enumerate}
\end{exercice}
