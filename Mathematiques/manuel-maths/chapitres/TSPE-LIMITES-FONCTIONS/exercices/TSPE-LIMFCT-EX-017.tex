\begin{exercice}{TSPE-LIMFCT-EX-017}{3}{18}

\begin{enumerate}
  \item Determiner le domaine de definition.
  \item Simplifier $f(x)$ en factorisant.
  \item Determiner toutes les limites aux bornes du domaine.
  \item En deduire les asymptotes.
\end{enumerate}
