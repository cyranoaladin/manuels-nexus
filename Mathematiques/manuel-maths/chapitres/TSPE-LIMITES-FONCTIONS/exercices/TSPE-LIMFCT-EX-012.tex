\begin{exercice}{TSPE-LIMFCT-EX-012}{2}{12}
Soit $f(x) = \dfrac{2x^2 + 3}{x^2 - 4}$.
\begin{enumerate}
  \item Determiner $\lim_{x \to +\infty} f(x)$.
  \item Determiner $\lim_{x \to 2^+} f(x)$ et $\lim_{x \to 2^-} f(x)$.
\end{enumerate}
\end{exercice}
