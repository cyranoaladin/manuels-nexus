\begin{exercice}{TSPE-LIMFCT-EX-047}{3}{20}
\textbf{Demonstration de la croissance comparee.}
\begin{enumerate}
  \item Soit $g(x) = \mathrm{e}^x - 1 - x$. Montrer que $g'(x) = \mathrm{e}^x - 1$ et que $g(x) \geq 0$ pour $x \geq 0$.
  \item En deduire que $\mathrm{e}^{x/2} \geq 1 + x/2$ pour $x \geq 0$.
  \item Montrer que $\mathrm{e}^x \geq x^2/4$ pour $x \geq 0$.
  \item Conclure que $\lim_{x \to +\infty} \dfrac{\mathrm{e}^x}{x} = +\infty$.
\end{enumerate}
\end{exercice}
