\begin{exercice}{TSPE-LIMFCT-EX-035}{3}{20}
Soit $f(x) = \dfrac{x^2 + 1}{x}$ definie sur $\mathbb{R}^*$.
\begin{enumerate}
  \item Montrer que $f(x) = x + \dfrac{1}{x}$.
  \item Determiner les limites en $+\infty$, $-\infty$, $0^+$ et $0^-$.
  \item Montrer que $y = x$ est asymptote oblique et $x = 0$ est AV.
  \item Montrer que $f(x) \geq 2$ pour tout $x > 0$.
\end{enumerate}
\end{exercice}
