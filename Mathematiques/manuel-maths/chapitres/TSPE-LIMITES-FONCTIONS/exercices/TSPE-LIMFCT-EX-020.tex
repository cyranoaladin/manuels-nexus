\begin{exercice}{TSPE-LIMFCT-EX-020}{3}{20}
Soit $f(x) = \dfrac{x^3 - 8}{x^2 - 4}$ definie sur $\mathbb{R} \setminus \{-2, 2\}$.
\begin{enumerate}
  \item Factoriser $x^3 - 8$ et $x^2 - 4$.
  \item En deduire une ecriture simplifiee de $f(x)$ pour $x \neq 2$.
  \item Determiner toutes les limites aux bornes du domaine.
  \item Preciser les asymptotes de $\mathcal{C}_f$.
\end{enumerate}
\end{exercice}
