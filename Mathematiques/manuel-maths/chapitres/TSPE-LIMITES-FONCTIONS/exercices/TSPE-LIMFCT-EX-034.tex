\begin{exercice}{TSPE-LIMFCT-EX-034}{3}{18}
Soit $f(x) = \dfrac{x\,\mathrm{e}^x}{\mathrm{e}^x - 1}$ definie sur $\mathbb{R}^*$.
\begin{enumerate}
  \item Determiner les limites de $f$ en $+\infty$, $-\infty$ et en $0$.
  \item En deduire les asymptotes de $\mathcal{C}_f$.
  \item La fonction $f$ est-elle prolongeable par continuite en $0$ ?
\end{enumerate}
\end{exercice}
