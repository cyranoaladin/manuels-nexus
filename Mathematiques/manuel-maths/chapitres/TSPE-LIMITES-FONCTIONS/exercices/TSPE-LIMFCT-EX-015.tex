\begin{exercice}{TSPE-LIMFCT-EX-015}{3}{15}
Soit $f(x) = (2x^2 - x + 3)\,\mathrm{e}^{-x}$.
\begin{enumerate}
  \item Determiner les limites de $f$ en $+\infty$ et en $-\infty$.
  \item Calculer $f'(x)$.

\end{enumerate}
