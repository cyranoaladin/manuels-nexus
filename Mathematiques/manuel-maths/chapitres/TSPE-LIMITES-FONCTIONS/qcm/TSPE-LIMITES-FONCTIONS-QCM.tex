\subsection*{C1 — Determiner des limites}
% ------------------------------------------------------------

\begin{enumerate}

\item \textbf{[Q1]} $\lim_{x \to +\infty} \dfrac{3x^2 - x}{x^2 + 5}$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $0$
    \item $3$
    \item $-\dfrac{1}{5}$
    \item $+\infty$
  \end{enumerate}

\bigskip

\item \textbf{[Q2]} $\lim_{x \to +\infty} (x^2 - 3x + 1)$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $1$
    \item $-\infty$
    \item $+\infty$
    \item On ne peut pas conclure
  \end{enumerate}

\bigskip

\item \textbf{[Q3]} La forme $\dfrac{+\infty}{+\infty}$ est :
  \begin{enumerate}[label=\Alph*.]
    \item egale a $1$
    \item egale a $+\infty$
    \item une forme indeterminee
    \item egale a $0$
  \end{enumerate}

\bigskip

\item \textbf{[Q4]} $\lim_{x \to 0^+} \dfrac{1}{x}$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $0$
    \item $1$
    \item $+\infty$
    \item $-\infty$
  \end{enumerate}

\bigskip

\item \textbf{[Q5]} $\lim_{x \to +\infty} x\,\mathrm{e}^{-x}$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $+\infty$
    \item $0$
    \item $1$
    \item $-\infty$
  \end{enumerate}

\bigskip

\subsection*{C2 — Asymptotes}
% ------------------------------------------------------------

\item \textbf{[Q6]} La droite $y = 2$ est asymptote horizontale a $\mathcal{C}_f$ si :
  \begin{enumerate}[label=\Alph*.]
    \item $f(2) = 0$
    \item $\lim_{x \to 2} f(x) = +\infty$
    \item $\lim_{x \to +\infty} f(x) = 2$ ou $\lim_{x \to -\infty} f(x) = 2$
    \item La courbe coupe la droite $y = 2$
  \end{enumerate}

\bigskip

\item \textbf{[Q7]} Soit $f(x) = \dfrac{x+1}{x-3}$. L'asymptote verticale est :
  \begin{enumerate}[label=\Alph*.]
    \item $x = -1$
    \item $x = 3$
    \item $y = 1$
    \item $x = 0$
  \end{enumerate}

\bigskip

\item \textbf{[Q8]} Soit $f(x) = \dfrac{4x}{x+1}$. L'asymptote horizontale en $+\infty$ est :
  \begin{enumerate}[label=\Alph*.]
    \item $y = 0$
    \item $y = 4$
    \item $y = 1$
    \item Il n'y a pas d'AH
  \end{enumerate}

\bigskip

\item \textbf{[Q9]} La fonction $f(x) = \mathrm{e}^{-x} + 3$ admet :
  \begin{enumerate}[label=\Alph*.]
    \item une AH $y = 0$ et une AV $x = 3$
    \item une AH $y = 3$ en $+\infty$ et pas d'AV
    \item une AH $y = 3$ en $-\infty$ et pas d'AV
    \item pas d'asymptote
  \end{enumerate}

\bigskip

\item \textbf{[Q10]} Si $\lim_{x \to 5^+} f(x) = -\infty$, alors :
  \begin{enumerate}[label=\Alph*.]
    \item $y = 5$ est AH
    \item $x = 5$ est AV
    \item $f(5) = -\infty$
    \item La courbe coupe l'axe des abscisses en $5$
  \end{enumerate}

\bigskip

\subsection*{C3 — Croissances comparees}
% ------------------------------------------------------------

\item \textbf{[Q11]} $\lim_{x \to +\infty} \dfrac{\mathrm{e}^x}{x^3}$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $0$
    \item $1$
    \item $+\infty$
    \item $\dfrac{1}{3}$
  \end{enumerate}

\bigskip

\item \textbf{[Q12]} $\lim_{x \to -\infty} x^2\,\mathrm{e}^x$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $+\infty$
    \item $0$
    \item $-\infty$
    \item $1$
  \end{enumerate}

\bigskip

\item \textbf{[Q13]} Pour determiner $\lim_{x \to +\infty} (\mathrm{e}^x - x^2)$, on :
  \begin{enumerate}[label=\Alph*.]
    \item ecrit $\mathrm{e}^x - x^2 = +\infty - \infty = 0$
    \item factorise par $\mathrm{e}^x$ et utilise $x^2/\mathrm{e}^x \to 0$
    \item factorise par $x^2$ et conclut $+\infty$
    \item dit que c'est une FI sans conclusion
  \end{enumerate}

\bigskip

\item \textbf{[Q14]} L'inegalite $\mathrm{e}^x \geq 1 + x$ pour $x \geq 0$ s'obtient par :
  \begin{enumerate}[label=\Alph*.]
    \item Recurrence
    \item Etude de $g(x) = \mathrm{e}^x - 1 - x$ (variations)
    \item Division euclidienne
    \item Formule de Taylor
  \end{enumerate}

\bigskip

\item \textbf{[Q15]} $\lim_{x \to +\infty} \dfrac{x^{100}}{\mathrm{e}^x}$ vaut :
  \begin{enumerate}[label=\Alph*.]
    \item $+\infty$
    \item $100$
    \item $0$
    \item $1$
  \end{enumerate}

\end{enumerate}
