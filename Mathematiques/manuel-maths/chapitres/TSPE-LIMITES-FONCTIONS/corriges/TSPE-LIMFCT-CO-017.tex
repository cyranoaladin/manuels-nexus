\begin{corrige}{TSPE-LIMFCT-EX-017}

\textbf{1.} $x^2-1 = 0 \iff x = \pm 1$. $D_f = \mathbb{R} \setminus \{-1, 1\}$.

\textbf{2.} $x^2+x-2 = (x-1)(x+2)$, $x^2-1 = (x-1)(x+1)$. Pour $x \neq 1$ : $f(x) = \dfrac{x+2}{x+1}$.

\textbf{3.} $\lim_{\pm\infty} f = 1$. $\lim_{x \to 1} f = 3/2$. $\lim_{x \to -1^+} f = -\infty$. $\lim_{x \to -1^-} f = +\infty$.

Correction du signe : en $-1^+$ : $(x+2) \to 1 > 0$, $(x+1) \to 0^+$, $f \to \boxed{+\infty}$. En $-1^-$ : $(x+1) \to 0^-$, $f \to \boxed{-\infty}$.

\textbf{4.} AH : $y = 1$ en $\pm\infty$. AV : $x = -1$. En $x = 1$ : pas d'AV (limite finie $3/2$, trou dans la courbe).

\end{corrige}
