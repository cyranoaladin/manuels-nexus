\begin{corrige}{TSPE-LIMFCT-EX-025}

\textbf{1.} $\mathrm{e}^{-x} \to 0$ en $+\infty$ : $\lim_{+\infty} f = \boxed{1}$. $\mathrm{e}^{-x} \to +\infty$ en $-\infty$ : $\lim_{-\infty} f = \boxed{+\infty}$.

\textbf{2.} AH en $+\infty$ : $y = 1$. Pas d'AH en $-\infty$. Pas d'AV ($f$ definie sur $\mathbb{R}$).
