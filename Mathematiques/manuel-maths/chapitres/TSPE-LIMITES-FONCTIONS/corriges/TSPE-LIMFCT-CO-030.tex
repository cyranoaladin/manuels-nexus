\begin{corrige}{TSPE-LIMFCT-EX-030}

\textbf{1.} En $+\infty$ : diviser par $\mathrm{e}^x$ : $\dfrac{2+1/\mathrm{e}^x}{1+3/\mathrm{e}^x} \to \boxed{2}$.

En $-\infty$ : $\mathrm{e}^x \to 0$. $\dfrac{0+1}{0+3} = \boxed{\dfrac{1}{3}}$.

\textbf{2.} AH en $+\infty$ : $y = 2$. AH en $-\infty$ : $y = 1/3$.

\textbf{3.} $\mathrm{e}^x + 3 > 0$ pour tout $x$ (car $\mathrm{e}^x > 0$). Pas d'AV.
