\begin{corrige}{TSPE-LIMFCT-EX-050}

\textbf{1.} $C(t) = t/\mathrm{e}^{t/2}$. Posons $u = t/2$ : $C = 2u/\mathrm{e}^u \to 0$ (croissances comparees). Le medicament est elimine.

\textbf{2.} $C'(t) = \mathrm{e}^{-t/2} + t(-1/2)\mathrm{e}^{-t/2} = (1-t/2)\mathrm{e}^{-t/2}$.

\textbf{3.} $C'(t) = 0 \iff t = 2$. $C$ croissante sur $[0, 2]$, decroissante sur $[2, +\infty[$.

\textbf{4.} Max en $t = 2$ : $C(2) = 2\mathrm{e}^{-1} \approx 0{,}74$ mg/L.

\textbf{5.} $C(0) = 0 < 0{,}5 < C(2) \approx 0{,}74$. Par TVI, une solution $t_1 \in ]0, 2[$. $C$ decroissante sur $[2, +\infty[$, de $0{,}74$ a $0$, avec $0 < 0{,}5 < 0{,}74$. Par TVI, une solution $t_2 \in ]2, +\infty[$. Exactement 2 solutions (stricte monotonie sur chaque intervalle).

\end{corrige}
