\begin{exercice}{1SPE-PROBCOND-RE-C5-EX1}{1}{10}
Dans un village, $40\,\%$ des habitants possedent un chien ($C$). Parmi les proprietaires de chien, $30\,\%$ ont aussi un chat ($H$). Parmi ceux sans chien, $50\,\%$ ont un chat.

\begin{enumerate}
  \item Identifier les evenements et ecrire les probabilites.
  \item Calculer $P(H)$ par la formule des probabilites totales.
  \item Calculer $P_H(C)$ : sachant qu'une personne a un chat, quelle est la probabilite qu'elle ait aussi un chien ?
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-PROBCOND-RE-C5-EX1}
\textbf{1.} $P(C) = \frac{2}{5}$, $P_C(H) = \frac{3}{10}$, $P_{\overline{C}}(H) = \frac{1}{2}$.

\textbf{2.} $P(H) = \frac{2}{5}\times\frac{3}{10} + \frac{3}{5}\times\frac{1}{2} = \frac{6}{50}+\frac{15}{50} = \frac{21}{50} = 42\,\%$.

\textbf{3.} $P_H(C) = \frac{P(C \cap H)}{P(H)} = \frac{6/50}{21/50} = \frac{6}{21} = \frac{2}{7} \approx 28{,}6\,\%$.
\end{corrige}
