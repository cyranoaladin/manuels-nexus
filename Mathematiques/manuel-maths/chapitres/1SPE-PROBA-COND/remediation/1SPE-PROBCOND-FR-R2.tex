\subsection*{Rappel --- Fractions et pourcentages}

\textbf{Conversion :} $\dfrac{a}{b} = \dfrac{a \times 100}{b}\,\%$ et $p\,\% = \dfrac{p}{100}$.

\begin{exercice}{1SPE-PROBCOND-FR-R2-EX1}{1}{5}
\begin{enumerate}
  \item Convertir $\dfrac{7}{20}$ en pourcentage.
  \item Ecrire $12{,}5\,\%$ sous forme de fraction irreductible.
  \item Calculer $\dfrac{3}{8} + \dfrac{5}{12}$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-PROBCOND-FR-R2-EX1}
\textbf{1.} $\frac{7}{20} = \frac{35}{100} = 35\,\%$.

\textbf{2.} $12{,}5\,\% = \frac{12{,}5}{100} = \frac{125}{1000} = \frac{1}{8}$.

\textbf{3.} $\frac{3}{8} + \frac{5}{12} = \frac{9}{24} + \frac{10}{24} = \frac{19}{24}$.
\end{corrige}
