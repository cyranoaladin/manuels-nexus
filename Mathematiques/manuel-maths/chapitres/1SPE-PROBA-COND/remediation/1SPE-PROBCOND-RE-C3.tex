\begin{exercice}{1SPE-PROBCOND-RE-C3-EX1}{1}{8}
$P(B) = \dfrac{1}{4}$, $P_B(A) = \dfrac{2}{3}$, $P_{\overline{B}}(A) = \dfrac{1}{3}$.

\begin{enumerate}
  \item Calculer $P(\overline{B})$.
  \item Calculer $P(B) \times P_B(A)$.
  \item Calculer $P(\overline{B}) \times P_{\overline{B}}(A)$.
  \item En deduire $P(A)$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-PROBCOND-RE-C3-EX1}
\textbf{1.} $P(\overline{B}) = 1 - \frac{1}{4} = \frac{3}{4}$.

\textbf{2.} $\frac{1}{4} \times \frac{2}{3} = \frac{2}{12} = \frac{1}{6}$.

\textbf{3.} $\frac{3}{4} \times \frac{1}{3} = \frac{3}{12} = \frac{1}{4}$.

\textbf{4.} $P(A) = \frac{1}{6} + \frac{1}{4} = \frac{2}{12} + \frac{3}{12} = \frac{5}{12}$.
\end{corrige}
