\subsection*{Rappel --- Calcul litteral}

\begin{exercice}{1SPE-PROBCOND-FR-R4-EX1}{1}{5}
\begin{enumerate}
  \item Resoudre $5x - 3 = 2x + 9$.
  \item Resoudre $\dfrac{x}{3} + \dfrac{1}{4} = \dfrac{7}{12}$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-PROBCOND-FR-R4-EX1}
\textbf{1.} $5x - 3 = 2x + 9 \Leftrightarrow 3x = 12 \Leftrightarrow x = 4$.

\textbf{2.} $\frac{x}{3} = \frac{7}{12} - \frac{1}{4} = \frac{7}{12} - \frac{3}{12} = \frac{4}{12} = \frac{1}{3}$, donc $x = 1$.
\end{corrige}
