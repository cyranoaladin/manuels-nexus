\begin{exercice}{1SPE-PROBCOND-EX-007}{2}{12}
On donne $P(A) = \dfrac{2}{5}$, $P(B) = \dfrac{3}{10}$ et $P(A \cup B) = \dfrac{1}{2}$.

\begin{enumerate}
  \item Calculer $P(A \cap B)$.
  \item En deduire $P_A(B)$.
  \item Calculer $P_B(A)$.
\end{enumerate}
\end{exercice}
