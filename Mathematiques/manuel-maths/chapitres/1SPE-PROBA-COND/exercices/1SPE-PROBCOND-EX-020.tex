\begin{exercice}{1SPE-PROBCOND-EX-020}{3}{15}
On connait les probabilites des quatre feuilles d'un arbre a deux niveaux :
\[
  P(A \cap B) = \frac{3}{20}, \quad P(A \cap \overline{B}) = \frac{1}{20}, \quad P(\overline{A} \cap B) = \frac{1}{4}, \quad P(\overline{A} \cap \overline{B}) = \frac{11}{20}.
\]

\begin{enumerate}
  \item Verifier que la somme de ces quatre probabilites vaut $1$.
  \item Reconstituer l'arbre pondere complet : determiner $P(A)$, $P_A(B)$, $P_{\overline{A}}(B)$.
\end{enumerate}
\end{exercice}
