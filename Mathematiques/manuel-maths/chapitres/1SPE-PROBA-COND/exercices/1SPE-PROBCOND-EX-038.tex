\begin{exercice}{1SPE-PROBCOND-EX-038}{2}{12}

\begin{enumerate}
  \item Calculer $P(A \cap B)$.
  \item Calculer $P(A \cup B)$.
  \item Calculer $P_A(B)$. Commenter.
\end{enumerate}
