\begin{exercice}{1SPE-PROBCOND-EX-009}{3}{15}
On donne $P(A) = p$ (inconnue), $P_A(B) = \dfrac{1}{3}$, $P(B) = \dfrac{1}{2}$ et $P_B(A) = \dfrac{1}{4}$.

\begin{enumerate}
  \item Exprimer $P(A \cap B)$ en fonction de $p$.
  \item En utilisant $P_B(A) = \dfrac{1}{4}$, determiner la valeur de $p$.
  \item Verifier la coherence des resultats.
\end{enumerate}
\end{exercice}
