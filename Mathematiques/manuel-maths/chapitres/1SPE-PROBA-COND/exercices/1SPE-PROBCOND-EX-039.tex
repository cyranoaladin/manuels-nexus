\begin{exercice}{1SPE-PROBCOND-EX-039}{3}{15}
$A$ et $B$ sont independants. On donne $P(A) = \dfrac{1}{3}$ et $P(A \cup B) = \dfrac{1}{2}$.

\begin{enumerate}
  \item Exprimer $P(A \cup B)$ en fonction de $P(A)$ et $P(B)$ en utilisant l'independance.
  \item En deduire $P(B)$.
  \item Verifier en calculant directement $P(A \cap B)$ et $P(A) + P(B) - P(A \cap B)$.
\end{enumerate}
\end{exercice}
