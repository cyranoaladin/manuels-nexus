\begin{exercice}{1SPE-PROBCOND-EX-033}{1}{10}
On donne $P(A) = \dfrac{1}{2}$, $P(B) = \dfrac{1}{6}$ et $P(A \cap B) = \dfrac{1}{12}$.

\begin{enumerate}
  \item Montrer que $A$ et $B$ sont independants.
  \item En deduire que $\overline{A}$ et $B$ sont aussi independants.
\end{enumerate}
\end{exercice}
