\begin{corrige}{1SPE-PROBCOND-EX-007}

\textbf{1.} $P(A \cap B) = P(A) + P(B) - P(A \cup B) = \dfrac{2}{5} + \dfrac{3}{10} - \dfrac{1}{2} = \dfrac{4}{10} + \dfrac{3}{10} - \dfrac{5}{10} = \dfrac{2}{10} = \dfrac{1}{5}$.

\medskip
\textbf{2.} $P_A(B) = \dfrac{P(A \cap B)}{P(A)} = \dfrac{\frac{1}{5}}{\frac{2}{5}} = \dfrac{1}{2}$.

\medskip
\textbf{3.} $P_B(A) = \dfrac{P(A \cap B)}{P(B)} = \dfrac{\frac{1}{5}}{\frac{3}{10}} = \dfrac{1}{5} \times \dfrac{10}{3} = \dfrac{2}{3}$.

\end{corrige}
