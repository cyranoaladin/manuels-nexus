\begin{corrige}{1SPE-PROBCOND-EX-033}

\textbf{1.} $P(A) \times P(B) = \frac{1}{2} \times \frac{1}{6} = \frac{1}{12} = P(A \cap B)$. Donc $A$ et $B$ sont independants.

\textbf{2.} Si $A$ et $B$ sont independants, alors $\overline{A}$ et $B$ le sont aussi (propriete du cours). Verification : $P(\overline{A} \cap B) = P(B) - P(A \cap B) = \frac{1}{6}-\frac{1}{12} = \frac{1}{12}$ et $P(\overline{A}) \times P(B) = \frac{1}{2} \times \frac{1}{6} = \frac{1}{12}$. \checkmark

\end{corrige}
