\begin{corrige}{1SPE-PROBCOND-EX-039}

\textbf{1.} $P(A \cup B) = P(A) + P(B) - P(A \cap B) = P(A) + P(B) - P(A)P(B)$ (independance).

$\frac{1}{2} = \frac{1}{3} + P(B) - \frac{1}{3}P(B) = \frac{1}{3} + \frac{2}{3}P(B)$.

\textbf{2.} $\frac{2}{3}P(B) = \frac{1}{6}$, donc $P(B) = \frac{1}{4}$.

\textbf{3.} $P(A \cap B) = \frac{1}{3} \times \frac{1}{4} = \frac{1}{12}$. $P(A)+P(B)-P(A \cap B) = \frac{1}{3}+\frac{1}{4}-\frac{1}{12} = \frac{4+3-1}{12} = \frac{1}{2}$ \checkmark.

\end{corrige}
