\begin{corrige}{1SPE-PROBCOND-EX-010}

\textbf{1.} $P(A \cap B) = P(A) + P(B) - P(A \cup B) = \dfrac{3}{5} + \dfrac{1}{2} - \dfrac{4}{5} = \dfrac{6}{10} + \dfrac{5}{10} - \dfrac{8}{10} = \dfrac{3}{10}$.

\medskip
\textbf{2.} $P_A(B) = \dfrac{3/10}{3/5} = \dfrac{1}{2}$ et $P_B(A) = \dfrac{3/10}{1/2} = \dfrac{3}{5}$.

\medskip
\textbf{3.} $P(\overline{A}) = \dfrac{2}{5}$. $P(\overline{A} \cap B) = P(B) - P(A \cap B) = \dfrac{1}{2} - \dfrac{3}{10} = \dfrac{1}{5}$.

$P_{\overline{A}}(B) = \dfrac{1/5}{2/5} = \dfrac{1}{2}$.

\medskip
\textbf{4.} $P_A(B) = P_{\overline{A}}(B) = \dfrac{1}{2}$. La realisation de $A$ ne modifie pas la probabilite de $B$.

\end{corrige}
