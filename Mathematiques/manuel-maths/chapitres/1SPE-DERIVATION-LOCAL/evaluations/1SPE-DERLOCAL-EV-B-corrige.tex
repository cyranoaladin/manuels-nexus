\subsection*{Exercice 1 — 6 points}
%---------------------------------------------------------------

\textbf{Question 1.}
$f(2) = 4 - 12 + 10 = 2$ et $f(5) = 25 - 30 + 10 = 5$.
\[
\tau = \frac{f(5)-f(2)}{5-2} = \frac{5-2}{3} = 1.
\]

\textbf{Question 2.}
$f(5) = 5$.
\begin{align*}
f(5+h) &= (5+h)^2 - 6(5+h) + 10 \\
        &= 25 + 10h + h^2 - 30 - 6h + 10 = h^2 + 4h + 5.
\end{align*}
\[
\frac{f(5+h)-f(5)}{h} = \frac{h^2+4h}{h} = h+4.
\]

\textbf{Question 3.}
$f'(5) = \lim_{h \to 0}(h+4) = 4.$

\textbf{Question 4.}
$f'(5) = 4$ signifie que la tangente à la courbe de $f$ au point d'abscisse $5$ a pour pente $4$ : la courbe monte localement avec un coefficient directeur de $4$.

\textbf{Question 1.}

\textbf{Question 2.}
$g(2) = 8$. La tangente $T_2$ en $x=2$ :
\[
T_2 : y = 12(x-2) + 8 = 12x - 16.
\]

\textbf{Question 3.}
$T_2(0) = 12 \times 0 - 16 = -16$. Le point $(0;-16)$ est bien sur $T_2$. \checkmark

\textbf{Question 4.}
Tangente horizontale $\Leftrightarrow g'(a) = 0 \Leftrightarrow 3a^2 = 0 \Leftrightarrow a = 0$.

\subsection*{Exercice 4 — 4 points}

\textbf{Question 1.}
$h(4) = \dfrac{1}{4}$ et $h(5) = \dfrac{1}{5}$.
\[
\tau = \frac{h(5)-h(4)}{5-4} = \frac{\frac{1}{5}-\frac{1}{4}}{1} = \frac{-\frac{1}{20}}{1} = -\frac{1}{20} = -0{,}05.
\]

\textbf{Question 2.}
$h'(5) = -\dfrac{1}{25}$.
\[
h(5{,}02) \approx h(5) + h'(5) \times 0{,}02 = \frac{1}{5} + \left(-\frac{1}{25}\right) \times 0{,}02 = 0{,}2 - 0{,}0008 = 0{,}1992.
\]
