\subsection*{C1 — Taux de variation et pente de sécante}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} Soit $f$ la fonction définie par $f(x)=x^2$. Quel est le taux de variation de $f$ entre $1$ et $3$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{f(3)}{3} = 3$
    \item $\dfrac{f(3)-f(1)}{3-1} = 4$
    \item $f(3)-f(1) = 8$
    \item $\dfrac{3-1}{f(3)-f(1)} = \dfrac{1}{4}$
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} Le taux de variation de $f$ entre $a$ et $a+h$ ($h\neq 0$) est :

  \begin{enumerate}[label=\Alph*.]
    \item $f(a+h) - f(a)$
    \item $\dfrac{f(a+h)-f(a)}{h}$
    \item $\dfrac{h}{f(a+h)-f(a)}$
    \item $f(a+h) \times h$
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} Un cycliste parcourt $f(t)$ mètres en $t$ secondes. Le taux de variation de $f$ entre $t=10$ et $t=20$ représente :

  \begin{enumerate}[label=\Alph*.]
    \item La distance parcourue entre $t=10$ et $t=20$.
    \item La vitesse à l'instant $t=15$.
    \item La vitesse moyenne entre $t=10$ et $t=20$.
    \item L'accélération entre $t=10$ et $t=20$.
  \end{enumerate}

\end{enumerate}

\bigskip

\subsection*{C2 — Nombre dérivé, pente et vitesse instantanée}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q4 ----
\item \textbf{[Q4]} Le nombre dérivé $f'(a)$ est :

  \begin{enumerate}[label=\Alph*.]
    \item La pente de la tangente à la courbe au point d'abscisse $a$.
    \item La pente de la sécante entre $a$ et $b$.
    \item L'ordonnée du point d'abscisse $a$ sur la courbe.
    \item La valeur de $f$ au point le plus proche de $a$.
  \end{enumerate}

\bigskip

% ---- Q5 ----
\item \textbf{[Q5]} Soit $f(x) = x^2$. On pose $t(h) = \dfrac{f(2+h)-f(2)}{h}$. Quelle est la forme simplifiée de $t(h)$ pour $h \neq 0$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $4+h$
    \item $2h$
    \item $h+2$
    \item $4h$
  \end{enumerate}

\bigskip

% ---- Q6 ----
\item \textbf{[Q6]} Un mobile a pour position $d(t) = 3t^2 + t$ (en mètres, $t$ en secondes). Le nombre $d'(2)$ représente :

  \begin{enumerate}[label=\Alph*.]
    \item La distance parcourue à $t=2$.
    \item La vitesse instantanée à $t=2$.
    \item La vitesse moyenne entre $t=0$ et $t=2$.
    \item L'accélération à $t=2$.
  \end{enumerate}

\end{enumerate}

\bigskip

\subsection*{C3 — Lecture graphique et construction de la tangente}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} Sur un graphique, la tangente à la courbe de $f$ au point $A(2;5)$ passe aussi par le point $B(4;9)$. Alors $f'(2)$ vaut :

  \begin{enumerate}[label=\Alph*.]
    \item $\dfrac{9}{4} = 2{,}25$
    \item $\dfrac{9-5}{4-2} = 2$
    \item $9-5 = 4$
    \item $\dfrac{4-2}{9-5} = 0{,}5$
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} La tangente en un point de la courbe est horizontale. Cela signifie que :

  \begin{enumerate}[label=\Alph*.]
    \item $f(a) = 0$
    \item $f'(a) = 0$
    \item $f'(a) = 1$
    \item $f(a) = f'(a)$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} Pour construire la tangente en $A(a;f(a))$, il faut connaître :

  \begin{enumerate}[label=\Alph*.]
    \item Uniquement $f(a)$.
    \item Uniquement $f'(a)$.
    \item Le point $A$ et la pente $f'(a)$.
    \item Deux points quelconques de la courbe.
  \end{enumerate}

\end{enumerate}

\bigskip

% ------------------------------------------------------------
\subsection*{C4 — Équation de la tangente}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q10 ----
\item \textbf{[Q10]} L'équation de la tangente à la courbe de $f$ au point d'abscisse $a$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $y = f'(a) \cdot x + f(a)$
    \item $y = f'(a)(x - a) + f(a)$
    \item $y = f(a)(x - a) + f'(a)$
    \item $y = f'(x)(x - a) + f(a)$
  \end{enumerate}

\bigskip

% ---- Q11 ----
\item \textbf{[Q11]} Soit $f(3) = 7$ et $f'(3) = -2$. L'équation de la tangente en $x=3$ est :

  \begin{enumerate}[label=\Alph*.]
    \item $y = -2x + 7$
    \item $y = -2(x-3) + 7$
    \item $y = 7(x-3) - 2$
    \item $y = -2x + 3$
  \end{enumerate}

\bigskip

% ---- Q12 ----
\item \textbf{[Q12]} Pour vérifier que la tangente en $a$ est correcte, on peut notamment :

  \begin{enumerate}[label=\Alph*.]
    \item Vérifier que $y(0)=f(0)$.
    \item Vérifier que $y(a) = f(a)$ et que le coefficient directeur est $f'(a)$.
    \item Vérifier que $y(a) = 0$.
    \item Vérifier que $y(a) = f'(a)$.
  \end{enumerate}

\end{enumerate}

\bigskip

\subsection*{C5 — Approximation linéaire}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q13 ----
\item \textbf{[Q13]} Si $f(a) = 4$ et $f'(a) = 3$, alors $f(a + h) \approx$ :

  \begin{enumerate}[label=\Alph*.]
    \item $3h$
    \item $4 + 3h$
    \item $3 + 4h$
    \item $7h$
  \end{enumerate}

\bigskip

% ---- Q14 ----
\item \textbf{[Q14]} L'approximation $f(a+h) \approx f(a) + f'(a) \cdot h$ est d'autant plus fiable que :

  \begin{enumerate}[label=\Alph*.]
    \item $h$ est grand.
    \item $h$ est proche de $0$.
    \item $f'(a)$ est nul.
    \item $a = 0$.
  \end{enumerate}

\bigskip

% ---- Q15 ----
\item \textbf{[Q15]} On sait que $\sqrt{4} = 2$ et que la dérivée de $\sqrt{x}$ en $x=4$ vaut $\dfrac{1}{4}$. Alors $\sqrt{4{,}06} \approx$ :

  \begin{enumerate}[label=\Alph*.]
    \item $2{,}06$
    \item $2{,}015$
    \item $2{,}06 \times \dfrac{1}{4}$
    \item $2{,}6$
  \end{enumerate}

\end{enumerate}
