\begin{exercice}{1SPE-DERLOCAL-EX-011}{3}{20}
On veut déterminer $f'(a)$ pour la fonction carré $f(x)=x^2$, où $a$ est un réel quelconque.
\begin{enumerate}
  \item Développer $(a+h)^2$ et en déduire $f(a+h)-f(a)$.
  \item Exprimer $\dfrac{f(a+h)-f(a)}{h}$ pour $h\ne 0$ et simplifier.
  \item En déduire $f'(a)$ pour tout réel $a$.
  \item Que vaut $f'(0)$ ? $f'(-1)$ ? $f'(5)$ ?
  \item Pour quelle valeur de $a$ a-t-on $f'(a)=0$ ? Interpréter géométriquement.
\end{enumerate}
\end{exercice}
