\begin{exercice}{1SPE-DERLOCAL-EX-020}{1}{8}
Soit $f$ la fonction définie sur $\mathbb{R}$ par $f(x) = x^2 + 1$. On admet que $f'(a) = 2a$ pour tout réel $a$.

\begin{enumerate}
  \item Calculer $f(-1)$ et $f'(-1)$.
  \item En déduire l'équation de la tangente à la courbe de $f$ au point d'abscisse $-1$.
\end{enumerate}
\end{exercice}
