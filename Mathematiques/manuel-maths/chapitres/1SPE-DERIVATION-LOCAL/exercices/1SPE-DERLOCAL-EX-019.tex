\begin{exercice}{1SPE-DERLOCAL-EX-019}{1}{8}
Soit $f$ une fonction définie et dérivable sur $\mathbb{R}$ telle que $f(1) = 3$ et $f'(1) = 2$.

\begin{enumerate}
  \item Écrire l'équation de la tangente $T$ à la courbe de $f$ au point d'abscisse $1$.
  \item Vérifier que la tangente passe bien par le point $(1\,;\,3)$.
\end{enumerate}
\end{exercice}
