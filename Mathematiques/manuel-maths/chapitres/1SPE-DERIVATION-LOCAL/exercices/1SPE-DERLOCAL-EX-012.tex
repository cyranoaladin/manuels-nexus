\begin{exercice}{1SPE-DERLOCAL-EX-012}{3}{20}
Soit $f(x)=\dfrac{1}{x}$ définie sur $]0\,;\,+\infty[$.
\begin{enumerate}
  \item Vérifier que, pour $h\ne 0$ et $1+h>0$ :
  \[
  \frac{f(1+h)-f(1)}{h}=\frac{-1}{1+h}.
  \]
  \item En déduire $f'(1)$.
  \item Déterminer l'équation de la tangente à la courbe de $f$ au point d'abscisse $1$.
  \item Cette tangente coupe-t-elle l'axe des abscisses ? Si oui, en quel point ?
\end{enumerate}
\end{exercice}
