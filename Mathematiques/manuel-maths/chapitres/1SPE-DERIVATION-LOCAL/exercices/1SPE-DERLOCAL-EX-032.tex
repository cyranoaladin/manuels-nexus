\begin{exercice}{1SPE-DERLOCAL-EX-032}{1}{8}
On considère une fonction $f$ dont on connaît les valeurs suivantes :

\begin{center}
\begin{tabular}{|c|c|c|c|c|}
\hline
$x$ & $1$ & $3$ & $5$ & $7$ \\
\hline
$f(x)$ & $2$ & $10$ & $26$ & $50$ \\
\hline
\end{tabular}
\end{center}

\begin{enumerate}
\item Calculer le taux de variation de $f$ entre $x=1$ et $x=5$.
\item Calculer le taux de variation de $f$ entre $x=3$ et $x=7$.
\end{enumerate}
\end{exercice}
