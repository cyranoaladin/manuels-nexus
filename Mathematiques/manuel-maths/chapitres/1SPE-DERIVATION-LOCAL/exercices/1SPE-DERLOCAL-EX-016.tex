\begin{exercice}{1SPE-DERLOCAL-EX-016}{2}{15}
Soit $f$ une fonction définie et dérivable sur $\mathbb{R}$. On donne :
\[
f(1) = 2, \quad f'(1) = -1, \quad f(3) = 4, \quad f'(3) = 2.
\]

\begin{enumerate}
  \item Déterminer l'équation de la tangente $T_1$ à la courbe de $f$ au point d'abscisse $1$.
  \item Déterminer l'équation de la tangente $T_2$ à la courbe de $f$ au point d'abscisse $3$.
  \item Les droites $T_1$ et $T_2$ sont-elles sécantes ? Si oui, déterminer les coordonnées de leur point d'intersection.
\end{enumerate}
\end{exercice}
