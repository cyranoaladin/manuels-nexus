\begin{exercice}{1SPE-DERLOCAL-EX-050}{3}{20}
\textbf{Analyse d'erreur.}

Karim affirme : \og Si $f(x) = \dfrac{1}{x}$, alors $f'(2) = -\dfrac{1}{2}$ \fg.

\begin{enumerate}
  \item Calculer le taux de variation $\dfrac{f(2+h) - f(2)}{h}$ pour $h \neq 0$ et simplifier.
  \item Karim s'est-il trompé ? Déterminer la valeur correcte de $f'(2)$.
  \item Écrire l'équation de la tangente à la courbe de $f$ au point d'abscisse $2$.
  \item Tracer la courbe de $f$ et la tangente sur un même repère pour $x \in [0{,}5\,;\,5]$.
  \item Conjecturer graphiquement si la courbe est au-dessus ou en dessous de la tangente au voisinage de $x = 2$. Justifier.
\end{enumerate}
\end{exercice}
