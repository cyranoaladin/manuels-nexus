\subsection*{Rappel — Vitesse moyenne}

La \textbf{vitesse moyenne} entre les instants $t_1$ et $t_2$ est :
\[
  v_{\text{moy}} = \frac{d(t_2) - d(t_1)}{t_2 - t_1}
\]
où $d(t)$ est la distance parcourue à l'instant $t$.

Unités : si $d$ est en mètres et $t$ en secondes, la vitesse est en m/s.

%---------------------------------------------------------------

\begin{exercice}{1SPE-DERLOCAL-FR-R5-EX1}{1}{5}
Un marcheur parcourt $d(t) = 80t$ mètres en $t$ minutes.
\begin{enumerate}
  \item Calculer la distance parcourue entre $t=0$ et $t=5$.
  \item Calculer sa vitesse moyenne sur cet intervalle, en m/min.
  \item Calculer sa vitesse moyenne entre $t=3$ et $t=7$, en m/min.
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% d = lambda t: 80*t
% assert d(5) - d(0) == 400
% assert (d(5)-d(0))/(5-0) == 80
% assert (d(7)-d(3))/(7-3) == 80
% END-VERIFY

\begin{corrige}{1SPE-DERLOCAL-FR-R5-EX1}

\textbf{Question 1.}
$d(5) - d(0) = 400 - 0 = 400$ m.

\textbf{Question 2.}
$v = \dfrac{400}{5} = 80$ m/min.

\textbf{Question 3.}
$v = \dfrac{d(7)-d(3)}{7-3} = \dfrac{560 - 240}{4} = \dfrac{320}{4} = 80$ m/min.

\end{corrige}

\begin{exercice}{1SPE-DERLOCAL-FR-R5-EX2}{1}{6}
Un cycliste a pour position $d(t) = 5t^2$ mètres, $t$ en secondes.
\begin{enumerate}
  \item Calculer la vitesse moyenne entre $t=2$ et $t=4$.
  \item Calculer la vitesse moyenne entre $t=2$ et $t=2{,}1$.
  \item Calculer la vitesse moyenne entre $t=2$ et $t=2{,}01$.
  \item Que conjecturez-vous pour la vitesse instantanée à $t=2$ ?
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% d = lambda t: 5*t**2
% v1 = (d(4)-d(2))/(4-2)
% assert v1 == 30
% from sympy import Rational
% v2 = Rational(d(Rational(21,10))-d(2), Rational(21,10)-2)
% assert v2 == Rational(205, 10)
% v3 = Rational(d(Rational(201,100))-d(2), Rational(201,100)-2)
% assert v3 == Rational(2005, 100)
% END-VERIFY

\begin{corrige}{1SPE-DERLOCAL-FR-R5-EX2}

\textbf{Question 1.}
$v = \dfrac{d(4)-d(2)}{4-2} = \dfrac{80-20}{2} = 30$ m/s.

\textbf{Question 2.}
$v = \dfrac{d(2{,}1)-d(2)}{0{,}1} = \dfrac{22{,}05-20}{0{,}1} = \dfrac{2{,}05}{0{,}1} = 20{,}5$ m/s.

\textbf{Question 3.}
$v = \dfrac{d(2{,}01)-d(2)}{0{,}01} = \dfrac{20{,}2005-20}{0{,}01} = \dfrac{0{,}2005}{0{,}01} = 20{,}05$ m/s.

\commentaireMarge{Plus l'intervalle est petit, plus la vitesse moyenne se rapproche de $20$ m/s.}

\textbf{Question 4.} Les vitesses moyennes se rapprochent de $20$ m/s. On conjecture que la vitesse instantanée à $t=2$ est $20$ m/s (c'est le nombre dérivé $d'(2)$).

\end{corrige}

\begin{exercice}{1SPE-DERLOCAL-FR-R5-EX3}{1}{6}
Un objet chute librement : sa hauteur est $h(t) = 100 - 5t^2$ (en mètres, $t$ en secondes).
\begin{enumerate}
  \item Quelle est la hauteur initiale ?
  \item Calculer la vitesse moyenne de chute entre $t=0$ et $t=2$.
  \item Le signe de cette vitesse moyenne est-il cohérent ? Justifier.
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% h = lambda t: 100 - 5*t**2
% assert h(0) == 100
% v = (h(2)-h(0))/(2-0)
% assert v == -10
% END-VERIFY

\begin{corrige}{1SPE-DERLOCAL-FR-R5-EX3}

\textbf{Question 1.}
$h(0) = 100 - 0 = 100$ m.

\textbf{Question 2.}
$v = \dfrac{h(2)-h(0)}{2-0} = \dfrac{80 - 100}{2} = \dfrac{-20}{2} = -10$ m/s.

\textbf{Question 3.} La vitesse moyenne est négative : la hauteur diminue (l'objet descend), ce qui est cohérent avec une chute. Le taux de variation négatif traduit la décroissance de la fonction $h$.

\end{corrige}
