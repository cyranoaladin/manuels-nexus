\subsection*{Rappel — Droites dans le plan}

Le \textbf{coefficient directeur} (ou pente) de la droite passant par $A(x_A;y_A)$ et $B(x_B;y_B)$ est :
\[
  m = \frac{y_B - y_A}{x_B - x_A} \quad (x_A \neq x_B).
\]

L'équation de la droite de pente $m$ passant par $A(x_A;y_A)$ est :
\[
  y = m(x - x_A) + y_A.
\]

%---------------------------------------------------------------

\begin{exercice}{1SPE-DERLOCAL-FR-R3-EX1}{1}{5}
Calculer le coefficient directeur de la droite passant par les points suivants.
\begin{enumerate}
  \item $A(1;3)$ et $B(4;9)$
  \item $C(2;5)$ et $D(6;5)$
  \item $E(0;-1)$ et $F(3;8)$
\end{enumerate}
\end{exercice}

% m3 = (8-(-1))/(3-0)

\begin{corrige}{1SPE-DERLOCAL-FR-R3-EX1}

\textbf{Question 1.}
$m = \dfrac{9-3}{4-1} = \dfrac{6}{3} = 2.$

\textbf{Question 2.}
$m = \dfrac{5-5}{6-2} = \dfrac{0}{4} = 0.$ La droite est horizontale.

\textbf{Question 3.}
$m = \dfrac{8-(-1)}{3-0} = \dfrac{9}{3} = 3.$

\end{corrige}

\begin{exercice}{1SPE-DERLOCAL-FR-R3-EX2}{1}{6}
Écrire l'équation de la droite de pente $m$ passant par le point $P$ indiqué. Mettre sous la forme $y = mx + p$.
\begin{enumerate}
  \item $m = 2$, $P(1;3)$.
  \item $m = -3$, $P(2;7)$.
  \item $m = \dfrac{1}{2}$, $P(4;5)$.

\begin{corrige}{1SPE-DERLOCAL-FR-R3-EX2}

\commentaireMarge{On utilise $y = m(x - x_P) + y_P$ puis on développe.}

\textbf{Question 1.}
$y = 2(x - 1) + 3 = 2x - 2 + 3 = 2x + 1.$

\textbf{Question 2.}
$y = -3(x - 2) + 7 = -3x + 6 + 7 = -3x + 13.$

\textbf{Question 3.}
$y = \dfrac{1}{2}(x - 4) + 5 = \dfrac{x}{2} - 2 + 5 = \dfrac{x}{2} + 3.$

\end{corrige}

  \item Vérifier que le point $C(3;6)$ appartient à $\mathcal{D}$.

\end{exercice}

\begin{corrige}{1SPE-DERLOCAL-FR-R3-EX3}

\textbf{Question 1.}
$m = \dfrac{10-4}{5-2} = \dfrac{6}{3} = 2.$

\textbf{Question 2.}
$y = 2(x - 2) + 4 = 2x - 4 + 4 = 2x.$

\commentaireMarge{On vérifie en remplaçant $x$ par $3$ dans l'équation.}

\textbf{Question 3.}
$y(3) = 2 \times 3 = 6$. L'ordonnée de $C$ est bien $6$, donc $C \in \mathcal{D}$. \checkmark

\end{corrige}
