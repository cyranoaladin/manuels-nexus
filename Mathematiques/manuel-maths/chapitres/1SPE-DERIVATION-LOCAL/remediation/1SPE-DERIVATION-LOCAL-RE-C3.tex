\begin{exercice}{1SPE-DERLOCAL-RE-C3-EX1}{1}{7}
Sur un graphique, la tangente à la courbe de $f$ au point $A(3;2)$ passe par le point $B(5;6)$.
\begin{enumerate}
  \item Déterminer $f'(3)$.
  \item La tangente au point $C(1;4)$ est horizontale. Que vaut $f'(1)$ ?
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Le nombre dérivé $f'(a)$ est le coefficient directeur de la tangente. Utilise la formule de la pente avec les deux points.}

% BEGIN-VERIFY
% m = (6-2)/(5-3)
% assert m == 2
% END-VERIFY

\begin{corrige}{1SPE-DERLOCAL-RE-C3-EX1}

\commentaireMarge{$f'(a)$ = pente de la tangente en $a$.}

\textbf{Question 1.}
$f'(3) = \dfrac{6-2}{5-3} = \dfrac{4}{2} = 2.$

\textbf{Question 2.}
Une tangente horizontale a une pente nulle, donc $f'(1) = 0$.

\end{corrige}

\begin{exercice}{1SPE-DERLOCAL-RE-C3-EX2}{1}{8}
On sait que $f(2) = 5$ et $f'(2) = -1$.
\begin{enumerate}
  \item Tracer la tangente au point $A(2;5)$ de pente $-1$ (sur un repère).
  \item Lire graphiquement l'ordonnée de la tangente pour $x=4$.
  \item Vérifier par le calcul : $y(4) = f'(2)(4-2)+f(2)$.
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% y4 = -1*(4-2) + 5
% assert y4 == 3
% END-VERIFY

\begin{corrige}{1SPE-DERLOCAL-RE-C3-EX2}

\textbf{Question 1.} On place $A(2;5)$. La pente est $-1$ : pour $1$ unité vers la droite, on descend de $1$. On obtient le point $(3;4)$. On trace la droite passant par $A$ et $(3;4)$.

\textbf{Question 2.} Sur le graphique, en $x=4$, la tangente donne $y \approx 3$.

\textbf{Question 3.}
$y(4) = -1 \times (4-2) + 5 = -2 + 5 = 3.$ \checkmark

\end{corrige}

\begin{exercice}{1SPE-DERLOCAL-RE-C3-EX3}{1}{8}
Sur un graphique, on lit les informations suivantes pour la courbe de $f$ :
\begin{itemize}
  \item En $A(0;1)$, la tangente passe par $(1;3)$.
  \item En $B(2;4)$, la tangente passe par $(4;4)$.
  \item En $C(4;0)$, la tangente passe par $(6;-4)$.
\end{itemize}
Déterminer $f'(0)$, $f'(2)$ et $f'(4)$.
\end{exercice}

% BEGIN-VERIFY
% assert (3-1)/(1-0) == 2
% assert (4-4)/(4-2) == 0
% assert (-4-0)/(6-4) == -2
% END-VERIFY

\begin{corrige}{1SPE-DERLOCAL-RE-C3-EX3}

$f'(0) = \dfrac{3-1}{1-0} = 2.$

$f'(2) = \dfrac{4-4}{4-2} = 0.$ La tangente en $B$ est horizontale.

$f'(4) = \dfrac{-4-0}{6-4} = \dfrac{-4}{2} = -2.$

\end{corrige}
