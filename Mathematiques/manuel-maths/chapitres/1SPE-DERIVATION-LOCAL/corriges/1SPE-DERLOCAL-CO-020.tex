\begin{corrige}{1SPE-DERLOCAL-EX-020}

\textbf{Question 1.} On calcule :
\[
f(-1) = (-1)^2 + 1 = 1 + 1 = 2 \qquad \text{et} \qquad f'(-1) = 2 \times (-1) = -2.
\]
Donc $\boxed{f(-1) = 2}$ et $\boxed{f'(-1) = -2}$.

\textbf{Question 2.} L'équation de la tangente au point d'abscisse $a = -1$ est :
\[
y = f'(-1)\bigl(x - (-1)\bigr) + f(-1) = -2(x + 1) + 2.
\]
On développe :
\[
y = -2x - 2 + 2 = -2x.
\]
L'équation de la tangente est donc $\boxed{y = -2x}$.

\end{corrige}
