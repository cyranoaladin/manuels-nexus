\item Le taux de variation de $f$ entre $x=1$ et $x=5$ est :
\[
\tau = \frac{f(5) - f(1)}{5 - 1} = \frac{26 - 2}{4} = \frac{24}{4} = \boxed{6}

\[

\]
\end{enumerate}
