\begin{corrige}{1SPE-DERLOCAL-EX-042}
\textbf{1.} On calcule les hauteurs d'eau :
\[
h(0) = 12 - \frac{0^2}{3} = 12 \qquad h(3) = 12 - \frac{9}{3} = 12 - 3 = 9 \qquad h(6) = 12 - \frac{36}{3} = 12 - 12 = 0
\]

\textbf{2.} Le taux de variation de $h$ entre $t = 0$ et $t = 3$ :
\[
\tau = \frac{h(3) - h(0)}{3 - 0} = \frac{9 - 12}{3} = \frac{-3}{3} = -1
\]
En moyenne, le niveau d'eau baisse de $1$ mètre par heure entre $t = 0$ et $t = 3$.

\textbf{3.} On calcule $h(3 + s)$ :
\[
h(3+s) = 12 - \frac{(3+s)^2}{3} = 12 - \frac{9 + 6s + s^2}{3} = 12 - 3 - 2s - \frac{s^2}{3} = 9 - 2s - \frac{s^2}{3}
\]
Le taux de variation est :
\[
\frac{h(3+s) - h(3)}{s} = \frac{9 - 2s - \dfrac{s^2}{3} - 9}{s} = \frac{-2s - \dfrac{s^2}{3}}{s} = -2 - \frac{s}{3}
\]
En faisant tendre $s$ vers $0$ :
\[
h'(3) = \lim_{s \to 0} \left(-2 - \frac{s}{3}\right) = -2
\]

\textbf{4.} Comme $h'(3) = \boxed{-2} < 0$, le niveau d'eau descend à l'instant $t = 3$, à un rythme instantané de $2$ mètres par heure.
\end{corrige}
