\begin{fichemethode}{M5}{Approcher une image avec la tangente}
\quandUtiliser{Mots-clés : valeur approchée, voisinage de $a$, $a+h$, approximation linéaire.}
\pasApas{\begin{enumerate}\item J'écris le nombre cherché sous la forme $a+h$ avec $h$ petit.\item Je calcule $f(a)$ et $f'(a)$.\item Je calcule $f(a)+f'(a)h$.\item Je conclus avec $\approx$.\end{enumerate}}
\exempleRedige{Sachant $f(4)=5$ et $f'(4)=-2$, approchons $f(4{,}03)$. Ici $h=0{,}03$, donc $f(4{,}03)\approx5-2\times0{,}03=4{,}94$.}
\pieges{Employer $=$ ; oublier $h$ ; choisir un $h$ non proche de zéro sans avertissement.}
\verifier{Si $f'(a)<0$ et $h>0$, l'approximation doit être inférieure à $f(a)$.}
\sEntrainer{\refExos{M5}}
\end{fichemethode}
