\begin{exercice}{1SPE-EXPO-EX-039}{3}{15}
Soit $f$ la fonction définie sur $\mathbb{R}$ par $f(x) = (x-1)\,\mathrm{e}^{x}$.

\begin{enumerate}
\item Calculer $f'(x)$ et montrer que $f'(x) = x\,\mathrm{e}^{x}$.
\item Étudier le signe de $f'(x)$ et dresser le tableau de variations de $f$ sur $\mathbb{R}$.
\item Déterminer les limites de $f$ en $-\infty$ et en $+\infty$.
\item Calculer $f(0)$, $f(1)$ et $f(-1)$.
\item Dresser le tableau de variations complet de $f$ et tracer l'allure de la courbe représentative de $f$.
\end{enumerate}
\end{exercice}
