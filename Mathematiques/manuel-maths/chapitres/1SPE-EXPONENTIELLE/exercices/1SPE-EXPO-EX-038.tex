\begin{exercice}{1SPE-EXPO-EX-038}{2}{10}
Soit $f$ la fonction définie sur $\mathbb{R}$ par $f(x) = (x^2-1)\,\mathrm{e}^{x}$.
\begin{enumerate}
\item Calculer $f'(x)$.
\item Montrer que $f'(x) = (x^2+2x-1)\,\mathrm{e}^{x}$.
\item Résoudre $x^2 + 2x - 1 = 0$ et en déduire les extremums de $f$.
\end{enumerate}
\end{exercice}
