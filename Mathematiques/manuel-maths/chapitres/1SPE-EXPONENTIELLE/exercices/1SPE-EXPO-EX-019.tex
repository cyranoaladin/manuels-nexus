\begin{exercice}{1SPE-EXPO-EX-019}{3}{15}
On admet la propriété : pour tous réels $a$ et $b$, $\exp(a+b) = \exp(a) \times \exp(b)$.

Démontrer par récurrence que pour tout entier naturel $n \geqslant 1$ et pour tout réel $a$ :
\[ \exp(na) = \bigl[\exp(a)\bigr]^n. \]
\begin{enumerate}
  \item \textbf{Initialisation :} Vérifier la propriété pour $n=1$.
  \item \textbf{Hérédité :} On suppose que pour un certain entier $k \geqslant 1$, $\exp(ka) = \bigl[\exp(a)\bigr]^k$. Montrer que $\exp\bigl((k+1)a\bigr) = \bigl[\exp(a)\bigr]^{k+1}$.
  \item \textbf{Conclusion :} Rédiger la conclusion du raisonnement par récurrence.
  \item En déduire que pour tout entier naturel $n$ et tout réel $a$, $\exp(-na) = \dfrac{1}{\bigl[\exp(a)\bigr]^n}$.
\end{enumerate}
\end{exercice}
