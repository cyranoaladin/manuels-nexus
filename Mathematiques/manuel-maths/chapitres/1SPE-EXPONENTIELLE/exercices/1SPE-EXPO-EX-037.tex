\begin{exercice}{1SPE-EXPO-EX-037}{2}{10}
Soit $f$ la fonction définie sur $]-1\,;\,+\infty[$ par $f(x) = \dfrac{\mathrm{e}^{x}}{x+1}$.
\begin{enumerate}
\item Calculer $f'(x)$ en utilisant la formule du quotient $\left(\dfrac{u}{v}\right)' = \dfrac{u'v - uv'}{v^2}$.
\item Montrer que $f'(x) = \dfrac{x\,\mathrm{e}^{x}}{(x+1)^2}$.
\item Étudier le signe de $f'(x)$ sur $]-1\,;\,+\infty[$ et dresser le tableau de variations de $f$.
\end{enumerate}
\end{exercice}
