\begin{exercice}{1SPE-EXPO-EX-023}{1}{5}

\begin{enumerate}
  \item $\displaystyle\lim_{x \to +\infty} \mathrm{e}^x$
  \item $\displaystyle\lim_{x \to -\infty} \mathrm{e}^x$

\end{enumerate}
