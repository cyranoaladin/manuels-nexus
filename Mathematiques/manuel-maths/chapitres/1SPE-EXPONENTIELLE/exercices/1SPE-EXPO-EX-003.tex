\begin{exercice}{1SPE-EXPO-EX-003}{1}{5}
\begin{enumerate}
\item Calculer $\exp(0)$.
\item La fonction exponentielle peut-elle s'annuler ? Justifier en utilisant la propriété $\exp(x) > 0$ pour tout réel $x$.
\item Compléter : pour tout réel $x$, $\mathrm{e}^x \ldots 0$ (choisir parmi $>$, $\geqslant$, $<$, $\leqslant$, $=$).
\item En déduire que l'équation $\mathrm{e}^x = 0$ n'a aucune solution dans $\mathbb{R}$.
\end{enumerate}
\end{exercice}
