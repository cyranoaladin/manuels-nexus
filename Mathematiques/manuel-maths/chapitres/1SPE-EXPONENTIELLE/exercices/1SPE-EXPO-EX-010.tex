\begin{exercice}{1SPE-EXPO-EX-010}{3}{12}
On considère la fonction $f$ définie sur $\mathbb{R}$ par $f(x) = x\mathrm{e}^x - \mathrm{e}^x + 1$.
\begin{enumerate}
\item Calculer $f(0)$.
\item Calculer $f'(x)$ et vérifier que $f'(x) = x\mathrm{e}^x$.
\item En déduire $f'(0)$.
\item Calculer $f''(x)$ et montrer que $f''(x) = (x+1)\mathrm{e}^x$.
\item Calculer $f''(0)$. En déduire la nature du point critique $x = 0$.
\item Dresser le tableau de variations de $f$ et montrer que $f(x) \geqslant 0$ pour tout réel $x$.
\item En déduire l'inégalité $x\mathrm{e}^x \geqslant \mathrm{e}^x - 1$ pour tout réel $x$.
\end{enumerate}
\end{exercice}
