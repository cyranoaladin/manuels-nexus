\begin{exercice}{1SPE-EXPO-EX-027}{2}{8}
Calculer les limites suivantes :
\begin{enumerate}
  \item $\displaystyle\lim_{x \to +\infty} \left(\mathrm{e}^x - x\right)$
  \item $\displaystyle\lim_{x \to -\infty} \left(\mathrm{e}^x + x\right)$
\end{enumerate}
\textit{Indication pour la question 1 : on admettra que $\mathrm{e}^x$ croît plus vite que $x$ quand $x \to +\infty$.}
\end{exercice}
