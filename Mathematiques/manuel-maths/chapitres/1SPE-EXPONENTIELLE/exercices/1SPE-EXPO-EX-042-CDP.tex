\begin{coupdepouce}{1SPE-EXPO-EX-042}
\coupDePouce{1}{Pour résoudre $\mathrm{e}^x = k$ avec $k > 0$, la solution est $x = \ln k$.}
\coupDePouce{2}{Pour $\mathrm{e}^{2x} = 7$, commencer par écrire $2x = \ln 7$, puis isoler $x$.}
\coupDePouce{3}{Pour $\mathrm{e}^{-x} = 5$, écrire $-x = \ln 5$ et en déduire $x$.}
