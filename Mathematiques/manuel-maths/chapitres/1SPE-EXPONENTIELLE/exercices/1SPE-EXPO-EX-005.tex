\begin{exercice}{1SPE-EXPO-EX-005}{2}{10}
On suppose qu'une fonction $f$, dérivable sur $\mathbb{R}$, vérifie $f' = f$.

On pose $g(x) = f(x) \cdot \mathrm{e}^{-x}$.
\begin{enumerate}
\item Calculer $g'(x)$ en utilisant la formule de dérivation d'un produit.
\item En utilisant l'hypothèse $f'(x) = f(x)$, montrer que $g'(x) = 0$ pour tout réel $x$.
\item En déduire que $g$ est une fonction constante sur $\mathbb{R}$.
\item Montrer que $f(x) = f(0) \cdot \mathrm{e}^x$ pour tout réel $x$.
\item En déduire que si $f$ vérifie $f' = f$, alors $f$ ne s'annule jamais ou bien $f$ est la fonction nulle.
\end{enumerate}
\end{exercice}
