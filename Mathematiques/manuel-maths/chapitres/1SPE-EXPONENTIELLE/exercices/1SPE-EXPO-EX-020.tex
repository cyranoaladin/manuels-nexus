\begin{exercice}{1SPE-EXPO-EX-020}{3}{12}
On définit la fonction $f$ sur $\mathbb{R}$ par :
\[ f(x) = \frac{\mathrm{e}^x + \mathrm{e}^{-x}}{2}. \]
\begin{enumerate}
  \item Montrer que pour tout réel $x$, $f(-x) = f(x)$. Comment se traduit cette propriété graphiquement ?
  \item Calculer $f(0)$.
  \item Calculer $f'(x)$ et montrer que $f'(x) = \dfrac{\mathrm{e}^x - \mathrm{e}^{-x}}{2}$.
  \item Étudier le signe de $f'(x)$ et en déduire les variations de $f$ sur $\mathbb{R}$.
  \item Montrer que pour tout réel $x$, $\bigl[f(x)\bigr]^2 - \left[\dfrac{\mathrm{e}^x - \mathrm{e}^{-x}}{2}\right]^2 = 1$.
\end{enumerate}
\end{exercice}
