\begin{exercice}{1SPE-EXPO-EX-049}{3}{15}
On considère la fonction $f$ définie sur $\mathbb{R}$ par $f(x) = (2x - 1)\mathrm{e}^x$.
\begin{enumerate}
\item Calculer $f'(x)$ et montrer que $f'(x) = (2x + 1)\mathrm{e}^x$.
\item Étudier le signe de $f'(x)$ et dresser le tableau de variations de $f$.
\item Résoudre l'équation $f(x) = 0$.
\item Déterminer le signe de $f(x)$ sur $\mathbb{R}$.
\item Déterminer les limites de $f$ en $+\infty$ et en $-\infty$.
\end{enumerate}
\end{exercice}
