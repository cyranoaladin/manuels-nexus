\begin{exercice}{1SPE-EXPO-EX-016}{2}{8}
En utilisant la propriété $\exp(a+b) = \exp(a) \times \exp(b)$, démontrer les résultats suivants.
\begin{enumerate}
  \item Pour tout réel $a$, $\exp(-a) = \dfrac{1}{\exp(a)}$.
  \item Pour tous réels $a$ et $b$, $\exp(a-b) = \dfrac{\exp(a)}{\exp(b)}$.
  \item Pour tout réel $a$, $\exp(a) \times \exp(-a) = 1$.
\end{enumerate}
\end{exercice}
