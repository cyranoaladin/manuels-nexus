\subsection*{Rappel — Puissances entieres et regles de calcul}

Pour tout reel $a \neq 0$ et tous entiers relatifs $m$, $n$ :
\begin{itemize}
  \item $a^m \times a^n = a^{m+n}$ \quad (produit de puissances de meme base)
  \item $\dfrac{a^m}{a^n} = a^{m-n}$ \quad (quotient de puissances de meme base)
  \item $(a^m)^n = a^{m \times n}$ \quad (puissance d'une puissance)
  \item $a^0 = 1$ \quad et \quad $a^{-n} = \dfrac{1}{a^n}$
\end{itemize}

Ces regles sont exactement celles que l'on retrouve pour les exposants de l'exponentielle :
$\mathrm{e}^a \times \mathrm{e}^b = \mathrm{e}^{a+b}$, $\dfrac{\mathrm{e}^a}{\mathrm{e}^b} = \mathrm{e}^{a-b}$, $(\mathrm{e}^a)^n = \mathrm{e}^{na}$.

\begin{exercice}{1SPE-EXPO-FR-R4-EX1}{1}{5}
Simplifier en utilisant les regles des puissances.
\begin{enumerate}
  \item $2^3 \times 2^5$
  \item $\dfrac{5^7}{5^3}$
  \item $(3^2)^4$
  \item $7^0$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Produit : on additionne les exposants. Quotient : on soustrait. Puissance de puissance : on multiplie.}

% BEGIN-VERIFY
% assert 2**3 * 2**5 == 2**8 == 256
% assert 5**7 // 5**3 == 5**4 == 625
% assert (3**2)**4 == 3**8 == 6561
% assert 7**0 == 1
% END-VERIFY

\begin{corrige}{1SPE-EXPO-FR-R4-EX1}

\textbf{Question 1.} $2^3 \times 2^5 = 2^{3+5} = 2^8 = 256$.

\textbf{Question 2.} $\dfrac{5^7}{5^3} = 5^{7-3} = 5^4 = 625$.

\textbf{Question 3.} $(3^2)^4 = 3^{2 \times 4} = 3^8 = 6561$.

\textbf{Question 4.} $7^0 = 1$ (par convention pour tout reel non nul).

\end{corrige}

\begin{exercice}{1SPE-EXPO-FR-R4-EX2}{1}{6}
Ecrire sous la forme $a^n$ (une seule puissance).
\begin{enumerate}
  \item $\dfrac{10^5 \times 10^3}{10^2}$
  \item $\left(\dfrac{2^3}{2}\right)^2$
  \item $3^{-2} \times 3^5 \times 3^{-1}$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Traiter etape par etape : d'abord le numerateur, puis le quotient, puis la puissance exterieure s'il y en a une.}

% BEGIN-VERIFY
% assert (10**5 * 10**3) // 10**2 == 10**6
% assert (2**3 // 2)**2 == (2**2)**2 == 2**4 == 16
% assert 3**(-2) * 3**5 * 3**(-1) == 3**2 == 9
% END-VERIFY

\begin{corrige}{1SPE-EXPO-FR-R4-EX2}

\textbf{Question 1.}
\[
\frac{10^5 \times 10^3}{10^2} = \frac{10^{5+3}}{10^2} = \frac{10^8}{10^2} = 10^{8-2} = 10^6.
\]

\textbf{Question 2.}
\[
\left(\frac{2^3}{2}\right)^2 = \left(2^{3-1}\right)^2 = (2^2)^2 = 2^{2 \times 2} = 2^4 = 16.
\]

\textbf{Question 3.}
\[
3^{-2} \times 3^5 \times 3^{-1} = 3^{-2+5+(-1)} = 3^{2} = 9.
\]

\end{corrige}

\begin{exercice}{1SPE-EXPO-FR-R4-EX3}{2}{6}
Par analogie avec les puissances, simplifier les expressions suivantes (ecrire sous la forme $\mathrm{e}^k$).
\begin{enumerate}
  \item $\mathrm{e}^2 \times \mathrm{e}^3 \times \mathrm{e}^{-1}$
  \item $\dfrac{(\mathrm{e}^3)^2}{\mathrm{e}^4}$
  \item $\dfrac{\mathrm{e}^5 \times \mathrm{e}^{-2}}{(\mathrm{e}^2)^3}$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Les regles sont les memes : $\mathrm{e}^a \times \mathrm{e}^b = \mathrm{e}^{a+b}$, $(\mathrm{e}^a)^n = \mathrm{e}^{na}$, $\mathrm{e}^a / \mathrm{e}^b = \mathrm{e}^{a-b}$.}

% BEGIN-VERIFY
% from sympy import exp, simplify, E
% e = E
% assert simplify(e**2 * e**3 * e**(-1) - e**4) == 0
% assert simplify((e**3)**2 / e**4 - e**2) == 0
% assert simplify(e**5 * e**(-2) / (e**2)**3 - e**(-3)) == 0
% END-VERIFY

\begin{corrige}{1SPE-EXPO-FR-R4-EX3}

\commentaireMarge{On additionne les exposants pour le produit, on multiplie pour la puissance.}

\textbf{Question 1.}
\[
\mathrm{e}^2 \times \mathrm{e}^3 \times \mathrm{e}^{-1} = \mathrm{e}^{2+3+(-1)} = \mathrm{e}^{4}.
\]

\textbf{Question 2.}
\[
\frac{(\mathrm{e}^3)^2}{\mathrm{e}^4} = \frac{\mathrm{e}^{6}}{\mathrm{e}^{4}} = \mathrm{e}^{6-4} = \mathrm{e}^{2}.
\]

\textbf{Question 3.}
\[
\frac{\mathrm{e}^5 \times \mathrm{e}^{-2}}{(\mathrm{e}^2)^3} = \frac{\mathrm{e}^{5+(-2)}}{\mathrm{e}^{6}} = \frac{\mathrm{e}^{3}}{\mathrm{e}^{6}} = \mathrm{e}^{3-6} = \mathrm{e}^{-3}.
\]

\end{corrige}
