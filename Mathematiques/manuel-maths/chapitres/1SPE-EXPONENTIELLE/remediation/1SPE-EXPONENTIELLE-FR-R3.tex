\subsection*{Rappel — Resolution d'equations et d'inequations du second degre}

Soit $ax^2 + bx + c = 0$ avec $a \neq 0$. On calcule le discriminant $\Delta = b^2 - 4ac$.
\begin{itemize}
  \item Si $\Delta > 0$ : deux solutions $x_1 = \dfrac{-b - \sqrt{\Delta}}{2a}$ et $x_2 = \dfrac{-b + \sqrt{\Delta}}{2a}$.
  \item Si $\Delta = 0$ : une solution double $x_0 = \dfrac{-b}{2a}$.
  \item Si $\Delta < 0$ : pas de solution reelle.
\end{itemize}

Pour les inequations : utiliser le signe du trinome (meme signe que $a$ a l'exterieur des racines, signe oppose entre les racines).

\begin{exercice}{1SPE-EXPO-FR-R3-EX1}{1}{6}
Resoudre chaque equation dans $\mathbb{R}$.
\begin{enumerate}
  \item $x^2 - 5x + 6 = 0$
  \item $2x^2 + 3x - 2 = 0$
  \item $x^2 + 4x + 5 = 0$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Calculer $\Delta = b^2 - 4ac$. Si $\Delta > 0$, utiliser la formule des racines. Si $\Delta < 0$, conclure directement.}

\begin{corrige}{1SPE-EXPO-FR-R3-EX1}

\commentaireMarge{$\Delta = b^2 - 4ac$.}

\textbf{Question 1.} $a = 1$, $b = -5$, $c = 6$.
$\Delta = 25 - 24 = 1 > 0$.
\[
x_1 = \frac{5 - 1}{2} = 2, \qquad x_2 = \frac{5 + 1}{2} = 3.
\]
Solutions : $\{2\,;\,3\}$.

\textbf{Question 2.} $a = 2$, $b = 3$, $c = -2$.
$\Delta = 9 + 16 = 25 > 0$.
\[
x_1 = \frac{-3 - 5}{4} = -2, \qquad x_2 = \frac{-3 + 5}{4} = \frac{1}{2}.
\]
Solutions : $\left\{-2\,;\,\frac{1}{2}\right\}$.

\textbf{Question 3.} $a = 1$, $b = 4$, $c = 5$.
$\Delta = 16 - 20 = -4 < 0$.
Pas de solution reelle.

\end{corrige}

\begin{exercice}{1SPE-EXPO-FR-R3-EX2}{1}{8}
Resoudre les inequations suivantes dans $\mathbb{R}$.
\begin{enumerate}
  \item $x^2 - 4 \geqslant 0$
  \item $-x^2 + 2x + 3 > 0$
  \item $x^2 + 2x + 1 \leqslant 0$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Pour chaque inequation, trouver d'abord les racines du trinome, puis utiliser la regle du signe (meme signe que $a$ a l'exterieur des racines).}

\begin{corrige}{1SPE-EXPO-FR-R3-EX2}

\textbf{Question 1.} $x^2 - 4 = (x-2)(x+2)$. Racines : $x = -2$ et $x = 2$.

Comme $a = 1 > 0$, le trinome est positif a l'exterieur des racines.

Solution : $x \in \,]-\infty\,;\,-2] \cup [2\,;\,+\infty[$.

\textbf{Question 2.} $-x^2 + 2x + 3 = 0$. $\Delta = 4 + 12 = 16$. Racines : $x = -1$ et $x = 3$.

Comme $a = -1 < 0$, le trinome est \textbf{positif entre les racines}.

\commentaireMarge{Strictement positif : les bornes sont exclues.}

Solution : $x \in \,]-1\,;\,3[$.

\textbf{Question 3.} $x^2 + 2x + 1 = (x+1)^2$. C'est un carre, toujours $\geqslant 0$.

$(x+1)^2 \leqslant 0$ si et seulement si $(x+1)^2 = 0$, c'est-a-dire $x = -1$.

Solution : $\{-1\}$.

\end{corrige}

\begin{exercice}{1SPE-EXPO-FR-R3-EX3}{2}{8}
Resoudre dans $\mathbb{R}$ :
\begin{enumerate}
  \item $x^2 - 4 = 0$ \quad (utile pour $\mathrm{e}^{x^2-4} = 1$)
  \item $x^2 - x - 6 = 0$ \quad (utile pour $\mathrm{e}^{x^2-x-6} = 1$)
  \item $2x^2 - 8x + 6 = 0$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Pour la question 1 : $x^2 = 4$ donne $x = 2$ ou $x = -2$. Ne pas oublier la racine negative !}

\begin{corrige}{1SPE-EXPO-FR-R3-EX3}

\textbf{Question 1.} $x^2 - 4 = 0 \Leftrightarrow x^2 = 4 \Leftrightarrow x = -2$ ou $x = 2$.

Solutions : $\{-2\,;\,2\}$.

\textbf{Question 2.} $\Delta = 1 + 24 = 25$. $x_1 = \frac{1-5}{2} = -2$, $x_2 = \frac{1+5}{2} = 3$.

Solutions : $\{-2\,;\,3\}$.

\textbf{Question 3.} On simplifie par $2$ : $x^2 - 4x + 3 = 0$. $\Delta = 16 - 12 = 4$.

$x_1 = \frac{4-2}{2} = 1$, $x_2 = \frac{4+2}{2} = 3$.

Solutions : $\{1\,;\,3\}$.

\end{corrige}
