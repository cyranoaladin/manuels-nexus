\begin{corrige}{1SPE-EXPO-RE-C4-EX1}

\commentaireMarge{Formule : $(\mathrm{e}^u)' = u' \times \mathrm{e}^u$.}

\textbf{Question 1.} $u(x) = 5x$, $u'(x) = 5$.
\[
f'(x) = 5\,\mathrm{e}^{5x}.
\]

\textbf{Question 2.} $u(x) = -2x + 3$, $u'(x) = -2$.
\[
g'(x) = -2\,\mathrm{e}^{-2x+3}.
\]

\textbf{Question 3.} $u(x) = x^2 - 1$, $u'(x) = 2x$.
\[
h'(x) = 2x\,\mathrm{e}^{x^2-1}.
\]

\end{corrige}

\begin{exercice}{1SPE-EXPO-RE-C4-EX2}{2}{8}
Calculer la derivee de chaque fonction (produit avec exponentielle).
\begin{enumerate}
  \item $f(x) = (2x + 1)\,\mathrm{e}^{x}$
  \item $g(x) = x^2\,\mathrm{e}^{-x}$
  \item $h(x) = (x - 3)\,\mathrm{e}^{2x}$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Utiliser la formule du produit $(uv)' = u'v + uv'$. Par exemple pour $f$ : $u = 2x+1$, $v = \mathrm{e}^x$, $u' = 2$, $v' = \mathrm{e}^x$.}

% fp = diff(f, x)

% g = x**2 * exp(-x)
% gp = diff(g, x)
% assert simplify(gp - (2*x - x**2)*exp(-x)) == 0
% h = (x - 3)*exp(2*x)
% hp = diff(h, x)
% assert simplify(hp - (2*x - 5)*exp(2*x)) == 0
% END-VERIFY

\begin{corrige}{1SPE-EXPO-RE-C4-EX2}

\commentaireMarge{$(uv)' = u'v + uv'$.}

\textbf{Question 1.} $u = 2x+1$, $v = \mathrm{e}^x$, $u' = 2$, $v' = \mathrm{e}^x$.
\[
f'(x) = 2\,\mathrm{e}^x + (2x+1)\,\mathrm{e}^x = (2 + 2x + 1)\,\mathrm{e}^x = (2x + 3)\,\mathrm{e}^x.
\]

\textbf{Question 2.} $u = x^2$, $v = \mathrm{e}^{-x}$, $u' = 2x$, $v' = -\mathrm{e}^{-x}$.
\[
g'(x) = 2x\,\mathrm{e}^{-x} + x^2 \times (-\mathrm{e}^{-x}) = (2x - x^2)\,\mathrm{e}^{-x} = x(2-x)\,\mathrm{e}^{-x}.
\]

\textbf{Question 3.} $u = x-3$, $v = \mathrm{e}^{2x}$, $u' = 1$, $v' = 2\,\mathrm{e}^{2x}$.
\[
h'(x) = 1 \times \mathrm{e}^{2x} + (x-3) \times 2\,\mathrm{e}^{2x} = (1 + 2x - 6)\,\mathrm{e}^{2x} = (2x - 5)\,\mathrm{e}^{2x}.
\]

\end{corrige}

\begin{corrige}{1SPE-EXPO-RE-C4-EX3}

\textbf{Question 1.} $u = x^2 - 4$, $v = \mathrm{e}^x$, $u' = 2x$, $v' = \mathrm{e}^x$.
\[
f'(x) = 2x\,\mathrm{e}^x + (x^2 - 4)\,\mathrm{e}^x = (x^2 + 2x - 4)\,\mathrm{e}^x.
\]

\textbf{Question 2.} Comme $\mathrm{e}^x > 0$, $f'(x) = 0 \Leftrightarrow x^2 + 2x - 4 = 0$.

$\Delta = 4 + 16 = 20$, $\sqrt{\Delta} = 2\sqrt{5}$.
\[
x_1 = \frac{-2 - 2\sqrt{5}}{2} = -1 - \sqrt{5} \approx -3{,}24, \qquad x_2 = -1 + \sqrt{5} \approx 1{,}24.
\]

\commentaireMarge{$\mathrm{e}^x > 0$ pour tout $x$, donc le signe de $f'$ est celui de $x^2+2x-4$.}

\textbf{Question 3.} Le trinome $x^2 + 2x - 4$ (avec $a = 1 > 0$) est :
\begin{itemize}
  \item positif pour $x < -1 - \sqrt{5}$,
  \item negatif pour $-1-\sqrt{5} < x < -1+\sqrt{5}$,
  \item positif pour $x > -1 + \sqrt{5}$.
\end{itemize}

$f$ est strictement croissante sur $]-\infty\,;\,-1-\sqrt{5}]$, strictement decroissante sur $[-1-\sqrt{5}\,;\,-1+\sqrt{5}]$, et strictement croissante sur $[-1+\sqrt{5}\,;\,+\infty[$.

\end{corrige}
