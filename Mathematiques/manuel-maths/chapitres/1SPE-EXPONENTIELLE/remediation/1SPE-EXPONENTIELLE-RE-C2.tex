\begin{exercice}{1SPE-EXPO-RE-C2-EX1}{1}{6}
Simplifier chaque expression sous la forme $\mathrm{e}^k$ ou $k$ est un entier.
\begin{enumerate}
  \item $\mathrm{e}^4 \times \mathrm{e}^{-1}$
  \item $\dfrac{\mathrm{e}^9}{\mathrm{e}^5}$
  \item $\left(\mathrm{e}^{-2}\right)^3$
  \item $\mathrm{e}^3 \times \mathrm{e}^{-3}$
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-EXPO-RE-C2-EX1}

\commentaireMarge{Regle : $\mathrm{e}^a \times \mathrm{e}^b = \mathrm{e}^{a+b}$.}

\textbf{Question 1.} $\mathrm{e}^4 \times \mathrm{e}^{-1} = \mathrm{e}^{4+(-1)} = \mathrm{e}^3$.

\commentaireMarge{Regle : $\mathrm{e}^a / \mathrm{e}^b = \mathrm{e}^{a-b}$.}

\textbf{Question 2.} $\dfrac{\mathrm{e}^9}{\mathrm{e}^5} = \mathrm{e}^{9-5} = \mathrm{e}^4$.

\commentaireMarge{Regle : $(\mathrm{e}^a)^n = \mathrm{e}^{na}$.}

\textbf{Question 3.} $\left(\mathrm{e}^{-2}\right)^3 = \mathrm{e}^{(-2) \times 3} = \mathrm{e}^{-6}$.

\textbf{Question 4.} $\mathrm{e}^3 \times \mathrm{e}^{-3} = \mathrm{e}^{3+(-3)} = \mathrm{e}^0 = 1$.

\end{corrige}

\begin{exercice}{1SPE-EXPO-RE-C2-EX2}{1}{8}
Simplifier les expressions suivantes (les exposants sont des expressions en $x$).
\begin{enumerate}
  \item $\mathrm{e}^{x+1} \times \mathrm{e}^{x-1}$
  \item $\dfrac{\mathrm{e}^{3x}}{\mathrm{e}^{x}}$
  \item $\left(\mathrm{e}^{x}\right)^2 \times \mathrm{e}^{-2x}$
  \item $\dfrac{\mathrm{e}^{2x} \times \mathrm{e}^{-x}}{\mathrm{e}^{x+3}}$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{Appliquer les memes regles que pour les nombres : additionner les exposants pour un produit, soustraire pour un quotient. Simplifier l'expression algebrique obtenue.}

\begin{corrige}{1SPE-EXPO-RE-C2-EX2}

\textbf{Question 1.}
\[
\mathrm{e}^{x+1} \times \mathrm{e}^{x-1} = \mathrm{e}^{(x+1)+(x-1)} = \mathrm{e}^{2x}.
\]

\textbf{Question 2.}
\[
\frac{\mathrm{e}^{3x}}{\mathrm{e}^{x}} = \mathrm{e}^{3x - x} = \mathrm{e}^{2x}.
\]

\textbf{Question 3.}
\[
\left(\mathrm{e}^{x}\right)^2 \times \mathrm{e}^{-2x} = \mathrm{e}^{2x} \times \mathrm{e}^{-2x} = \mathrm{e}^{2x - 2x} = \mathrm{e}^0 = 1.
\]

\textbf{Question 4.}
\[
\frac{\mathrm{e}^{2x} \times \mathrm{e}^{-x}}{\mathrm{e}^{x+3}} = \frac{\mathrm{e}^{2x-x}}{\mathrm{e}^{x+3}} = \frac{\mathrm{e}^{x}}{\mathrm{e}^{x+3}} = \mathrm{e}^{x-(x+3)} = \mathrm{e}^{-3}.
\]

\end{corrige}

\begin{exercice}{1SPE-EXPO-RE-C2-EX3}{2}{8}
Ecrire sous la forme $A \times \mathrm{e}^{f(x)}$ la plus simple possible.
\begin{enumerate}
  \item $3\mathrm{e}^{2x} \times 2\mathrm{e}^{x}$
  \item $\dfrac{4\mathrm{e}^{5x}}{2\mathrm{e}^{2x}}$
  \item $\left(2\mathrm{e}^{x}\right)^3$
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-EXPO-RE-C2-EX3}

\commentaireMarge{On separe les coefficients numeriques et les exponentielles.}

\textbf{Question 1.}
\[
3\mathrm{e}^{2x} \times 2\mathrm{e}^{x} = (3 \times 2) \times \mathrm{e}^{2x+x} = 6\mathrm{e}^{3x}.
\]

\textbf{Question 2.}
\[
\frac{4\mathrm{e}^{5x}}{2\mathrm{e}^{2x}} = \frac{4}{2} \times \mathrm{e}^{5x-2x} = 2\mathrm{e}^{3x}.
\]

\textbf{Question 3.}
\[
\left(2\mathrm{e}^{x}\right)^3 = 2^3 \times \left(\mathrm{e}^{x}\right)^3 = 8\mathrm{e}^{3x}.
\]

\end{corrige}
