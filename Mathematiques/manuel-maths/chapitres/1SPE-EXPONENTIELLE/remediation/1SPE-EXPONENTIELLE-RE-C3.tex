\begin{exercice}{1SPE-EXPO-RE-C3-EX1}{1}{6}
\begin{enumerate}
  \item Justifier que la fonction exponentielle est strictement croissante sur $\mathbb{R}$.
  \item Donner $\displaystyle\lim_{x \to +\infty} \mathrm{e}^x$ et $\displaystyle\lim_{x \to -\infty} \mathrm{e}^x$.
  \item La droite $y = 0$ est-elle asymptote a la courbe de $\exp$ ? Si oui, en $+\infty$ ou en $-\infty$ ?
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% from sympy import exp, diff, limit, symbols, oo
% x = symbols('x')
% # exp'(x) = exp(x) > 0
% assert diff(exp(x), x) == exp(x)
% assert limit(exp(x), x, oo) == oo
% assert limit(exp(x), x, -oo) == 0
% END-VERIFY

\begin{corrige}{1SPE-EXPO-RE-C3-EX1}

\commentaireMarge{$\exp'(x) = \exp(x) > 0$ pour tout $x$.}

\textbf{Question 1.} La derivee de $\exp$ est $\exp'(x) = \exp(x)$. Or $\exp(x) > 0$ pour tout $x \in \mathbb{R}$, donc $\exp$ est strictement croissante sur $\mathbb{R}$.

\textbf{Question 2.}
\[
\lim_{x \to +\infty} \mathrm{e}^x = +\infty \qquad \text{et} \qquad \lim_{x \to -\infty} \mathrm{e}^x = 0.
\]

\textbf{Question 3.} Oui, la droite $y = 0$ (l'axe des abscisses) est asymptote horizontale a la courbe en $-\infty$, car $\displaystyle\lim_{x \to -\infty} \mathrm{e}^x = 0$.

\end{corrige}

% EXERCICE 2 — Avec coup de pouce
%---------------------------------------------------------------

\begin{exercice}{1SPE-EXPO-RE-C3-EX2}{1}{6}
Comparer sans calculatrice les nombres suivants, en utilisant la croissance de l'exponentielle.
\begin{enumerate}
  \item $\mathrm{e}^3$ et $\mathrm{e}^{\pi}$
  \item $\mathrm{e}^{-1}$ et $\mathrm{e}^{-2}$
  \item $\mathrm{e}^{\sqrt{5}}$ et $\mathrm{e}^{2{,}3}$
\end{enumerate}
\end{exercice}

\coupDePouce{1}{La fonction $\exp$ est strictement croissante : si $a < b$ alors $\mathrm{e}^a < \mathrm{e}^b$. Il suffit de comparer les exposants.}

% BEGIN-VERIFY
% from sympy import exp, pi, sqrt, Rational, N
% # (1) 3 < pi, donc e^3 < e^pi
% assert 3 < pi
% assert exp(3) < exp(pi)
% # (2) -2 < -1, donc e^(-2) < e^(-1)
% assert -2 < -1
% assert exp(-2) < exp(-1)
% # (3) sqrt(5) approx 2.236 < 2.3, donc e^sqrt(5) < e^2.3
% assert float(sqrt(5)) < 2.3
% assert exp(sqrt(5)) < exp(Rational(23, 10))
% END-VERIFY

\begin{corrige}{1SPE-EXPO-RE-C3-EX2}

\commentaireMarge{Si $a < b$, alors $\mathrm{e}^a < \mathrm{e}^b$ (croissance stricte).}

\textbf{Question 1.} On sait que $3 < \pi$ (car $\pi \approx 3{,}14$). Par croissance de $\exp$ :
\[
\mathrm{e}^3 < \mathrm{e}^{\pi}.
\]

\textbf{Question 2.} $-2 < -1$, donc par croissance de $\exp$ :
\[
\mathrm{e}^{-2} < \mathrm{e}^{-1}.
\]

\textbf{Question 3.} $\sqrt{5} \approx 2{,}236$ et $2{,}3 > 2{,}236$, donc $\sqrt{5} < 2{,}3$.

Par croissance de $\exp$ : $\mathrm{e}^{\sqrt{5}} < \mathrm{e}^{2{,}3}$.

\end{corrige}

\begin{exercice}{1SPE-EXPO-RE-C3-EX3}{2}{8}
Soit $f(x) = \mathrm{e}^x - x - 1$ definie sur $\mathbb{R}$.
\begin{enumerate}
  \item Calculer $f(0)$.
  \item Calculer $f'(x)$ et montrer que $f'(x) = 0$ pour $x = 0$.
  \item Etudier le signe de $f'(x)$ et en deduire le tableau de variations de $f$.
  \item En deduire que pour tout $x \in \mathbb{R}$, $\mathrm{e}^x \geqslant x + 1$.
\end{enumerate}
\end{exercice}

% BEGIN-VERIFY
% from sympy import exp, diff, symbols, solve, oo, limit
% x = symbols('x')
% f = exp(x) - x - 1
% assert f.subs(x, 0) == 0
% fp = diff(f, x)

% # f(0) = 0 est le minimum, donc f(x) >= 0 pour tout x
% assert f.subs(x, 0) == 0
% assert limit(f, x, oo) == oo
% assert limit(f, x, -oo) == oo
% END-VERIFY

\begin{corrige}{1SPE-EXPO-RE-C3-EX3}

\textbf{Question 1.} $f(0) = \mathrm{e}^0 - 0 - 1 = 1 - 0 - 1 = 0$.

\textbf{Question 2.} $f'(x) = \mathrm{e}^x - 1$. En $x = 0$ : $f'(0) = \mathrm{e}^0 - 1 = 1 - 1 = 0$.

\commentaireMarge{$\mathrm{e}^x - 1 < 0 \Leftrightarrow \mathrm{e}^x < 1 \Leftrightarrow x < 0$.}

\textbf{Question 3.}
\begin{itemize}
  \item $f'(x) < 0$ pour $x < 0$ (car $\mathrm{e}^x < 1$),
  \item $f'(0) = 0$,
  \item $f'(x) > 0$ pour $x > 0$ (car $\mathrm{e}^x > 1$).
\end{itemize}

$f$ est strictement decroissante sur $]-\infty\,;\,0]$ et strictement croissante sur $[0\,;\,+\infty[$.

Le minimum de $f$ est $f(0) = 0$, atteint en $x = 0$.

\textbf{Question 4.} Comme $f(x) \geqslant f(0) = 0$ pour tout $x$, on a :
\[
\mathrm{e}^x - x - 1 \geqslant 0 \quad \text{soit} \quad \mathrm{e}^x \geqslant x + 1 \quad \text{pour tout } x \in \mathbb{R}.
\]

\end{corrige}
