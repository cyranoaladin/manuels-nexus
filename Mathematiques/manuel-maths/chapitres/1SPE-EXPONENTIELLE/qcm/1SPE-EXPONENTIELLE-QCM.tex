\subsection*{C1 — Definition de l'exponentielle, exp(0)=1, exp toujours positive}
% ------------------------------------------------------------

\begin{enumerate}

% ---- Q1 ----
\item \textbf{[Q1]} Parmi les proprietes suivantes, laquelle caracterise la fonction
  exponentielle ?

  \begin{enumerate}[label=\Alph*.]
    \item C'est la seule fonction derivable sur $\mathbb{R}$ telle que $f'(x) = f(x)$
      pour tout $x$ et $f(1) = \mathrm{e}$.
    \item C'est la seule fonction derivable sur $\mathbb{R}$ telle que $f'(x) = f(x)$
      pour tout $x$ et $f(0) = 1$.
    \item C'est la seule fonction derivable sur $\mathbb{R}$ telle que $f'(x) = f(x)$
      pour tout $x$ et $f(0) = 0$.
    \item C'est la seule fonction derivable sur $\mathbb{R}$ telle que $f'(x) = x f(x)$
      pour tout $x$ et $f(0) = 1$.
  \end{enumerate}

\bigskip

% ---- Q2 ----
\item \textbf{[Q2]} Quelle est la valeur de $\mathrm{e}^0$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $0$
    \item $\mathrm{e}$
    \item $1$
    \item La valeur n'est pas definie.
  \end{enumerate}

\bigskip

% ---- Q3 ----
\item \textbf{[Q3]} Pour tout reel $x$, on a :

  \begin{enumerate}[label=\Alph*.]
    \item $\mathrm{e}^x > 0$
    \item $\mathrm{e}^x \geqslant 0$ (avec $\mathrm{e}^x = 0$ possible)
    \item $\mathrm{e}^x$ peut etre negatif si $x < 0$
    \item $\mathrm{e}^x > 1$ pour tout $x \in \mathbb{R}$
  \end{enumerate}

\end{enumerate}

  \end{enumerate}

% ------------------------------------------------------------
\subsection*{C3 — Variations, limites, comparaison}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q7 ----
\item \textbf{[Q7]} La fonction exponentielle est :

  \begin{enumerate}[label=\Alph*.]
    \item strictement decroissante sur $\mathbb{R}$
    \item strictement croissante sur $\mathbb{R}$
    \item croissante sur $]0\,;+\infty[$ et decroissante sur $]-\infty\,;0[$
    \item constante sur $\mathbb{R}$
  \end{enumerate}

\bigskip

% ---- Q8 ----
\item \textbf{[Q8]} Quelle est la limite de $\mathrm{e}^x$ lorsque $x \to +\infty$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $0$
    \item $1$
    \item $\mathrm{e}$
    \item $+\infty$
  \end{enumerate}

\bigskip

% ---- Q9 ----
\item \textbf{[Q9]} Sachant que la fonction exponentielle est strictement croissante,
  comparer $\mathrm{e}^{2{,}5}$ et $\mathrm{e}^{\sqrt{7}}$.

  \begin{enumerate}[label=\Alph*.]
    \item $\mathrm{e}^{2{,}5} > \mathrm{e}^{\sqrt{7}}$
    \item $\mathrm{e}^{2{,}5} < \mathrm{e}^{\sqrt{7}}$
    \item $\mathrm{e}^{2{,}5} = \mathrm{e}^{\sqrt{7}}$
    \item On ne peut pas comparer sans calculatrice.
  \end{enumerate}

\end{enumerate}

\subsection*{C4 — Derivation de $\mathrm{e}^{u(x)}$}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q10 ----
\item \textbf{[Q10]} Soit $f(x) = \mathrm{e}^{3x+1}$. Quelle est $f'(x)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\mathrm{e}^{3x+1}$
    \item $(3x+1)\,\mathrm{e}^{3x+1}$
    \item $3\,\mathrm{e}^{3x+1}$
    \item $3\,\mathrm{e}^{3}$
  \end{enumerate}

\bigskip

% ---- Q11 ----
\item \textbf{[Q11]} Soit $g(x) = \mathrm{e}^{x^2}$. Quelle est $g'(x)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\mathrm{e}^{x^2}$
    \item $x^2\,\mathrm{e}^{x^2}$
    \item $2x\,\mathrm{e}^{x^2}$
    \item $2x\,\mathrm{e}^{2x}$
  \end{enumerate}

\bigskip

% ---- Q12 ----
\item \textbf{[Q12]} Soit $h(x) = x\,\mathrm{e}^{x}$. Quelle est $h'(x)$ ?

  \begin{enumerate}[label=\Alph*.]
    \item $\mathrm{e}^x$
    \item $x\,\mathrm{e}^x$
    \item $(1+x)\,\mathrm{e}^x$
    \item $(x-1)\,\mathrm{e}^x$
  \end{enumerate}

\end{enumerate}

% ------------------------------------------------------------
\subsection*{C5 — Equations et inequations avec l'exponentielle}
% ------------------------------------------------------------

\begin{enumerate}[resume]

% ---- Q13 ----
\item \textbf{[Q13]} Resoudre $\mathrm{e}^{x} = \mathrm{e}^{5}$.

  \begin{enumerate}[label=\Alph*.]
    \item $x = \mathrm{e}^5$
    \item $x = 5$
    \item $x = \ln 5$
    \item Pas de solution.
  \end{enumerate}

\bigskip

% ---- Q14 ----
\item \textbf{[Q14]} Resoudre l'inequation $\mathrm{e}^{2x-1} \geqslant 1$.

  \begin{enumerate}[label=\Alph*.]
    \item $x \geqslant 1$
    \item $x \geqslant \dfrac{1}{2}$
    \item $x \leqslant \dfrac{1}{2}$
    \item $x \geqslant 0$
  \end{enumerate}

\bigskip

% ---- Q15 ----
\item \textbf{[Q15]} Resoudre $\mathrm{e}^{x^2 - 4} = 1$.

  \begin{enumerate}[label=\Alph*.]
    \item $x = 4$
    \item $x = 2$
    \item $x = -2$ ou $x = 2$
    \item $x = 0$
  \end{enumerate}

\end{enumerate}
