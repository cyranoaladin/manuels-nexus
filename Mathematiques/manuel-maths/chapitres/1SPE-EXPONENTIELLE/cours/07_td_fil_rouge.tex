\section*{TD fil rouge — Des intérêts composés au nombre $\mathrm{e}$}

\textit{Durée : 50 minutes — Toutes les capacités C1 à C5 sont mobilisées.}

\bigskip

\begin{center}
\textbf{Accroche}

\medskip
\parbox{0.9\textwidth}{On place un capital de $1\,000$~\euro{} sur un compte rémunéré au taux annuel de $100\,\%$. Si les intérêts sont calculés une seule fois en fin d'année, le capital double et vaut $2\,000$~\euro{}. Mais que se passe-t-il si les intérêts sont calculés plus fréquemment : chaque mois, chaque jour, chaque seconde ? Le capital peut-il croître sans limite ?}
\end{center}

\bigskip

\subsection*{Partie A — Intérêts composés et découverte de $\mathrm{e}$ (C1) — 10 min}
%---------------------------------------------------------------

\begin{enumerate}

  \item Calculer $C_n$ pour $n = 1$, $n = 2$, $n = 4$, $n = 12$, $n = 365$.
  \item Conjecturer la valeur de $\displaystyle\lim_{n \to +\infty} C_n$. Que vaut $\displaystyle\lim_{n \to +\infty} \left(1 + \frac{1}{n}\right)^n$ ?
\end{enumerate}

\subsection*{Partie D — Temps de doublement et synthèse (C5) — 15 min}

\end{enumerate}
