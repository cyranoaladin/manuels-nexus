\section{Propriétés algébriques de la fonction exponentielle}

\margeAppui{La relation fonctionnelle $\exp(a+b) = \exp(a) \times \exp(b)$ est le résultat central : elle transforme une somme en produit. C'est cette propriété qui justifie la notation $\mathrm{e}^x$.}

\theoreme[Relation fonctionnelle — Démonstration exigible]{
Pour tous réels $a$ et $b$ :
\[
\exp(a + b) = \exp(a) \times \exp(b).
\]
}

\demonstration{
Soit $a \in \mathbb{R}$ fixé. Posons $f(x) = \exp(x + a)$ et $g(x) = \exp(a) \times \exp(x)$.

\textbf{Étape 1 : $f$ et $g$ vérifient la même équation différentielle.}

$f'(x) = \exp(x + a) = f(x)$ \quad (car $\exp' = \exp$).

$g'(x) = \exp(a) \times \exp(x) = g(x)$ \quad (car $\exp(a)$ est une constante et $\exp' = \exp$).

\textbf{Étape 2 : $f$ et $g$ ont la même valeur en $0$.}

$f(0) = \exp(0 + a) = \exp(a)$.

$g(0) = \exp(a) \times \exp(0) = \exp(a) \times 1 = \exp(a)$.

\textbf{Conclusion.} Considérons $h = \dfrac{f}{g}$. On a $g > 0$ sur $\mathbb{R}$ donc $h$ est bien définie.

$h'(x) = \dfrac{f'g - fg'}{g^2} = \dfrac{fg - fg}{g^2} = 0$.

Donc $h$ est constante, et $h(0) = \dfrac{f(0)}{g(0)} = 1$. Ainsi $f = g$ sur $\mathbb{R}$, c'est-à-dire :
\[
\exp(x + a) = \exp(a) \times \exp(x) \quad \text{pour tout } x \in \mathbb{R}.
\]
En posant $x = b$, on obtient $\exp(a + b) = \exp(a) \times \exp(b)$.
}

\propriete[Conséquences algébriques]{
Pour tous réels $a$ et $b$, et pour tout entier relatif $n$ :
\begin{enumerate}
  \item $\exp(-a) = \dfrac{1}{\exp(a)}$
  \item $\exp(a - b) = \dfrac{\exp(a)}{\exp(b)}$
  \item $\exp(na) = [\exp(a)]^n$
\end{enumerate}
}

\demonstration{
\textbf{Preuve de $\exp(-a) = \dfrac{1}{\exp(a)}$.}

On applique la relation fonctionnelle avec $b = -a$ :
\[
\exp(a + (-a)) = \exp(a) \times \exp(-a), \quad \text{soit} \quad \exp(0) = \exp(a) \times \exp(-a).
\]
Or $\exp(0) = 1$, donc $\exp(-a) = \dfrac{1}{\exp(a)}$.

\textbf{Preuve de $\exp(a - b) = \dfrac{\exp(a)}{\exp(b)}$.}

$\exp(a - b) = \exp(a + (-b)) = \exp(a) \times \exp(-b) = \exp(a) \times \dfrac{1}{\exp(b)} = \dfrac{\exp(a)}{\exp(b)}$.

\textbf{Preuve de $\exp(na) = [\exp(a)]^n$ pour $n \in \mathbb{N}$.}

Par récurrence : le cas $n = 0$ donne $\exp(0) = 1 = [\exp(a)]^0$. Si $\exp(na) = [\exp(a)]^n$, alors :
\[
\exp((n+1)a) = \exp(na + a) = \exp(na) \times \exp(a) = [\exp(a)]^n \times \exp(a) = [\exp(a)]^{n+1}.
\]
Le cas $n < 0$ se déduit de $\exp(na) = \dfrac{1}{\exp(-na)} = \dfrac{1}{[\exp(a)]^{-n}} = [\exp(a)]^n$.
}

\propriete[Justification de la notation $\mathrm{e}^x$]{
La relation $\exp(na) = [\exp(a)]^n$ montre que l'exponentielle se comporte comme une puissance. En posant $a = 1$ :
\[
\exp(n) = [\exp(1)]^n = \mathrm{e}^n.
\]
On étend alors cette notation à tout réel $x$ : $\exp(x) = \mathrm{e}^x$.
}

\exemple{
Simplifier $A = \mathrm{e}^3 \times \mathrm{e}^{-5}$.

On applique $\exp(a+b) = \exp(a) \times \exp(b)$ :
\[
A = \mathrm{e}^{3+(-5)} = \mathrm{e}^{-2}.
\]
}

\exemple{
Simplifier $C = (\mathrm{e}^{2})^4 \times \mathrm{e}^{-3}$.

\[
C = \mathrm{e}^{2 \times 4} \times \mathrm{e}^{-3} = \mathrm{e}^{8} \times \mathrm{e}^{-3} = \mathrm{e}^{8-3} = \mathrm{e}^{5}.
\]
}

\exemple{
Simplifier $D = \dfrac{\mathrm{e}^{2x+1} \times \mathrm{e}^{3-x}}{\mathrm{e}^{x+2}}$ pour $x \in \mathbb{R}$.

Au numérateur : $\mathrm{e}^{2x+1} \times \mathrm{e}^{3-x} = \mathrm{e}^{(2x+1)+(3-x)} = \mathrm{e}^{x+4}$.

Puis : $D = \dfrac{\mathrm{e}^{x+4}}{\mathrm{e}^{x+2}} = \mathrm{e}^{(x+4)-(x+2)} = \mathrm{e}^{2}$.
}

\erreurFrequente{
Écrire $\mathrm{e}^a \times \mathrm{e}^b = \mathrm{e}^{ab}$. L'exponentielle transforme une \emph{somme} en produit : $\mathrm{e}^a \times \mathrm{e}^b = \mathrm{e}^{a+b}$. Les exposants s'additionnent, ils ne se multiplient pas.
}

\erreurFrequente{
Écrire $\mathrm{e}^{a+b} = \mathrm{e}^a + \mathrm{e}^b$. C'est faux : l'exponentielle d'une somme est un \emph{produit}, pas une somme.
}

\contreexemple{
On peut vérifier numériquement : $\mathrm{e}^{1+2} = \mathrm{e}^3 \approx 20{,}09$, tandis que $\mathrm{e}^1 + \mathrm{e}^2 \approx 2{,}718 + 7{,}389 = 10{,}107$. On a bien $\mathrm{e}^{1+2} \neq \mathrm{e}^1 + \mathrm{e}^2$.
}

\approfondissement{
La relation fonctionnelle $\exp(a+b) = \exp(a) \exp(b)$ caractérise les morphismes du groupe $(\mathbb{R}, +)$ vers le groupe $(\mathbb{R}^*_+, \times)$. La fonction exponentielle est le seul morphisme continu vérifiant $\exp(1) = \mathrm{e}$. Sa réciproque, le logarithme népérien, transforme les produits en sommes : $\ln(AB) = \ln A + \ln B$.
}
