\begin{fichemethode}{M5}{Résoudre une inéquation avec l'exponentielle}
\quandUtiliser{J'utilise cette méthode lorsque l'énoncé demande de résoudre une inéquation faisant intervenir l'exponentielle, du type $\mathrm{e}^A \geq \mathrm{e}^B$ ou $\mathrm{e}^A \geq k$.}
\pasApas{\begin{enumerate}
  \item Je ramène l'inéquation à l'une des deux formes suivantes :
    \begin{itemize}
      \item $\mathrm{e}^A \geq \mathrm{e}^B$ (ou $>$, $\leq$, $<$) ;
      \item $\mathrm{e}^A \geq k$, où $k$ est un réel.
    \end{itemize}
  \item J'applique la règle correspondante :
    \begin{itemize}
      \item La fonction exponentielle est strictement croissante sur $\mathbb{R}$, donc $\mathrm{e}^A \geq \mathrm{e}^B \Leftrightarrow A \geq B$ (le sens de l'inégalité est conservé).
      \item Si $k > 0$ : $\mathrm{e}^A \geq k \Leftrightarrow A \geq \ln(k)$.
      \item Si $k \leq 0$ : l'inéquation est toujours vérifiée (car $\mathrm{e}^A > 0$ pour tout réel $A$).
    \end{itemize}
  \item Je résous l'inéquation algébrique obtenue.
\end{enumerate}}
\exempleRedige{Résolvons $\mathrm{e}^{-x+2} \leq \mathrm{e}^{3x-1}$ dans $\mathbb{R}$. \commentaireMarge{Forme $\mathrm{e}^A \leq \mathrm{e}^B$.} La fonction exponentielle est strictement croissante, donc :
\[\mathrm{e}^{-x+2} \leq \mathrm{e}^{3x-1} \Leftrightarrow -x + 2 \leq 3x - 1.\]
\commentaireMarge{Résolution de l'inéquation du premier degré.} $2 + 1 \leq 3x + x$, soit $3 \leq 4x$, d'où $x \geq \dfrac{3}{4}$.
\[\mathcal{S} = \left[\dfrac{3}{4}\,;\,+\infty\right[.\]}
\pieges{\begin{itemize}
  \item Inverser le sens de l'inégalité : la fonction exponentielle est croissante, donc le sens est conservé.
  \item Oublier que $\mathrm{e}^A > 0$ toujours : si $k \leq 0$, l'inéquation $\mathrm{e}^A \geq k$ est vérifiée pour tout réel.
  \item Confondre avec le cas d'une fonction décroissante (comme $x \mapsto -\mathrm{e}^x$) où le raisonnement sur le signe change.
\end{itemize}}
\verifier{Je teste une valeur dans l'ensemble solution et une valeur en dehors : par exemple, pour $x = 1 \geq \dfrac{3}{4}$, $\mathrm{e}^{1} \leq \mathrm{e}^{2}$ est vrai ; pour $x = 0 < \dfrac{3}{4}$, $\mathrm{e}^{2} \leq \mathrm{e}^{-1}$ est faux.}
\sEntrainer{\refExos{M5}}
\end{fichemethode}
