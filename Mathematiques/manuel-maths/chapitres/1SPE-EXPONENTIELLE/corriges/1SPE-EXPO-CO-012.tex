\begin{corrige}{1SPE-EXPO-EX-012}
On applique les propriétés algébriques de l'exponentielle.

\textbf{1.} $\mathrm{e}^x \times \mathrm{e}^{2x} = \mathrm{e}^{x+2x} = \mathrm{e}^{3x}$.

\textbf{2.} $\dfrac{\mathrm{e}^{3x}}{\mathrm{e}^x} = \mathrm{e}^{3x-x} = \mathrm{e}^{2x}$.

\textbf{3.} $\left(\mathrm{e}^x\right)^5 = \mathrm{e}^{5x}$.

\textbf{4.} $\mathrm{e}^{2x} \times \mathrm{e}^{-x} \times \mathrm{e}^{3x} = \mathrm{e}^{2x+(-x)+3x} = \mathrm{e}^{4x}$.
\end{corrige}
