\begin{corrige}{1SPE-EXPO-EX-049}
\begin{enumerate}
\item On dérive $f(x) = (2x - 1)\mathrm{e}^x$ avec la règle du produit :
\[f'(x) = 2\,\mathrm{e}^x + (2x - 1)\mathrm{e}^x = (2 + 2x - 1)\mathrm{e}^x = (2x + 1)\mathrm{e}^x.\]

\item Comme $\mathrm{e}^x > 0$ pour tout réel $x$, le signe de $f'(x)$ est celui de $2x + 1$ :
\begin{itemize}
\item $f'(x) < 0$ si $x < -\dfrac{1}{2}$ ;
\item $f'(x) = 0$ si $x = -\dfrac{1}{2}$ ;
\item $f'(x) > 0$ si $x > -\dfrac{1}{2}$.
\end{itemize}

Donc $f$ est strictement décroissante sur $\left]-\infty\,;-\dfrac{1}{2}\right]$ et strictement croissante sur $\left[-\dfrac{1}{2}\,;+\infty\right[$.

Le minimum de $f$ est atteint en $x = -\dfrac{1}{2}$ :
\[f\!\left(-\dfrac{1}{2}\right) = \left(-1 - 1\right)\mathrm{e}^{-1/2} = -2\,\mathrm{e}^{-1/2} = \dfrac{-2}{\sqrt{\mathrm{e}}}.\]

\item $f(x) = 0 \iff (2x - 1)\mathrm{e}^x = 0$. Or $\mathrm{e}^x \neq 0$, donc :
\[2x - 1 = 0 \iff x = \dfrac{1}{2}.\]

L'unique solution est $\boxed{x = \dfrac{1}{2}}$.

\item Le signe de $f(x)$ est celui de $2x - 1$ (car $\mathrm{e}^x > 0$) :
\begin{itemize}
\item $f(x) < 0$ si $x < \dfrac{1}{2}$ ;
\item $f(x) = 0$ si $x = \dfrac{1}{2}$ ;
\item $f(x) > 0$ si $x > \dfrac{1}{2}$.
\end{itemize}

\item \textbf{Limite en $+\infty$ :} $\lim_{x \to +\infty}(2x - 1) = +\infty$ et $\lim_{x \to +\infty}\mathrm{e}^x = +\infty$, donc $\lim_{x \to +\infty} f(x) = +\infty$.

\textbf{Limite en $-\infty$ :} on écrit $f(x) = (2x - 1)\mathrm{e}^x$. Or $\lim_{x \to -\infty} x\,\mathrm{e}^x = 0$ (croissance comparée), donc $\lim_{x \to -\infty} f(x) = 0$.
\end{enumerate}
\end{corrige}
