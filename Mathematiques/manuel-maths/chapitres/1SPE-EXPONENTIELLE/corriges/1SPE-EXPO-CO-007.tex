\begin{corrige}{1SPE-EXPO-EX-007}
\begin{enumerate}
\item On dérive :
\[h'(x) = 3\mathrm{e}^x.\]

\item On compare $h'(x)$ et $h(x)$ :
\[h'(x) = 3\mathrm{e}^x \quad \text{et} \quad h(x) = 3\mathrm{e}^x - 2.\]
Comme $h'(x) \neq h(x)$ (ils diffèrent de $2$), $h$ n'est \textbf{pas} solution de $y' = y$.

\item On calcule :
\[h'(x) - h(x) = 3\mathrm{e}^x - (3\mathrm{e}^x - 2) = 2.\]
Donc $h'(x) = h(x) + 2$, c'est-à-dire que $h$ est solution de l'équation différentielle $y' = y + 2$.

\item $h(0) = 3\mathrm{e}^0 - 2 = 3 \times 1 - 2 = 1$.
La condition $h(0) = 1$ est bien vérifiée.

\item Bien que $h(0) = 1$, la fonction $h$ ne vérifie pas l'équation $y' = y$. Elle vérifie $y' = y + 2$, qui est une équation différentielle \textbf{différente}. La fonction exponentielle est l'unique solution de $y' = y$ avec $y(0) = 1$. Les \textbf{deux} conditions sont nécessaires : il faut $y' = y$ (pas $y' = y + 2$) et $y(0) = 1$.
\end{enumerate}
\end{corrige}
