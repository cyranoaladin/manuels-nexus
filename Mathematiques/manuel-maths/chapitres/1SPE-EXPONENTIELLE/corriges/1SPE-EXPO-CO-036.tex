\begin{corrige}{1SPE-EXPO-EX-036}

\textbf{Question 1.} On pose $u(x) = 2x - 1$ et $v(x) = \mathrm{e}^{-x}$.

On a $u'(x) = 2$ et $v'(x) = -\mathrm{e}^{-x}$.

Par la formule du produit :
\[f'(x) = 2\,\mathrm{e}^{-x} + (2x-1) \times (-\mathrm{e}^{-x}) = 2\,\mathrm{e}^{-x} - (2x-1)\,\mathrm{e}^{-x}.\]

\textbf{Question 2.} On factorise par $\mathrm{e}^{-x}$ :
\[f'(x) = \bigl[2 - (2x-1)\bigr]\,\mathrm{e}^{-x} = \boxed{(3-2x)\,\mathrm{e}^{-x}}.\]

\textbf{Question 3.} Comme $\mathrm{e}^{-x} > 0$ pour tout réel $x$, le signe de $f'(x)$ est celui de $3 - 2x$.
\begin{itemize}
\item $3 - 2x > 0 \iff x < \dfrac{3}{2}$ : $f$ est croissante sur $\left]-\infty\,;\,\dfrac{3}{2}\right]$.
\item $3 - 2x < 0 \iff x > \dfrac{3}{2}$ : $f$ est décroissante sur $\left[\dfrac{3}{2}\,;\,+\infty\right[$.
\end{itemize}
La fonction $f$ admet un maximum en $x = \dfrac{3}{2}$, avec $f\!\left(\dfrac{3}{2}\right) = (3-1)\,\mathrm{e}^{-3/2} = 2\,\mathrm{e}^{-3/2}$.
\end{corrige}
