\begin{corrige}{1SPE-EXPO-EX-013}
On utilise les propriétés : $\mathrm{e}^a \times \mathrm{e}^b = \mathrm{e}^{a+b}$ et $\dfrac{\mathrm{e}^a}{\mathrm{e}^b} = \mathrm{e}^{a-b}$.

\textbf{1.} $\mathrm{e}^{x+1} \times \mathrm{e}^{x-1} = \mathrm{e}^{(x+1)+(x-1)} = \mathrm{e}^{2x}$.

\textbf{2.} $\dfrac{\mathrm{e}^{2x+3}}{\mathrm{e}^{x+1}} = \mathrm{e}^{(2x+3)-(x+1)} = \mathrm{e}^{x+2}$.

\textbf{3.} $\dfrac{\left(\mathrm{e}^x\right)^3}{\mathrm{e}^{2x}} = \dfrac{\mathrm{e}^{3x}}{\mathrm{e}^{2x}} = \mathrm{e}^{3x-2x} = \mathrm{e}^x$.

\textbf{4.} $\mathrm{e}^{x+2} \times \mathrm{e}^{-(x-2)} = \mathrm{e}^{(x+2)+(-x+2)} = \mathrm{e}^{4}$.
\end{corrige}
