\begin{corrige}{1SPE-EXPO-EX-002}
\begin{enumerate}
\item On dérive : $g'(x) = 2\mathrm{e}^x$.

\item On a $g'(x) = 2\mathrm{e}^x = g(x)$ pour tout réel $x$.
Donc $g$ vérifie bien l'équation $y' = y$.

\item On calcule $g(0) = 2\mathrm{e}^0 = 2 \times 1 = 2$.
La condition $y(0) = 1$ n'est \textbf{pas} vérifiée car $g(0) = 2 \neq 1$.

\item La fonction exponentielle est l'unique fonction vérifiant \textbf{simultanément} $y' = y$ et $y(0) = 1$. La fonction $g$ vérifie $g' = g$ mais pas $g(0) = 1$. Elle ne peut donc pas être la fonction exponentielle. C'est un multiple de l'exponentielle, mais pas l'exponentielle elle-même.
\end{enumerate}
\end{corrige}
