\begin{corrige}{1SPE-SECDEG-EX-011}

\textbf{1. Racines de $f$.}

\commentaireMarge{$f(x) = 0 \iff (x-2)(x+5) = 0$.}

$f$ est déjà sous forme factorisée. Par la règle du produit nul :
\[(x-2)(x+5) = 0 \iff x - 2 = 0 \text{ ou } x + 5 = 0 \iff x = 2 \text{ ou } x = -5.\]

Les racines de $f$ sont $x_1 = -5$ et $x_2 = 2$.

\medskip
\textbf{2. Tableau de signes.}

\commentaireMarge{$a = 1 > 0$ : $f > 0$ à l'extérieur des racines.}

\[
\begin{array}{|c|ccccccc|}
\hline
x & -\infty & & -5 & & 2 & & +\infty \\
\hline
x+5 & & - & 0 & + & & + & \\
x-2 & & - & & - & 0 & + & \\
\hline
f(x) & & + & 0 & - & 0 & + & \\
\hline
\end{array}
\]

\medskip
\textbf{3. Solutions des inéquations.}

\textbf{a)} $f(x) \geqslant 0$ : d'après le tableau, $f(x) \geqslant 0$ sur $\left]-\infty\,;\,-5\right] \cup \left[2\,;\,+\infty\right[$.

\[\boxed{\mathcal{S}_a = \left]-\infty\,;\,-5\right] \cup \left[2\,;\,+\infty\right[.}\]

\textbf{b)} $f(x) < 0$ : d'après le tableau, $f(x) < 0$ sur $\left]-5\,;\,2\right[$.

\[\boxed{\mathcal{S}_b = \left]-5\,;\,2\right[.}\]

\medskip
\textbf{4. Développement et calcul de $\Delta$.}

\commentaireMarge{$(x-2)(x+5) = x^2+3x-10$.}

\[f(x) = (x-2)(x+5) = x^2 + 5x - 2x - 10 = x^2 + 3x - 10.\]

On a $a = 1$, $b = 3$, $c = -10$.

$\Delta = 3^2 - 4 \times 1 \times (-10) = 9 + 40 = \mathbf{49}$. \hfill $\square$

\end{corrige}
