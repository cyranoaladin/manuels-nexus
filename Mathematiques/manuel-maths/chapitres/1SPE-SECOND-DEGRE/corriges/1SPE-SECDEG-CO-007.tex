\begin{corrige}{1SPE-SECDEG-EX-007}

\textbf{Rappel de la méthode :} pour résoudre $ax^2 + bx + c = 0$, on calcule $\Delta = b^2 - 4ac$.
\begin{itemize}
  \item Si $\Delta > 0$ : deux racines réelles $x_{1,2} = \dfrac{-b \pm \sqrt{\Delta}}{2a}$.
  \item Si $\Delta = 0$ : une racine double $x_0 = \dfrac{-b}{2a}$.
  \item Si $\Delta < 0$ : aucune racine réelle.
\end{itemize}

\medskip
\textbf{a) $x^2 - 5x + 6 = 0$.}

\commentaireMarge{$a=1$, $b=-5$, $c=6$.}

$\Delta = (-5)^2 - 4 \times 1 \times 6 = 25 - 24 = 1 > 0$.

\[\sqrt{\Delta} = 1.\]

\[x_1 = \frac{5 - 1}{2} = 2, \qquad x_2 = \frac{5 + 1}{2} = 3.\]

\[\boxed{\mathcal{S} = \{2\,;\,3\}.}\]

\medskip
\textbf{b) $2x^2 + 3x - 2 = 0$.}

\commentaireMarge{$a=2$, $b=3$, $c=-2$.}

$\Delta = 3^2 - 4 \times 2 \times (-2) = 9 + 16 = 25 > 0$.

\[\sqrt{\Delta} = 5.\]

\[x_1 = \frac{-3 - 5}{4} = \frac{-8}{4} = -2, \qquad x_2 = \frac{-3 + 5}{4} = \frac{2}{4} = \frac{1}{2}.\]

\[\boxed{\mathcal{S} = \left\{-2\,;\,\frac{1}{2}\right\}.}\]

\medskip
\textbf{c) $x^2 - 2x + 5 = 0$.}

\commentaireMarge{$\Delta < 0$ : pas de solution dans $\mathbb{R}$.}

$\Delta = (-2)^2 - 4 \times 1 \times 5 = 4 - 20 = -16 < 0$.

Puisque $\Delta < 0$, l'équation n'admet \textbf{aucune solution réelle}.

\[\boxed{\mathcal{S} = \emptyset.}\]

\end{corrige}
