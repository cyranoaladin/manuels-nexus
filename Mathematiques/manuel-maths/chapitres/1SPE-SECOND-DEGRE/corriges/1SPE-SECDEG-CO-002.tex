\begin{corrige}{1SPE-SECDEG-EX-002}

\textbf{1. Passage de la forme canonique à la forme développée : $f(x) = 3(x-2)^2 - 1$.}

\commentaireMarge{Développer $(x-2)^2 = x^2 - 4x + 4$.}

\[f(x) = 3(x-2)^2 - 1 = 3(x^2 - 4x + 4) - 1 = 3x^2 - 12x + 12 - 1.\]

\[\boxed{f(x) = 3x^2 - 12x + 11.}\]

\medskip
\textbf{2. Passage de la forme factorisée à la forme développée : $g(x) = -2(x-1)(x+4)$.}

\commentaireMarge{Développer le produit, puis multiplier par $-2$.}

\[(x-1)(x+4) = x^2 + 4x - x - 4 = x^2 + 3x - 4.\]

\[g(x) = -2(x^2 + 3x - 4) = -2x^2 - 6x + 8.\]

\[\boxed{g(x) = -2x^2 - 6x + 8.}\]

\medskip
\textbf{3. Passage de la forme développée à la forme canonique : $h(x) = x^2 + 6x + 5$.}

\commentaireMarge{$\alpha = -\dfrac{b}{2a}$, $\beta = h(\alpha)$.}

Ici $a = 1$, $b = 6$, $c = 5$.

Abscisse du sommet :
\[\alpha = -\frac{b}{2a} = -\frac{6}{2} = -3.\]

Ordonnée du sommet :
\[\beta = h(-3) = (-3)^2 + 6 \times (-3) + 5 = 9 - 18 + 5 = -4.\]

Donc :
\[\boxed{h(x) = (x + 3)^2 - 4.}\]

\textit{Vérification :} $(x+3)^2 - 4 = x^2 + 6x + 9 - 4 = x^2 + 6x + 5$. \hfill $\square$

\end{corrige}
