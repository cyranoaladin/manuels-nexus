\begin{exercice}{1SPE-SECDEG-EX-003}{1}{10}
\begin{enumerate}
  \item Parmi les expressions suivantes, identifier celles qui sont des trinômes du second degré en $x$. Préciser alors les valeurs de $a$, $b$ et $c$.
  \begin{align*}
    p(x) &= 4x^2 - x + 7 & q(x) &= 3x - 2 & r(x) &= -x^2 + 5
  \end{align*}

  \item On considère $f(x) = 2\!\left(x - \dfrac{1}{2}\right)\!(x - 3)$.
  \begin{enumerate}
    \item Développer $f(x)$ pour obtenir sa forme développée.
    \item En déduire les coordonnées du sommet $S(\alpha\,;\,\beta)$ de la parabole.
    \item Écrire $f$ sous forme canonique.
  \end{enumerate}
\end{enumerate}
\end{exercice}
