\begin{exercice}{1SPE-SECDEG-EX-012}{1}{12}
On considère le trinôme $g(x) = -2x^2 + 2x + 12$.

\begin{enumerate}
  \item Calculer $\Delta$ et déterminer les racines de $g$.
  \item Factoriser $g(x)$.
  \item Dresser le tableau de signes de $g$ sur $\mathbb{R}$. \textit{(On rappelle que le signe de $g$ est celui de $a$ à l'extérieur des racines.)}
  \item Résoudre $g(x) \leqslant 0$.
\end{enumerate}
\end{exercice}
