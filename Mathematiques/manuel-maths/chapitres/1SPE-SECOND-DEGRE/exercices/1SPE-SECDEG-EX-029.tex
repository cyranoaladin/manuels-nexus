\begin{exercice}{1SPE-SECDEG-EX-029}{2}{20}
\begin{enumerate}
  \item \textbf{(C3)} Résoudre dans $\mathbb{R}$ l'équation $x^2 - 5x + 3 = 0$. Donner les solutions sous forme exacte.
  \item \textbf{(C5)} En déduire la solution de l'inéquation $x^2 - 5x + 3 \leqslant 0$.

  \textit{On pourra noter $x_1 = \dfrac{5 - \sqrt{13}}{2}$ et $x_2 = \dfrac{5 + \sqrt{13}}{2}$.}

  \item \textbf{(C3)} Calculer le discriminant de $g(x) = x^2 + 4x + 5$.
  \item \textbf{(C5)} En déduire, sans résoudre d'équation, le signe de $g(x)$ sur $\mathbb{R}$ et la solution de $g(x) > 0$.
\end{enumerate}
\end{exercice}
