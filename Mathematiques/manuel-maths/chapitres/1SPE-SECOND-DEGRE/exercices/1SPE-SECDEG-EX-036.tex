\begin{exercice}{1SPE-SECDEG-EX-036}{3}{35}
On considère les deux polynômes du second degré :
\[
  f(x) = x^2 - 5x + 4 \qquad \text{et} \qquad g(x) = -x^2 + 3x + 4.
\]

\begin{enumerate}
  \item \textbf{Étude de $f$.}
  \begin{enumerate}
    \item Calculer le discriminant $\Delta_f$ et déterminer les racines de $f$.
    \item Dresser le tableau de signes de $f(x)$ sur $\mathbb{R}$.
  \end{enumerate}

  \item \textbf{Étude de $g$.}
  \begin{enumerate}
    \item Calculer le discriminant $\Delta_g$ et déterminer les racines de $g$.
    \item Dresser le tableau de signes de $g(x)$ sur $\mathbb{R}$.
    \item Remarquer qu'une des racines de $g$ est commune avec une racine de $f$ : que peut-on en déduire sur les paraboles représentant $f$ et $g$ ?
  \end{enumerate}

  \item \textbf{Système d'inéquations.} Résoudre le système :
  \[
    \begin{cases} f(x) < 0 \\ g(x) > 0 \end{cases}
  \]
  en combinant les tableaux de signes établis. Exprimer la solution sous forme d'un intervalle.

  \item \textbf{Inéquation produit.} On souhaite résoudre $f(x) \times g(x) \leq 0$.
  \begin{enumerate}
    \item Dresser le tableau de signes du produit $f(x) \times g(x)$.
    \item En déduire l'ensemble des solutions de $f(x) \times g(x) \leq 0$.
  \end{enumerate}

  \item \textbf{Question ouverte.} On considère l'inéquation $f(x) + g(x) \leq 0$.
  \begin{enumerate}
    \item Calculer $h(x) = f(x) + g(x)$ et simplifier.
    \item Résoudre l'inéquation $h(x) \leq 0$ en détaillant la méthode choisie.
    \item Comparer les solutions obtenues avec celles de la question 3. Que remarque-t-on ?
  \end{enumerate}
\end{enumerate}
\end{exercice}
