\begin{exercice}{1SPE-SECDEG-RE-C3-EX1}{1}{8}
Resoudre l'equation $2x^2 - 7x + 3 = 0$.
\begin{enumerate}
  \item Identifier $a$, $b$, $c$.
  \item Calculer $\Delta = b^2 - 4ac$.
  \item Determiner le nombre de solutions et les calculer si elles existent.
  \item Verifier en substituant les solutions dans l'equation.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-RE-C3-EX1}

\commentaireMarge{Identifier $a$, $b$, $c$ dans $ax^2 + bx + c$.}

\textbf{Question 1.} $a = 2$, $b = -7$, $c = 3$.

\commentaireMarge{Formule : $\Delta = b^2 - 4ac = (-7)^2 - 4 \times 2 \times 3$.}

\textbf{Question 2.}
\[
\Delta = (-7)^2 - 4 \times 2 \times 3 = 49 - 24 = 25.
\]

\commentaireMarge{$\Delta = 25 > 0$ : deux racines reelles distinctes. $\sqrt{25} = 5$.}

\textbf{Question 3.} $\Delta = 25 > 0$, donc deux racines :
\[
x_1 = \frac{7 - 5}{4} = \frac{2}{4} = \frac{1}{2}, \qquad
x_2 = \frac{7 + 5}{4} = \frac{12}{4} = 3.
\]

\textbf{Question 4.} Verification :
\[
f\!\left(\tfrac{1}{2}\right) = 2 \times \tfrac{1}{4} - 7 \times \tfrac{1}{2} + 3 = \tfrac{1}{2} - \tfrac{7}{2} + 3 = -3 + 3 = 0. \checkmark
\]
\[
f(3) = 2 \times 9 - 7 \times 3 + 3 = 18 - 21 + 3 = 0. \checkmark
\]

\end{corrige}

\begin{corrige}{1SPE-SECDEG-RE-C3-EX2}

\commentaireMarge{Trois cas : $\Delta > 0$ (deux racines), $\Delta = 0$ (une racine double), $\Delta < 0$ (pas de racine reelle).}

\textbf{Question 1.} $a = 1$, $b = -6$, $c = 9$. $\Delta = 36 - 36 = 0$.
Racine double : $x_0 = \dfrac{6}{2} = 3$. L'ensemble des solutions est $\{3\}$.

\textbf{Question 2.} $a = 1$, $b = 1$, $c = 1$. $\Delta = 1 - 4 = -3 < 0$.
Aucune racine reelle. L'ensemble des solutions est $\emptyset$.

\textbf{Question 3.} $a = 3$, $b = -5$, $c = -2$. $\Delta = 25 + 24 = 49 > 0$. $\sqrt{49} = 7$.
\[
x_1 = \frac{5 - 7}{6} = \frac{-2}{6} = -\frac{1}{3}, \qquad
x_2 = \frac{5 + 7}{6} = \frac{12}{6} = 2.
\]

\end{corrige}

\begin{corrige}{1SPE-SECDEG-RE-C3-EX3}

\textbf{Question 1.} $\Delta = 25 - 16 = 9 > 0$. $x_1 = (5-3)/2 = 1$, $x_2 = (5+3)/2 = 4$.

\textbf{Question 2.} $2x^2 + 4x + 2 = 2(x^2 + 2x + 1) = 2(x+1)^2$. On peut aussi calculer $\Delta = 16 - 16 = 0$, racine double $x_0 = -4/4 = -1$.

\textbf{Question 3.} $\Delta = (-4)^2 - 4 \times 1 \times k = 16 - 4k$. Deux racines distinctes $\Leftrightarrow \Delta > 0 \Leftrightarrow 16 - 4k > 0 \Leftrightarrow k < 4$.

\textbf{Question 4.} $x(7-x) = 21 \Leftrightarrow 7x - x^2 = 21 \Leftrightarrow x^2 - 7x + 21 = 0$.
$\Delta = 49 - 84 = -35 < 0$ : pas de solution reelle. Un tel rectangle n'existe pas.

\end{corrige}
