\subsection*{Rappel — Inegalites et tableau de signes}

\textbf{Regles de calcul sur les inegalites} :
\begin{itemize}
  \item On peut additionner/soustraire le meme reel des deux membres sans changer le sens.
  \item On peut multiplier/diviser par un reel \textbf{strictement positif} sans changer le sens.
  \item On \textbf{inverse le sens} si on multiplie/divise par un reel \textbf{strictement negatif}.
\end{itemize}

\textbf{Tableau de signes d'un produit} : le signe du produit $A \times B$ se determine colonne par colonne.

\[
\text{Signe de } AB : \quad
\begin{array}{|c|ccc|}
\hline
 & A < 0 & A = 0 & A > 0 \\
\hline
B > 0 & - & 0 & + \\
B = 0 & 0 & 0 & 0 \\
B < 0 & + & 0 & - \\
\hline
\end{array}
\]

\end{corrige}

\begin{corrige}{1SPE-SECDEG-FR-R5-EX2}

\textbf{Question 1.} Racines : $x = -3$ et $x = 2$.

\begin{center}
\begin{tabular}{|c|cccccc|}
\hline
$x$ & $-\infty$ & & $-3$ & & $2$ & $+\infty$ \\
\hline
$x - 2$ & & $-$ & $-$ & $-$ & $0$ & $+$ \\
$x + 3$ & & $-$ & $0$ & $+$ & $+$ & $+$ \\
\hline
$(x-2)(x+3)$ & & $+$ & $0$ & $-$ & $0$ & $+$ \\
\hline
\end{tabular}
\end{center}

Solution de $\geq 0$ : $x \in \,]-\infty\,;\,-3] \cup [2\,;\,+\infty[$.

\textbf{Question 2.} Racines : $x = -1$ et $x = 4$. Le produit est negatif strictement entre les racines.

Solution de $< 0$ : $x \in \,]-1\,;\,4[$.

\textbf{Question 3.} Racines de $(x-1)(x+2)$ : $x = -2$ et $x = 1$. Le facteur $-3 < 0$ inverse le signe.

$(x-1)(x+2) \geq 0$ sur $]-\infty\,;\,-2] \cup [1\,;\,+\infty[$, donc $-3(x-1)(x+2) \leq 0$ sur le meme ensemble.

Solution : $x \in \,]-\infty\,;\,-2] \cup [1\,;\,+\infty[$.

\end{corrige}

\begin{corrige}{1SPE-SECDEG-FR-R5-EX4}

\textbf{Question 1.} Racines de $(x-1)(x+5)$ : $x = -5$ et $x = 1$. Le produit est negatif entre les racines.
Solution : $x \in \,[-5\,;\,1]$.

\textbf{Question 2.} $x^2 - 4 = (x-2)(x+2) > 0$ : le produit est positif a l'exterieur des racines.
Solution : $x \in \,]-\infty\,;\,-2[ \cup \,]2\,;\,+\infty[$.

\textbf{Question 3.} $x^2 - 3x = x(x-3) \geq 0$ : racines $0$ et $3$. Le produit est positif a l'exterieur.
Solution : $x \in \,]-\infty\,;\,0] \cup [3\,;\,+\infty[$.

\commentaireMarge{$(x-2)^2 \geq 0$ pour tout $x$, avec egalite uniquement en $x = 2$.}

\textbf{Question 4.} $(x-2)^2 > 0$ est vrai pour tout $x \neq 2$ (le carre est toujours $\geq 0$, et nul seulement en $x = 2$).
Solution : $x \in \mathbb{R} \setminus \{2\} = \,]-\infty\,;\,2[ \cup \,]2\,;\,+\infty[$.

\end{corrige}
