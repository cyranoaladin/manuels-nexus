\begin{exercice}{1SPE-SECDEG-RE-C4-EX1}{1}{8}
Soit $f(x) = 2x^2 - 2x - 12$.
\begin{enumerate}
  \item Calculer $\Delta$ et les racines $x_1$, $x_2$.
  \item Ecrire la forme factorisee $f(x) = a(x - x_1)(x - x_2)$.
  \item Dresser le tableau de signes de $f$.
  \item Resoudre $f(x) \leq 0$.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-RE-C4-EX1}

\textbf{Question 1.} $a = 2$, $b = -2$, $c = -12$.
\[
\Delta = 4 + 96 = 100. \quad \sqrt{\Delta} = 10.
\]
\[
x_1 = \frac{2 - 10}{4} = -2, \qquad x_2 = \frac{2 + 10}{4} = 3.
\]

\commentaireMarge{Forme factorisee : $f(x) = a(x - x_1)(x - x_2)$. Ne pas oublier $a = 2$.}

\textbf{Question 2.}
\[
f(x) = 2(x - (-2))(x - 3) = 2(x + 2)(x - 3).
\]
Verification : $2(x+2)(x-3) = 2(x^2 - x - 6) = 2x^2 - 2x - 12$. \checkmark

\textbf{Question 3. Tableau de signes :}

\hline

\textbf{Question 4.} $f(x) \leq 0$ sur $[-2\,;\,3]$.

\begin{exercice}{1SPE-SECDEG-RE-C4-EX2}{1}{8}
Soit $g(x) = x^2 + 2x + 5$.
\begin{enumerate}
  \item Calculer $\Delta$.
  \item Que conclure sur le signe de $g(x)$ ?
  \item Resoudre $g(x) > 0$.
  \item Justifier que $g(x) + 1 > 0$ pour tout $x \in \mathbb{R}$.
\end{enumerate}
\end{exercice}

\coupDePouce{1}{$\Delta = 4 - 20 = -16 < 0$ et $a = 1 > 0$. Rappel : si $\Delta < 0$ et $a > 0$, le trinome est du meme signe que $a$ partout.}

% g = x**2 + 2*x + 5
% Delta = 2**2 - 4*1*5
% assert Delta == -16
% assert Delta < 0
% # g positif partout : verifier quelques valeurs
% assert g.subs(x, 0) == 5   # > 0
% assert g.subs(x, -1) == 4  # > 0
% assert g.subs(x, 10) == 115 # > 0
% # g + 1 positif
% assert (g + 1).subs(x, 0) == 6 > 0
% END-VERIFY

\begin{corrige}{1SPE-SECDEG-RE-C4-EX2}

\textbf{Question 1.} $a = 1$, $b = 2$, $c = 5$. $\Delta = 4 - 20 = -16 < 0$.

\commentaireMarge{Regle : $\Delta < 0$ et $a > 0$ $\Rightarrow$ $f(x) > 0$ pour tout $x \in \mathbb{R}$.}

\textbf{Question 2.} Comme $\Delta < 0$ et $a = 1 > 0$, le trinome $g(x)$ est strictement positif pour tout $x \in \mathbb{R}$.

\textbf{Question 3.} $g(x) > 0$ pour tout $x \in \mathbb{R}$, donc l'ensemble solution est $\mathbb{R}$.

\textbf{Question 4.} Comme $g(x) > 0$ pour tout $x$, on a $g(x) + 1 > 0 + 1 = 1 > 0$ pour tout $x$.

\end{corrige}

\textbf{Question 2.} $h(x) \geq 0$ sur $]-\infty\,;\,2] \cup [5\,;\,+\infty[$. \quad $h(x) < 0$ sur $]2\,;\,5[$.
