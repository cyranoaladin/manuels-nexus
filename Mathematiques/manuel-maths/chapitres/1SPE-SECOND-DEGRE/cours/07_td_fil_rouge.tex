\section*{TD fil rouge — La boite en carton}

\subsection*{Mise en situation}

\begin{quotation}
Un artisan fabrique des boites en carton pour les cadeaux. Il part d'une feuille rectangulaire de $30$~cm sur $20$~cm, decoupe un carre de cote $x$ (en cm) aux quatre coins, puis replie les bords pour former une boite ouverte. Quelle valeur de $x$ maximise le volume de la boite ?
\end{quotation}

Ce probleme d'optimisation reunit toutes les capacites du chapitre.

\bigskip

\subsection*{Partie A — Mise en equation}

\begin{exercice}{1SPE-SECDEG-TD-FIL-A}{2}{10}
\begin{enumerate}
  \item Exprimer en fonction de $x$ la longueur, la largeur et la hauteur de la boite.
  \item Justifier que $x$ doit verifier $0 < x < 10$ (domaine de validite).
  \item Montrer que le volume de la boite est :
  \[
    V(x) = x(30 - 2x)(20 - 2x).
  \]
  \item Developper $V(x)$ et identifier les coefficients du polynome obtenu.
  \item Calculer $V(0)$, $V(5)$ et $V(10)$. Commenter.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-TD-FIL-A}

\textbf{Question 1.} Apres decoupe des carres de cote $x$ et pliage :
\begin{itemize}
  \item Hauteur : $x$~cm.
  \item Longueur : $30 - 2x$~cm.
  \item Largeur : $20 - 2x$~cm.
\end{itemize}

\textbf{Question 2.} Pour que les dimensions soient strictement positives :
$x > 0$, $30 - 2x > 0 \Rightarrow x < 15$, $20 - 2x > 0 \Rightarrow x < 10$.
La contrainte la plus restrictive est $x < 10$, d'ou $x \in \,]0\,;\,10[$.

\textbf{Question 3.} Volume = longueur $\times$ largeur $\times$ hauteur :
$V(x) = x \times (30-2x) \times (20-2x)$.

\textbf{Question 4.}
\begin{align*}
V(x) &= x(30-2x)(20-2x) \\
&= x\bigl[30 \times 20 - 30 \times 2x - 2x \times 20 + 4x^2\bigr] \\
&= x\bigl[600 - 60x - 40x + 4x^2\bigr] \\
&= x(4x^2 - 100x + 600) \\
&= 4x^3 - 100x^2 + 600x.
\end{align*}

C'est un polynome de degre $3$ (pas du second degre directement).

\textbf{Question 5.}
\[
V(0) = 0 \text{ cm}^3 \quad (\text{pas de hauteur}),
\]
\[
V(5) = 5 \times 20 \times 10 = 1\,000 \text{ cm}^3,
\]
\[
V(10) = 10 \times 10 \times 0 = 0 \text{ cm}^3 \quad (\text{pas de largeur}).
\]
Le volume est nul aux extremites du domaine et positif a l'interieur.

\end{corrige}

\bigskip

\subsection*{Partie B — Etude sur un sous-domaine (approche second degre)}

\begin{exercice}{1SPE-SECDEG-TD-FIL-B}{2}{12}
On note $W(x) = \dfrac{V(x)}{x} = (30-2x)(20-2x)$ pour $x \in \,]0\,;\,10[$.
\begin{enumerate}
  \item Montrer que $W(x) = 4x^2 - 100x + 600$.
  \item Calculer l'abscisse $\alpha$ et l'ordonnee $\beta$ du sommet de la parabole representant $W$.
  \item La valeur $\alpha$ appartient-elle au domaine $]0\,;\,10[$ ?
    Que represente $W(\alpha)$ en termes de volume ?
  \item Dresser le tableau de variations de $W$ sur $]0\,;\,10[$.
  \item En deduire les valeurs de $x$ pour lesquelles $W(x)$ est croissante ou decroissante.
    Interpreter geometriquement.
\end{enumerate}
\end{exercice}

\begin{corrige}{1SPE-SECDEG-TD-FIL-B}

\textbf{Question 1.}
\[
W(x) = (30-2x)(20-2x) = 600 - 60x - 40x + 4x^2 = 4x^2 - 100x + 600.
\]

\textbf{Question 2.} $a = 4$, $b = -100$.
\[
\alpha = \frac{-(-100)}{2 \times 4} = \frac{100}{8} = \frac{25}{2} = 12{,}5.
\]
$\beta = W(12{,}5) = 4 \times 156{,}25 - 100 \times 12{,}5 + 600 = 625 - 1250 + 600 = -25$.

\textbf{Question 3.} $\alpha = 12{,}5 \notin \,]0\,;\,10[$ : le sommet de $W$ est hors du domaine. Sur $]0\,;\,10[$, on est toujours a gauche du sommet (puisque $\alpha = 12{,}5 > 10$). Comme $a = 4 > 0$, $W$ est \textbf{decroissante} sur tout $]0\,;\,10[$.

$W(\alpha) = -25 < 0$ n'a pas de sens physique (surface negative) : le sommet est hors du domaine de validite.

\textbf{Question 4. Tableau de variations de $W$ sur $]0\,;\,10[$ :}

\begin{center}
\begin{tabular}{|c|ccc|}
\hline
$x$ & $0$ & & $10$ \\
\hline
$W(x)$ & $600$ & $\searrow$ & $0$ \\
\hline
\end{tabular}
\end{center}

\textbf{Question 5.} $W$ est strictement decroissante sur $]0\,;\,10[$ : plus on decoupe un grand carre (grande hauteur $x$), plus la base de la boite est petite. Le volume $V(x) = x \cdot W(x)$ resulte d'un compromis entre hauteur et surface de la base.

\end{corrige}

\textbf{Question 2.} Le maximum semble se situer entre $x = 3$ et $x = 5$, avec le plus grand volume observe en $x = 4$ ($V(4) = 1\,056$).

\textbf{Question 3.} $V(3{,}5) = 3{,}5 \times 23 \times 13 = 3{,}5 \times 299 = 1\,046{,}5$~cm$^3$.
$V(4) = 1\,056 > 1\,046{,}5 = V(3{,}5)$ : pour $x = 4$, le volume est plus grand.

\commentaireMarge{La valeur exacte du maximum (par derivation en Terminale) est $x = \frac{50 - \sqrt{700}}{6} \approx 3{,}92$~cm.}

\textbf{Question 4. Bilan :}
\begin{itemize}
  \item \textbf{Forme developpee} : $V(x) = 4x^3 - 100x^2 + 600x$ (polynome de degre 3).
  \item \textbf{Forme factorisee} : $V(x) = x(30-2x)(20-2x)$ — permet de lire le domaine.
  \item \textbf{Polynome du second degre} : $W(x) = (30-2x)(20-2x)$ — etude de la base.
  \item Outils mobilises : formes d'un polynome, sommet, tableau de variations, domaine de validite, etude numerique.
\end{itemize}

\textbf{Question 5.} Le volume $V(x) = x(30-2x)(20-2x)$ est un produit de trois facteurs lineaires, donc un polynome de \textbf{degre 3}. Pour l'optimiser directement, il faudrait utiliser la derivation (programme de Terminale). Dans ce chapitre, on se contente d'une approche numerique ou d'une etude de la fonction auxiliaire $W(x) = V(x)/x$ (polynome du second degre).

\end{corrige}
